20 !> @brief This programs tests mpp_copy_domains and makes sure the domain was

56 subroutine compare_domains(Domain, Domain2, supergrid)
57  type(domain2d), intent(inout) :: Domain !< Domain
58  type(domain2d), intent(inout) :: Domain2 !< Domain
59  logical, intent(in) :: supergrid !< Flag indicating if comparing supergrid
60 
61  integer, allocatable :: pe_indices (:,:) !< Indices for current pe
62  integer, allocatable :: pe_indices2(:,:) !< Indices for current pe
63  integer, allocatable :: pes_indices(:,:) !< Indices for all pes in the pelist
64  integer, allocatable :: pes_indices2(:,:) !< Indices for all pes in the pelist
65 
66  allocate(pes_indices(mpp_npes(),6))
67  allocate(pes_indices2(mpp_npes(),6))
68  allocate(pe_indices(1,6))
69  allocate(pe_indices2(1,6))
70 
71  !< Check if the pe_indices are the same for the two domains
72  !> Compute Domain:
73  call mpp_get_compute_domain(domain, xbegin=pe_indices(1,1), xend=pe_indices(1,2), &
74  & ybegin=pe_indices(1,3), yend=pe_indices(1,4), &
75  & xsize =pe_indices(1,5), ysize=pe_indices(1,6))
76  if(supergrid) call get_expected_indices(pe_indices)
77 
78  call mpp_get_compute_domain(domain2, xbegin=pe_indices2(1,1), xend=pe_indices2(1,2), &
79  & ybegin=pe_indices2(1,3), yend=pe_indices2(1,4), &
80  & xsize =pe_indices2(1,5), ysize=pe_indices2(1,6))
81  call compare_pe_indices(pe_indices, pe_indices2, "compute domain")
82 
83  !> Data Domain:
84  call mpp_get_data_domain(domain, xbegin=pe_indices(1,1), xend=pe_indices(1,2), &
85  & ybegin=pe_indices(1,3), yend=pe_indices(1,4), &
86  & xsize =pe_indices(1,5), ysize=pe_indices(1,6))
87  if(supergrid) call get_expected_indices(pe_indices)
88 
89  call mpp_get_data_domain(domain2, xbegin=pe_indices2(1,1), xend=pe_indices2(1,2), &
90  & ybegin=pe_indices2(1,3), yend=pe_indices2(1,4), &
91  & xsize =pe_indices2(1,5), ysize=pe_indices2(1,6))
92  call compare_pe_indices(pe_indices, pe_indices2, "data domain")
93 
94  !> Global Domain:
95  call mpp_get_global_domain(domain, xbegin=pe_indices(1,1), xend=pe_indices(1,2), &
96  & ybegin=pe_indices(1,3), yend=pe_indices(1,4), &
97  & xsize =pe_indices(1,5), ysize=pe_indices(1,6))
98  if(supergrid) call get_expected_indices(pe_indices)
99 
100  call mpp_get_global_domain(domain2, xbegin=pe_indices2(1,1), xend=pe_indices2(1,2), &
101  & ybegin=pe_indices2(1,3), yend=pe_indices2(1,4), &
102  & xsize =pe_indices2(1,5), ysize=pe_indices2(1,6))
103  call compare_pe_indices(pe_indices, pe_indices2, "global domain")
104 
105  !> Compute Domains (get the indices for all of the ranks)
106  call mpp_get_compute_domains(domain, xbegin=pes_indices(:,1), xend=pes_indices(:,2), &
107  & ybegin=pes_indices(:,3), yend=pes_indices(:,4), &
108  & xsize =pes_indices(:,5), ysize=pes_indices(:,6))
109  if(supergrid) call get_expected_indices(pes_indices)
110 
111  call mpp_get_compute_domains(domain2, xbegin=pes_indices2(:,1), xend=pes_indices2(:,2), &
112  & ybegin=pes_indices2(:,3), yend=pes_indices2(:,4), &
113  & xsize =pes_indices2(:,5), ysize=pes_indices2(:,6))
114  call compare_pe_indices(pes_indices, pes_indices2, "compute domains")
115 
116  !> Global Domains (get the indices for all of the ranks)
117  call mpp_get_global_domains(domain, xbegin=pes_indices(:,1), xend=pes_indices(:,2), &
118  & ybegin=pes_indices(:,3), yend=pes_indices(:,4), &
119  & xsize =pes_indices(:,5), ysize=pes_indices(:,6))
120  if(supergrid) call get_expected_indices(pes_indices)
121 
122  call mpp_get_global_domains(domain2, xbegin=pes_indices2(:,1), xend=pes_indices2(:,2), &
123  & ybegin=pes_indices2(:,3), yend=pes_indices2(:,4), &
124  & xsize =pes_indices2(:,5), ysize=pes_indices2(:,6))
125  call compare_pe_indices(pes_indices, pes_indices2, "global domains")
126 
127 end subroutine compare_domains

129 subroutine get_expected_indices(pe_indices)
130  integer, intent(inout) :: pe_indices(:,:) !< Indices from original domain
131 
132  integer :: i !< For do loop
133 
134  do i=1, size(pe_indices,1)
135  pe_indices(i,1) = 2*pe_indices(i,1)-1 !< xbegin
136  pe_indices(i,2) = 2*pe_indices(i,2)+1 !< xend
137  pe_indices(i,3) = 2*pe_indices(i,3)-1 !< ybegin
138  pe_indices(i,4) = 2*pe_indices(i,4)+1 !< yend
139  pe_indices(i,5) = pe_indices(i,2) - pe_indices(i,1) +1 !< xsize
140  pe_indices(i,6) = pe_indices(i,4) - pe_indices(i,3) +1 !< ysize
141  enddo
142 
143 end subroutine get_expected_indices
144 
145 subroutine compare_pe_indices(pe_indices, pe_indices2, domain_type)
146  integer, intent(in) :: pe_indices(:,:) !< Indices from original domain
147  integer, intent(in) :: pe_indices2(:,:) !< Indices from new domain
148  character(len=*), intent(in) :: domain_type !< Type of domain "data", "compute", and "global"
149 
150  integer :: i,j !< For do loop
151 
152  do i = 1, size(pe_indices, 1)
153  do j = 1, size(pe_indices,2)
154  if (pe_indices(i, j) .ne. pe_indices2(i, j)) then
155  call mpp_error(fatal, "Compare pe_indices: the "//trim(domain_type)//" pe_indices are not the same ")
156  endif
157  enddo
158  enddo
159 end subroutine compare_pe_indices
