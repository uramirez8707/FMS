25 !> @author Miguel Zuniga
26 !!
27 !! <TT>fms_diag_dlinked_list_mod</TT> defines a generic doubly linked
28 !! list class and an iterator class for traversing the list. It is
29 !! generic in the sense that the elements or objects it contains are
30 !! "class(*)" objects. If additional typecheking or psossibly a
31 !! slightly different user interface is desired, consider creating
32 !! a wrapper or another class with this one for a memeber element and
33 !! procedures that are trivially implemeted by using this class.
34 !!
35 !! This version is roughly a fortran translation of the C++ doubly linked list
36 !! class in the book ``Data Structures And Algorithm Analysis in C++", 3rd Edition,
37 !! by Mark Allen Weiss.

103  !> @brief Insert data d in a new node to be placed in front of the
104  !! target node t_nd.
105  !! @return Returns an iterator that starts with the newly inserted node.
106  function insert_data( this, t_nd, d ) result(liter)
107  class(fmsdllist_t), intent(in out) :: this !<The instance of the class that this function is bound to.
108  type(fmsdllistnode_t), pointer, intent (in) :: t_nd !< The target node.
109  class(*), target, intent(in) :: d !< The data to insert.
110  class(fmsdlliterator_t), allocatable :: liter !< A linked list iterator.
111  type(fmsdllistnode_t), pointer :: nd !< The new node that is to "hold" the data.
112  allocate(nd)
113  nd%data_ptr => d
114  !! Insert nd into list so that list section [prev node <--> target node ] looks like
115  !! [prev node <--> new nd <--> target node]. The four pointers pointing to and/or
116  !! from "new nd" need to be set. Therefore :
117  !! a) The new nd's prev needs to be whatever was the targets prev:
118  nd%prev => t_nd%prev
119  !! b) New node nd's next is obviously the target node:
120  nd%next => t_nd
121  !! c) the next of the prev node needs to point to the new node nd:
122  t_nd%prev%next => nd
123  !! d) target node's prev needs to point to the new node :
124  t_nd%prev => nd
125  this%the_size = this%the_size + 1
126  liter = fmsdlliterator_t(nd, this%tail)
127  end function insert_data
128 
129  !> @brief Remove Node nd from the linked tree.
130  !! @return Return the iterator that begins with the next node after nd, and ends with
131  !! the list end node. Returns the list iterator if the node cannot be removed.
132  function remove_node( this, nd ) result( litr)
133  class(fmsdllist_t), intent(in out) :: this !<The instance of the class that this function is bound to.
134  type(fmsdllistnode_t), pointer, intent(in out) :: nd !< The node to remove from the list.
135  class(fmsdlliterator_t), ALLOCATABLE :: litr !< The iterator starting from the node that was
136  !< following the removed node.
137  !Dont even try to remove the head and tail nodes!
138  if (.not. ( associated (this%head , nd ) .or. &
139  associated (this%tail , nd ) )) THEN
140  litr = fmsdlliterator_t( nd%next, this%tail )
141  nd%prev%next => nd%next
142  nd%next%prev => nd%prev
143  deallocate(nd)
144  this%the_size = this%the_size - 1
145  else
146  litr = this%get_literator()
147  endif
148  end function remove_node

298  logical, optional, intent(in) :: data_dealloc_flag !< If not present or .true., client data is deallocated.

315  'In clear_all; linked node contains node with unassociated data pointer', &
316  warning)
317  else

320  endif
321  end do
