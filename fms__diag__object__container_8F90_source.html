22 !> @brief fms_diag_object_container_mod defines a container class and iterator class
23 !! for inserting, removing and searching for <TT>fmsDiagField_type</TT> instances
24 !!
25 !> @author Miguel Zuniga
26 !!
27 !! <TT>fms_diag_object_container_mod</TT> defines a container for inserting, removing and
28 !! searching for <TT>fmsDiagField_type</TT> instances. It also defined an iterator for
29 !! the data in the container. The value returned by the fms_diag_object function get_id()
30 !! is used for search key comparison.
31 !!
32 !! Most of the functions in class FmsDiagObjectContainer_t are simple wrappers over
33 !! those of the underlying <TT>fms_doubly_linked_list_mod</TT> class. The find/search
34 !! are a little more than that, and what FmsDiagObjectContainer_t provides over the
35 !! underlying liked list is the search function, type checking, convenience, and a
36 !! fixed user interface defined for the intended use.
37 !!

94  !> @brief Returns an empty iterator if a diag object with this ID was not found.
95  !! If the diag object was found, return an iterator with the current object being
96  !! the found object, ad the last/anchor being the last/anchor of the container.
97  !! Note that this routine can accept an optional iterator as input, which
98  !! is useful for chaining searches, which may be needed if there are key duplicates.
99  !! @return In iterator that starts from the inserted object.
100  function find_diag_object (this, id , iiter) result (riter)
101  class(fmsdiagobjectcontainer_t), intent (in out) :: this
102  !<The instance of the class that this function is bound to.
103  integer , intent (in) :: id !< The id of the object to find.
104  class(fmsdiagobjiterator_t), intent (in), optional :: iiter
105  !< An (optional) iterator over the searchable set.
106  class(fmsdiagobjiterator_t) , allocatable :: riter !< The resultant iterator to the object.
107  class(fmsdiagfield_type), pointer:: ptdo !< A pointer to temporaty diagnostic object
108  integer :: status !< A status from iterator operations.
109  !!
110  if(present (iiter)) then
111  riter = iiter
112  else
113  riter = this%iterator() !!An iterator over the entire container
114  endif
115  do while( riter%has_data() .eqv. .true.)
116  ptdo => riter%get()
117  if(id == ptdo%get_id() ) then
118  EXIT
119  end if
120  status = riter%next()
121  end do
122  end function find_diag_object
123 
124  !> @brief insert diagnostic object obj with given id.
125  !! Objects are inserted at the back / end of the list
126  !! This version of the container also enforces that the
127  !! objects ID is equal the input id.
128  !! @return A status of -1 if there was an error, and 0 otherwise.
129  function insert_diag_object (this, id, obj) result (status)
130  class(fmsdiagobjectcontainer_t), intent (in out) :: this
131  integer, intent (in) :: id !< The id of the object to insert.
132  class(fmsdiagfield_type) , intent (in out) :: obj !< The object to insert
133  integer :: status !< The returned status. 0 for success.
134  class(fmsdlliterator_t), allocatable :: tliter !< A temporary iterator.
135 
136  status = -1
137  if ( id .ne. obj%get_id() ) then
138  !!TODO: log error
139  endif
140  tliter = this%the_linked_list%push_back( obj )
141  if(tliter%has_data() .eqv. .true. ) then
142  status = 0
143  endif
144  end function

146  !> @brief Remove and return the first object in the container with the corresponding id .
147  !! Note that if the client code does not already have a reference to the object being
148  !! removed, then the client may want to to use procedure find before using procedure remove.
149  !! If procedure find is used, consider calling remove with the iterator returned from find.
150  !! @return In iterator starting from the node that was following the removed node.
151  function remove_diag_object (this, id, iiter ) result (riter)
152  class(fmsdiagobjectcontainer_t), intent (in out) :: this
153  !<The instance of the class that this function is bound to.
154  integer , intent (in) :: id !< The id of the object to remove.
155  class(fmsdiagobjiterator_t), intent (in), optional :: iiter
156  !< An (optional) iterator over the searchable set.
157  class(fmsdiagobjiterator_t), allocatable:: riter !< The resultant iterator
158  class(fmsdlliterator_t), allocatable :: temp_liter !< A temporary iterator
159  type(fmsdllistnode_t), pointer :: pn
160  if(present (iiter)) then
161  riter = iiter
162  else
163  riter = this%iterator() !!An iterator over the entire container
164  endif
165  !Find the object in the container.
166  riter = this%find ( id , riter)
167  pn => riter%liter%get_current_node_pointer()
168  temp_liter = this%the_linked_list%remove( pn )
169  riter = fmsdiagobjiterator_t(temp_liter)
170  end function

242  !> @brief Get the current data the iterator is pointing to.
243  !! Note the common use case is to call function has_data to decide if
244  !! this function should be called (again).
245  !! @return Returns a pointer to the current data.
246  function literator_data( this ) result( rdo )
247  class(fmsdiagobjiterator_t), intent(in) :: this

249  class(fmsdiagfield_type), pointer :: rdo !< The resultant diagnostic object.
250  class(*), pointer :: gp !< A eneric typed object in the container.

252  rdo => null()
