157 !> @brief Registers a field.
158 !! @description This to avoid having duplicate code in each of the _scalar, _array and _static register calls
159 !! @return field index for subsequent call to send_data.
160 integer function fms_register_diag_field_obj &
161  (this, modname, varname, axes, init_time, &
162  longname, units, missing_value, varrange, mask_variant, standname, &
163  do_not_log, err_msg, interp_method, tile_count, area, volume, realm, static)
164 
165  class(fmsdiagobject_type),TARGET,INTENT(inout):: this !< Diaj_obj to fill
166  CHARACTER(len=*), INTENT(in) :: modname !< The module name
167  CHARACTER(len=*), INTENT(in) :: varname !< The variable name
168  TYPE(time_type), OPTIONAL, INTENT(in) :: init_time !< Initial time
169  INTEGER, TARGET, OPTIONAL, INTENT(in) :: axes(:) !< The axes indicies
170  CHARACTER(len=*), OPTIONAL, INTENT(in) :: longname !< THe variables long name
171  CHARACTER(len=*), OPTIONAL, INTENT(in) :: units !< The units of the variables
172  CHARACTER(len=*), OPTIONAL, INTENT(in) :: standname !< The variables stanard name
173  class(*), OPTIONAL, INTENT(in) :: missing_value !< Missing value to add as a attribute
174  class(*), OPTIONAL, INTENT(in) :: varrange(2) !< Range to add as a attribute
175  LOGICAL, OPTIONAL, INTENT(in) :: mask_variant !< Mask
176  LOGICAL, OPTIONAL, INTENT(in) :: do_not_log !< if TRUE, field info is not logged
177  CHARACTER(len=*), OPTIONAL, INTENT(out) :: err_msg !< Error message to be passed back up
178  CHARACTER(len=*), OPTIONAL, INTENT(in) :: interp_method !< The interp method to be used when
179  !! regridding the field in post-processing.
180  !! Valid options are "conserve_order1",
181  !! "conserve_order2", and "none".
182  INTEGER, OPTIONAL, INTENT(in) :: tile_count !< the number of tiles
183  INTEGER, OPTIONAL, INTENT(in) :: area !< diag_field_id of the cell area field
184  INTEGER, OPTIONAL, INTENT(in) :: volume !< diag_field_id of the cell volume field
185  CHARACTER(len=*), OPTIONAL, INTENT(in) :: realm !< String to set as the value to the
186  !! modeling_realm attribute
187  LOGICAL, OPTIONAL, INTENT(in) :: static !< True if the variable is static
188 #ifdef use_yaml
189 
190  class(fmsdiagfile_type), pointer :: fileptr => null() !< Pointer to the diag_file
191  class(fmsdiagfield_type), pointer :: fieldptr => null() !< Pointer to the diag_field
192  integer, allocatable :: file_ids(:) !< The file IDs for this variable
193  integer :: i !< For do loops
194  integer, allocatable :: diag_field_indices(:) !< indices where the field was found in the yaml
195 #endif
196 #ifndef use_yaml
197 fms_register_diag_field_obj = diag_field_not_found
198 CALL mpp_error(fatal,"You can not use the modern diag manager without compiling with -Duse_yaml")
199 #else
200  diag_field_indices = find_diag_field(varname, modname)
201  if (diag_field_indices(1) .eq. diag_null) then
202  !< The field was not found in the table, so return diag_null
203  fms_register_diag_field_obj = diag_null
204  deallocate(diag_field_indices)
205  return
206  endif
207 
208  this%registered_variables = this%registered_variables + 1
209  fms_register_diag_field_obj = this%registered_variables
210 
211  call this%FMS_diag_fields(this%registered_variables)%&
212  &setid(this%registered_variables)
213 
214 !> Use pointers for convenience
215  fieldptr => this%FMS_diag_fields(this%registered_variables)
216 !> Register the data for the field
217  call fieldptr%register(modname, varname, diag_field_indices, fms_diag_object%diag_axis, &
218  axes=axes, longname=longname, units=units, missing_value=missing_value, varrange= varrange, &
219  mask_variant= mask_variant, standname=standname, do_not_log=do_not_log, err_msg=err_msg, &
220  interp_method=interp_method, tile_count=tile_count, area=area, volume=volume, realm=realm, &
221  static=static)
222 !> Get the file IDs from the field indicies from the yaml
223  file_ids = get_diag_files_id(diag_field_indices)
224  call fieldptr%set_file_ids(file_ids)
225 !> Add the axis information, initial time, and field IDs to the files
226  if (present(axes) .and. present(init_time)) then
227  do i = 1, size(file_ids)
228  fileptr => this%FMS_diag_files(file_ids(i))%FMS_diag_file
229  call fileptr%add_field_and_yaml_id(fieldptr%get_id(), diag_field_indices(i))
230  call fileptr%set_file_domain(fieldptr%get_domain(), fieldptr%get_type_of_domain())
231  call fileptr%init_diurnal_axis(this%diag_axis, this%registered_axis, diag_field_indices(i))
232  call fileptr%add_axes(axes, this%diag_axis, this%registered_axis, diag_field_indices(i))
233  call fileptr%add_start_time(init_time, this%current_model_time)
234  call fileptr%set_file_time_ops (fieldptr%diag_field(i), fieldptr%is_static())
235  enddo
236  elseif (present(axes)) then !only axes present
237  do i = 1, size(file_ids)
238  fileptr => this%FMS_diag_files(file_ids(i))%FMS_diag_file
239  call fileptr%add_field_and_yaml_id(fieldptr%get_id(), diag_field_indices(i))
240  call fileptr%init_diurnal_axis(this%diag_axis, this%registered_axis, diag_field_indices(i))
241  call fileptr%set_file_domain(fieldptr%get_domain(), fieldptr%get_type_of_domain())
242  call fileptr%add_axes(axes, this%diag_axis, this%registered_axis, diag_field_indices(i))
243  call fileptr%set_file_time_ops (fieldptr%diag_field(i), fieldptr%is_static())
244  enddo
245  elseif (present(init_time)) then !only inti time present
246  do i = 1, size(file_ids)
247  fileptr => this%FMS_diag_files(file_ids(i))%FMS_diag_file
248  call fileptr%add_field_and_yaml_id(fieldptr%get_id(), diag_field_indices(i))
249  call fileptr%add_start_time(init_time, this%current_model_time)
250  call fileptr%set_file_time_ops (fieldptr%diag_field(i), fieldptr%is_static())
251  enddo
252  else !no axis or init time present
253  do i = 1, size(file_ids)
254  fileptr => this%FMS_diag_files(file_ids(i))%FMS_diag_file
255  call fileptr%add_field_and_yaml_id(fieldptr%get_id(), diag_field_indices(i))
256  call fileptr%set_file_time_ops (fieldptr%diag_field(i), fieldptr%is_static())
257  enddo
258  endif
259  nullify (fileptr)
260  nullify (fieldptr)
261  deallocate(diag_field_indices)
262 #endif
263 end function fms_register_diag_field_obj
