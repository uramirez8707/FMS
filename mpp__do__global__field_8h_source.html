20  subroutine MPP_DO_GLOBAL_FIELD_3D_( domain, local, global, tile, ishift, jshift, flags, default_data)
21 !get a global field from a local field
22 !local field may be on compute OR data domain
23  type(domain2D), intent(in) :: domain
24  MPP_TYPE_, intent(in) :: local(:,:,:)
25  integer, intent(in) :: tile, ishift, jshift
26  MPP_TYPE_, intent(out) :: global(domain%x(tile)%global%begin:,domain%y(tile)%global%begin:,:)
27  integer, intent(in), optional :: flags
28  MPP_TYPE_, intent(in), optional :: default_data
29 
30  integer :: i, j, k, m, n, nd, nwords, lpos, rpos, ioff, joff, from_pe, root_pe, tile_id
31  integer :: ke, isc, iec, jsc, jec, is, ie, js, je, nword_me
32  integer :: ipos, jpos
33  logical :: xonly, yonly, root_only, global_on_this_pe
34  MPP_TYPE_ :: clocal ((domain%x(1)%compute%size+ishift) *(domain%y(1)%compute%size+jshift) *size(local,3))
35  MPP_TYPE_ :: cremote((domain%x(1)%compute%max_size+ishift)*(domain%y(1)%compute%max_size+jshift)*size(local,3))
36  integer :: stackuse
37  character(len=8) :: text
38 
39  pointer( ptr_local, clocal )
40  pointer( ptr_remote, cremote )
41 
42  stackuse = size(clocal(:))+size(cremote(:))
43  if( stackuse.GT.mpp_domains_stack_size )then
44  write( text, '(i8)' )stackuse
45  call mpp_error( FATAL, &
46  'MPP_DO_GLOBAL_FIELD user stack overflow: call mpp_domains_set_stack_size('//trim(text)//') from all PEs.' )
47  end if
48  mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, stackuse )
49 
50  ptr_local = LOC(mpp_domains_stack)
51  ptr_remote = LOC(mpp_domains_stack(size(clocal(:))+1))
52 
53  if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_GLOBAL_FIELD: must first call mpp_domains_init.' )
54 
55  xonly = .FALSE.
56  yonly = .FALSE.
57  root_only = .FALSE.
58  if( PRESENT(flags) ) then
59  xonly = BTEST(flags,EAST)
60  yonly = BTEST(flags,SOUTH)
61  if( .NOT.xonly .AND. .NOT.yonly )call mpp_error( WARNING, &
62  'MPP_GLOBAL_FIELD: you must have flags=XUPDATE, YUPDATE or XUPDATE+YUPDATE' )
63  if(xonly .AND. yonly) then
64  xonly = .false.; yonly = .false.
65  endif
66  root_only = BTEST(flags, ROOT_GLOBAL)
67  if( (xonly .or. yonly) .AND. root_only ) then
68  call mpp_error( WARNING, 'MPP_GLOBAL_FIELD: flags = XUPDATE+GLOBAL_ROOT_ONLY or ' // &
69  'flags = YUPDATE+GLOBAL_ROOT_ONLY is not supported, will ignore GLOBAL_ROOT_ONLY' )
70  root_only = .FALSE.
71  endif
72  endif
73 
74  global_on_this_pe = .NOT. root_only .OR. domain%pe == domain%tile_root_pe
75  ipos = 0; jpos = 0
76  if(global_on_this_pe ) then
77  if(size(local,3).NE.size(global,3) ) call mpp_error( FATAL, &
78  'MPP_GLOBAL_FIELD: mismatch of third dimension size of global and local')
79  if( size(global,1).NE.(domain%x(tile)%global%size+ishift) .OR. size(global,2).NE.(domain%y(tile)%global%size+jshift))then
80  if(xonly) then
81  if(size(global,1).NE.(domain%x(tile)%global%size+ishift) .OR. &
82  size(global,2).NE.(domain%y(tile)%compute%size+jshift)) &
83  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD: incoming arrays do not match domain for xonly global field.' )
84  jpos = -domain%y(tile)%compute%begin + 1
85  else if(yonly) then
86  if(size(global,1).NE.(domain%x(tile)%compute%size+ishift) .OR. &
87  size(global,2).NE.(domain%y(tile)%global%size+jshift)) &
88  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD: incoming arrays do not match domain for yonly global field.' )
89  ipos = -domain%x(tile)%compute%begin + 1
90  else
91  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD: incoming arrays do not match domain.' )
92  endif
93  endif
94  endif
95 
96  if( size(local,1).EQ.(domain%x(tile)%compute%size+ishift) .AND. size(local,2).EQ.(domain%y(tile)%compute%size+jshift) )then
97  !local is on compute domain
98  ioff = -domain%x(tile)%compute%begin + 1
99  joff = -domain%y(tile)%compute%begin + 1
100  else if( size(local,1).EQ.(domain%x(tile)%memory%size+ishift) .AND. size(local,2).EQ.(domain%y(tile)%memory%size+jshift) )then
101  !local is on data domain
102  ioff = -domain%x(tile)%data%begin + 1
103  joff = -domain%y(tile)%data%begin + 1
104  else
105  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD_: incoming field array must match either compute domain or memory domain.' )
106  end if
107 
108  ke = size(local,3)
109  isc = domain%x(tile)%compute%begin; iec = domain%x(tile)%compute%end+ishift
110  jsc = domain%y(tile)%compute%begin; jec = domain%y(tile)%compute%end+jshift
111 
112  nword_me = (iec-isc+1)*(jec-jsc+1)*ke
113 
114 ! make contiguous array from compute domain
115  m = 0
116  if(global_on_this_pe) then
117  !z1l: initialize global = 0 to support mask domain
118  if(PRESENT(default_data)) then
119  global = default_data
120  else
121 #ifdef LOGICAL_VARIABLE
122  global = .false.
123 #else
124  global = 0
125 #endif
126  endif
127 
128  do k = 1, ke
129  do j = jsc, jec
130  do i = isc, iec
131  m = m + 1
132  clocal(m) = local(i+ioff,j+joff,k)
133  global(i+ipos,j+jpos,k) = clocal(m) !always fill local domain directly
134  end do
135  end do
136  end do
137  else
138  do k = 1, ke
139  do j = jsc, jec
140  do i = isc, iec
141  m = m + 1
142  clocal(m) = local(i+ioff,j+joff,k)
143  end do
144  end do
145  end do
146  endif
147 
148 ! if there is more than one tile on this pe, then no decomposition for all tiles on this pe, so we can just return
149  if(size(domain%x(:))>1) then
150  !--- the following is needed to avoid deadlock.
151  if( tile == size(domain%x(:)) ) call mpp_sync_self( )
152  return
153  end if
154 
155  root_pe = mpp_root_pe()
156 
157 !fill off-domains (note loops begin at an offset of 1)
158  if( xonly )then
159  nd = size(domain%x(1)%list(:))
160  do n = 1,nd-1
161  lpos = mod(domain%x(1)%pos+nd-n,nd)
162  rpos = mod(domain%x(1)%pos +n,nd)
163  from_pe = domain%x(1)%list(rpos)%pe
164  rpos = from_pe - root_pe ! for concurrent run, root_pe may not be 0.
165  if (from_pe == NULL_PE) then
166  nwords = 0
167  else
168  nwords = (domain%list(rpos)%x(1)%compute%size+ishift) &
169  * (domain%list(rpos)%y(1)%compute%size+jshift) * ke
170  endif
171  ! Force use of scalar, integer ptr interface
172  call mpp_transmit( put_data=clocal(1), plen=nword_me, to_pe=domain%x(1)%list(lpos)%pe, &
173  get_data=cremote(1), glen=nwords, from_pe=from_pe )
174  m = 0
175  if (from_pe /= NULL_PE) then
176  is = domain%list(rpos)%x(1)%compute%begin; ie = domain%list(rpos)%x(1)%compute%end+ishift
177  do k = 1, ke
178  do j = jsc, jec
179  do i = is, ie
180  m = m + 1
181  global(i,j+jpos,k) = cremote(m)
182  end do
183  end do
184  end do
185  endif
186  call mpp_sync_self() !-ensure MPI_ISEND is done.
187  end do
188  else if( yonly )then
189  nd = size(domain%y(1)%list(:))
190  do n = 1,nd-1
191  lpos = mod(domain%y(1)%pos+nd-n,nd)
192  rpos = mod(domain%y(1)%pos +n,nd)
193  from_pe = domain%y(1)%list(rpos)%pe
194  rpos = from_pe - root_pe
195  if (from_pe == NULL_PE) then
196  nwords = 0
197  else
198  nwords = (domain%list(rpos)%x(1)%compute%size+ishift) &
199  * (domain%list(rpos)%y(1)%compute%size+jshift) * ke
200  endif
201  ! Force use of scalar, integer pointer interface
202  call mpp_transmit( put_data=clocal(1), plen=nword_me, to_pe=domain%y(1)%list(lpos)%pe, &
203  get_data=cremote(1), glen=nwords, from_pe=from_pe )
204  m = 0
205  if (from_pe /= NULL_PE) then
206  js = domain%list(rpos)%y(1)%compute%begin; je = domain%list(rpos)%y(1)%compute%end+jshift
207  do k = 1,ke
208  do j = js, je
209  do i = isc, iec
210  m = m + 1
211  global(i+ipos,j,k) = cremote(m)
212  end do
213  end do
214  end do
215  endif
216  call mpp_sync_self() !-ensure MPI_ISEND is done.
217  end do
218  else
219  tile_id = domain%tile_id(1)
220  nd = size(domain%list(:))
221  if(root_only) then
222  if(domain%pe .NE. domain%tile_root_pe) then
223  call mpp_send( clocal(1), plen=nword_me, to_pe=domain%tile_root_pe, tag=COMM_TAG_1 )
224  else
225  do n = 1,nd-1
226  rpos = mod(domain%pos+n,nd)
227  if( domain%list(rpos)%tile_id(1) .NE. tile_id ) cycle
228  nwords = (domain%list(rpos)%x(1)%compute%size+ishift) * (domain%list(rpos)%y(1)%compute%size+jshift) * ke
229  call mpp_recv(cremote(1), glen=nwords, from_pe=domain%list(rpos)%pe, tag=COMM_TAG_1 )
230  m = 0
231  is = domain%list(rpos)%x(1)%compute%begin; ie = domain%list(rpos)%x(1)%compute%end+ishift
232  js = domain%list(rpos)%y(1)%compute%begin; je = domain%list(rpos)%y(1)%compute%end+jshift
233 
234  do k = 1,ke
235  do j = js, je
236  do i = is, ie
237  m = m + 1
238  global(i,j,k) = cremote(m)
239  end do
240  end do
241  end do
242  end do
243  endif
244  else
245  do n = 1,nd-1
246  lpos = mod(domain%pos+nd-n,nd)
247  if( domain%list(lpos)%tile_id(1).NE. tile_id ) cycle ! global field only within tile
248  call mpp_send( clocal(1), plen=nword_me, to_pe=domain%list(lpos)%pe, tag=COMM_TAG_2 )
249  end do
250  do n = 1,nd-1
251  rpos = mod(domain%pos+n,nd)
252  if( domain%list(rpos)%tile_id(1) .NE. tile_id ) cycle ! global field only within tile
253  nwords = (domain%list(rpos)%x(1)%compute%size+ishift) * (domain%list(rpos)%y(1)%compute%size+jshift) * ke
254  call mpp_recv( cremote(1), glen=nwords, from_pe=domain%list(rpos)%pe, tag=COMM_TAG_2 )
255  m = 0
256  is = domain%list(rpos)%x(1)%compute%begin; ie = domain%list(rpos)%x(1)%compute%end+ishift
257  js = domain%list(rpos)%y(1)%compute%begin; je = domain%list(rpos)%y(1)%compute%end+jshift
258 
259  do k = 1,ke
260  do j = js, je
261  do i = is, ie
262  m = m + 1
263  global(i,j,k) = cremote(m)
264  end do
265  end do
266  end do
267  end do
268  endif
269  end if
270 
271  call mpp_sync_self()
272 
273  return
274  end subroutine MPP_DO_GLOBAL_FIELD_3D_

277  subroutine MPP_DO_GLOBAL_FIELD_A2A_3D_( domain, local, global, tile, ishift, jshift, flags, default_data)
278 !get a global field from a local field
279 !local field may be on compute OR data domain
280  type(domain2D), intent(in) :: domain
281  integer, intent(in) :: tile, ishift, jshift
282  MPP_TYPE_, intent(in), contiguous, target :: local(:,:,:)
283  MPP_TYPE_, intent(out), contiguous, target :: global(domain%x(tile)%global%begin:,domain%y(tile)%global%begin:,:)
284  integer, intent(in), optional :: flags
285  MPP_TYPE_, intent(in), optional :: default_data
286 
287  integer :: i, j, k, m, n, nd, nwords, lpos, rpos, ioff, joff, from_pe, root_pe, tile_id
288  integer :: ke, isc, iec, jsc, jec, is, ie, js, je
289  integer :: ipos, jpos
290  logical :: xonly, yonly, root_only, global_on_this_pe
291 
292  ! Alltoallw vectors
293  MPP_TYPE_, dimension(:), pointer :: plocal, pglobal
294 
295  integer, dimension(:), allocatable :: sendcounts(:), recvcounts(:)
296  integer, dimension(:), allocatable :: sdispls(:), rdispls(:)
297  type(mpp_type), allocatable :: sendtypes(:), recvtypes(:)
298  integer, dimension(3) :: array_of_subsizes, array_of_starts
299  integer :: n_sends, n_ax, pe
300  integer :: isg, jsg
301  integer, allocatable :: pelist(:), axis_pelist(:), pelist_idx(:)
302 
303  if (.NOT.module_is_initialized) &
304  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD: must first call mpp_domains_init.' )
305 
306  ! Validate flag consistency and configure the function
307  xonly = .FALSE.
308  yonly = .FALSE.
309  root_only = .FALSE.
310  if( PRESENT(flags) ) then
311  xonly = BTEST(flags,EAST)
312  yonly = BTEST(flags,SOUTH)
313  if( .NOT.xonly .AND. .NOT.yonly )call mpp_error( WARNING, &
314  'MPP_GLOBAL_FIELD: you must have flags=XUPDATE, YUPDATE or XUPDATE+YUPDATE' )
315  if(xonly .AND. yonly) then
316  xonly = .false.; yonly = .false.
317  endif
318  root_only = BTEST(flags, ROOT_GLOBAL)
319  if( (xonly .or. yonly) .AND. root_only ) then
320  call mpp_error( WARNING, 'MPP_GLOBAL_FIELD: flags = XUPDATE+GLOBAL_ROOT_ONLY or ' // &
321  'flags = YUPDATE+GLOBAL_ROOT_ONLY is not supported, will ignore GLOBAL_ROOT_ONLY' )
322  root_only = .FALSE.
323  endif
324  endif
325 
326  global_on_this_pe = .NOT. root_only .OR. domain%pe == domain%tile_root_pe
327 
328  ! Calculate offset for truncated global fields
329  ! NOTE: We do not check contiguity of global subarrays, and assume that
330  ! they have been copied to a contigous array.
331  ipos = 0; jpos = 0
332  if(global_on_this_pe ) then
333  if(size(local,3).NE.size(global,3) ) call mpp_error( FATAL, &
334  'MPP_GLOBAL_FIELD: mismatch of third dimension size of global and local')
335  if( size(global,1).NE.(domain%x(tile)%global%size+ishift) .OR. size(global,2).NE.(domain%y(tile)%global%size+jshift))then
336  if(xonly) then
337  if(size(global,1).NE.(domain%x(tile)%global%size+ishift) .OR. &
338  size(global,2).NE.(domain%y(tile)%compute%size+jshift)) &
339  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD: incoming arrays do not match domain for xonly global field.' )
340  jpos = -domain%y(tile)%compute%begin + 1
341  else if(yonly) then
342  if(size(global,1).NE.(domain%x(tile)%compute%size+ishift) .OR. &
343  size(global,2).NE.(domain%y(tile)%global%size+jshift)) &
344  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD: incoming arrays do not match domain for yonly global field.' )
345  ipos = -domain%x(tile)%compute%begin + 1
346  else
347  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD: incoming arrays do not match domain.' )
348  endif
349  endif
350  endif
351 
352  ! NOTE: Since local is assumed to contiguously match the data domain, this
353  ! is not a useful check. But maybe someday we can support compute
354  ! domains.
355  if( size(local,1).EQ.(domain%x(tile)%compute%size+ishift) .AND. size(local,2).EQ.(domain%y(tile)%compute%size+jshift) )then
356  !local is on compute domain
357  ioff = -domain%x(tile)%compute%begin
358  joff = -domain%y(tile)%compute%begin
359  else if( size(local,1).EQ.(domain%x(tile)%memory%size+ishift) .AND. size(local,2).EQ.(domain%y(tile)%memory%size+jshift) )then
360  !local is on data domain
361  ioff = -domain%x(tile)%data%begin
362  joff = -domain%y(tile)%data%begin
363  else
364  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD_: incoming field array must match either compute domain or memory domain.' )
365  end if
366 
367  ke = size(local,3)
368  isc = domain%x(tile)%compute%begin; iec = domain%x(tile)%compute%end+ishift
369  jsc = domain%y(tile)%compute%begin; jec = domain%y(tile)%compute%end+jshift
370  isg = domain%x(1)%global%begin; jsg = domain%y(1)%global%begin
371 
372  if(global_on_this_pe) then
373  !z1l: initialize global = 0 to support mask domain
374  if(PRESENT(default_data)) then
375  global = default_data
376  else
377 #ifdef LOGICAL_VARIABLE
378  global = .false.
379 #else
380  global = 0
381 #endif
382  endif
383  endif
384 
385  ! if there is more than one tile on this pe, then no decomposition for
386  ! all tiles on this pe, so we can just return
387  if(size(domain%x(:))>1) then
388  !--- the following is needed to avoid deadlock.
389  if( tile == size(domain%x(:)) ) call mpp_sync_self( )
390  return
391  end if
392 
393  root_pe = mpp_root_pe()
394 
395  ! Generate the pelist
396  ! TODO: Add these to the domain API
397  if (xonly) then
398  n_ax = size(domain%x(1)%list(:))
399  allocate(axis_pelist(n_ax))
400  axis_pelist = [ (domain%x(1)%list(i)%pe, i = 0, n_ax-1) ]
401 
402  nd = count(axis_pelist >= 0)
403  allocate(pelist(nd), pelist_idx(0:nd-1))
404  pelist = pack(axis_pelist, mask=(axis_pelist >= 0))
405  pelist_idx = pack([(i, i=0, n_ax-1)], mask=(axis_pelist >= 0))
406 
407  deallocate(axis_pelist)
408  else if (yonly) then
409  n_ax = size(domain%y(1)%list(:))
410  allocate(axis_pelist(n_ax))
411  axis_pelist = [ (domain%y(1)%list(i)%pe, i = 0, n_ax-1) ]
412 
413  nd = count(axis_pelist >= 0)
414  allocate(pelist(nd), pelist_idx(0:nd-1))
415  pelist = pack(axis_pelist, mask=(axis_pelist >= 0))
416  pelist_idx = pack([(i, i=0, n_ax-1)], mask=(axis_pelist >= 0))
417 
418  deallocate(axis_pelist)
419  else
420  nd = size(domain%list(:))
421  allocate(pelist(nd), pelist_idx(0:nd-1))
422  call mpp_get_pelist(domain, pelist)
423  pelist_idx = [ (i, i=0, nd-1) ]
424  end if
425 
426  ! Allocate message data buffers
427  allocate(sendcounts(0:nd-1))
428  allocate(sdispls(0:nd-1))
429  allocate(sendtypes(0:nd-1))
430  sendcounts(:) = 0
431  sdispls(:) = 0
432  sendtypes(:) = mpp_byte
433 
434  allocate(recvcounts(0:nd-1))
435  allocate(rdispls(0:nd-1))
436  allocate(recvtypes(0:nd-1))
437  recvcounts(:) = 0
438  rdispls(:) = 0
439  recvtypes(:) = mpp_byte
440 
441  array_of_subsizes = [iec - isc + 1, jec - jsc + 1, size(local, 3)]
442  array_of_starts = [isc + ioff, jsc + joff, 0]
443 
444  n_sends = merge(1, nd, root_only) ! 1 if root_only else nd
445  do n = 0, n_sends - 1
446  sendcounts(n) = 1
447 
448  call mpp_type_create( &
449  local, &
450  array_of_subsizes, &
451  array_of_starts, &
452  sendtypes(n) &
453  )
454  end do
455 
456  ! Receive configuration
457  if (global_on_this_pe) then
458  do n = 0, nd - 1
459  recvcounts(n) = 1
460  pe = pelist_idx(n)
461 
462  if (xonly) then
463  is = domain%x(1)%list(pe)%compute%begin
464  ie = domain%x(1)%list(pe)%compute%end + ishift
465  js = jsc; je = jec
466  else if (yonly) then
467  is = isc; ie = iec
468  js = domain%y(1)%list(pe)%compute%begin
469  je = domain%y(1)%list(pe)%compute%end + jshift
470  else
471  is = domain%list(pe)%x(1)%compute%begin
472  ie = domain%list(pe)%x(1)%compute%end + ishift
473  js = domain%list(pe)%y(1)%compute%begin
474  je = domain%list(pe)%y(1)%compute%end + jshift
475  end if
476 
477  array_of_subsizes = [ie - is + 1, je - js + 1, ke]
478  array_of_starts = [is - isg + ipos, js - jsg + jpos, 0]
479 
480  call mpp_type_create( &
481  global, &
482  array_of_subsizes, &
483  array_of_starts, &
484  recvtypes(n) &
485  )
486  end do
487  end if
488 
489  plocal(1:size(local)) => local
490  pglobal(1:size(global)) => global
491 
492  call mpp_alltoall(plocal, sendcounts, sdispls, sendtypes, &
493  pglobal, recvcounts, rdispls, recvtypes, &
494  pelist)
495 
496  plocal => null()
497  pglobal => null()
498 
499  ! Cleanup
500  deallocate(pelist)
501  deallocate(sendcounts, sdispls, sendtypes)
502  deallocate(recvcounts, rdispls, recvtypes)
503 
504  call mpp_sync_self()
505 
506  end subroutine MPP_DO_GLOBAL_FIELD_A2A_3D_
field_manager_mod::initialize
