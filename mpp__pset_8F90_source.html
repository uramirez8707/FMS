99  subroutine mpp_pset_create(npset,pset,stacksize,pelist, commID)
100 !create PSETs
101 ! called by all PEs in parent pelist
102 ! mpset must be exact divisor of npes
103  integer, intent(in) :: npset !number of PSETs per set
104  type(mpp_pset_type), intent(inout) :: pset
105  integer, intent(in), optional :: stacksize
106  integer, intent(in), optional :: pelist(:)
107  integer, intent(in), optional :: commID
108 
109  integer :: npes, my_commID
110  integer :: i, j, k, out_unit, errunit
111  integer, allocatable :: my_pelist(:), root_pelist(:)
112 
113  call mpp_init()
114  call mpp_pset_init()
115 
116 #ifdef PSET_DEBUG
117  verbose=.true.
118 #endif
119  out_unit = stdout()
120  errunit = stderr()
121  pe = mpp_pe()
122  if(present(pelist)) then
123  npes = size(pelist(:))
124  else
125  npes = mpp_npes()
126  endif
127  if( mod(npes,npset).NE.0 )then
128  write( text,'(a,2i6)' ) &
129  'MPP_PSET_CREATE: PSET size (npset) must divide npes exactly:'// &
130  ' npset, npes=', npset, npes
131  call mpp_error( fatal, text )
132  end if
133 
134  !configure out root_pelist
135  allocate(my_pelist(0:npes-1) )
136  allocate(root_pelist(0:npes/npset-1) )
137  if(present(pelist)) then
138  if(.not. present(commid)) call mpp_error(fatal, &
139  'MPP_PSET_CREATE: when pelist is present, commID should also be present')
140  my_pelist = pelist
141  my_commid = commid
142  else
143  call mpp_get_current_pelist(my_pelist, commid = my_commid)
144  endif
145  do i = 0,npes/npset-1
146  root_pelist(i) = my_pelist(npset*i)
147  enddo
148  write( out_unit,'(a,i6)' )'MPP_PSET_CREATE creating PSETs... npset=', npset
149  if(any(my_pelist == pe) ) then
150  if( pset%initialized )call mpp_error( fatal, &
151  'MPP_PSET_CREATE: PSET already initialized!' )
152  pset%npset = npset
153  allocate( pset%pelist(0:npes-1) )
154  allocate( pset%root_pelist(0:npes/npset-1) )
155  pset%commID = my_commid
156  pset%pelist = my_pelist
157 !create the root PElist
158  pset%root_pelist = root_pelist
159  allocate( pset%pset(0:npset-1) )
160  do i = 0,npes/npset-1
161  k = npset*i
162 !designate the root PE, next PE, prev PE
163  do j = 0,npset-1
164  if( pe.EQ.pset%pelist(k+j) )then
165  pset%pset(:) = pset%pelist(k:k+npset-1)
166  pset%pos = j
167  pset%root_in_pset = pset%root_pelist(i)
168  if( j.EQ.0 )then
169  pset%prev_in_pset = pset%pelist(k+npset-1)
170  else
171  pset%prev_in_pset = pset%pelist(k+j-1)
172  end if
173  if( j.EQ.npset-1 )then
174  pset%next_in_pset = pset%pelist(k)
175  else
176  pset%next_in_pset = pset%pelist(k+j+1)
177  end if
178  end if
179  end do
180  end do
181 
182  pset%root = pe.EQ.pset%root_in_pset
183 
184 !stack
185  pset%hiWM = 0 !initialize hi-water-mark
186  pset%maxstack = 1000000 !default
187  if( PRESENT(stacksize) )pset%maxstack = stacksize
188  write( out_unit,'(a,i8)' ) &
189  'MPP_PSET_CREATE: setting stacksize=', pset%maxstack
190  if( pset%root )then
191  allocate( pset%stack(pset%maxstack) )
192 #ifdef use_CRI_pointers
193  pset%p_stack = loc(pset%stack)
194 #endif
195  end if
196  pset%initialized = .true. !must be called before using pset
197  call mpp_pset_broadcast_ptr(pset,pset%p_stack)
198  endif
199 
200  call mpp_declare_pelist(root_pelist)
201 
202  if( verbose )then
203  write( errunit,'(a,4i6)' )'MPP_PSET_CREATE: pe, root, next, prev=', &
204  pe, pset%root_in_pset, pset%next_in_pset, pset%prev_in_pset
205  write( errunit,* )'PE ', pe, ' pset=', pset%pset(:)
206  write( out_unit,* )'root pelist=', pset%root_pelist(:)
207  end if
208  end subroutine mpp_pset_create
209 
210  subroutine mpp_pset_delete(pset)
211  type(mpp_pset_type), intent(inout) :: pset
212  integer :: out_unit
213 
214  out_unit = stdout()
215  if( .NOT.pset%initialized )call mpp_error( fatal, &
216  'MPP_PSET_DELETE: called with uninitialized PSET.' )
217 !deallocate arrays...
218  deallocate( pset%pelist )
219  deallocate( pset%root_pelist )
220  deallocate( pset%pset )
221  if( pset%root )deallocate( pset%stack )
222  write( out_unit, '(a,i10)' ) &
223  'Deleting PSETs... stack high-water-mark=', pset%hiWM
224 !... and set status flag
225  pset%initialized = .false.
226  end subroutine mpp_pset_delete

287 !broadcast value on the root to its sub-threads

289  real, intent(inout) :: a
290  integer :: i
291 
292  if( .NOT.pset%initialized )call mpp_error( fatal, &

347  subroutine mpp_pset_check_ptr(pset,ptr)
348 !checks if the supplied pointer is indeed shared
349  type(mpp_pset_type), intent(in) :: pset
350 #ifdef use_CRI_pointers
351  real :: dummy
352  pointer( ptr, dummy )
353 #else
354  integer(POINTER_KIND), intent(in) :: ptr
355 #endif
356 #ifdef PSET_DEBUG
357  integer(POINTER_KIND) :: p
358  integer :: i
359  if( .NOT.pset%initialized )call mpp_error( fatal, &
360  'MPP_PSET_CHECK_PTR: called with uninitialized PSET.' )
361  commid = pset%commID !pass to mpp_translate_remote_ptr
362 !check if this is a shared pointer
363  p = ptr
364  if( pset%root )then
365  do i = 1,pset%npset-1
366  call mpp_send_ptr( p, pset%pset(i) )
367  end do
368  else
369  call mpp_recv_ptr( p, pset%root_in_pset )
370  end if
371  call mpp_pset_sync(pset)
372  if( p.NE.ptr )call mpp_error( fatal, &
373  'MPP_PSET_CHECK_PTR: pointers do not match!' )
374 #else
375 !do nothing if the debug CPP flag isn't on
376 #endif
377  end subroutine mpp_pset_check_ptr

379  subroutine mpp_pset_segment_array( pset, ls, le, lsp, lep )
380 !given input indices ls, le, returns indices lsp, lep
381 !so that segments span the range ls:le with no overlaps.
382 !attempts load balance: also some PSETs might get lsp>lep
383 !so that do-loops will be null
384  type(mpp_pset_type), intent(in) :: pset
385  integer, intent(in) :: ls, le
386  integer, intent(out) :: lsp, lep
387  integer :: i
388 
389  if( .NOT.pset%initialized )call mpp_error( fatal, &
390  'MPP_PSET_SEGMENT_ARRAY: called with uninitialized PSET.' )
391 #ifdef PSET_DEBUG
392  if( le-ls+1.LT.pset%npset )then
393  write( text,'(3(a,i6))' ) &
394  'MPP_PSET_ARRAY_SEGMENT: parallel range (', ls, ',', le, &
395  ') is smaller than the number of threads:', pset%npset
396  call mpp_error( warning, text )
397  end if
398 #endif
399  lep = ls-1 !initialize so that lsp is correct on first pass
400  do i = 0,pset%pos
401  lsp = lep + 1
402  lep = lsp + ceiling( real(le-lsp+1)/(pset%npset-i) ) - 1
403  end do
404  end subroutine mpp_pset_segment_array

406  subroutine mpp_pset_stack_push( pset, ptr, len )
407 !mpp_malloc specialized for shared arrays
408 !len is the length of the required array
409 !lstack is the stack already in play
410 !user should zero lstack (call mpp_pset_stack_reset) when the stack is to be cleared
411  type(mpp_pset_type), intent(inout) :: pset
412  integer, intent(in) :: len
413 #ifdef use_CRI_pointers
414  real :: dummy
415  pointer( ptr, dummy )
416  real :: stack(pset%maxstack)
417  pointer( p, stack )
418 
419  if( .NOT.pset%initialized )call mpp_error( fatal, &
420  'MPP_PSET_STACK_PUSH: called with uninitialized PSET.' )
421  if( pset%lstack+len.GT.pset%maxstack )then
422  write( text, '(a,3i12)' ) &
423  'MPP_PSET_STACK_PUSH: mpp_pset_stack overflow: '// &
424  .GT.'len+lstackmaxstack. len, lstack, maxstack=', &
425  len, pset%lstack, pset%maxstack
426  call mpp_error( fatal, text )
427  end if
428  p = pset%p_stack !point stack to shared stack pointer
429  ptr = loc( stack(pset%lstack+1) )
430  call mpp_pset_check_ptr(pset,ptr) !make sure ptr is the same across PSETs
431  pset%lstack = pset%lstack + len
432  pset%hiWM = max( pset%hiWM, pset%lstack )
433 #else
434  integer(POINTER_KIND), intent(out) :: ptr
435  call mpp_error( fatal, &
436  'MPP_PSET_STACK_PUSH only works with Cray pointers.' )
437 #endif
438  end subroutine mpp_pset_stack_push

537 !accessor functions
538  function mpp_pset_root(pset)
539  logical :: mpp_pset_root
540  type(mpp_pset_type), intent(in) :: pset
