20 !> @brief This programs tests diag_manager with the following diag_table
21 !! test_diag_manager
22 !! 2 1 1 0 0 0
23 !! "ocn%4yr%2mo%2dy%2hr", 1, "days", 1, "days", "time", 1, "days", "2 1 1 0 0 0"
24 !! "test_diag_manager_mod", "sst", "sst", "ocn%4yr%2mo%2dy%2hr", "all", .true., "none", 2
25 
26 program test_diag_manager_ug
27 
28 use mpp_domains_mod
29 use diag_manager_mod
30 use fms_mod
31 use time_manager_mod, only: time_type, set_calendar_type, set_date, noleap, julian, operator(+), set_time, print_time
32 use mpp_mod
33 
34 implicit none
35 
36 type(time_type) :: Time !< Simulation time
37 integer :: nlon, nlat !< Number of lon/lat in each tile
38 type(domain2d) :: Domain !< Cubesphere domain
39 type(domainug) :: land_domain !< Unstructured domain
40 integer :: i !< For do loops
41 real, allocatable :: sst(:) !< Data in the unstructured grid
42 logical :: used !< Flag indicating if data was sucessfully sent
43 integer :: id_x, id_y, id_ug !< ids for the x,y, and unstructured grid axis
44 integer :: id_sst !< id for the data in the unstructured grid
45 integer :: ug_dim_size ! Size of the unstructured axis
46 integer,allocatable :: ug_dim_data(:) ! Unstructured axis data.
47 integer :: npes !< Number of PE ranks the test is running with
48 integer,dimension(2) :: io_layout=(/1, 1/) !< Io_layout for the cubed_sphere/lat-lon grids
49 integer, parameter :: ntiles=6 !< Number of tiles for the test
50 integer, dimension(4,ntiles) :: global_indices !< The global indices for each tile
51 integer, dimension(2,ntiles) :: layout !< The layout for each tile
52 integer, dimension(ntiles) :: pe_start !< The starting pe in the pelist for each tile
53 integer, dimension(ntiles) :: pe_end !< The ending pe in the pelist for each tile
54 integer :: npes_group=1 !< For the land layout
55 
56 call fms_init
57 call set_calendar_type(julian)
58 call diag_manager_init
59 
60 time = set_date(2,1,1,0,0,0)
61 nlon = 20
62 nlat = 30
63 npes = mpp_npes()
64 
65 call mpp_domains_set_stack_size(17280000)
66 
67 !< Set up the domain
68 do i = 1,ntiles
69  global_indices(:, i) = (/1, nlon, 1, nlat/)
70  layout(:, i) = (/1, npes/ntiles/)
71  pe_start(i) = (i-1)*(npes/ntiles)
72  pe_end(i) = i*(npes/ntiles) - 1
73 enddo
74 
75 call create_atmosphere_domain((/nlon, nlon, nlon, nlon, nlon, nlon/), &
76  (/nlat, nlat, nlat, nlat, nlat, nlat/), &
77  global_indices, layout, pe_start, pe_end, &
78  io_layout, domain)
79 call create_land_domain(domain, nlon, nlat, ntiles, land_domain, npes_group)
80 
81 call mpp_get_ug_compute_domain(land_domain, size=ug_dim_size)
82 allocate(ug_dim_data(ug_dim_size))
83 call mpp_get_ug_domain_grid_index(land_domain, ug_dim_data)
84 ug_dim_data = ug_dim_data - 1
85 
86 !< Register the axis and the data
87 id_ug = diag_axis_init("grid_index", real(ug_dim_data), "none", "U", long_name="grid indices", &
88  set_name="land", domainu=land_domain, aux="geolon_t geolat_t")
89 id_x = diag_axis_init( 'grid_xt', (/(real(i),i=1,nlon)/), 'degrees_E', 'X', &
90  'T-cell longitude', set_name='land' )
91 id_y = diag_axis_init( 'grid_yt', (/(real(i),i=1,nlat)/), 'degrees_N', 'Y', &
92  'T-cell latitude', set_name='land' )
93 id_sst = register_diag_field('land_mod', 'sst', (/id_ug/), time, 'SST', 'K')
94 
95 call diag_axis_add_attribute(id_ug,'compress','grid_xt grid_yt')
96 
97 !< Set up the data
98 allocate(sst(ug_dim_size))
99 sst = 999.
100 
101 ! Increase the time and send data
102 do i=1,23
103  time = set_date(2,1,1,i,0,0)
104  sst = real(i)
105 
106  if(id_sst > 0) used = send_data(id_sst, sst, time)
107 enddo
108 
109 call diag_manager_end(time)
110 call fms_end
111 
112 contains
113 include "../fms2_io/create_atmosphere_domain.inc"
114 include "../fms2_io/create_land_domain.inc"
115 
116 end program test_diag_manager_ug
