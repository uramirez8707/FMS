20 program test_io_with_mask
21 !> @brief This programs tests fms2io/include/domain_write ability to write
22 !! data when the domain contains a mask table. For the points that are
23 !! masked out, no data should be writen.
24 !! It also tests fms2io/include/domain_read ability to read the data when
25 !! the domain contains a mask table. For this case the masked data should
26 !! not be read.
27 
28 use mpp_domains_mod, only: mpp_domains_set_stack_size, mpp_define_domains, mpp_define_io_domain, &
29  mpp_get_compute_domain,domain2d
30 use mpp_mod, only: mpp_pe, mpp_root_pe, mpp_error, fatal
31 use fms2_io_mod, only: open_file, register_axis, register_variable_attribute, close_file, &
32  fmsnetcdfdomainfile_t, write_data, register_field, read_data, &
33  parse_mask_table
34 use fms_mod, only: fms_init, fms_end
35 use netcdf, only: nf90_open, nf90_get_var, nf90_nowrite, nf90_noerr, nf90_get_var, &
36  nf90_close
37 use mpi, only: mpi_barrier, mpi_comm_world
38 use platform_mod
39 
40 implicit none
41 
42 integer, dimension(2) :: layout = (/2,3/) !< Domain layout
43 integer :: nlon !< Number of points in x axis
44 integer :: nlat !< Number of points in y axis
45 type(domain2d) :: Domain !< Domain with mask table
46 real, dimension(:), allocatable :: x !< x axis data
47 real, dimension(:), allocatable :: y !< y axis data
48 real(kind=r8_kind), allocatable, dimension(:,:) :: sst !< Data to be written
49 real(kind=r8_kind), allocatable, dimension(:,:) :: sst_in !< Buffer where data will be read with netcdf
50 real(kind=r8_kind), allocatable, dimension(:,:) :: sst_in2 !< Buffer where data will be read with fms2io
51 logical, allocatable, dimension(:,:) :: parsed_mask !< Parsed masked
52 character(len=6), dimension(2) :: names !< Dimensions names
53 type(FmsNetcdfDomainFile_t) :: fileobj !< fms2io fileobj for domain decomposed
54 integer :: err !< Return code.
55 integer :: ncid !< File ID for checking file.
56 integer :: i, j !< Helper integers
57 integer :: is !< Starting x index
58 integer :: ie !< Ending x index
59 integer :: js !< Starting y index
60 integer :: je !< Ending y index
61 
62 call fms_init
63 
64 nlon = 60
65 nlat = 60
66 
67 !< Parse the mask table
68 allocate(parsed_mask(layout(1), layout(2)))
69 call parse_mask_table("the_mask", parsed_mask, 'test_io_with_mask')
70 
71 !< Create a domain nlonXnlat with mask
72 call mpp_domains_set_stack_size(17280000)
73 call mpp_define_domains( (/1,nlon,1,nlat/), layout, domain, name='test_io_with_mask', maskmap=parsed_mask)
74 call mpp_define_io_domain(domain, (/1,1/))
75 call mpp_get_compute_domain(domain, is, ie, js, je)
76 
77 !< Set up the data
78 allocate(x(nlon), y(nlat))
79 allocate(sst(is:ie,js:je))
80 allocate(sst_in2(is:ie,js:je))
81 
82 do i=1,nlon
83  x(i) = i
84 enddo
85 do j=1,nlat
86  y(j) = j
87 enddo
88 
89 sst = real(7., kind=r8_kind)
90 
91 !< Open a netCDF file and initialize the file object.
92 if (open_file(fileobj, "test_io_with_mask.nc", "overwrite", domain)) then
93  !< Register the axis
94  names(1) = "lon"
95  names(2) = "lat"
96  call register_axis(fileobj, "lon", "x")
97  call register_axis(fileobj, "lat", "y")
98 
99  !< Register the variable and Write out the data
100  call register_field(fileobj, "sst", "double", names(1:2))
101  call register_variable_attribute(fileobj, "sst", "_FillValue", real(999., kind=r8_kind))
102  call write_data(fileobj, "sst", sst)
103 
104  !< Close the file
105  call close_file(fileobj)
106 else
107  call mpp_error(fatal, "test_io_with_mask: error opening the file for writting")
108 endif
109 
110 !< Read the file back to check if it was done right!
111 if (mpp_pe() .eq. mpp_root_pe()) then
112  err = nf90_open("test_io_with_mask.nc", nf90_nowrite, ncid)
113  if (err .ne. nf90_noerr) call mpp_error(fatal, "test_io_with_mask: error opening the file for reading")
114 
115  allocate(sst_in(nlon,nlat))
116  err = nf90_get_var(ncid, 1, sst_in)
117  if (err .ne. nf90_noerr) call mpp_error(fatal, "test_io_with_mask: error reading from the file")
118 
119  !< x: 1-30 y: 1-20 are masked out, so sst_in for this values has to be equal to the fill value
120  !! For the other points the data should be equal to real(7., kind=r8_kind)
121 
122  do i=1,nlon
123  do j=1,nlat
124  if (i > 30 .or. j > 20) then
125  if (sst_in(i,j) .ne. real(7., kind=r8_kind)) then
126  print *, 'i=', i, ' j=', j, ' sst_in=', sst_in(i,j)
127  call mpp_error(fatal, "test_io_with_mask: the unmasked data read in is not correct")
128  endif
129  else
130  if (sst_in(i,j) .ne. real(999., kind=r8_kind)) then
131  print *, 'i=', i, ' j=', j, ' sst_in=', sst_in(i,j)
132  call mpp_error(fatal, "test_io_with_mask: the masked data read in is not correct")
133  endif
134  endif
135  enddo
136  enddo
137 
138  err = nf90_close(ncid)
139  if (err .ne. nf90_noerr) call mpp_error(fatal, "test_io_with_mask: error closing the file")
140 endif
141 
142 !< Wait for the root pe to catch up!
143 call mpi_barrier(mpi_comm_world, err)
144 
145 !< Read the file back using fms2io
146 sst_in2 = 0.
147 if (open_file(fileobj, "test_io_with_mask.nc", "read", domain)) then
148  names(1) = "lon"
149  names(2) = "lat"
150  call register_axis(fileobj, "lon", "x")
151  call register_axis(fileobj, "lat", "y")
152 
153  !< Register the variable and read out the data
154  call register_field(fileobj, "sst", "double", names(1:2))
155 
156  call read_data(fileobj, "sst", sst_in2)
157  call close_file(fileobj)
158 
159  do i=is,ie
160  do j=js,je
161  if (sst_in2(i,j) .ne. real(7., kind=r8_kind)) then
162  print *, 'i=', i, ' j=', j, ' sst_in=', sst_in2(i,j)
163  call mpp_error(fatal, "test_io_with_mask: the unmasked data read in is not correct")
164  endif
165  enddo
166  enddo
167 
168 endif
169 
170 
171 call fms_end
172 end program test_io_with_mask
