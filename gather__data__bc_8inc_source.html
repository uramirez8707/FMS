22 !> @brief gathers the 2d vdata from all of the relevant pes into the root_pe and saves it to a
23 !! buffer.
24 subroutine gather_data_bc_2d(fileobj, vdata, bc_info)
25  class(fmsnetcdffile_t), intent(inout) :: fileobj !< Fms2io netcdf fileobj
26  class(*), dimension(:,:), intent(in) :: vdata !< Data to be gather
27  type(bc_information), intent(inout) :: bc_info !< information about the boundary condition variable
28 
29  integer :: i_glob !< Size of the global domain in x
30  integer :: j_glob !< Size of the global domain in y
31 
32  integer :: isc, iec, jsc, jec !< current PE's indices
33  integer :: i1, i2, j1, j2 !< current PE's indices relative to the global domain
34  integer :: i_add, j_add !< current PE's shift
35 
36  real(kind=r4_kind), dimension(:,:), allocatable, target :: global_buf_r4_kind !< buffer with a data gathered
37  real(kind=r4_kind), dimension(:,:), allocatable, target :: local_buf_r4_kind !< current PEs data
38  real(kind=r8_kind), dimension(:,:), allocatable, target :: global_buf_r8_kind !< buffer with a data gathered
39  real(kind=r8_kind), dimension(:,:), allocatable, target :: local_buf_r8_kind !< current PEs data
40 
41  integer(kind=i8_kind) :: chksum_val !< Checksum value calculated from the gathered data
42  character(len=32) :: chksum !< Cheksum value converted to a string
43 
44  !> Set the indices
45  isc = bc_info%indices(1)
46  iec = bc_info%indices(2)
47  jsc = bc_info%indices(3)
48  jec = bc_info%indices(4)
49 
50  !> This is the section of the PE that will actually be added to the global_buffer
51  i1 = 1 + bc_info%x_halo
52  i2 = i1 + (iec-isc)
53  j1 = 1 + bc_info%y_halo
54  j2 = j1 + (jec-jsc)
55 
56  !> Set up index shifts for global array
57  i_add = bc_info%ishift
58  j_add = bc_info%jshift
59 
60  !> Only root allocates a global_buffer that will be written
61  if (fileobj%is_root) then
62  i_glob = bc_info%global_size(1)
63  j_glob = bc_info%global_size(2)
64  endif
65 
66  !> Gather the data and calculate the checksum for the resulting array.
67  select type(vdata)
68  type is (real(kind=r4_kind))
69  !> If the fileobj's root does not have data for this variable
70  if (fileobj%is_root .and. .not. bc_info%data_on_file_root) then
71  !> Allocate global_buf_r4_kind to be one size bigger, global_buf_r4_kind(i_glob+1,,:) is just dummy data
72  allocate(global_buf_r4_kind(i_glob+1, j_glob))
73  !> Allocate a temp local buffer to the fileobj's root. This is needed because the data needs to be send
74  !! to the fileobj's root, but because fileobj's root doesn't have any data, we just create dummy data and
75  !! not write it later
76  allocate(local_buf_r4_kind(1,1))
77  local_buf_r4_kind = 0.
78  isc = 1+i_glob; i_add=0; iec=1+i_glob; jsc=j_glob; j_add=0; jec=j_glob
79  i1=1; i2=1; j1=1; j2=1
80  else
81  !! In this case there is data in fileobj's root, so there is no need for the dummy data
82  if(fileobj%is_root) allocate(global_buf_r4_kind(i_glob, j_glob))
83  allocate(local_buf_r4_kind(size(vdata,1), size(vdata,2)))
84  local_buf_r4_kind = vdata
85  endif
86 
87  call mpp_gather(isc+i_add, iec+i_add, jsc+j_add, jec+j_add, bc_info%pelist, &
88  local_buf_r4_kind(i1:i2,j1:j2), &
89  global_buf_r4_kind, fileobj%is_root)
90 
91  deallocate(local_buf_r4_kind)
92  !> If you are on fileobj's root, calculate the checksum and save the gathered data in a buffer
93  if (fileobj%is_root) then
94  chksum_val = mpp_chksum(global_buf_r4_kind(1:i_glob,1:j_glob), (/mpp_pe()/))
95  allocate(bc_info%globaldata2d_r4(i_glob, j_glob))
96  bc_info%globaldata2d_r4=global_buf_r4_kind(1:i_glob,1:j_glob)
97  deallocate(global_buf_r4_kind)
98  endif
99  type is (real(kind=r8_kind))
100  !> If the fileobj's root does not have data for this variable
101  if (fileobj%is_root .and. .not. bc_info%data_on_file_root) then
102  !> Allocate global_buf_r8_kind to be one size bigger, global_buf_r8_kind(i_glob+1,,:) is just dummy data
103  allocate(global_buf_r8_kind(i_glob+1, j_glob))
104  !> Allocate a temp local buffer to the fileobj's root. This is needed because the data needs to be send
105  !! to the fileobj's root, but because fileobj's root doesn't have any data, we just create dummy data and
106  !! not write it later
107  allocate(local_buf_r8_kind(1,1))
108  local_buf_r8_kind = 0.
109  isc = 1+i_glob; i_add=0; iec=1+i_glob; jsc=j_glob; j_add=0; jec=j_glob
110  i1=1; i2=1; j1=1; j2=1
111  else
112  !! In this case there is data in fileobj's root, so there is no need for the dummy data
113  if(fileobj%is_root) allocate(global_buf_r8_kind(i_glob, j_glob))
114  allocate(local_buf_r8_kind(size(vdata,1), size(vdata,2)))
115  local_buf_r8_kind = vdata
116  endif
117 
118  call mpp_gather(isc+i_add, iec+i_add, jsc+j_add, jec+j_add, bc_info%pelist, &
119  local_buf_r8_kind(i1:i2,j1:j2), &
120  global_buf_r8_kind, fileobj%is_root)
121  deallocate(local_buf_r8_kind)
122  !> If you are on fileobj's root, calculate the checksum and save the gathered data in a buffer
123  if (fileobj%is_root) then
124  chksum_val = mpp_chksum(global_buf_r8_kind(1:i_glob,1:j_glob), (/mpp_pe()/))
125  allocate(bc_info%globaldata2d_r8(i_glob, j_glob))
126  bc_info%globaldata2d_r8=global_buf_r8_kind(1:i_glob,1:j_glob)
127  deallocate(global_buf_r8_kind)
128  endif
129  class default
130  call error("gather_data_bc_2d: unsupported type. Currently only r8_kind and r4_kinds are supported")
131  end select
132 
133  !> Save the checksum, so you can write it later
134  if (fileobj%is_root) then
135  chksum = ""
136  write(chksum, "(Z16)") chksum_val
137  bc_info%chksum = chksum
138  endif
139 
140 end subroutine gather_data_bc_2d
