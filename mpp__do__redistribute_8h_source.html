19  subroutine MPP_DO_REDISTRIBUTE_3D_( f_in, f_out, d_comm, d_type )
20  integer(i8_kind), intent(in) :: f_in(:), f_out(:)
21  type(DomainCommunicator2D), intent(in) :: d_comm
22  MPP_TYPE_, intent(in) :: d_type
23  MPP_TYPE_ :: field_in(d_comm%domain_in%x(1)%data%begin:d_comm%domain_in%x(1)%data%end, &
24  d_comm%domain_in%y(1)%data%begin:d_comm%domain_in%y(1)%data%end,d_comm%ke)
25  pointer( ptr_field_in, field_in)
26  MPP_TYPE_ :: field_out(d_comm%domain_out%x(1)%data%begin:d_comm%domain_out%x(1)%data%end, &
27  d_comm%domain_out%y(1)%data%begin:d_comm%domain_out%y(1)%data%end,d_comm%ke)
28  pointer( ptr_field_out, field_out)
29  type(domain2D), pointer :: domain_in, domain_out
30  integer :: i, j, k, l, n, l_size
31  integer :: is, ie, js, je
32  integer :: ke
33  integer :: list, pos, msgsize
34  integer :: to_pe, from_pe
35  MPP_TYPE_ :: buffer(size(mpp_domains_stack(:)))
36  pointer( ptr, buffer )
37  integer :: buffer_pos, wordlen, errunit
38 
39 !fix ke
40  errunit = stderr()
41  l_size = size(f_out(:)) ! equal to size(f_in(:))
42  ke = d_comm%ke
43  domain_in =>d_comm%domain_in; domain_out =>d_comm%domain_out
44 
45  buffer_pos = 0
46  ptr = LOC(mpp_domains_stack)
47  wordlen = size(TRANSFER(buffer(1),mpp_domains_stack))
48 
49 !pre-post recv
50  n = d_comm%Rlist_size
51  do list = 0,n-1
52  if( .NOT. d_comm%R_do_buf(list) )cycle
53  from_pe = d_comm%cfrom_pe(list)
54  msgsize = d_comm%R_msize(list)*l_size
55  call mpp_recv( buffer(buffer_pos+1), glen=msgsize, from_pe=from_pe, block=.FALSE., tag=COMM_TAG_1 )
56  buffer_pos = buffer_pos + msgsize
57  enddo
58 
59 !send
60  n = d_comm%Slist_size
61  do list = 0,n-1
62  if( .NOT. d_comm%S_do_buf(list) )cycle
63  to_pe = d_comm%cto_pe(list)
64  is=d_comm%sendis(1,list); ie=d_comm%sendie(1,list)
65  js=d_comm%sendjs(1,list); je=d_comm%sendje(1,list)
66  pos = buffer_pos
67  do l=1,l_size ! loop over number of fields
68  ptr_field_in = f_in(l)
69  do k = 1,ke
70  do j = js,je
71  do i = is,ie
72  pos = pos+1
73  buffer(pos) = field_in(i,j,k)
74  end do
75  end do
76  end do
77  end do
78  if( debug )write( errunit,* )'PE', pe, ' to PE ', to_pe, 'is,ie,js,je=', is, ie, js, je
79  msgsize = pos - buffer_pos
80  call mpp_send( buffer(buffer_pos+1), plen=msgsize, to_pe=to_pe, tag=COMM_TAG_1 )
81  buffer_pos = pos
82  end do
83 
84  call mpp_sync_self(check=EVENT_RECV)
85 
86 !unpack buffer
87  buffer_pos = 0
88  n = d_comm%Rlist_size
89  do list = 0,n-1
90  if( .NOT. d_comm%R_do_buf(list) )cycle
91  from_pe = d_comm%cfrom_pe(list)
92  is=d_comm%recvis(1,list); ie=d_comm%recvie(1,list)
93  js=d_comm%recvjs(1,list); je=d_comm%recvje(1,list)
94  if( debug )write( errunit,* )'PE', pe, ' from PE ', from_pe, 'is,ie,js,je=', is, ie, js, je
95  pos = buffer_pos
96  do l=1,l_size ! loop over number of in/out fields
97  ptr_field_out = f_out(l)
98  do k = 1,ke
99  do j = js,je
100  do i = is,ie
101  pos = pos+1
102  field_out(i,j,k) = buffer(pos)
103  end do
104  end do
105  end do
106  end do
107  buffer_pos = pos
108  end do
109 
110  call mpp_sync_self()
111  end subroutine MPP_DO_REDISTRIBUTE_3D_
