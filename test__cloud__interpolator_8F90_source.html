178  subroutine test_get_node_values
179  integer, parameter :: nd = 3, n1=6, n2=5, n3=4
180  real, dimension(n1, n2, n3) :: fnodes
181  real :: fvals(2**nd), fexact(2**nd)
182  real x, y, z
183  integer i, j, k, ier, indices(nd)
184  real :: error_tot = 0.
185  do k = 1, n3
186  do j = 1, n2
187  do i = 1, n1
188  x = 1* real(i-1)/real(n1-1)
189  y = 2* real(j-1)/real(n2-1)
190  z = 3* real(k-1)/real(n3-1)
191  fnodes(i,j,k) = x + y*z**2
192  enddo
193  enddo
194  enddo
195  indices = (/1,1,1/)
196  call cld_ntrp_get_cell_values((/n1,n2,n3/), pack(fnodes, .true.) , indices, fvals, ier)
197  fexact = (/0.0, 0.2, 0.0, 0.2, 0.0, 0.2, 0.5, 0.7/)
198  if(ier/=0) print *,'ERROR flag ier=', ier
199  print *,'indices ', indices
200  print *,'fvals=', fvals, ' ==? ', fexact
201  error_tot = error_tot + abs(sum(fvals - fexact))
202 
203  indices = (/5,4,2/)
204  call cld_ntrp_get_cell_values((/n1,n2,n3/), pack(fnodes, .true.) , indices, fvals, ier)
205  fexact = (/2.3, 2.5, 2.8, 3.0, 6.8, 7.0, 8.8, 9.0/)
206  if(ier/=0) print *,'ERROR flag ier=', ier
207  print *,'indices ', indices
208  print *,'fvals=', fvals, ' ==? ', fexact
209  error_tot = error_tot + abs(sum(fvals - fexact))
210 
211  if(error_tot /= 0) then
212  print *,'ERROR flag error_tot=', error_tot
213  call mpp_error(fatal,'ERROR: get node values test failed (error_tot/=0)')
214  endif
215 
216 end subroutine test_get_node_values
