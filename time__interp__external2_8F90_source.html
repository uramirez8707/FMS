777 
778  integer, intent(in) :: index
779  type(time_type), intent(in) :: time
780  real, dimension(:,:,:), intent(inout) :: data
781  integer, intent(in), optional :: interp
782  logical, intent(in), optional :: verbose
783  type(horiz_interp_type), intent(in), optional :: horz_interp

786  integer, intent(in), optional :: window_id

789  integer :: i1, i2, isc, iec, jsc, jec, mod_time
790  integer :: yy, mm, dd, hh, min, ss
791  character(len=256) :: err_msg, filename
792 
793  integer :: isw, iew, jsw, jew, nxw, nyw
794  ! these are boundaries of the updated portion of the "data" argument
795  ! they are calculated using sizes of the "data" and isc,iec,jsc,jsc
796  ! fileds from respective input field, to center the updated portion
797  ! in the output array

800  logical :: verb

804  ny = size(data,2)
805  nz = size(data,3)
806 
807  interp_method = linear_time_interp
808  if (PRESENT(interp)) interp_method = interp
809  verb=.false.
810  if (PRESENT(verbose)) verb=verbose
811  if (debug_this_module) verb = .true.
812 
813  if (index < 1.or.index > num_fields) &
814  call mpp_error(fatal,'invalid index in call to time_interp_ext -- field was not initialized or failed to initialize')
815 
816  isc=field(index)%isc;iec=field(index)%iec
817  jsc=field(index)%jsc;jec=field(index)%jec
818 
819  if( field(index)%numwindows == 1 ) then
820  nxw = iec-isc+1
821  nyw = jec-jsc+1
822  else
823  if( .not. present(is_in) .or. .not. present(ie_in) .or. .not. present(js_in) .or. .not. present(je_in) ) then
824  call mpp_error(fatal, 'time_interp_external: is_in, ie_in, js_in and je_in must be present ' // &
825  'when numwindows > 1, field='//trim(field(index)%name))
826  endif
827  nxw = ie_in - is_in + 1
828  nyw = je_in - js_in + 1
829  isc = isc + is_in - 1
830  iec = isc + ie_in - is_in
831  jsc = jsc + js_in - 1
832  jec = jsc + je_in - js_in
833  endif
834 
835  isw = (nx-nxw)/2+1; iew = isw+nxw-1
836  jsw = (ny-nyw)/2+1; jew = jsw+nyw-1
837 
838  if (nx < nxw .or. ny < nyw .or. nz < field(index)%siz(3)) then

841  ' Array "data" is too small. shape(data)='//message1)
842  endif

844  if (size(mask_out,1) /= nx .or. size(mask_out,2) /= ny .or. size(mask_out,3) /= nz) then
845  write(message1,'(i6,2i5)') nx,ny,nz
846  write(message2,'(i6,2i5)') size(mask_out,1),size(mask_out,2),size(mask_out,3)

851  endif

854  ! only one record in the file => time-independent field

857  if( field(index)%region_type == no_region ) then

862  data(isw:iew,jsw:jew,:) = field(index)%missing
863  end where
864  else
865  where(field(index)%mask(isc:iec,jsc:jec,:,i1))
866  data(isw:iew,jsw:jew,:) = field(index)%data(isc:iec,jsc:jec,:,i1)
867  end where
868  endif
869  if(PRESENT(mask_out)) &
870  mask_out(isw:iew,jsw:jew,:) = field(index)%mask(isc:iec,jsc:jec,:,i1)
871  else
872  if(field(index)%have_modulo_times) then
873  call time_interp(time,field(index)%modulo_time_beg, field(index)%modulo_time_end, field(index)%time(:), &
874  w2, t1, t2, field(index)%correct_leap_year_inconsistency, err_msg=err_msg)

878  ",file="//trim(filename)//",field="//trim(field(index)%name) )
879  endif
880  else
881  if(field(index)%modulo_time) then
882  mod_time=1
883  else
884  mod_time=0
885  endif
886  call time_interp(time,field(index)%time(:),w2,t1,t2,modtime=mod_time, err_msg=err_msg)
887  if(err_msg .NE. '') then
888  filename = trim(field(index)%fileobj%path)
889  call mpp_error(fatal,"time_interp_external 2: "//trim(err_msg)//&
890  ",file="//trim(filename)//",field="//trim(field(index)%name) )
891  endif
892  endif

894  if (verb) then

896  write(outunit,'(a,i4,a,i2,a,i2,1x,i2,a,i2,a,i2)') &
897  'target time yyyy/mm/dd hh:mm:ss= ',yy,'/',mm,'/',dd,hh,':',min,':',ss
898  write(outunit,*) 't1, t2, w1, w2= ', t1, t2, w1, w2
899  endif
900 
901  call load_record(field(index),t1,horz_interp, is_in, ie_in ,js_in, je_in, window_id)
902  call load_record(field(index),t2,horz_interp, is_in, ie_in ,js_in, je_in, window_id)
903  i1 = find_buf_index(t1,field(index)%ibuf)
904  i2 = find_buf_index(t2,field(index)%ibuf)
905  if(i1<0.or.i2<0) &
906  call mpp_error(fatal,'time_interp_external : records were not loaded correctly in memory')
907

909  write(outunit,*) 'ibuf= ',field(index)%ibuf
910  write(outunit,*) 'i1,i2= ',i1, i2
911  endif
912 
913  if( field(index)%region_type == no_region ) then
914  where(field(index)%mask(isc:iec,jsc:jec,:,i1).and.field(index)%mask(isc:iec,jsc:jec,:,i2))
915  data(isw:iew,jsw:jew,:) = field(index)%data(isc:iec,jsc:jec,:,i1)*w1 + &
916  field(index)%data(isc:iec,jsc:jec,:,i2)*w2
917  elsewhere
918 ! data(isw:iew,jsw:jew,:) = time_interp_missing !field(index)%missing? Balaji
919  data(isw:iew,jsw:jew,:) = field(index)%missing
920  end where
921  else
922  where(field(index)%mask(isc:iec,jsc:jec,:,i1).and.field(index)%mask(isc:iec,jsc:jec,:,i2))
923  data(isw:iew,jsw:jew,:) = field(index)%data(isc:iec,jsc:jec,:,i1)*w1 + &
924  field(index)%data(isc:iec,jsc:jec,:,i2)*w2
925  end where
926  endif
927  if(PRESENT(mask_out)) &
928  mask_out(isw:iew,jsw:jew,:) = &
929  field(index)%mask(isc:iec,jsc:jec,:,i1).and.&
930  field(index)%mask(isc:iec,jsc:jec,:,i2)
931  endif
