19 !> @defgroup diag_table_mod diag_table_mod
20 !> @ingroup diag_manager
21 !! @brief <TT>diag_table_mod</TT> is a set of subroutines use to parse out the data from a <TT>diag_table</TT>. This module
22 !! will also setup the arrays required to store the information by counting the number of input fields, output files, and
23 !! files.
24 !! @author Seth Underwood
25 !!
26 !! <TT>diag_table_mod</TT> parses the <TT>diag_table</TT> file, and sets up the required arrays to hold the information
27 !! needed for the <TT>diag_manager_mod</TT> to correctly write out the model history files.
28 !!
29 !! The <I>diagnostics table</I> allows users to specify sampling rates and the choice of fields at run time. The
30 !! <TT>diag_table</TT> file consists of comma-separated ASCII values. The <TT>diag_table</TT> essentially has three sections:
31 !! <B>Global</B>, <B>File</B>, and <B>Field</B> sections. The <B>Global</B> section must be the first two lines of the file,
32 !! whereas the <B>File</B> and <B>Field</B> sections can be inter mixed to allow the file to be organized as desired.
33 !! Comments can be added to the <TT>diag_table</TT> file by using the hash symbol (#) as the first character in the line.
34 !!
35 !! All errors in the <TT>diag_table</TT> will throw a <TT>FATAL</TT> error. A simple utility <TT>diag_table_chk</TT>has been
36 !! added to the FRE tools suite to check a <TT>diag_table</TT> for errors. A brief usage statement can be obtained by running
37 !! <TT>diag_table_chk --help</TT>, and a man page like description can views by running <TT>perldoc diag_table_chk</TT>.
38 !!
39 !! Below is a description of the three sections.
40 !! <OL>
41 !! <LI>
42 !! <B>Global Section:</B> The first two lines of the <TT>diag_table</TT> must contain the <I>title</I> and the <I>base
43 !! date</I> of the experiment respectively. The <I>title</I> must be a Fortran CHARACTER string. The <I>base date</I>
44 !! is the reference time used for the time units, and must be greater than or equal to the model start time.
45 !! The <I>base date</I> consists of six space-separated integer in the following format.<BR />
46 !! <TT> year month day hour minute second </TT><BR />
47 !! </LI>
48 !! <LI>
49 !! <B>File Section:</B> File lines contain 6 required and 5 optional fields (optional fields are surrounded with
50 !! square brackets ([]). File lines can be intermixed with the field lines, but the file must be defined before any
51 !! fields that are to be written to the file. File lines have the following format:<BR />
52 !! "file_name", output_freq, "output_freq_units", file_format, "time_axis_units", "time_axis_name"
53 !! [, new_file_freq, "new_file_freq_units"[, "start_time"[, file_duration, "file_duration_units"]]]
54 !! <BR />
55 !! with the following descriptions.
56 !! <DL>
57 !! <DT><TT>CHARACTER(len=128) :: file_name</TT></DT>
58 !! <DD>
59 !! Output file name without the trailing <TT>".nc"</TT>.
60 !!
61 !! A single file description can produce multiple files using special time string suffix keywords. This time string
62 !! will append the time strings to the base file name each time a new file is opened. They syntax for the time string
63 !! suffix keywords are <TT>%#tt</TT> Where <TT>#</TT> is a mandatory single digit number specifying the width of the
64 !! field, and <TT>tt</TT> can be as follows:
65 !! <UL>
66 !! <LI><TT>yr</TT> Years</LI>
67 !! <LI><TT>mo</TT> Months</LI>
68 !! <LI><TT>dy</TT> Days</LI>
69 !! <LI><TT>hr</TT> Hours</LI>
70 !! <LI><TT>mi</TT> Minutes</LI>
71 !! <LI><TT>sc</TT> Seconds</LI>
72 !! </UL>
73 !! Thus, a file name of <TT>file2_yr_dy%1yr%3dy</TT> will have a base file name of <TT>file2_yr_dy_1_001</TT> if the
74 !! file is created on year 1 day 1 of the model run. <B><I>NOTE:</I></B> The time suffix keywords must be used if the
75 !! optional fields <TT>new_file_freq</TT> and <TT>new_file_freq_units</TT> are used, otherwise a <TT>FATAL</TT> error
76 !! will occur.
77 !! </DD>
78 !!
79 !! <DT><TT>INTEGER :: output_freq</TT></DT>
80 !! <DD>How often to write fields to file.
81 !! <UL>
82 !! <LI><TT>> 0</TT> <EN /> Output frequency in <TT>output_freq_units</TT>.</LI>
83 !! <LI><TT>= 0</TT> <EN /> Output frequency every time set. (<TT>output_freq_units</TT> is ignored.)</LI>
84 !! <LI><TT>=-1</TT> <EN /> Output at end of run only. (<TT>output_freq_units</TT> is ignored.)</LI>
85 !! </UL>
86 !! </DD>
87 !! <DT><TT>CHARACTER(len=10) :: output_freq_units</TT></DT>
88 !! <DD>
89 !! Time units for output. Can be either <TT>years</TT>, <TT>months</TT>, <TT>days</TT>, <TT>minutes</TT>,
90 !! <TT>hours</TT>, or <TT>seconds</TT>.
91 !! </DD>
92 !! <DT><TT>INTEGER :: file_format</TT></DT>
93 !! <DD>
94 !! Output file format. Currently only the <I>netCDF</I> file format is supported.
95 !! <UL>
96 !! <LI><TT>= 1</TT> <EN /> netCDF</LI>
97 !! </UL>
98 !! </DD>
99 !! <DT><TT>CHARACTER(len=10) :: time_axis_units</TT></DT>
100 !! <DD>
101 !! Time units for the output file time axis. Can be either <TT>years</TT>, <TT>months</TT>, <TT>days</TT>,
102 !! <TT>minutes</TT>, <TT>hours</TT>, or <TT>seconds</TT>.
103 !! </DD>
104 !! <DT><TT>CHARACTER(len=128) :: time_axis_name</TT></DT>
105 !! <DD>
106 !! Axis name for the output file time axis. The character sting must contain the string 'time'. (mixed upper and
107 !! lowercase allowed.)
108 !! </DD>
109 !! <DT><TT>INTEGER, OPTIONAL :: new_file_freq</TT></DT>
110 !! <DD>
111 !! Frequency for closing the existing file, and creating a new file in <TT>new_file_freq_units</TT>.
112 !! </DD>
113 !! <DT><TT>CHARACTER(len=10), OPTIONAL :: new_file_freq_units</TT></DT>
114 !! <DD>
115 !! Time units for creating a new file. Can be either <TT>years</TT>, <TT>months</TT>, <TT>days</TT>,
116 !! <TT>minutes</TT>, <TT>hours</TT>, or <TT>seconds</TT>. <B><I>NOTE:</I></B> If the <TT>new_file_freq</TT> field is
117 !! present, then this field must also be present.
118 !! </DD>
119 !! <DT><TT>CHARACTER(len=25), OPTIONAL :: start_time</TT></DT>
120 !! <DD>
121 !! Time to start the file for the first time. The format of this string is the same as the <I>global date</I>. <B><I>
122 !! NOTE:</I></B> The <TT>new_file_freq</TT> and the <TT>new_file_freq_units</TT> fields must be present to use this field.
123 !! </DD>
124 !! <DT><TT>INTEGER, OPTIONAL :: file_duration</TT></DT>
125 !! <DD>
126 !! How long file should receive data after start time in <TT>file_duration_units</TT>. This optional field can only
127 !! be used if the <TT>start_time</TT> field is present. If this field is absent, then the file duration will be equal
128 !! to the frequency for creating new files. <B><I>NOTE:</I></B> The <TT>file_duration_units</TT> field must also be
129 !! present if this field is present.
130 !! </DD>
131 !! <DT><TT>CHARACTER(len=10), OPTIONAL :: file_duration_units</TT></DT>
132 !! <DD>
133 !! File duration units. Can be either <TT>years</TT>, <TT>months</TT>, <TT>days</TT>,
134 !! <TT>minutes</TT>, <TT>hours</TT>, or <TT>seconds</TT>. <B><I>NOTE:</I></B> If the <TT>file_duration</TT> field is
135 !! present, then this field must also be present.
136 !! </DD>
137 !! </DL>
138 !! </LI>
139 !! <LI>
140 !! <B>Field Section:</B> Field lines contain 8 fields. Field lines can be intermixed with file lines. Fields line can contain
141 !! fields that are not written to any files. The file name for these fields is <TT>null</TT>.
142 !!
143 !! Field lines have the following format:<BR />
144 !! <PRE>
145 !! "module_name", "field_name", "output_name", "file_name", "time_sampling", "reduction_method", "regional_section", packing
146 !! </PRE>
147 !! with the following descriptions.
148 !! <DL>
149 !! <DT><TT>CHARACTER(len=128) :: module_name</TT></DT>
150 !! <DD>Module that contains the <TT>field_name</TT> variable. (e.g. <TT>atmos_mod</TT>, <TT>land_mod</TT>)</DD>
151 !! <DT><TT>CHARACTER(len=128) :: field_name</TT></DT>
152 !! <DD>Module variable name that has data to be written to file.</DD>
153 !! <DT><TT>CHARACTER(len=128) :: output_name</TT></DT>
154 !! <DD>Name of the field as written in <TT>file_name</TT>.</DD>
155 !! <DT><TT>CHARACTER(len=128) :: file_name</TT></DT>
156 !! <DD>
157 !! Name of the file where the field is to be written. <B><I>NOTE:</I></B> The file <TT>file_name</TT> must be
158 !! defined first.
159 !! </DD>
160 !! <DT><TT>CHARACTER(len=50) :: time_sampling</TT></DT>
161 !! <DD>Currently not used. Please use the string "all".</DD>
162 !! <DT><TT>CHARACTER(len=50) :: reduction_method</TT></DT>
163 !! <DD>
164 !! The data reduction method to perform prior to writing data to disk. Valid options are (redundant names are
165 !! separated with commas):
166 !! <DL>
167 !! <DT><TT>.TRUE.</TT>, average, avg, mean</DT>
168 !! <DD>Average from the last time written to the current time.</DD>
169 !! <DT><TT>.FALSE.</TT>, none</DT>
170 !! <DD>No reduction performed. Write current time step value only.</DD>
171 !! <DT>rms</DT> <DD>Calculate the root mean square from the last time written to the current time.</DD>
172 !! <DT>pow##</DT> <DD>Calculate the mean of the power ## from the last time written to the current time.</DD>
173 !! <DT>min</DT> <DD>Minimum value from last write to current time.</DD>
174 !! <DT>max</DT> <DD>Maximum value from last write to current time.</DD>
175 !! <DT>diurnal##</DT> <DD>## diurnal averages</DD>
176 !! </DL>
177 !! </DD>
178 !! <DT><TT>CHARACTER(len=50) :: regional_section</TT></DT>
179 !! <DD>
180 !! Bounds of the regional section to capture. A value of <TT>none</TT> indicates a global region. The regional
181 !! section has the following format:<BR />
182 !! <TT>lat_min, lat_max, lon_min, lon_max, vert_min, vert_max</TT><BR />
183 !! Use <TT>vert_min = -1</TT> and <TT>vert_max = -1</TT> to get the entire vertical axis. <B><I>NOTE:</I></B>
184 !! Currently, the defined region <I>MUST</I> be confined to a single tile.
185 !! </DD>
186 !! <DT><TT>INTEGER :: packing</TT></DT>
187 !! <DD>
188 !! Fortran number <TT>KIND</TT> of the data written. Valid values:
189 !! - <TT>= 1</TT> <EN /> double precision
190 !! - <TT>= 2</TT> <EN /> float
191 !! - <LI><TT>= 4</TT> <EN /> packed 16-bit integers
192 !! - <TT>= 8</TT> <EN /> packed 1-byte (not tested)
193 !! </DD>
194 !! </DL>
195 !! </LI>
196 !! </OL>
197 !!
198 !! <H4><B>Sample <TT>diag_table</TT></B></H4>
199 !! <LI>
200 !! <PRE>
201 !! "diag manager test"
202 !! 1999 1 1 0 0 0
203 !!
204 !! #output files
205 !! 10_days, 10, "days", 1, "hours", "Time"
206 !! "file1_hr%hr3", 5, "days", 1, "hours", "Time", 15, "days"
207 !! "file2_yr_dy%yr1%dy3", 5, "days", 1, "hours", "Time", 10, "days", "1 1 7 0 0 0"
208 !! "file3_yr_dy%yr1%dy3", 5, "days", 1, "hours", "Time", 20, "days", "1 1 7 0 0 0", 5, "years"
209 !!
210 !! #output variables
211 !! "ice_mod", "ice", "ice", "10_days", "all", .false., "none", 2
212 !!
213 !! # temp_local file and fields.
214 !! temp_local, 1, "days", 1, "hours", "Time"
215 !! "ocean_mod", "temp", "temp", "temp_local", "all", .FALSE., "5 259.5 -59.5 59.5 1 1", 2
216 !! </PRE>
217 !! </LI>

279  !> @brief Parse the <TT>diag_table</TT> in preparation for diagnostic output.
280  !! @details <TT>parse_diag_table</TT> is the public interface to parse the diag_table, and setup the arrays needed to store the
281  !! requested diagnostics from the <TT>diag_table</TT>. <TT>parse_diag_table</TT> will return a non-zero <TT>istat</TT> if
282  !! a problem parsing the <TT>diag_table</TT>.
283  !!
284  !! NOT YET IMPLEMENTED: <TT>parse_diag_table</TT> will parse through the <TT>diag_table</TT> twice. The first pass, will be
285  !! to get a good "guess" of array sizes. These arrays, that will hold the requested diagnostic fields and files, will then be
286  !! allocated to the size of the "guess" plus a slight increase.
287  SUBROUTINE parse_diag_table(diag_subset, istat, err_msg)
288  INTEGER, INTENT(in), OPTIONAL :: diag_subset !< Diagnostic sampling subset.
289  INTEGER, INTENT(out), OPTIONAL, TARGET :: istat !< Status of parsing the <TT>diag_table</TT>.
290  !! A non-zero status indicates a problem parsing the table.
291  CHARACTER(len=*), INTENT(out), OPTIONAL :: err_msg !< Error message corresponding to the
292  !! <TT>istat</TT> return value.
293 
294  INTEGER, PARAMETER :: dt_line_length = 256
295 
296  INTEGER :: stdlog_unit !< Fortran file unit number for the stdlog file.
297  INTEGER :: record_len !< String length of the diag_table line read in.
298  INTEGER :: num_lines !< Number of lines in diag_table
299  INTEGER :: line_num !< Integer representation of the line number.
300  INTEGER :: commentstart !< Index location of first '#' on line
301  INTEGER :: diag_subset_output !< local value of diag_subset
302  INTEGER :: nfields, nfiles !< Number of fields and files. Not used yet.
303  INTEGER :: npass !< number of passes done while parsing the diag_table (1 for files, 2 for fields)
304  INTEGER, TARGET :: mystat !< variable to hold return status of function/subroutine calls.
305  INTEGER, POINTER :: pstat !< pointer that points to istat if preset, otherwise, points to mystat.
306 
307  CHARACTER(len=5) :: line_number !< String representation of the line number.
308  CHARACTER(len=9) :: amonth !< Month name
309  CHARACTER(len=256) :: record_line !< Current line from the diag_table.
310  CHARACTER(len=256) :: local_err_msg !< Sting to hold local error messages.
311  CHARACTER(len=:), DIMENSION(:), ALLOCATABLE :: diag_table
312 
313  TYPE(file_description_type) :: temp_file
314  TYPE(field_description_type) :: temp_field
315 
316  ! set up the pstat pointer
317  IF ( PRESENT(istat) ) THEN
318  pstat => istat
319  ELSE
320  pstat => mystat
321  END IF
322  ! Default return value (success)
323  pstat = 0
324 
325  IF ( PRESENT(diag_subset) ) THEN
326  diag_subset_output = diag_subset
327  ELSE
328  diag_subset_output = diag_all
329  END IF
330 
331  ! get the stdlog unit number
332  stdlog_unit = stdlog()
333 
334  call ascii_read('diag_table', diag_table, num_lines=num_lines)
335 
336  ! Read in the global file labeling string
337  READ (unit=diag_table(1), fmt=*, iostat=mystat) global_descriptor
338  IF ( mystat /= 0 ) THEN
339  pstat = mystat
340  IF ( fms_error_handler('diag_table_mod::parse_diag_table', 'Error reading the global descriptor from the diagnostic table.',&
341  & err_msg) ) RETURN
342  END IF
343 
344  ! Read in the base date
345  READ (unit=diag_table(2), fmt=*, iostat=mystat) base_year, base_month, base_day, base_hour, base_minute, base_second
346  IF ( mystat /= 0 ) THEN
347  pstat = mystat
348  IF ( fms_error_handler('diag_manager_init', 'Error reading the base date from the diagnostic table.', err_msg) ) RETURN
349  END IF
350 
351  ! Set up the time type for base time
352  IF ( get_calendar_type() /= no_calendar ) THEN
353  IF ( base_year==0 .OR. base_month==0 .OR. base_day==0 ) THEN
354  pstat = 101
355  IF ( fms_error_handler('diag_table_mod::parse_diag_table', 'The base_year/month/day can not equal zero', err_msg) ) RETURN
356  END IF
357  base_time = set_date(base_year, base_month, base_day, base_hour, base_minute, base_second)
358  amonth = month_name(base_month)
359  ELSE
360  ! No calendar - ignore year and month
361  base_time = set_time(nint(base_hour*seconds_per_hour)+nint(base_minute*seconds_per_minute)+base_second, base_day)
362  base_year = 0
363  base_month = 0
364  amonth = 'day'
365  END IF
366 
367  IF ( mpp_pe() == mpp_root_pe() ) THEN
368  WRITE (stdlog_unit,'("base date used = ",I4,1X,A,2I3,2(":",I2.2)," gmt")') base_year, trim(amonth), base_day, &
369  & base_hour, base_minute, base_second
370  END IF
371 
372  nfiles=0
373  nfields=0
374  pass: DO npass = 1, 2
375  parser: DO line_num=3, num_lines
376  ! Read in the entire line from the file.
377  ! If there is a read error, give a warning, and
378  ! cycle the parser loop.
379  READ (diag_table(line_num), fmt='(A)', iostat=mystat) record_line
380  ! Increase line counter, and put in string for use in warning/error messages.
381  WRITE (line_number, '(I5)') line_num
382 
383  IF ( mystat > 0 ) THEN
384  IF ( mpp_pe() == mpp_root_pe() ) &
385  & CALL error_mesg("diag_table_mod::parse_diag_table",&
386  & "Problem reading the diag_table (line:" //line_number//").", fatal)
387  cycle parser
388  ELSE IF ( mystat < 0 ) THEN
389  EXIT parser
390  END IF
391 
392  ! How long is the read in string?
393  record_len = len_trim(record_line)
394 
395  ! ignore blank lines and lines with comments only (comment marker '#')
396  commentstart = index(record_line,'#')
397  IF ( commentstart .NE. 0 ) record_line = record_line(1:commentstart-1)
398  IF ( len_trim(record_line) == 0 .OR. record_len == 0 ) cycle parser
399 
400  init: IF ( npass == 1 ) THEN ! Checking for files only
401  IF ( is_a_file(trim(record_line)) ) THEN
402  temp_file = parse_file_line(line=record_line, istat=mystat, err_msg=local_err_msg)
403 
404  IF ( mystat > 0 ) THEN
405  CALL error_mesg("diag_table_mod::parse_diag_table",&
406  & trim(local_err_msg)//" (line:" //trim(line_number)//").", fatal)
407  ELSE IF ( mystat < 0 ) THEN
408  IF ( mpp_pe() == mpp_root_pe() )&
409  & CALL error_mesg("diag_table_mod::parse_diag_table",&
410  & trim(local_err_msg)//" (line: "//trim(line_number)//").", warning)
411  cycle parser
412  ELSE IF ( (diag_subset_output == diag_other .AND. index(lowercase(temp_file%file_name), "ocean") .NE. 0).OR.&
413  & (diag_subset_output == diag_ocean .AND. index(lowercase(temp_file%file_name), "ocean") .EQ. 0) ) THEN
414  cycle parser
415  ELSE IF ( temp_file%new_file_freq > 0 ) THEN ! Call the init_file subroutine. The '1' is for the tile_count
416  CALL init_file(temp_file%file_name, temp_file%output_freq, temp_file%iOutput_freq_units, temp_file%file_format,&
417  & temp_file%iTime_units, temp_file%long_name, 1, temp_file%new_file_freq, temp_file%iNew_file_freq_units,&
418  & temp_file%start_time, temp_file%file_duration, temp_file%iFile_duration_units, temp_file%filename_time_bounds)
419  ELSE
420  CALL init_file(temp_file%file_name, temp_file%output_freq, temp_file%iOutput_freq_units, temp_file%file_format,&
421  & temp_file%iTime_units, temp_file%long_name, 1)
422  END IF
423 
424  ! Increment number of files
425  nfiles = nfiles + 1
426  END IF
427  ELSE ! Looking for fields
428  IF ( .NOT.is_a_file(trim(record_line)) ) THEN
429  temp_field = parse_field_line(line=record_line, istat=mystat, err_msg=local_err_msg)
430 
431  ! Check for errors, then initialize the input and output field
432  IF ( mystat > 0 ) THEN
433  CALL error_mesg("diag_table_mod::parse_diag_table",&
434  & trim(local_err_msg)//" (line: "//trim(line_number)//").",fatal)
435  ELSE IF ( mystat < 0 ) THEN
436  IF ( mpp_pe() == mpp_root_pe() )&
437  & CALL error_mesg("diag_table_mod::Parse_diag_table",&
