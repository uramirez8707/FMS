69  ! Content check

71  do i=1, num_lines-1
72  read(2, '(A)', iostat=stat) line
73  if (stat.eq.-1) then
74  call mpp_error(fatal, "Problem reading the ascii file")
75  end if

81  ! Tests meant to raise errors
