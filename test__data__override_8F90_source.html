313  subroutine get_grid
314  real, allocatable, dimension(:,:,:) :: lon_vert_glo, lat_vert_glo
315  real, allocatable, dimension(:,:) :: lon_global, lat_global
316  integer, dimension(4) :: siz
317  character(len=128) :: message
318 
319 
320  if(field_exist(grid_file, 'x_T')) then
321  call field_size(grid_file, 'x_T', siz)
322  if(siz(1) /= nlon .or. siz(2) /= nlat) then
323  write(message,'(a,2i4)') 'x_T is wrong shape. shape(x_T)=',siz(1:2)
324  call error_mesg('test_data_override', trim(message), fatal)
325  endif
326  allocate(lon_vert_glo(nlon,nlat,4), lat_vert_glo(nlon,nlat,4) )
327  allocate(lon_global(nlon,nlat ), lat_global(nlon,nlat ) )
328  call read_data(trim(grid_file), 'x_vert_T', lon_vert_glo, no_domain=.true.)
329  call read_data(trim(grid_file), 'y_vert_T', lat_vert_glo, no_domain=.true.)
330  lon_global(:,:) = (lon_vert_glo(:,:,1) + lon_vert_glo(:,:,2) + lon_vert_glo(:,:,3) + lon_vert_glo(:,:,4))*0.25
331  lat_global(:,:) = (lat_vert_glo(:,:,1) + lat_vert_glo(:,:,2) + lat_vert_glo(:,:,3) + lat_vert_glo(:,:,4))*0.25
332  else if(field_exist(grid_file, "geolon_t" ) ) then
333  call field_size(grid_file, 'geolon_vert_t', siz)
334  if(siz(1) /= nlon+1 .or. siz(2) /= nlat+1) then
335  write(message,'(a,2i4)') 'geolon_vert_t is wrong shape. shape(geolon_vert_t)=',siz(1:2)
336  call error_mesg('test_data_override', trim(message), fatal)
337  endif
338  allocate(lon_vert_glo(nlon+1,nlat+1,1), lat_vert_glo(nlon+1,nlat+1,1))
339  allocate(lon_global(nlon, nlat ), lat_global(nlon, nlat ))
340  call read_data(trim(grid_file), 'geolon_vert_t', lon_vert_glo, no_domain=.true.)
341  call read_data(trim(grid_file), 'geolat_vert_t', lat_vert_glo, no_domain=.true.)
342 
343  do i = 1, nlon
344  do j = 1, nlat
345  lon_global(i,j) = (lon_vert_glo(i,j,1) + lon_vert_glo(i+1,j,1) + &
346  lon_vert_glo(i+1,j+1,1) + lon_vert_glo(i,j+1,1))*0.25
347  lat_global(i,j) = (lat_vert_glo(i,j,1) + lat_vert_glo(i+1,j,1) + &
348  lat_vert_glo(i+1,j+1,1) + lat_vert_glo(i,j+1,1))*0.25
349  enddo
350  enddo
351  else if( field_exist(grid_file, "ocn_mosaic_file") ) then ! reading from mosaic file
352  call field_size(tile_file, 'area', siz)
353  if(siz(1) /= nlon*2 .or. siz(2) /= nlat*2) then
354  write(message,'(a,2i4)') 'area is wrong shape. shape(area)=',siz(1:2)
355  call error_mesg('test_data_override', trim(message), fatal)
356  endif
357  allocate(lon_vert_glo(siz(1)+1,siz(2)+1,1), lat_vert_glo(siz(1)+1,siz(2)+1,1))
358  allocate(lon_global(nlon, nlat ), lat_global(nlon, nlat ))
359  call read_data( tile_file, 'x', lon_vert_glo, no_domain=.true.)
360  call read_data( tile_file, 'y', lat_vert_glo, no_domain=.true.)
361  do j = 1, nlat
362  do i = 1, nlon
363  lon_global(i,j) = lon_vert_glo(i*2,j*2,1)
364  lat_global(i,j) = lat_vert_glo(i*2,j*2,1)
365  end do
366  end do
367  end if
368 
369  allocate(lon(is:ie,js:je), lat(is:ie,js:je))
370  lon = lon_global(is:ie,js:je)
371  lat = lat_global(is:ie,js:je)
372 
373  deallocate(lon_vert_glo)
374  deallocate(lat_vert_glo)
375  deallocate(lon_global)
376  deallocate(lat_global)
377 
378  end subroutine get_grid

380  subroutine test_unstruct_grid( type, Time )
