31 subroutine mpp_error_basic( errortype, errormsg )
32  !a very basic error handler
33  !uses ABORT and FLUSH calls, may need to use cpp to rename
34  integer, intent(in) :: errortype
35  character(len=*), intent(in), optional :: errormsg
36  character(len=512) :: text
37  logical :: opened
38  integer :: istat, errunit, outunit
39 
40  if( .NOT.module_is_initialized )call abort()
41 
42  select case( errortype )
43  case(note)
44  text = 'NOTE' !just FYI
45  case(warning)
46  text = 'WARNING' !probable error
47  case(fatal)
48  text = 'FATAL' !fatal error
49  case default
50  text = 'WARNING: non-existent errortype (must be NOTE|WARNING|FATAL)'
51  end select
52 
53  if( npes.GT.1 )write( text,'(a,i5)' )trim(text)//' from PE', pe !this is the mpp part
54  if( PRESENT(errormsg) )text = trim(text)//': '//trim(errormsg)
55 
56  errunit = stderr()
57  outunit = stdout()
58 
59  select case( errortype )
60  case(note)
61  write( outunit,'(a)' )trim(text)
62  case default
63  write( errunit,'(/a/)' )trim(text)
64  write( outunit,'(/a/)' )trim(text)
65  if( errortype.EQ.fatal .OR. warnings_are_fatal )then
66  FLUSH(outunit)
67  call abort() !automatically calls traceback on Cray systems
68  end if
69  end select
70 
71  error_state = errortype
72  return
73 end subroutine mpp_error_basic
