19  ! <SUBROUTINE NAME="MPP_SCATTER_PELIST_2D_">
20  ! <OVERVIEW>
21  ! Scatter data from one pe to the specified pes.
22  ! </OVERVIEW>
23  ! <TEMPLATE>
24  ! SUBROUTINE MPP_SCATTER_PELIST_2D_(is, ie, js, je, pelist, array_seg, data, is_root_pe, &
25  ! ishift, jshift)
26  ! </TEMPLATE>
27  ! <DESCRIPTION>
28  ! Scatter (ie - is) * (je - js) contiguous elements of array data from the designated root pe
29  ! into contigous members of array segment in each pe that is included in the pelist argument.
30  ! </DESCRIPTION>
31  ! <IN NAME="is, ie" TYPE="INTEGER">Start and end index of the first dimension of the segment array</IN>
32  ! <IN NAME="js, je" TYPE="INTEGER">Start and end index of the second dimension of the segment array</IN>
33  ! <IN NAME="pelist" TYPE="INTEGER()">The PE list of of target pes, Needs to be in monotonic increasing order.
34  ! The root pe is allowed to be included (see input is_root_pe). If a pe is absent in this list then
35  ! its segment array is not updated. </IN>
36  ! <IN NAME="array_seg" TYPE="REAL(:)">The 2D array that the data is to be copied into</IN>
37  ! <IN NAME="data" TYPE="REAL(:)">The source array.</IN>
38  ! <IN NAME="is_root_pe" TYPE="LOGICAL" > True if the calee is root pe, false otherwise. </IN>
39  ! <IN NAME="ishift, jshift" TYPE="INTEGER" > Offsets specifying the first element in data array.</IN>
40 subroutine MPP_SCATTER_PELIST_2D_(is, ie, js, je, pelist, array_seg, data, is_root_pe, &
41  ishift, jshift)
42  integer, intent(in) :: is, ie, js, je
43  integer, dimension(:), intent(in) :: pelist
44  MPP_TYPE_, dimension(is:ie,js:je), intent(inout) :: array_seg
45  MPP_TYPE_, dimension(:,:), intent(in) :: data
46  logical, intent(in) :: is_root_pe
47  integer, optional, intent(in) :: ishift, jshift
48 
49  MPP_TYPE_ :: arr3D(size(array_seg,1),size(array_seg,2),1)
50  MPP_TYPE_ :: data3D(size( data,1),size( data,2),1)
51  pointer( aptr, arr3D )
52  pointer( dptr, data3D )
53  aptr = LOC(array_seg)
54  dptr = LOC( data)
55 
56  call mpp_scatter(is, ie, js, je, 1, pelist, arr3D, data3D, is_root_pe, &
57  ishift, jshift)
58  return

60 end subroutine MPP_SCATTER_PELIST_2D_

63 subroutine MPP_SCATTER_PELIST_3D_(is, ie, js, je, nk, pelist, array_seg, data, is_root_pe, &
64  ishift, jshift)
65  integer, intent(in) :: is, ie, js, je, nk
66  integer, dimension(:), intent(in) :: pelist
67  MPP_TYPE_, dimension(is:ie,js:je,1:nk), intent(inout) :: array_seg
68  MPP_TYPE_, dimension(:,:,:), intent(in) :: data
69  logical, intent(in) :: is_root_pe
70  integer, optional, intent(in) :: ishift, jshift
71 
72  integer :: i, msgsize, root_pe, root_pe_test
73  integer :: i1, i2, j1, j2, ioff, joff
74  integer :: my_ind(4), gind(4,size(pelist))
75  type array3D
76  MPP_TYPE_, dimension(:,:,:), allocatable :: data
77  endtype array3D
78  type(array3d), dimension(size(pelist)) :: temp
79 
80  if (.not.ANY(mpp_pe().eq.pelist(:))) return
81 
82  if (is_root_pe) then
83  root_pe = mpp_pe()
84  root_pe_test = 999
85  if (.not.ANY(pelist(:).eq.root_pe)) call mpp_error(FATAL, &
86  "fms_io(mpp_scatter_pelist): root_pe not a member of pelist")
87  else
88  root_pe = 0
89  root_pe_test = -999
90  endif
91 ! need this check in case MPI-rank 0 is a member of the pelist
92  call mpp_max(root_pe_test, pelist)
93  if (root_pe_test.lt.0) call mpp_error(FATAL, &
94  "fms_io(mpp_scatter_pelist): root_pe not specified or not a member of the pelist")
95 ! need to make sure only one root_pe has been specified
96  call mpp_sum(root_pe, pelist)
97  if ((is_root_pe) .and. (mpp_pe().ne.root_pe)) call mpp_error(FATAL, &
98  "fms_io(mpp_scatter_pelist): too many root_pes specified")
99 
100 
101  ioff=0
102  joff=0
103  if (present(ishift)) ioff=ishift
104  if (present(jshift)) joff=jshift
105 
106  my_ind(1) = is
107  my_ind(2) = ie
108  my_ind(3) = js
109  my_ind(4) = je
110 
111 ! scatter indices into global index on root_pe
112  if (is_root_pe) then
113  do i = 1, size(pelist)
114 ! root_pe data copy - no send to self
115  if (pelist(i).eq.root_pe) then
116  gind(:,i) = my_ind(:)
117  else
118  call mpp_recv(gind(:,i:i), 4, pelist(i), .FALSE., COMM_TAG_1)
119  endif
120  enddo
121  call mpp_sync_self(check=EVENT_RECV)
122  gind(1,:)=gind(1,:)+ioff
123  gind(2,:)=gind(2,:)+ioff
124  gind(3,:)=gind(3,:)+joff
125  gind(4,:)=gind(4,:)+joff
126 ! check indices to make sure they are within the range of "data"
127  if ((minval(gind).lt.1) .OR. (maxval(gind(1:2,:)).gt.size(data,1)) .OR. (maxval(gind(3:4,:)).gt.size(data,2))) &
128  call mpp_error(FATAL,"fms_io(mpp_scatter_pelist): specified indices (with shift) are outside of the &
129  &range of the receiving array")
130  else
131 ! non root_pe's send indices to root_pe
132  call mpp_send(my_ind(:), 4, root_pe, COMM_TAG_1)
133  call mpp_sync_self(check=EVENT_SEND)
134  endif
135 
136 ! scatter segments into data based on indices
137  if (is_root_pe) then
138  do i = 1, size(pelist)
139  if (pelist(i).ne.root_pe) then ! no send to self
140  i1 = gind(1,i)
141  i2 = gind(2,i)
142  j1 = gind(3,i)
143  j2 = gind(4,i)
144  msgsize = (i2-i1+1)*(j2-j1+1)*nk
145 ! allocate and copy data into a contiguous memory space
146  allocate(temp(i)%data(i1:i2,j1:j2,1:nk))
147  temp(i)%data(i1:i2,j1:j2,1:nk)=data(i1:i2,j1:j2,1:nk)
148  call mpp_send(temp(i)%data, msgsize, pelist(i), COMM_TAG_2)
149  else
150 ! data copy - no send to self
151  array_seg(is:ie,js:je,1:nk) = data(is+ioff:ie+ioff,js+joff:je+joff,1:nk)
152  endif
153  enddo
154  call mpp_sync_self(check=EVENT_SEND)
155 ! deallocate the temporary array used for the send
156  do i = 1, size(pelist)
157  if (allocated(temp(i)%data)) deallocate(temp(i)%data)
158  enddo
159  else
160 ! non root_pe's recv data from root_pe
161  msgsize = (my_ind(2)-my_ind(1)+1) * (my_ind(4)-my_ind(3)+1) * nk
162  call mpp_recv(array_seg, msgsize, root_pe, .FALSE., COMM_TAG_2)
163  call mpp_sync_self(check=EVENT_RECV)
164  endif
165 
166  call mpp_sync_self()
167 
168  return
169 
170 end subroutine MPP_SCATTER_PELIST_3D_
