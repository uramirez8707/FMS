20 !> @brief This programs tests test_get_mosaic_tile_grid.
21 program test_get_mosaic_tile_grid
22 use mpp_mod, only: mpp_npes, mpp_pe, mpp_root_pe, mpp_error, fatal
23 use mpp_domains_mod, only: domain2d, mpp_define_mosaic, mpp_define_io_domain
24 use fms2_io_mod, only: get_mosaic_tile_grid
25 use fms_mod, only: fms_init, fms_end
26 use netcdf, only: nf90_create, nf90_def_dim, nf90_clobber, nf90_64bit_offset, &
27  nf90_def_var, nf90_enddef, nf90_put_var, nf90_close, nf90_char, &
28  nf90_noerr
29 use mpp_domains_mod, only: domain2d, mpp_define_mosaic, mpp_define_io_domain
30 use mpp_mod, only: mpp_npes, mpp_pe, mpp_root_pe, mpp_error, fatal
31 use mpi, only: mpi_barrier, mpi_comm_world
32 
33 
34 implicit none
35 
36 type(domain2d) :: domain !< 6 tile cube sphere domain
37 integer, parameter :: ntiles = 6 !< Number of tiles
38 integer :: nx=60 !< Number of grid points in x
39 integer :: ny=60 !< Number of grid points in y
40 integer, dimension(4,ntiles) :: global_indices !< Global indices for each tile
41 integer, dimension(2,ntiles) :: layout !< Domain layout for each title
42 integer, dimension(2) :: io_layout !< Io layout for each tile
43 integer, dimension(ntiles) :: pe_start !< Starting PE for the given domain
44 integer, dimension(ntiles) :: pe_end !< Ending PE for the given domain
45 integer :: npes !< Number of PEs
46 character(len=256) :: expected_grid_file !< Expected grid_file name
47 character(len=256) :: grid_file !< Grid file read with the get_mosaic_tile_grid call
48 character(len=256) :: mosaic_file !< Filename where the grid_file is read from
49 integer :: i !< Index
50 integer :: err, ncid, dim1d, dim2d, varid
51 character (len = 255), dimension (7) :: char_in = (/'C48_grid.tile1.nc', &
52  'C48_grid.tile2.nc', &
53  'C48_grid.tile3.nc', &
54  'C48_grid.tile4.nc', &
55  'C48_grid.tile5.nc', &
56  'C48_grid.tile6.nc', &
57  'C48_grid.tile7.nc' /)
58 call fms_init
59 
60 !< Create a sample grid_spec file with the needed information
61 if (mpp_pe() .eq. mpp_root_pe()) then
62  err = nf90_create('grid_spec.nc', ior(nf90_clobber, nf90_64bit_offset), ncid)
63  if (err .ne. nf90_noerr) call mpp_error(fatal, "test_get_mosaic_tile_grid: Error creating file")
64 
65  err = nf90_def_dim(ncid, 'str', 255, dim1d)
66  if (err .ne. nf90_noerr) call mpp_error(fatal, "test_get_mosaic_tile_grid: Error creating dimension")
67 
68  err = nf90_def_dim(ncid, 'ntiles', 7, dim2d)
69  if (err .ne. nf90_noerr) call mpp_error(fatal, "test_get_mosaic_tile_grid: Error creating dimension")
70 
71  err = nf90_def_var(ncid, 'gridfiles', nf90_char, (/dim1d, dim2d/), varid)
72  if (err .ne. nf90_noerr) call mpp_error(fatal, "test_get_mosaic_tile_grid: Error defining variable")
73 
74  err = nf90_enddef(ncid)
75  if (err .ne. nf90_noerr) call mpp_error(fatal, "test_get_mosaic_tile_grid: Error leaving define mode")
76 
77  err = nf90_put_var(ncid, varid, char_in, start=(/ 1, 1/), count=(/ 255, 7 /))
78  if (err .ne. nf90_noerr) call mpp_error(fatal, "test_get_mosaic_tile_grid: Error adding variable")
79 
80  err = nf90_close(ncid)
81  if (err .ne. nf90_noerr) call mpp_error(fatal, "test_get_mosaic_tile_grid: Error closing file")
82 endif
83 
84 call mpi_barrier(mpi_comm_world, err)
85 
86 npes = mpp_npes()
87 
88 !< Create a 6 tile cube sphere domain
89 do i = 1,ntiles
90  global_indices(:, i) = (/1, nx, 1, ny/)
91  layout(:, i) = (/1, npes/ntiles/)
92  pe_start(i) = (i-1)*(npes/ntiles)
93  pe_end(i) = i*(npes/ntiles) - 1
94 enddo
95 io_layout(:) = 1
96 
97 call create_atmosphere_domain((/nx, nx, nx, nx, nx, nx/), &
98  (/ny, ny, ny, ny, ny, ny/), &
99  global_indices, layout, pe_start, pe_end, &
100  io_layout, domain)
101 
102 !< Try to read in the grid_spec.nc file to get the name of the grid file for whatever tile
103 !! you are in
104 mosaic_file = 'grid_spec.nc'
105 call get_mosaic_tile_grid(grid_file,mosaic_file, domain)
106 
107 !> Check that the grid_file was read correctly
108 
109 !> Because this domain has 6 tiles, and you are running on 6 pes. Each pe will be on its
110 !! own tile. Pe 0 will be in tile1, pe 1 will be in tile 2, etc ..
111 
112 i = mpp_pe()+1
113 write(expected_grid_file,'(A,I1,A)') "INPUT/C48_grid.tile", i, ".nc"
114 if (trim(grid_file) .ne. trim(expected_grid_file)) then
115  print *, "grid_file: ", trim(grid_file), " expected: ", trim(expected_grid_file)
116  call mpp_error(fatal, "test_get_mosaic_tile_grid: the grid_file read is not correct")
117 endif
118 
119 call fms_end
120 
121 contains
122 
123 include "create_atmosphere_domain.inc"
124 end program
