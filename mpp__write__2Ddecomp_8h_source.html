19  subroutine WRITE_RECORD_( unit, field, nwords, data, time_in, domain, tile_count)
20 !routine that is finally called by all mpp_write routines to perform the write
21 !a non-netCDF record contains:
22 ! field ID
23 ! a set of 4 coordinates (is:ie,js:je) giving the data subdomain
24 ! a timelevel and a timestamp (=NULLTIME if field is static)
25 ! 3D real data (stored as 1D)
26 !if you are using direct access I/O, the RECL argument to OPEN must be large enough for the above
27 !in a global direct access file, record position on PE is given by %record.
28 
29 !Treatment of timestamp:
30 ! We assume that static fields have been passed without a timestamp.
31 ! Here that is converted into a timestamp of NULLTIME.
32 ! For non-netCDF fields, field is treated no differently, but is written
33 ! with a timestamp of NULLTIME. There is no check in the code to prevent
34 ! the user from repeatedly writing a static field.
35 
36  integer, intent(in) :: unit, nwords
37  type(fieldtype), intent(in) :: field
38  MPP_TYPE_, intent(in) :: data(nwords)
39  MPP_TYPE_, intent(in), optional :: time_in
40  type(domain2D), intent(in), optional :: domain
41  integer, intent(in), optional :: tile_count
42  integer, dimension(size(field%axes(:))) :: start, axsiz
43  real(r8_kind) :: time
44  integer :: time_level
45  logical :: newtime
46  integer :: subdomain(4)
47  integer :: packed_data(nwords)
48  integer :: i, is, ie, js, je
49 
50  real(r4_kind) :: data_r4(nwords)
51  pointer( ptr1, data_r4)
52  pointer( ptr2, packed_data)
53 
54  if (mpp_io_stack_size < nwords) call mpp_io_set_stack_size(nwords)
55 
56  if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_WRITE: must first call mpp_io_init.' )
57  if( .NOT.mpp_file(unit)%write_on_this_pe) return
58  if( .NOT.mpp_file(unit)%opened )call mpp_error( FATAL, 'MPP_WRITE: invalid unit number.' )
59  if( .NOT.mpp_file(unit)%initialized )then
60 !this is the first call to mpp_write
61 !we now declare the file to be initialized
62 !if this is netCDF we switch file from DEFINE mode to DATA mode
63  if( mpp_file(unit)%format.EQ.MPP_NETCDF )then
64 #ifdef use_netCDF
65 !NOFILL is probably required for parallel: any circumstances in which not advisable?
66  error = NF_SET_FILL( mpp_file(unit)%ncid, NF_NOFILL, i ); call netcdf_err( error, mpp_file(unit) )
67  if( mpp_file(unit)%action.EQ.MPP_WRONLY )then
68  if(header_buffer_val>0) then
69  error = NF__ENDDEF(mpp_file(unit)%ncid,header_buffer_val,4,0,4)
70  else
71  error = NF_ENDDEF(mpp_file(unit)%ncid)
72  endif
73  endif
74  call netcdf_err( error, mpp_file(unit) )
75 #endif
76  else
77  call mpp_write_meta( unit, 'END', cval='metadata' )
78  end if
79  mpp_file(unit)%initialized = .TRUE.
80  if( verbose )print '(a,i6,a)', 'MPP_WRITE: PE=', pe, ' initialized file '//trim(mpp_file(unit)%name)//'.'
81  end if
82 
83 !initialize time: by default assume NULLTIME
84  time = NULLTIME
85  time_level = -1
86  newtime = .FALSE.
87  if( PRESENT(time_in) )time = time_in
88 !increment time level if new time
89  if( time.GT.mpp_file(unit)%time+EPSILON(time) )then !new time
90  mpp_file(unit)%time_level = mpp_file(unit)%time_level + 1
91  mpp_file(unit)%time = time
92  newtime = .TRUE.
93  end if
94  if( verbose )print '(a,2i6,2i5,es13.5)', 'MPP_WRITE: PE, unit, %id, %time_level, %time=',&
95  pe, unit, mpp_file(unit)%id, mpp_file(unit)%time_level, mpp_file(unit)%time
96 
97  if( mpp_file(unit)%format.EQ.MPP_NETCDF )then
98  ptr2 = LOC(mpp_io_stack(1))
99 !define netCDF data block to be written:
100 ! time axis: START = time level
101 ! AXSIZ = 1
102 ! space axis: if there is no domain info
103 ! START = 1
104 ! AXSIZ = field%size(axis)
105 ! if there IS domain info:
106 ! start of domain is compute%start_index for multi-file I/O
107 ! global%start_index for all other cases
108 ! this number must be converted to 1 for NF_PUT_VAR
109 ! (netCDF fortran calls are with reference to 1),
110 ! So, START = compute%start_index - <start of domain> + 1
111 ! AXSIZ = usually compute%size
112 ! However, if compute%start_index-compute%end_index+1.NE.compute%size,
113 ! we assume that the call is passing a subdomain.
114 ! To pass a subdomain, you must pass a domain2D object that satisfies the following:
115 ! global%start_index must contain the <start of domain> as defined above;
116 ! the data domain and compute domain must refer to the subdomain being passed.
117 ! In this case, START = compute%start_index - <start of domain> + 1
118 ! AXSIZ = compute%start_index - compute%end_index + 1
119 ! NOTE: passing of subdomains will fail for multi-PE single-threaded I/O,
120 ! since that attempts to gather all data on PE 0.
121  start = 1
122  do i = 1,size(field%axes(:))
123  axsiz(i) = field%size(i)
124  if( i.EQ.field%time_axis_index )start(i) = mpp_file(unit)%time_level
125  start(i) = max(start(i),1)
126  end do
127 
128  if( debug )print '(a,2i6,12i6)', 'WRITE_RECORD: PE, unit, start, axsiz=', pe, unit, start, axsiz
129 #ifdef use_netCDF
130 !write time information if new time
131  if( newtime )then
132  if( KIND(time).EQ.r8_kind )then
133  error = NF_PUT_VAR1_DOUBLE( mpp_file(unit)%ncid, mpp_file(unit)%id, mpp_file(unit:unit)%time_level, time )
134  else if( KIND(time).EQ.r4_kind )then
135  error = NF90_PUT_VAR ( mpp_file(unit)%ncid, mpp_file(unit)%id, time)
136  end if
137  end if
138  if( field%pack == 0 )then
139  packed_data = CEILING(data)
140  error = NF_PUT_VARA_INT ( mpp_file(unit)%ncid, field%id, start, axsiz, packed_data )
141  elseif( field%pack.GT.0 .and. field%pack.LE.2 )then
142  error = NF90_PUT_VAR ( mpp_file(unit)%ncid, field%id, data, start=start, count=axsiz )
143  else !convert to integer using scale and add: no error check on packed data representation
144  packed_data = nint((data-field%add)/field%scale)
145  error = NF_PUT_VARA_INT ( mpp_file(unit)%ncid, field%id, start, axsiz, packed_data )
146  end if
147  call netcdf_err( error, mpp_file(unit), field=field )
148 #endif
149  else !non-netCDF
150  ptr1 = LOC(mpp_io_stack(1))
151 !subdomain contains (/is,ie,js,je/)
152  if( PRESENT(domain) )then
153  call mpp_get_compute_domain(domain, is, ie, js, je)
154  subdomain(:) = (/ is, ie, js, je /)
155  else
156  subdomain(:) = -1 ! -1 means use global value from axis metadata
157  end if
158  if( mpp_file(unit)%format.EQ.MPP_ASCII )then
159 !implies sequential access
160  write( unit,* )field%id, subdomain, time_level, time, data
161  else !MPP_IEEE32 or MPP_NATIVE
162  if( mpp_file(unit)%access.EQ.MPP_SEQUENTIAL )then
163  write(unit)field%id, subdomain, time_level, time, data
164  else !MPP_DIRECT
165  write( unit, rec=mpp_file(unit)%record )field%id, subdomain, time_level, time, data
166  if( debug )print '(a,i6,a,i6)', 'MPP_WRITE: PE=', pe, ' wrote record ', mpp_file(unit)%record
167  end if
168  end if
169  end if
170 
171 !recompute current record for direct access I/O
172  if( mpp_file(unit)%access.EQ.MPP_DIRECT )then
173  if( mpp_file(unit)%fileset.EQ.MPP_SINGLE )then
174 !assumes all PEs participate in I/O: modify later
175  mpp_file(unit)%record = mpp_file(unit)%record + records_per_pe*npes
176  else
177  mpp_file(unit)%record = mpp_file(unit)%record + records_per_pe
178  end if
179  end if
180 
181  return
182  end subroutine WRITE_RECORD_

201  subroutine MPP_WRITE_2DDECOMP_3D_( unit, field, domain, data, tstamp, tile_count, default_data)
202 !mpp_write writes <data> which has the domain decomposition <domain>
203  integer, intent(in) :: unit
204  type(fieldtype), intent(in) :: field
205  type(domain2D), intent(inout) :: domain
206  MPP_TYPE_, intent(inout) :: data(:,:,:)
207  MPP_TYPE_, intent(in), optional :: tstamp
208  integer, intent(in), optional :: tile_count
209  MPP_TYPE_, intent(in), optional :: default_data
210 
211 !cdata is used to store compute domain as contiguous data
212 !gdata is used to globalize data for multi-PE single-threaded I/O
213  MPP_TYPE_, allocatable, dimension(:,:,:) :: cdata, gdata
214 !NEW: data may be on compute OR data domain
215  logical :: data_has_halos, halos_are_global, x_is_global, y_is_global
216  integer :: is, ie, js, je, isd, ied, jsd, jed, isg, ieg, jsg, jeg, ism, iem, jsm, jem
217  integer :: position, errunit
218  type(domain2d), pointer :: io_domain=>NULL()
219 
220  call mpp_clock_begin(mpp_write_clock)
221 
222  errunit = stderr()
223  if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_WRITE: must first call mpp_io_init.' )
224  if( .NOT.mpp_file(unit)%valid )call mpp_error( FATAL, 'MPP_WRITE: invalid unit number.' )
225 
226  position = field%position
227 
228  call mpp_get_compute_domain( domain, is, ie, js, je, tile_count=tile_count, position=position )
229  call mpp_get_data_domain ( domain, isd, ied, jsd, jed, x_is_global=x_is_global, &
230  y_is_global=y_is_global, tile_count=tile_count, position=position )
231  call mpp_get_memory_domain ( domain, ism, iem, jsm, jem, position=position )
232 
233  if( size(data,1).EQ.ie-is+1 .AND. size(data,2).EQ.je-js+1 )then
234  data_has_halos = .FALSE.
235  else if( size(data,1).EQ.iem-ism+1 .AND. size(data,2).EQ.jem-jsm+1 )then
236  data_has_halos = .TRUE.
237  else
238  write( errunit,'(a,10i5)' )'MPP_WRITE_2DDECOMP fails on field '//trim(field%name)// &
239  ': is,ie,js,je, ism,iem,jsm,jem, size(data,1), size(data,2)=', &
240  is,ie,js,je, ism,iem,jsm,jem, size(data,1), size(data,2)
241  call mpp_error( FATAL, 'MPP_WRITE: data must be either on compute domain or data domain.' )
242  end if
243  halos_are_global = x_is_global .AND. y_is_global
244  if( npes.GT.1 .AND. mpp_file(unit)%threading.EQ.MPP_SINGLE )then
245  if( halos_are_global )then
246  call mpp_update_domains( data, domain, position = position )
247 !all non-0 PEs have passed their data to PE 0 and may now exit
248  if(mpp_file(unit)%write_on_this_pe ) then
249  call WRITE_RECORD_( unit, field, size(data(:,:,:)), data, tstamp)
250  endif
251  else
252 !put field onto global domain
253  call mpp_get_global_domain ( domain, isg, ieg, jsg, jeg, tile_count=tile_count, position=position )
254  if(mpp_file(unit)%write_on_this_pe .OR. .NOT. global_field_on_root_pe) then
255  allocate( gdata(isg:ieg,jsg:jeg,size(data,3)) )
256  else
257  allocate( gdata(1,1,1))
258  endif
259  if(global_field_on_root_pe) then
260  call mpp_global_field( domain, data, gdata, position = position, &
261  flags=XUPDATE+YUPDATE+GLOBAL_ROOT_ONLY, &
262  default_data=default_data)
263  else
264  call mpp_global_field( domain, data, gdata, position = position, &
265  default_data=default_data)
266  endif
267 !all non-0 PEs have passed their data to PE 0 and may now exit
268  if(mpp_file(unit)%write_on_this_pe ) then
269  call WRITE_RECORD_( unit, field, size(gdata(:,:,:)), gdata, tstamp)
270  endif
271  deallocate(gdata)
272  end if
273  else if(mpp_file(unit)%io_domain_exist ) then
274  if( halos_are_global )then
275  call mpp_update_domains( data, domain, position = position )
276  if(mpp_file(unit)%write_on_this_pe ) then
277  call WRITE_RECORD_( unit, field, size(data(:,:,:)), data, tstamp)
278  endif
279  else
280  io_domain=>mpp_get_io_domain(mpp_file(unit)%domain)
281  call mpp_get_global_domain ( io_domain, isg, ieg, jsg, jeg, tile_count=tile_count, position=position )
282  if(mpp_file(unit)%write_on_this_pe .OR. .NOT. global_field_on_root_pe) then
283  allocate( gdata(isg:ieg,jsg:jeg,size(data,3)) )
284  else
285  allocate( gdata(1,1,1))
286  endif
287  if(global_field_on_root_pe) then
288  call mpp_global_field( io_domain, data, gdata, position = position, &
289  flags=XUPDATE+YUPDATE+GLOBAL_ROOT_ONLY, &
290  default_data=default_data)
291  else
292  call mpp_global_field( io_domain, data, gdata, position = position, &
293  default_data=default_data)
294  endif
295  io_domain => NULL()
296  if(mpp_file(unit)%write_on_this_pe ) then
297  call WRITE_RECORD_( unit, field, size(gdata(:,:,:)), gdata, tstamp)
298  endif
299  deallocate( gdata )
300  endif
301  else if( data_has_halos )then
302 !store compute domain as contiguous data and pass to write_record
303  allocate( cdata(is:ie,js:je,size(data,3)) )
304  cdata(:,:,:) = data(is-isd+1:ie-isd+1,js-jsd+1:je-jsd+1,:)
305  call WRITE_RECORD_( unit, field, size(cdata(:,:,:)), cdata, tstamp, domain, tile_count )
306  else
307 !data is already contiguous
308  call WRITE_RECORD_( unit, field, size(data(:,:,:)), data, tstamp, domain, tile_count )
309  end if
310 
311  call mpp_clock_end(mpp_write_clock)
312 
313  return
314  end subroutine MPP_WRITE_2DDECOMP_3D_
