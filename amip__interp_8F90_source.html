20 !> @defgroup amip_interp_mod amip_interp_mod
21 !> @ingroup amip_interp
22 !> @brief Provides observed sea surface temperature and ice mask data sets that have been
23 !! interpolated onto your model's grid.
24 !!
25 !> @author Bruce Wyman
26 !!
27 !> When using these routines three possible data sets are available:
28 !!
29 !! 1. AMIP @link http://www-pcmdi.llnl.gov/amip @endlink from Jan 1979 to Jan 1989 (2 deg x 2 deg)
30 !! 2. Reynolds OI @link amip_interp.rey_oi.txt @endlink from Nov 1981 to Jan 1999 (1 deg x 1 deg)
31 !! 3. Reynolds EOF @link ftp://podaac.jpl.nasa.gov/pub/sea_surface_temperature/reynolds/rsst/doc/rsst.html @endlink from Jan 1950 to Dec 1998 (2 deg x 2 deg)
32 !!
33 !! All original data are observed monthly means. This module
34 !! interpolates linearly in time between pairs of monthly means.
35 !! Horizontal interpolation is done using the horiz_interp module.
36 !!
37 !! When a requested date falls outside the range of dates available
38 !! a namelist option allows for use of the climatological monthly
39 !! mean values which are computed from all of the data in a particular
40 !! data set. \n
41 !! \n AMIP 1:\n
42 !! from Jan 1979 to Jan 1989 (2 deg x 2 deg).\n\n
43 !! Reynolds OI:\n
44 !! from Nov 1981 to Jan 1999 (1 deg x 1 deg)\n
45 !! The analysis uses in situ and satellite SST's plus
46 !! SST's simulated by sea-ice cover.\n\n
47 !! Reynold's EOF:\n
48 !! from Jan 1950 to Dec 1998 (2 deg x 2 deg)\n
49 !! NCEP Reynolds Historical Reconstructed Sea Surface Temperature
50 !! The analysis uses both in-situ SSTs and satellite derived SSTs
51 !! from the NOAA Advanced Very High Resolution Radiometer.
52 !! In-situ data is used from 1950 to 1981, while both AVHRR derived
53 !! satellite SSTs and in-situ data are used from 1981 to the
54 !! end of 1998.
55 !!
56 !> @note The data set used by this module have been reformatted as 32-bit IEEE.
57 !! The data values are packed into 16-bit integers.
58 !!
59 !! The data sets are read from the following files:
60 !!
61 !! amip1 INPUT/amip1_sst.data
62 !! reynolds_io INPUT/reyoi_sst.data
63 !! reynolds_eof INPUT/reynolds_sst.data

366 !> Retrieve sea surface temperature data and interpolated grid
367 subroutine get_amip_sst (Time, Interp, sst, err_msg, lon_model, lat_model)
368 
369  type (time_type), intent(in) :: time !< Time to interpolate
370  type (amip_interp_type), intent(inout) :: interp !< Holds data for interpolation
371  real, intent(out) :: sst(:,:) !< Sea surface temperature data
372  character(len=*), optional, intent(out) :: err_msg !< Holds error message string if present
373 
374  real, dimension(mobs,nobs) :: sice
375 
376  integer :: year1, year2, month1, month2
377  real :: fmonth
378  type (date_type) :: date1, date2, udate1, udate2
379 
380  type(time_type) :: amip_time
381  integer :: tod(3),dum(3)
382 
383 ! add by JHC
384  real, intent(in), dimension(:,:), optional :: lon_model, lat_model
385  real :: pert
386  integer :: i, j, mobs_sst, nobs_sst
387  integer :: jhctod(6)
388  type (time_type) :: udate
389  character(len=4) :: yyyy
390  integer :: nrecords, ierr, k, yr, mo, dy
391  integer :: siz(4)
392  integer, dimension(:), allocatable :: ryr, rmo, rdy
393  character(len=30) :: time_unit
394  real, dimension(:), allocatable :: timeval
395  character(len=maxc) :: ncfilename
396  type(fmsnetcdffile_t) :: fileobj
397  logical :: the_file_exists
398 ! end add by JHC
399  logical, parameter :: debug = .false. !> switch for debugging output
400  !> These are fms_io specific
401  integer :: unit
402 
403  if(present(err_msg)) err_msg = ''
404  if(.not.interp%I_am_initialized) then
405  if(fms_error_handler('get_amip_sst','The amip_interp_type variable is not initialized',err_msg)) return
406  endif
407 
408 !-----------------------------------------------------------------------
409 !----- compute zonally symetric sst ---------------
410 
411  if ( use_ncep_sst .and. forecast_mode ) no_anom_sst = .false.
412 
413  if (all(amip_date>0)) then
414  call get_date(time,dum(1),dum(2),dum(3),tod(1),tod(2),tod(3))
415  amip_time = set_date(amip_date(1),amip_date(2),amip_date(3),tod(1),tod(2),tod(3))
416  else
417  amip_time = time
418  endif
419 
420 ! add by JHC
421 if ( .not.use_daily ) then
422 ! end add by JHC
423 
424  if ( .not. allocated(temp1) ) allocate (temp1(mobs,nobs))
425  if ( .not. allocated(temp2) ) allocate (temp2(mobs,nobs))
426 
427  if (use_zonal) then
428  call zonal_sst (amip_time, sice, temp1)
429  call horiz_interp ( interp%Hintrp, temp1, sst )
430  else
431 
432 !-----------------------------------------------------------------------
433 !---------- get new observed sea surface temperature -------------------
434 
435 ! ---- time interpolation for months -----
436  call time_interp (amip_time, fmonth, year1, year2, month1, month2)
437 ! ---- force climatology ----
438  if (interp % use_climo) then
439  year1=0; year2=0
440  endif
441  if (interp % use_annual) then
442  year1=0; year2=0
443  month1=0; month2=0
444  endif
445 ! ---------------------------
446 
447  date1 = date_type( year1, month1, 0 )
448  date2 = date_type( year2, month2, 0 )
449 
450 ! -- open/rewind file --
451  unit = -1
452 !-----------------------------------------------------------------------
453 
454 
455  if (date1 /= interp % Date1) then
456 ! ---- use Date2 for Date1 ----
457  if (date1 == interp % Date2) then
458  interp % Date1 = interp % Date2
459  interp % data1 = interp % data2
460  temp1(:,:) = temp2(:,:) ! SJL BUG fix: June 24, 2011
461  else
462  call read_record ('sst', date1, udate1, temp1)
463  if ( use_ncep_sst .and. (.not. no_anom_sst) ) then
464  temp1(:,:) = temp1(:,:) + sst_anom(:,:)
465  endif
466  call horiz_interp ( interp%Hintrp, temp1, interp%data1 )
467  call clip_data ('sst', interp%data1)
468  interp % Date1 = date1
469  endif
470  endif
471 
472 !-----------------------------------------------------------------------
473 
474  if (date2 /= interp % Date2) then
475  call read_record ('sst', date2, udate2, temp2)
476  if ( use_ncep_sst .and. (.not. no_anom_sst) ) then
477  temp2(:,:) = temp2(:,:) + sst_anom(:,:)
478  endif
479  call horiz_interp ( interp%Hintrp, temp2, interp%data2 )
480  call clip_data ('sst', interp%data2)
481  interp % Date2 = date2
482  endif
483 
484 !-----------------------------------------------------------------------
485 !---------- time interpolation (between months) of sst's ---------------
486 !-----------------------------------------------------------------------
487  sst = interp % data1 + fmonth * (interp % data2 - interp % data1)
488 
489 !-------------------------------------------------------------------------------
490 ! SJL mods for NWP and TCSF ---
491 ! Nudging runs: (Note: NCEP SST updated only every 6-hr)
492 ! Compute SST anomaly from global SST datasets for subsequent forecast runs
493 !-------------------------------------------------------------------------------
494  if ( use_ncep_sst .and. no_anom_sst ) then
495  sst_anom(:,:) = sst_ncep(:,:) - (temp1(:,:) + fmonth*(temp2(:,:) - temp1(:,:)) )
496  call horiz_interp ( interp%Hintrp, sst_ncep, sst )
497  call clip_data ('sst', sst)
498  endif
499 
500 !! DEBUG CODE
501  if (debug) then
502  call get_date(amip_time,jhctod(1),jhctod(2),jhctod(3),jhctod(4),jhctod(5),jhctod(6))
503  if (mpp_pe() == 0) then
504  write (*,200) 'JHC: use_daily = F, AMIP_Time: ',jhctod(1),jhctod(2),jhctod(3),jhctod(4),jhctod(5),jhctod(6)
505  write (*,300) 'JHC: use_daily = F, interped SST: ', sst(1,1),sst(5,5),sst(10,10)
506  endif
507  endif
508 
509 
510  endif
511 
512 ! add by JHC
513 else
514  call get_date(amip_time,jhctod(1),jhctod(2),jhctod(3),jhctod(4),jhctod(5),jhctod(6))
515  if (mpp_pe() == mpp_root_pe()) write(*,200) 'amip_interp_mod: use_daily = T, Amip_Time = ',jhctod(1),jhctod(2),jhctod(3),jhctod(4),jhctod(5),jhctod(6)
516 
517  yr = jhctod(1); mo = jhctod(2); dy = jhctod(3)
518 
519  write (yyyy,'(i4)') jhctod(1)
520 
521  file_name_sst = 'INPUT/' // 'sst.day.mean.'//yyyy//'.v2.nc'
522  ncfilename = trim(file_name_sst)
523  time_unit = 'days since 1978-01-01 00:00:00'
524 
525  mobs_sst = 1440; nobs_sst = 720
526 
527  call set_sst_grid_edges_daily(mobs_sst, nobs_sst)
528  call horiz_interp_new ( interp%Hintrp2, lon_bnd, lat_bnd, &
529  lon_model, lat_model, interp_method="bilinear" )
530 
531  the_file_exists = fms2_io_file_exists(ncfilename)
532 
533  if ( (.NOT. the_file_exists) ) then
534  call mpp_error ('amip_interp_mod', &
535  'cannot find daily SST input data file: '//trim(ncfilename), note)
536  else
537  if (mpp_pe() == mpp_root_pe()) call mpp_error ('amip_interp_mod', &
538  'Reading NetCDF formatted daily SST from: '//trim(ncfilename), note)
539 
540  if(.not. open_file(fileobj, trim(ncfilename), 'read')) &
541  call error_mesg ('get_amip_sst', 'Error in opening file '//trim(ncfilename), fatal)
542 
543  call get_dimension_size(fileobj, 'TIME', nrecords)
544  if (nrecords < 1) call mpp_error('amip_interp_mod', &
545  'Invalid number of SST records in daily SST data file: '//trim(ncfilename), fatal)
546  allocate(timeval(nrecords), ryr(nrecords), rmo(nrecords), rdy(nrecords))
547  call fms2_io_read_data(fileobj, 'TIME', timeval)
548 !!! DEBUG CODE
549  if(debug) then
550  if (mpp_pe() == 0) then
