37  subroutine MPP_GLOBAL_FIELD_UG_3D_( domain, local, global, flags, default_data)
38 !get a global field from a local field
39 !local field may be on compute OR data domain
40  type(domainUG), intent(in) :: domain
41  MPP_TYPE_, intent(in) :: local(domain%compute%begin:,:)
42  MPP_TYPE_, intent(out) :: global(domain%global%begin:,:)
43  integer, intent(in), optional :: flags
44  MPP_TYPE_, intent(in), optional :: default_data
45 
46  integer :: l, k, m, n, nd, num_words, lpos, rpos, tile_id
47  integer :: ke, lsc, lec, ls, le, num_word_me
48  integer :: ipos, jpos
49  logical :: root_only, global_on_this_pe
50  MPP_TYPE_ :: clocal (domain%compute%size*size(local,2))
51  MPP_TYPE_ :: cremote(domain%compute%max_size*size(local,2))
52  integer :: stackuse
53  character(len=8) :: text
54 
55  pointer( ptr_local, clocal )
56  pointer( ptr_remote, cremote )
57 
58  stackuse = size(clocal(:))+size(cremote(:))
59  if( stackuse.GT.mpp_domains_stack_size )then
60  write( text, '(i8)' )stackuse
61  call mpp_error( FATAL, &
62  'MPP_DO_GLOBAL_FIELD_UG user stack overflow: call mpp_domains_set_stack_size('//trim(text)// &
63  & ') from all PEs.' )
64  end if
65  mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, stackuse )
66 
67  ptr_local = LOC(mpp_domains_stack)
68  ptr_remote = LOC(mpp_domains_stack(size(clocal(:))+1))
69 
70  if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_GLOBAL_FIELD_UG: must first call mpp_domains_init.' )
71 
72  root_only = .FALSE.
73  if( PRESENT(flags) ) root_only = BTEST(flags, ROOT_GLOBAL)
74 
75  global_on_this_pe = .NOT. root_only .OR. domain%pe == domain%tile_root_pe
76  ipos = 0; jpos = 0
77  if(global_on_this_pe ) then
78  if(size(local,2).NE.size(global,2) ) call mpp_error( FATAL, &
79  'MPP_GLOBAL_FIELD_UG: mismatch of third dimension size of global and local')
80  if( size(global,1).NE.domain%global%size)then
81  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD_UG: incoming arrays do not match domain.' )
82  endif
83  endif
84 
85  if( size(local,1).NE.domain%compute%size )then
86  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD_UG_: incoming field array must match compute domain ' )
87  end if
88 
89  ke = size(local,2)
90  lsc = domain%compute%begin; lec = domain%compute%end
91 
92  num_word_me = (lec-lsc+1)*ke
93 
94 ! make contiguous array from compute domain
95  m = 0
96  if(global_on_this_pe) then
97  if(PRESENT(default_data)) then
98  global = default_data
99  else
100  global = 0
101  endif
102 
103  do k = 1, ke
104  do l = lsc, lec
105  m = m + 1
106  clocal(m) = local(l,k)
107  global(l,k) = clocal(m) !always fill local domain directly
108  end do
109  end do
110  else
111  do k = 1, ke
112  do l = lsc, lec
113  m = m + 1
114  clocal(m) = local(l,k)
115  end do
116  end do
117  endif
118 
119  !fill off-domains (note loops begin at an offset of 1)
120  tile_id = domain%tile_id
121  nd = size(domain%list(:))
122  if(root_only) then
123  if(domain%pe .NE. domain%tile_root_pe) then
124  call mpp_send( clocal(1), plen=num_word_me, to_pe=domain%tile_root_pe, tag=COMM_TAG_1 )
125  else
126  do n = 1,nd-1
127  rpos = mod(domain%pos+n,nd)
128  if( domain%list(rpos)%tile_id .NE. tile_id ) cycle
129  num_words = domain%list(rpos)%compute%size * ke
130  call mpp_recv(cremote(1), glen=num_words, from_pe=domain%list(rpos)%pe, tag=COMM_TAG_1 )
131  m = 0
132  ls = domain%list(rpos)%compute%begin; le = domain%list(rpos)%compute%end
133 
134  do k = 1,ke
135  do l = ls, le
136  m = m + 1
137  global(l,k) = cremote(m)
138  end do
139  end do
140  end do
141  endif
142  else
143  do n = 1,nd-1
144  lpos = mod(domain%pos+nd-n,nd)
145  if( domain%list(lpos)%tile_id.NE. tile_id ) cycle ! global field only within tile
146  call mpp_send( clocal(1), plen=num_word_me, to_pe=domain%list(lpos)%pe, tag=COMM_TAG_2 )
147  end do
148  do n = 1,nd-1
149  rpos = mod(domain%pos+n,nd)
150  if( domain%list(rpos)%tile_id .NE. tile_id ) cycle ! global field only within tile
151  num_words = domain%list(rpos)%compute%size * ke
152  call mpp_recv( cremote(1), glen=num_words, from_pe=domain%list(rpos)%pe, tag=COMM_TAG_2 )
153  m = 0
154  ls = domain%list(rpos)%compute%begin; le = domain%list(rpos)%compute%end
155 
156  do k = 1,ke
157  do l = ls, le
158  m = m + 1
159  global(l,k) = cremote(m)
160  end do
161  end do
162  enddo
163  endif
164 
165  call mpp_sync_self()
166 
167  end subroutine MPP_GLOBAL_FIELD_UG_3D_
