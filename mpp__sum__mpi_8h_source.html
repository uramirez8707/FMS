19  subroutine MPP_SUM_( a, length, pelist )
20 !sums array a over the PEs in pelist (all PEs if this argument is omitted)
21 !result is also automatically broadcast: all PEs have the sum in a at the end
22  !we are using f77-style call: array passed by address and not descriptor; further,
23  !the f90 conformance check is avoided.
24  integer, intent(in) :: length
25  integer, intent(in), optional :: pelist(:)
26  MPP_TYPE_, intent(inout) :: a(*)
27  integer :: n, errunit
28  MPP_TYPE_ :: work(length)
29 
30  if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_SUM: You must first call mpp_init.' )
31  n = get_peset(pelist); if( peset(n)%count.EQ.1 )return
32 
33  if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
34  if( verbose )call mpp_error( NOTE, 'MPP_SUM: using MPI_ALLREDUCE...' )
35  if( debug ) then
36  errunit = stderr()
37  write( errunit,* )'pe, n, peset(n)%id=', pe, n, peset(n)%id
38  endif
39  call MPI_ALLREDUCE( a, work, length, MPI_TYPE_, MPI_SUM, peset(n)%id, error )
40  a(1:length) = work(1:length)
41  if( debug .and. (current_clock.NE.0) )call increment_current_clock( EVENT_ALLREDUCE, length*MPP_TYPE_BYTELEN_ )
42  return
43  end subroutine MPP_SUM_
