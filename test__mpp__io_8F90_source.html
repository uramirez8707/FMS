175  subroutine test_netcdf_io(type)
176  character(len=*), intent(in) :: type
177  integer :: ndim, nvar, natt, ntime
178  integer :: is, ie, js, je, isd, ied, jsd, jed, ism, iem, jsm, jem
179  integer :: position, msize(2), ioff, joff, nxg, nyg
180  logical :: symmetry
181  type(atttype), allocatable :: atts(:)
182  type(fieldtype), allocatable :: vars(:)
183  type(axistype), allocatable :: axes(:)
184  real, allocatable :: tstamp(:)
185  real, dimension(:,:,:), allocatable :: data, gdata, rdata
186 
187  !--- determine the shift and symmetry according to type,
188  select case(type)
189  case('Simple')
190  position = center; symmetry = .false.
191  case('Symmetry_T_cell', 'Symmetry_T_cell_memory')
192  position = center; symmetry = .true.
193  case('Symmetry_E_cell', 'Symmetry_E_cell_memory')
194  position = east; symmetry = .true.
195  case('Symmetry_N_cell', 'Symmetry_N_cell_memory')
196  position = north; symmetry = .true.
197  case('Symmetry_C_cell', 'Symmetry_C_cell_memory')
198  position = corner; symmetry = .true.
199  case default
200  call mpp_error(fatal, "type = "//type//" is not a valid test type")
201  end select
202 
203 !define domain decomposition
204  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
205  if(index(type,"memory") == 0) then
206  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, xhalo=halo, yhalo=halo, symmetry = symmetry )
207  else ! on memory domain
208  msize(1) = nx/layout(1) + 2*halo + 2
209  msize(2) = ny/layout(2) + 2*halo + 2
210  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, xhalo=halo, yhalo=halo, symmetry = symmetry, &
211  memory_size = msize )
212  end if
213 
214  call mpp_get_compute_domain( domain, is, ie, js, je, position=position )
215  call mpp_get_data_domain ( domain, isd, ied, jsd, jed, position=position )
216  call mpp_get_memory_domain ( domain, ism, iem, jsm, jem, position=position )
217  call mpp_get_global_domain ( domain, xsize=nxg, ysize=nyg, position=position )
218  call mpp_get_domain_components( domain, xdom, ydom )
219 
220 !define global data array
221  allocate( gdata(nxg,nyg,nz) )
222  gdata = 0.
223  do k = 1,nz
224  do j = 1,nyg
225  do i = 1,nxg
226  gdata(i,j,k) = k + i*1e-3 + j*1e-6
227  end do
228  end do
229  end do
230 
231  ioff = ism - isd; joff = jsm - jsd
232  allocate( data(ism:iem,jsm:jem,nz) )
233  data = 0
234  data(is+ioff:ie+ioff,js+joff:je+joff,:) = gdata(is:ie,js:je,:)
235 
236 !tests
237 
238 !sequential write: single-threaded formatted: only if small
239  if( nx*ny*nz*nt.LT.1000 .AND. index(type,"memory") .NE. 0 )then
240  if( pe.EQ.mpp_root_pe() )print *, 'sequential write: single-threaded formatted'
241 !here the only test is a successful write: please look at test.txt for verification.
242  call mpp_open( unit, trim(file)//'s.txt', action=mpp_overwr, form=mpp_ascii, threading=mpp_single )
243  call mpp_write_meta( unit, x, 'X', 'km', 'X distance', domain=xdom, data=(/(i-1.,i=1,nxg)/) )
244  call mpp_write_meta( unit, y, 'Y', 'km', 'Y distance', domain=ydom, data=(/(i-1.,i=1,nyg)/) )
245  call mpp_write_meta( unit, z, 'Z', 'km', 'Z distance', data=(/(i-1.,i=1,nz)/) )
246  call mpp_write_meta( unit, t, 'T', 'sec', 'Time' )
247  call mpp_write_meta( unit, f, (/x,y,z,t/), 'Data', 'metres', 'Random data' )
248  call mpp_write( unit, x )
249  call mpp_write( unit, y )
250  call mpp_write( unit, z )
251  do i = 0,nt-1
252  time = i*10.
253  call mpp_write( unit, f, domain, data, time )
254  end do
255  call mpp_close(unit)
256  end if
257 
258 !netCDF distributed write
259  if( pe.EQ.mpp_root_pe() )print *, 'netCDF distributed write'
260  call mpp_open( unit, trim(type)//"_"//trim(file)//'d', action=mpp_overwr, &
261  form=mpp_netcdf, threading=mpp_multi, fileset=mpp_multi )
262  call mpp_write_meta( unit, x, 'X', 'km', 'X distance', 'X', domain=xdom, data=(/(i-1.,i=1,nxg)/) )
263  call mpp_write_meta( unit, y, 'Y', 'km', 'Y distance', 'Y', domain=ydom, data=(/(i-1.,i=1,nyg)/) )
264  call mpp_write_meta( unit, z, 'Z', 'km', 'Z distance', 'Z', data=(/(i-1.,i=1,nz)/) )
265  call mpp_write_meta( unit, t, 'T', 'sec', 'Time', 'T' )
266  call mpp_write_meta( unit, f, (/x,y,z,t/), 'Data', 'metres', 'Random data', pack=pack_size )
267  call mpp_write( unit, x )
268  call mpp_write( unit, y )
269  call mpp_write( unit, z )
270  do i = 0,nt-1
271  time = i*10.
272  call mpp_write( unit, f, domain, data, time )
273  end do
274  call mpp_close(unit)
275 
276 !netCDF single-threaded write
277  if( pe.EQ.mpp_root_pe() )print *, 'netCDF single-threaded write'
278  call mpp_open( unit, trim(type)//"_"//trim(file)//'s', action=mpp_overwr, form=mpp_netcdf, threading=mpp_single )
279 
280  call mpp_write_meta( unit, x, 'X', 'km', 'X distance', 'X', domain=xdom, data=(/(i-1.,i=1,nxg)/) )
281 
282  call mpp_write_meta( unit, y, 'Y', 'km', 'Y distance', 'Y', domain=ydom, data=(/(i-1.,i=1,nyg)/) )
283  call mpp_write_meta( unit, z, 'Z', 'km', 'Z distance', 'Z', data=(/(i-1.,i=1,nz)/) )
284  call mpp_write_meta( unit, t, 'T', 'sec', 'Time', 'T' )
285  call mpp_write_meta( unit, f, (/x,y,z,t/), 'Data', 'metres', 'Random data', pack=pack_size )
286 
287  call mpp_write( unit, x )
288  call mpp_write( unit, y )
289  call mpp_write( unit, z )
290 
291  do i = 0,nt-1
292  time = i*10.
293  call mpp_write( unit, f, domain, data, time)
294  end do
295  call mpp_close(unit)
296  allocate( rdata(is:ie,js:je,nz) )
297 
298 !netCDF multi-threaded read
299  if( pe.EQ.mpp_root_pe() )print *, 'netCDF multi-threaded read'
300  call mpp_sync()
301  call mpp_open( unit, trim(type)//"_"//trim(file)//'s', action=mpp_rdonly, &
302  form=mpp_netcdf, threading=mpp_multi, fileset=mpp_single )
303  call mpp_get_info( unit, ndim, nvar, natt, ntime )
304  allocate( atts(natt) )
305  allocate( axes(ndim) )
306  allocate( vars(nvar) )
307  allocate( tstamp(ntime) )
308  call mpp_get_atts ( unit, atts(:) )
309  call mpp_get_axes ( unit, axes(:) )
310  call mpp_get_fields ( unit, vars(:) )
311  call mpp_get_times( unit, tstamp(:) )
312 
313  call mpp_get_atts(vars(1),name=varname)
314 
315  if( varname.NE.'Data' )call mpp_error( fatal, 'File being read is not the expected one.' )
316  call mpp_read( unit, vars(1), domain, rdata, 1 )
317  rchk = mpp_chksum(rdata(is:ie,js:je,:))
318  chk = mpp_chksum( data(is+ioff:ie+ioff,js+joff:je+joff,:))
319  if( pe.EQ.mpp_root_pe() )print '(a,2z18)', trim(type)//' checksum=', rchk, chk
320  if( rchk == chk ) then
321  if( pe.EQ.mpp_root_pe() )call mpp_error( note, trim(type)//': single-fileset: data comparison are OK.' )
322  else
323  call mpp_error( fatal, 'Checksum error on multi-threaded/single-fileset netCDF read for type ' &
324  //trim(type) )
325  end if
326 
327  deallocate( atts, axes, vars, tstamp )
328 
329 !netCDF distributed read
330  if( pe.EQ.mpp_root_pe() )print *, 'netCDF multi-threaded read'
331  call mpp_sync() !wait for previous write to complete
332  call mpp_open( unit, trim(type)//"_"//trim(file)//'d', action=mpp_rdonly, &
333  form=mpp_netcdf, threading=mpp_multi, fileset=mpp_multi )
334  call mpp_get_info( unit, ndim, nvar, natt, ntime )
335  allocate( atts(natt) )
336  allocate( axes(ndim) )
337  allocate( vars(nvar) )
338  allocate( tstamp(ntime) )
339  call mpp_get_atts ( unit, atts(:) )
340  call mpp_get_axes ( unit, axes(:) )
341  call mpp_get_fields ( unit, vars(:) )
342  call mpp_get_times( unit, tstamp(:) )
343 
344  call mpp_get_atts(vars(1),name=varname)
345  rdata = 0
346 
347  if( varname.NE.'Data' )call mpp_error( fatal, 'File being read is not the expected one.' )
348 
349  call mpp_read( unit, vars(1), domain, rdata, 1 )
350 
351  rchk = mpp_chksum(rdata(is:ie,js:je,:))
352  chk = mpp_chksum( data(is+ioff:ie+ioff,js+joff:je+joff,:))
353  if( pe.EQ.mpp_root_pe() )print '(a,2z18)', trim(type)//' checksum=', rchk, chk
354  if( rchk == chk ) then
355  if( pe.EQ.mpp_root_pe() )call mpp_error( note, trim(type)//': multi-fileset: data comparison are OK.' )
356  else
357  call mpp_error( fatal, 'Checksum error on multi-threaded/multi-fileset netCDF read for type ' &
358  //trim(type) )
359  end if
360 
361  deallocate( atts, axes, vars, tstamp )
362 
363  deallocate( rdata, gdata, data)
364 
365  end subroutine test_netcdf_io

367  subroutine test_netcdf_io_append
368  integer :: ndim, nvar, natt, ntime

371  logical :: symmetry
372  real :: data

374 
375 !netCDF distributed write

378  form=mpp_netcdf, threading=mpp_single, fileset=mpp_single )
379  call mpp_write_meta( unit, t, 'T', 'sec', 'Time', 'T' )

398  endif

400  endif
401

403  time = i*1.

414  subroutine test_netcdf_io_mosaic(type, layout, ntiles_x, ntiles_y, io_layout)
415  character(len=*), intent(in) :: type
416  integer, intent(in) :: layout(:)
417  integer, intent(in) :: io_layout(:)
418  integer, intent(in) :: ntiles_x, ntiles_y
419 
420  integer :: ndim, nvar, natt, ntime
421  integer :: isc, iec, jsc, jec, nlon, nlat, n, i, j
422  integer :: my_tile, ncontacts, npes_per_tile, ntiles
423  integer, dimension(:), allocatable :: tile1, istart1, iend1, jstart1, jend1
424  integer, dimension(:), allocatable :: tile2, istart2, iend2, jstart2, jend2
425  integer, dimension(:), allocatable :: pe_start, pe_end
426  integer, dimension(:,:), allocatable :: layout2D, global_indices
427  character(len=64) :: output_file
428  logical :: is_root_pe
429  real, dimension(:,:,:), allocatable :: data, rdata
430  type(fieldtype), save :: vars(1)
431  integer :: id_clock_read, id_clock_write
432 
433  ! first get number of tiles of this mosaic. when there is one tile,
434  ! the file will be read/write using distributed file.
435  ! when there is more than one tile, single fileset will be used
436  npes = mpp_npes()
437 
438  id_clock_read = mpp_clock_id(trim(type)//" read", flags=mpp_clock_sync)
439  id_clock_write = mpp_clock_id(trim(type)//" write", flags=mpp_clock_sync)
440 
441  ncontacts = 0
442  ntiles = ntiles_x*ntiles_y
