102 !> Returns a surface height field that consists
103 !! of the sum of one or more Gaussian-shaped mountains.
104 !!
105 !> Returns a land surface topography that consists of a "set" of
106 !! simple Gaussian-shaped mountains. The height, position,
107 !! width, and elongation of the mountains can be controlled
108 !! by variables in the namelist.
109 subroutine gaussian_topog_init ( lon, lat, zsurf )
110 
111 real, intent(in) :: lon(:) !< The mean grid box longitude in radians
112 real, intent(in) :: lat(:) !< The mean grid box latitude in radians
113 real, intent(out) :: zsurf(:,:) !< The surface height (meters). Size must be size(lon) by size(lat)
114 
115 integer :: n
116 
117  if (.not.module_is_initialized) then
118  call write_version_number("GAUSSIAN_TOPOG_MOD", version)
119  endif
120 
121  if(any(shape(zsurf) /= (/size(lon(:)),size(lat(:))/))) then
122  call error_mesg ('get_gaussian_topog in topography_mod', &
123  'shape(zsurf) is not equal to (/size(lon),size(lat)/)', fatal)
124  endif
125 
126  if (do_nml) call read_namelist
127 
128 ! compute sum of all non-zero mountains
129  zsurf(:,:) = 0.
130  do n = 1, maxmts
131  if ( height(n) == 0. ) cycle
132  zsurf = zsurf + get_gaussian_topog( lon, lat, height(n), &
133  olon(n), olat(n), wlon(n), wlat(n), rlon(n), rlat(n))
134  enddo
135  module_is_initialized = .true.
136 
137 end subroutine gaussian_topog_init

141 !> @brief Returns a simple surface height field that consists of a single
142 !! Gaussian-shaped mountain.
143 !!
144 !> The height, position, width, and elongation of the mountain
145 !! is controlled by optional arguments.
146 !! @param real lon The mean grid box longitude in radians.
147 !! @param real lat The mean grid box latitude in radians.
148 !! @param real height Maximum surface height in meters.
149 !! @param real olond, olatd Position/origin of mountain in degrees longitude and latitude.
150 !! This is the location of the maximum height.
151 !! @param real wlond, wlatd Gaussian half-width of mountain in degrees longitude and latitude.
152 !! @param real rlond, rlatd Ridge half-width of mountain in degrees longitude and latitude.
153 !! This is the elongation of the maximum height.
154 !! @param real zsurf The surface height (in meters).
155 !! The size of the returned field is size(lon) by size(lat).
156 !! </OUT>
157 !!
158 !! @throws FATAL shape(zsurf) is not equal to (/size(lon),size(lat)/)
159 !! Check the input grid size and output field size.
160 !! The input grid is defined at the midpoint of grid boxes.
161 !!
162 !! @note
163 !! Mountains do not wrap around the poles.
164 !
165 !! <br>Example usage:
166 !! @code{.F90} zsurf = <B>get_gaussian_topog</B> ( lon, lat, height
167 !! [, olond, olatd, wlond, wlatd, rlond, rlatd ] )@endcode
168 function get_gaussian_topog ( lon, lat, height, &
169  olond, olatd, wlond, wlatd, rlond, rlatd ) &
170  result( zsurf )
171 
172 real, intent(in) :: lon(:), lat(:)
173 real, intent(in) :: height
174 real, intent(in), optional :: olond, olatd, wlond, wlatd, rlond, rlatd
175 real :: zsurf(size(lon,1),size(lat,1))
176 
177 integer :: i, j
178 real :: olon, olat, wlon, wlat, rlon, rlat
179 real :: tpi, dtr, dx, dy, xx, yy
180 
181  if (do_nml) call read_namelist
182 
183 ! no need to compute mountain if height=0
184  if ( height == 0. ) then
185  zsurf(:,:) = 0.
186  return
187  endif
188 
189  tpi = 2.0*pi
190  dtr = tpi/360.
191 
192 ! defaults and convert degrees to radians (dtr)
193  olon = 90.*dtr; if (present(olond)) olon=olond*dtr
194  olat = 45.*dtr; if (present(olatd)) olat=olatd*dtr
195  wlon = 15.*dtr; if (present(wlond)) wlon=wlond*dtr
196  wlat = 15.*dtr; if (present(wlatd)) wlat=wlatd*dtr
197  rlon = 0. ; if (present(rlond)) rlon=rlond*dtr
198  rlat = 0. ; if (present(rlatd)) rlat=rlatd*dtr
199 
200 ! compute gaussian-shaped mountain
201  do j=1,size(lat(:))
202  dy = abs(lat(j) - olat) ! dist from y origin
203  yy = max(0., dy-rlat)/wlat
204  do i=1,size(lon(:))
205  dx = abs(lon(i) - olon) ! dist from x origin
206  dx = min(dx, abs(dx-tpi)) ! To ensure that: -pi <= dx <= pi
207  xx = max(0., dx-rlon)/wlon
208  zsurf(i,j) = height*exp(-xx**2 - yy**2)
209  enddo
210  enddo
211 
212 end function get_gaussian_topog
