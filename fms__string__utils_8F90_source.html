125  !> @brief Converts a character array to an array of c pointers!
126  !! @return An array of c pointers
127  function fms_array_to_pointer(my_array) &
128  result(my_pointer)
129  character(len=*), target :: my_array(:) !!< Array of strings to convert
130  type(c_ptr), allocatable :: my_pointer(:)
131 
132  integer :: i !< For do loops
133 
134  if (allocated(my_pointer)) call mpp_error(fatal, "The c pointer array is &
135  already allocated. Deallocated before calling fms_array_to_pointer")

162  !> @brief Searches through a SORTED array of pointers for a string
163  !! @return the indices where the array was found

172  integer, intent(in) :: narray !< Length of the array
173  character(len=*), intent(in) :: string_to_find !< string to find

202  !> \brief Converts a C-string returned from a TYPE(C_PTR) function to
203  !! a fortran string with type character.
204  function cpointer_fortran_conversion (cstring) result(fstring)
205  type (c_ptr), intent(in) :: cstring !< Input C-pointer
206  character(len=:), allocatable :: fstring !< The fortran string returned
207  character(len=:,kind=c_char), pointer :: string_buffer !< A temporary pointer to between C and Fortran
208  integer(c_size_t) :: length !< The string length
209 
210  length = c_strlen(cstring)
211  allocate (character(len=length, kind=c_char) :: string_buffer)
212  block
213  character(len=length,kind=c_char), pointer :: s
214  call c_f_pointer(cstring,s) ! Recovers a view of the C string
215  string_buffer = s ! Copies the string contents
216  end block
217 
218  allocate(character(len=length) :: fstring) !> Set the length of fstring
219  fstring = string_buffer
220  deallocate(string_buffer)
221  end function cpointer_fortran_conversion

248  !> @brief Safely copy a string from one buffer to another.
249  subroutine string_copy(dest, source, check_for_null)
250  character(len=*), intent(inout) :: dest !< Destination string.
251  character(len=*), intent(in) :: source !< Source string.
252  logical, intent(in), optional :: check_for_null !<Flag indicating to test for null character
253 
254  integer :: i
255  logical :: check_null
256 
257  check_null = .false.
258  if (present(check_for_null)) check_null = check_for_null
259 
260  i = 0
261  if (check_null) then
262  i = index(source, char(0)) - 1
263  endif
264 
265  if (i < 1 ) i = len_trim(source)
266 
267  if (len_trim(source(1:i)) .gt. len(dest)) then
268  call mpp_error(fatal, "The input destination string is not big enough to" &
269  //" to hold the input source string.")
270  endif
271  dest = ""
272  dest = adjustl(trim(source(1:i)))
273  end subroutine string_copy
