27  function mpp_redistribute_init_comm(domain_in,l_addrs_in, domain_out,l_addrs_out, &
28  isize_in,jsize_in,ksize_in,isize_out,jsize_out,ksize_out) RESULT(d_comm)
29  type(DomainCommunicator2D), pointer :: d_comm
30  type(domain2D),target, intent(in) :: domain_in
31  integer(i8_kind), intent(in) :: l_addrs_in(:)
32  type(domain2D),target, intent(in) :: domain_out
33  integer(i8_kind), intent(in) :: l_addrs_out(:)
34  integer, intent(in) :: isize_in
35  integer, intent(in) :: jsize_in
36  integer, intent(in) :: ksize_in
37  integer, intent(in) :: isize_out
38  integer, intent(in) :: jsize_out
39  integer, intent(in) :: ksize_out
40 
41  integer(i8_kind) :: domain_id
42  integer :: m, list
43  integer :: is, ie, js, je, ke, ioff, joff, list_size
44  integer :: isc, iec, jsc, jec, mytile
45  integer :: lsize,rsize,msgsize,to_pe,from_pe
46  integer, allocatable,dimension(:) :: isL, jsL
47  integer(i8_kind),allocatable,dimension(:,:) :: slist_addr
48  character(len=8) :: text
49 
50 
51  ! This test determines whether input fields are from allocated memory (LOC gets global
52  ! address) or "static" memory (need shmem_ptr). This probably needs to be generalized
53  ! to determine appropriate mechanism for each incoming address.
54 
55  ! "Concurrent" run mode may leave field_in or field_out unassociated if pe does not
56  ! contain in/out data. Use of STATIC option for ocean complicates this as ocean component
57  ! always defined. Field_out is always a boundary structure and so is always allocated or
58  ! not depending on whether it's used. If field out is defined (>0), then it is used otherwise
59  ! field in must be defined.
60 
61 !fix ke
62  ke = 0
63  if( domain_in%pe /= null_pe )ke = ksize_in
64  if( domain_out%pe /= null_pe )then
65  if( ke /= 0 .AND. ke /= ksize_out ) &
66  call mpp_error( fatal, 'MPP_REDISTRIBUTE_INIT_COMM: mismatch between field_in and field_out.' )
67  ke = ksize_out
68  end if
69  if( ke == 0 )call mpp_error( fatal, 'MPP_REDISTRIBUTE_INIT_COMM: either domain_in or domain_out must be native.' )
70 !check sizes
71  if( domain_in%pe /= null_pe )then
72  if( isize_in /= domain_in%x(1)%data%size .OR. jsize_in /= domain_in%y(1)%data%size ) &
73  call mpp_error( fatal, 'MPP_REDISTRIBUTE_INIT_COMM: field_in must be on data domain of domain_in.' )
74  end if
75  if( domain_out%pe /= null_pe )then
76  if( isize_out /= domain_out%x(1)%data%size .OR. jsize_out /= domain_out%y(1)%data%size ) &
77  call mpp_error( fatal, 'MPP_REDISTRIBUTE_INIT_COMM: field_out must be on data domain of domain_out.' )
78  end if
79 
80 
81  ! Create unique domain identifier
82  list_size = size(l_addrs_in(:))
83  if(l_addrs_out(1) > 0)then
84  domain_id = set_domain_id(domain_out%id,ke+list_size)
85  else
86  domain_id = set_domain_id(domain_in%id,ke+list_size)
87  endif
88 
89  d_comm =>get_comm(domain_id,l_addrs_in(1),l_addrs_out(1))
90 
91  if(d_comm%initialized)return ! Found existing field/domain communicator
92 
93  d_comm%l_addr = l_addrs_in(1)
94  d_comm%domain_in =>domain_in
95  d_comm%Slist_size = size(domain_out%list(:))
96  d_comm%isize_in = isize_in
97  d_comm%jsize_in = jsize_in
98  d_comm%ke = ke
99 
100 !send
101  lsize = d_comm%Slist_size-1
102  allocate(d_comm%sendis(1,0:lsize), d_comm%sendie(1,0:lsize), &
103  d_comm%sendjs(1,0:lsize), d_comm%sendje(1,0:lsize), &
104  d_comm%S_msize(0:lsize),isl(0:lsize),jsl(0:lsize))
105  allocate(slist_addr(list_size,0:lsize))
106  allocate(d_comm%cto_pe(0:lsize), d_comm%S_do_buf(0:lsize))
107 
108  isl=0;jsl=0
109  slist_addr = -9999
110  d_comm%cto_pe=-1
111  d_comm%sendis=0; d_comm%sendie=0
112  d_comm%sendjs=0; d_comm%sendje=0;
113  d_comm%S_msize=0
114  d_comm%S_do_buf=.false.
115 
116  ioff = domain_in%x(1)%data%begin
117  joff = domain_in%y(1)%data%begin
118  mytile = domain_in%tile_id(1)
119 
120  call mpp_get_compute_domain( domain_in, isc, iec, jsc, jec )
121  do list = 0,lsize
122  m = mod( domain_out%pos+list+lsize+1, lsize+1 )
123  if( mytile .NE. domain_out%list(m)%tile_id(1) ) cycle
124  d_comm%cto_pe(list) = domain_out%list(m)%pe
125  to_pe = d_comm%cto_pe(list)
126  is = domain_out%list(m)%x(1)%compute%begin
127  ie = domain_out%list(m)%x(1)%compute%end
128  js = domain_out%list(m)%y(1)%compute%begin
129  je = domain_out%list(m)%y(1)%compute%end
130  is = max(is,isc); ie = min(ie,iec)
131  js = max(js,jsc); je = min(je,jec)
132  if( ie >= is .AND. je >= js )then
133  d_comm%S_do_buf(list) = .true.
134  d_comm%sendis(1,list)=is; d_comm%sendie(1,list)=ie
135  d_comm%sendjs(1,list)=js; d_comm%sendje(1,list)=je
136  d_comm%S_msize(list) = (ie-is+1)*(je-js+1)*ke
137  isl(list) = is-ioff+1; jsl(list) = js-joff+1
138  end if
139  end do
140 
141  call mpp_sync_self()
142 !recv
143  d_comm%domain_out =>domain_out
144  d_comm%Rlist_size = size(domain_in%list(:))
145  d_comm%isize_out = isize_out
146  d_comm%jsize_out = jsize_out
147 
148  rsize = d_comm%Rlist_size-1
149  allocate(d_comm%recvis(1,0:rsize), d_comm%recvie(1,0:rsize), &
150  d_comm%recvjs(1,0:rsize), d_comm%recvje(1,0:rsize), &
151  d_comm%R_msize(0:rsize))
152  allocate(d_comm%cfrom_pe(0:rsize), d_comm%R_do_buf(0:rsize))
153  allocate(d_comm%isizeR(0:rsize), d_comm%jsizeR(0:rsize))
154  allocate(d_comm%sendisR(1,0:rsize), d_comm%sendjsR(1,0:rsize))
155  allocate(d_comm%rem_addrl(list_size,0:rsize))
156  d_comm%rem_addrl=-9999
157  d_comm%cfrom_pe=-1
158  d_comm%recvis=0; d_comm%recvie=0
159  d_comm%recvjs=0; d_comm%recvje=0;
160  d_comm%R_msize=0
161  d_comm%R_do_buf=.false.
162  d_comm%isizeR=0; d_comm%jsizeR=0
163  d_comm%sendisR=0; d_comm%sendjsR=0
164 
165  mytile = domain_out%tile_id(1)
166  call mpp_get_compute_domain( domain_out, isc, iec, jsc, jec )
167  do list = 0,rsize
168  m = mod( domain_in%pos+rsize+1-list, rsize+1 )
169  if( mytile .NE. domain_in%list(m)%tile_id(1) ) cycle
170  d_comm%cfrom_pe(list) = domain_in%list(m)%pe
171  from_pe = d_comm%cfrom_pe(list)
172  is = domain_in%list(m)%x(1)%compute%begin
173  ie = domain_in%list(m)%x(1)%compute%end
174  js = domain_in%list(m)%y(1)%compute%begin
175  je = domain_in%list(m)%y(1)%compute%end
176  is = max(is,isc); ie = min(ie,iec)
177  js = max(js,jsc); je = min(je,jec)
178  if( ie >= is .AND. je >= js )then
179  d_comm%R_do_buf(list) = .true.
180  d_comm%recvis(1,list)=is; d_comm%recvie(1,list)=ie
181  d_comm%recvjs(1,list)=js; d_comm%recvje(1,list)=je
182  d_comm%R_msize(list) = (ie-is+1)*(je-js+1)*ke
183  end if
184  end do
185 
186  d_comm%isize_max = isize_in; call mpp_max(d_comm%isize_max)
187  d_comm%jsize_max = jsize_in; call mpp_max(d_comm%jsize_max)
188 
189  ! Handles case where S_msize and/or R_msize are 0 size array
190  msgsize = ( maxval( (/0,sum(d_comm%S_msize(:))/) ) + maxval( (/0,sum(d_comm%R_msize(:))/) ) ) * list_size
191  if(msgsize>0)then
192  mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, msgsize )
193  if( mpp_domains_stack_hwm.GT.mpp_domains_stack_size )then
194  write( text,'(i8)' )mpp_domains_stack_hwm
195  call mpp_error( fatal, 'MPP_REDISTRIBUTE_INIT_COMM: mpp_domains_stack overflow, call mpp_domains_set_stack_size(' &
196  //trim(text)//') from all PEs.' )
197  end if
198  end if
199 
200  DEALLOCATE(slist_addr,isl,jsl)
201 
202  d_comm%initialized = .true.
203 
204  end function mpp_redistribute_init_comm

207  function mpp_global_field_init_comm(domain,l_addr,isize_g,jsize_g,isize_l, &
208  jsize_l, ksize,l_addr2,flags, position) RESULT(d_comm)
209  type(DomainCommunicator2D), pointer :: d_comm
210  type(domain2D),target, intent(in) :: domain
211  integer(i8_kind), intent(in) :: l_addr
212  integer, intent(in) :: isize_g
213  integer, intent(in) :: jsize_g
214  integer, intent(in) :: isize_l
215  integer, intent(in) :: jsize_l
216  integer, intent(in) :: ksize
217  integer(i8_kind),optional,intent(in) :: l_addr2
218  integer, optional, intent(in) :: flags
219  integer, optional, intent(in) :: position
220 
221  integer(i8_kind) :: domain_id
222  integer :: n, lpos, rpos, list, nlist, tile_id
223  integer :: update_flags
224  logical :: xonly, yonly
225  integer :: is, ie, js, je, ioff, joff, ishift, jshift
226  integer :: lsize,msgsize,from_pe
227  integer, allocatable,dimension(:) :: isL, jsL
228  integer(i8_kind),allocatable,dimension(:,:) :: slist_addr
229  integer(i8_kind),save ,dimension(2) :: rem_addr
230  character(len=8) :: text
231 
232  if( .NOT.module_is_initialized )call mpp_error( fatal, 'MPP_GLOBAL_FIELD: must first call mpp_domains_init.' )
233  update_flags=xupdate+yupdate; xonly = .false.; yonly = .false.
234  if( PRESENT(flags) )then
235  update_flags = flags
236  xonly = btest(flags,east)
237  yonly = btest(flags,south)
238  if( .NOT.xonly .AND. .NOT.yonly )call mpp_error( warning, &
239  'MPP_GLOBAL_FIELD: you must have flags=XUPDATE, YUPDATE or XUPDATE+YUPDATE' )
240  if(xonly .AND. yonly) then
241  xonly = .false.; yonly = .false.
242  endif
243  end if
244 
245  call mpp_get_domain_shift(domain, ishift, jshift, position=position)
246 
247  if( isize_g /= (domain%x(1)%global%size+ishift) .OR. jsize_g /= (domain%y(1)%global%size+jshift) ) &
248  call mpp_error( fatal, 'MPP_GLOBAL_FIELD_INIT_COMM: incoming arrays do not match domain.' )
249 
250  if( isize_l == (domain%x(1)%compute%size+ishift) .AND. jsize_l == (domain%y(1)%compute%size+jshift) )then
251 !local is on compute domain
252  ioff = -domain%x(1)%compute%begin + 1
253  joff = -domain%y(1)%compute%begin + 1
254  elseif( isize_l == (domain%x(1)%memory%size+ishift) .AND. jsize_l == (domain%y(1)%memory%size+jshift) )then
255 !local is on data domain
256  ioff = -domain%x(1)%data%begin + 1
257  joff = -domain%y(1)%data%begin + 1
258  else
259  call mpp_error(fatal,'MPP_GLOBAL_FIELD_INIT_COMM: incoming field array must match either compute domain or data domain.')
260  endif
261 
262  ! Create unique domain identifier
263  domain_id=set_domain_id(domain%id,ksize,update_flags, position=position)
264  d_comm =>get_comm(domain_id,l_addr,l_addr2)
265 
266  if(d_comm%initialized)return ! Found existing field/domain communicator
267  d_comm%domain => domain
268  d_comm%isize_in = isize_l; d_comm%isize_out = isize_g
269  d_comm%jsize_in = jsize_l; d_comm%jsize_out = jsize_g
270  d_comm%ke = ksize
271  d_comm%gf_ioff=ioff; d_comm%gf_joff=joff
272 
273 !fill off-domains (note loops begin at an offset of 1)
274  if( xonly )then
275  lsize = size(domain%x(1)%list(:))
276 !send
277  allocate(d_comm%cto_pe(0:lsize-1))
278  d_comm%cto_pe=-1
279  do list = 0,lsize-1
280  lpos = mod(domain%x(1)%pos+lsize-list,lsize)
281  d_comm%cto_pe(list) = domain%x(1)%list(lpos)%pe
282  end do
283 !recv
284  allocate(d_comm%cfrom_pe(0:lsize-1))
285  allocate(d_comm%recvis(1,0:lsize-1), d_comm%recvie(1,0:lsize-1), &
286  d_comm%recvjs(1,0:lsize-1), d_comm%recvje(1,0:lsize-1), &
287  d_comm%R_msize(0:lsize-1))
288  d_comm%cfrom_pe=-1
289  d_comm%recvis=0; d_comm%recvie=0
290  d_comm%recvjs=0; d_comm%recvje=0;
291  d_comm%R_msize=0
292  do list = 0,lsize-1
293  rpos = mod(domain%x(1)%pos+list,lsize)
294  from_pe = domain%x(1)%list(rpos)%pe
295  d_comm%cfrom_pe(list) = from_pe
296  is = domain%list(from_pe)%x(1)%compute%begin; ie = domain%list(from_pe)%x(1)%compute%end+ishift
297  js = domain%y(1)%compute%begin; je = domain%y(1)%compute%end+jshift
298  d_comm%recvis(1,list)=is; d_comm%recvie(1,list)=ie
299  d_comm%recvjs(1,list)=js; d_comm%recvje(1,list)=je
300  d_comm%R_msize(list) = (ie-is+1) * (je-js+1) * ksize
301  end do
302 
303  elseif( yonly )then
304  lsize = size(domain%y(1)%list(:))
305 !send
306  allocate(d_comm%cto_pe(0:lsize-1))
307  d_comm%cto_pe=-1
308  do list = 0,lsize
309  lpos = mod(domain%y(1)%pos+lsize-list,lsize)
310  d_comm%cto_pe(list) = domain%y(1)%list(lpos)%pe
311  end do
312 !recv
313  allocate(d_comm%cfrom_pe(0:lsize-1))
314  allocate(d_comm%recvis(1,0:lsize-1), d_comm%recvie(1,0:lsize-1), &
315  d_comm%recvjs(1,0:lsize-1), d_comm%recvje(1,0:lsize-1), &
316  d_comm%R_msize(0:lsize-1))
317  d_comm%cfrom_pe=-1
318  d_comm%recvis=0; d_comm%recvie=0
319  d_comm%recvjs=0; d_comm%recvje=0;
320  d_comm%R_msize=0
321  do list = 0,lsize-1
322  rpos = mod(domain%y(1)%pos+list,lsize)
323  from_pe = domain%y(1)%list(rpos)%pe
324  d_comm%cfrom_pe(list) = from_pe
325  is = domain%x(1)%compute%begin; ie = domain%x(1)%compute%end+ishift
326  js = domain%list(from_pe)%y(1)%compute%begin; je = domain%list(from_pe)%y(1)%compute%end+jshift
327  d_comm%recvis(1,list)=is; d_comm%recvie(1,list)=ie
328  d_comm%recvjs(1,list)=js; d_comm%recvje(1,list)=je
329  d_comm%R_msize(list) = (ie-is+1) * (je-js+1) * ksize
330  end do
331 
332  else
333  nlist = size(domain%list(:))
334  tile_id = domain%tile_id(1)
335 
336  lsize = 0
337  do list = 0,nlist-1
338  if( domain%list(list)%tile_id(1) .NE. tile_id ) cycle
339  lsize = lsize+1
340  end do
341 
342  !send
343  allocate(d_comm%cto_pe(0:lsize-1))
344  d_comm%cto_pe=-1
345  n = 0
346  do list = 0,nlist-1
347  lpos = mod(domain%pos+nlist-list,nlist)
348  if( domain%list(lpos)%tile_id(1) .NE. tile_id ) cycle
349  d_comm%cto_pe(n) = domain%list(lpos)%pe
350  n = n + 1
351  end do
352  !recv
353  allocate(d_comm%cfrom_pe(0:lsize-1))
354  allocate(d_comm%recvis(1,0:lsize-1), d_comm%recvie(1,0:lsize-1), &
355  d_comm%recvjs(1,0:lsize-1), d_comm%recvje(1,0:lsize-1), &
356  d_comm%R_msize(0:lsize-1))
357  d_comm%cfrom_pe=-1
358  d_comm%recvis=0; d_comm%recvie=0
359  d_comm%recvjs=0; d_comm%recvje=0;
360  d_comm%R_msize=0
361  n = 0
362  do list = 0,nlist-1
363  rpos = mod(domain%pos+list,nlist)
364  if( domain%list(rpos)%tile_id(1) .NE. tile_id ) cycle
365  d_comm%cfrom_pe(n) = domain%list(rpos)%pe
366  is = domain%list(rpos)%x(1)%compute%begin; ie = domain%list(rpos)%x(1)%compute%end+ishift
367  js = domain%list(rpos)%y(1)%compute%begin; je = domain%list(rpos)%y(1)%compute%end+jshift
368  d_comm%recvis(1,n)=is; d_comm%recvie(1,n)=ie
369  d_comm%recvjs(1,n)=js; d_comm%recvje(1,n)=je
370  d_comm%R_msize(n) = (je-js+1) * (ie-is+1) * ksize
371  n = n+1
372  end do
373 
374  endif
375 
376  d_comm%Slist_size = lsize
377  d_comm%Rlist_size = lsize
378 
379 !send
380  allocate(d_comm%sendis(1,0:lsize-1), d_comm%sendie(1,0:lsize-1), &
381  d_comm%sendjs(1,0:lsize-1), d_comm%sendje(1,0:lsize-1), &
382  d_comm%S_msize(0:lsize-1),isl(0:lsize-1),jsl(0:lsize-1))
383  allocate(slist_addr(2,0:lsize-1))
384  isl=0; jsl=0
385  slist_addr = -9999
386  d_comm%sendis=0; d_comm%sendie=0
387  d_comm%sendjs=0; d_comm%sendje=0;
388  d_comm%S_msize=0
389  do list = 0,lsize-1
390  is=domain%x(1)%compute%begin; ie=domain%x(1)%compute%end+ishift
391  js=domain%y(1)%compute%begin; je=domain%y(1)%compute%end+jshift
392  d_comm%sendis(1,list)=is; d_comm%sendie(1,list)=ie
393  d_comm%sendjs(1,list)=js; d_comm%sendje(1,list)=je
394  d_comm%S_msize(list) = (je-js+1) * (ie-is+1) * ksize
395  isl(list) = ioff+domain%x(1)%compute%begin; jsl(list) = joff+domain%y(1)%compute%begin
396  end do
397 
398 !recv
399  allocate(d_comm%isizeR(0:lsize-1), d_comm%jsizeR(0:lsize-1))
400  allocate(d_comm%sendisR(1,0:lsize-1), d_comm%sendjsR(1,0:lsize-1))
401  if(.not.PRESENT(l_addr2))then
402  allocate(d_comm%rem_addr(0:lsize-1))
403  d_comm%rem_addr=-9999
404  else
405  allocate(d_comm%rem_addrx(0:lsize-1),d_comm%rem_addry(0:lsize-1))
406  d_comm%rem_addrx=-9999; d_comm%rem_addry=-9999
407  endif
408  d_comm%isizeR=0; d_comm%jsizeR=0
409  d_comm%sendisR=0; d_comm%sendjsR=0
410  rem_addr = -9999
411 
412  ! Handles case where S_msize and/or R_msize are 0 size array
413  msgsize = maxval( (/0,sum(d_comm%S_msize(:))/) ) + maxval( (/0,sum(d_comm%R_msize(:))/) )
414  if(msgsize>0)then
415  mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, msgsize )
416  if( mpp_domains_stack_hwm.GT.mpp_domains_stack_size )then
417  write( text,'(i8)' )mpp_domains_stack_hwm
418  call mpp_error( fatal, 'MPP_GLOBAL_FIELD_INIT_COMM: mpp_domains_stack overflow, call mpp_domains_set_stack_size(' &
419  //trim(text)//') from all PEs.' )
420  end if
421  end if
422 
423  DEALLOCATE(slist_addr,isl,jsl)
424 
425  d_comm%initialized = .true.
426 
427  end function mpp_global_field_init_comm

430  subroutine mpp_redistribute_free_comm(domain_in,l_addr,domain_out,l_addr2,ksize,lsize)
431  ! Since initialization of the d_comm type is expensive, freeing should be a rare
432  ! event. Thus no attempt is made to salvage freed d_comm's.
433  type(domain2D), intent(in) :: domain_in
434  integer(i8_kind), intent(in) :: l_addr
435  type(domain2D), intent(in) :: domain_out
436  integer(i8_kind), intent(in) :: l_addr2
437  integer, intent(in) :: ksize,lsize
438 
439  integer(i8_kind) :: domain_id

607  function find_key(key,sorted,insert) RESULT(n)
608  ! The algorithm used here requires monotonic keys w/out repetition.
609  integer(i8_kind),intent(in) :: key ! new address to be found in list
610  integer(i8_kind),dimension(:),intent(in) :: sorted ! list of sorted local addrs
611  integer, intent(out) :: insert
612  integer :: n, n_max, n_min, n_key
613  logical :: not_found
614 
615  n_key = size(sorted(:))
616  insert = 1
617  n = -1 ! value not in list
618  if(n_key == 0)return ! first call
619 
620  if(key < sorted(1))then
621  insert = 1; return
622  elseif(key > sorted(n_key))then
623  insert = n_key+1; return
624  endif
625 
626  if(key == sorted(1))then
627  n = 1; return
628  elseif(key == sorted(n_key))then
629  n = n_key; return
630  endif
631 
632  not_found = .true.
633  n = n_key/2 + 1
634  n_min=1; n_max=n_key
635  do while(not_found)
636  if(key == sorted(n))then
637  not_found = .false.
638  elseif(key > sorted(n))then
639  if(key < sorted(n+1))then
640  insert = n+1; exit
641  endif
642  n_min = n
643  n = (n+1+n_max)/2
644  else
645  if(key > sorted(n-1))then
646  insert = n; exit
647  endif
648  n_max = n
649  n = (n+n_min)/2
650  endif
651  if(n==1 .or. n==n_key)exit
652  end do
653  if(not_found)n = -1 ! value not in list
654  end function find_key
