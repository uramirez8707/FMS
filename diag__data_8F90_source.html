21 !! @brief Type descriptions and global variables for the diag_manager modules.
22 !! @author Seth Underwood <seth.underwood@noaa.gov>
23 !!
24 !! Notation:
25 !! <DL>
26 !! <DT>input field</DT>
27 !! <DD>The data structure describing the field as
28 !! registered by the model code.</DD>
29 !!
30 !! <DT>output field</DT>
31 !! <DD>The data structure describing the actual
32 !! diagnostic output with requested frequency and
33 !! other options.</DD>
34 !! </DL>
35 !!
36 !! Input fields, output fields, and output files are gathered in arrays called
37 !! "input_fields", "output_fields", and "files", respectively. Indices in these
38 !! arrays are used as pointers to create associations between various data
39 !! structures.
40 !!
41 !! Each input field associated with one or several output fields via array of
42 !! indices output_fields; each output field points to the single "parent" input
43 !! field with the input_field index, and to the output file with the output_file
44 !! index.

464  !> @brief Set the module variable base_time
465  subroutine set_base_time(base_time_int)
466  integer :: base_time_int(6) !< base_time as an array [year month day hour min sec]
467 
468  CHARACTER(len=9) :: amonth !< Month name
469  INTEGER :: stdlog_unit !< Fortran file unit number for the stdlog file.
470 
471  if (.not. module_is_initialized) call mpp_error(fatal, "set_base_time: diag_data is not initialized")
472  if (base_time_set) call mpp_error(fatal, "set_base_time: the base_time is already set!")
473 
474  base_year = base_time_int(1)
475  base_month = base_time_int(2)
476  base_day = base_time_int(3)
477  base_hour = base_time_int(4)
478  base_minute = base_time_int(5)
479  base_second = base_time_int(6)
480 
481  ! Set up the time type for base time
482  IF ( get_calendar_type() /= no_calendar ) THEN
483  IF ( base_year==0 .OR. base_month==0 .OR. base_day==0 ) THEN
484  call mpp_error(fatal, 'diag_data_mod::set_base_time'//&
485  & 'The base_year/month/day can not equal zero')
