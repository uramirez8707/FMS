21 program test_drifters_comm
22 
23  use drifters_core_mod
24  use drifters_comm_mod
25  use mpp_mod
26  use mpp_domains_mod
27  use fms_mod
28 
29  implicit none
30 
31  integer, parameter :: nd=2, npmax = 4
32  integer :: nx, ny, halox, haloy, layout(2), i, j, npes, pe, it, nt
33 ! _TYPE_DOMAIN2D :: domain
34  type(domain2d) :: domain
35  type(drifters_core_type) :: drfts
36  type(drifters_comm_type) :: drfts_com
37  real, parameter :: xmin=0., xmax=1., ymin=0., ymax=1.
38  real :: dx, dy, u0, v0, dt, Lx, Ly
39  real, allocatable :: x(:), y(:)
40  character(len=128) :: ermsg = ''
41  integer :: ids(npmax)
42  real :: positions(nd, npmax), velocity(nd, npmax)
43  integer :: io_status
44 !!$ integer :: stackmax=4000000
45 
46  namelist /drifters_comm_nml/ nx, ny, halox, haloy, u0, v0, dt, nt
47 
48  call mpp_init !(MPP_DEBUG)
49  !call mpp_set_stack_size(3145746)
50 
51  ! default input values
52  nx = 11
53  ny = 21
54  halox = 2
55  haloy = 2
56  u0 = 1.0
57  v0 = 0.0
58  dt = 0.1
59  nt = 10
60 
61  ! read input
62  read (input_nml_file, drifters_comm_nml, iostat=io_status)
63  io_status = check_nml_error(io_status, 'test_drifters_comm')
64 
65  ! create global domain
66  lx = xmax - xmin
67  ly = ymax - ymin
68  dx = lx/real(nx-1)
69  dy = ly/real(ny-1)
70  allocate(x(nx), y(ny))
71  x = xmin + (/ ( i*dx, i=0, nx-1) /)
72  y = ymin + (/ ( j*dy, j=0, ny-1) /)
73 
74  ! decompose domain
75  call mpp_domains_init ! (MPP_DEBUG)
76 !!$ call mpp_domains_set_stack_size(stackmax)
77 
78  npes = mpp_npes()
79  call mpp_define_layout( (/1,nx, 1,ny/), npes, layout )
80  if(mpp_pe()==0) print *,'LAYOUT: ', layout
81  call mpp_define_domains((/1,nx, 1,ny/), layout, domain, xhalo=halox, yhalo=haloy,&
82  & xflags=cyclic_global_domain, yflags=cyclic_global_domain)
83 
84  ! set up drifters' communicator
85  call drifters_comm_new(drfts_com)
86  call drifters_comm_set_domain(drfts_com, domain, x, y, 0,0)
87  ! repeated calls just for the fun of it
88  call drifters_comm_set_domain(drfts_com, domain, x, y, 0,0)
89  call drifters_comm_set_domain(drfts_com, domain, x, y, 0,0)
90 
91  ! create drifters
92  call drifters_core_new(drfts, nd, npmax, ermsg)
93  if(ermsg /= '') then
94  print *,'ERROR',ermsg
95  call mpp_error(fatal, 'test_drifters_comm: Error')
96  endif
97 
98  pe = mpp_pe()
99  ids = (/ (i+100*pe, i=1,npmax) /)
100  call drifters_core_set_ids(drfts, ids, ermsg)
101  if(ermsg /= '') then
102  print *,'ERROR', ermsg
103  call mpp_error(fatal, 'test_drifters_comm: Error')
104  endif
105 
106  ! position particles
107  if(pe == 0) then
108  positions(:, 1) = (/ (drfts_com%xcmin + drfts_com%xcmax)/2., &
109  & (drfts_com%ycmin + drfts_com%ycmax)/2. /)
110  !positions(:, 2) = (/0.,0.01/)
111  call drifters_core_set_positions(drfts, positions(:, 1:1), ermsg)
112  if(ermsg /= '') then
113  print *,'ERROR',ermsg
114  call mpp_error(fatal, 'test_drifters_comm: Error')
115  endif
116  endif
117 
118  ! push drifters
119  velocity(:,1) = (/u0, v0/)
120  do it = 1, nt
121  positions(:,1:drfts%np) = xmin + &
122  & modulo(drfts%positions(:,1:drfts%np) + dt*velocity(:,1:drfts%np)-xmin, xmax-xmin)
123  ! this will redistribute the drifters and update the positions
124  call drifters_comm_update(drfts_com, drfts, positions(:,1:drfts%np))
125 
126  if(drfts%np > 0) then
127  do i=1,drfts%np
128  print '(a,i6,a,i3,a,i3,a, i3, a,2f10.6)', 'PE: ',pe, ' it=', it, ' np=', drfts%np, ' ip=', i, &
129  & ' x,y=', drfts%positions(1,i), drfts%positions(2,i)
130  enddo
131  endif
132 !!$ call drifters_print(drfts, ermsg)
133 !!$ if(ermsg /= '') print *,ermsg
134  enddo
135 
136  ! clean up
137  call drifters_core_del(drfts, ermsg)
138  if(ermsg /= '') then
139  print *,ermsg, '<--- THIS'
140  call mpp_error(fatal, 'test_drifters_comm: Error')
141  endif
142  call drifters_comm_del(drfts_com)
143  call mpp_domains_exit
144  call mpp_exit
145 
146 end program test_drifters_comm
