144  subroutine test_netcdf_io_mosaic_r4(type, layout, ntiles_x, ntiles_y, io_layout)
145  character(len=*), intent(in) :: type
146  integer, intent(in) :: layout(:)
147  integer, intent(in) :: io_layout(:)
148  integer, intent(in) :: ntiles_x, ntiles_y
149 
150  integer :: ndim, nvar, natt, ntime
151  integer :: isc, iec, jsc, jec, nlon, nlat, n, i, j
152  integer :: my_tile, ncontacts, npes_per_tile, ntiles
153  integer, dimension(:), allocatable :: tile1, istart1, iend1, jstart1, jend1
154  integer, dimension(:), allocatable :: tile2, istart2, iend2, jstart2, jend2
155  integer, dimension(:), allocatable :: pe_start, pe_end
156  integer, dimension(:,:), allocatable :: layout2D, global_indices
157  character(len=64) :: output_file
158  logical :: is_root_pe
159  real(r4_kind), dimension(:,:,:), allocatable :: data, rdata
160  type(fieldtype), save :: vars(1)
161 
162  npes = mpp_npes()
163 
164  ncontacts = 0
165  ntiles = ntiles_x*ntiles_y
166 
167  npes_per_tile = npes/ntiles
168  my_tile = mpp_pe()/npes_per_tile + 1
169  is_root_pe = .false.
170  if(mpp_pe() == (my_tile-1)*npes_per_tile ) is_root_pe = .true.
171 
172  allocate(layout2d(2,ntiles), global_indices(4,ntiles), pe_start(ntiles), pe_end(ntiles) )
173  !--- for simplify purpose, we assume all the tiles have the same size.
174  do n = 1, ntiles
175  pe_start(n) = (n-1)*npes_per_tile
176  pe_end(n) = n*npes_per_tile-1
177  end do
178  if(ntiles>1) then
179  nlon = nx/ntiles_x
180  nlat = ny/ntiles_y
181  else
182  nlon = nx
183  nlat = ny
184  endif
185 
186  do n = 1, ntiles
187  global_indices(:,n) = (/1,nlon,1,nlat/)
188  layout2d(1,n) = layout(1)/ntiles_x
189  layout2d(2,n) = layout(2)/ntiles_y
190  end do
191 
192  call mpp_define_mosaic(global_indices, layout2d, domain, ntiles, ncontacts, tile1, tile2, &
193  istart1, iend1, jstart1, jend1, istart2, iend2, jstart2, jend2, pe_start, pe_end, &
194  name = type)
195  call mpp_get_compute_domain( domain, isc, iec, jsc, jec )
196  call mpp_get_domain_components(domain, xdom, ydom)
197  allocate( data (isc:iec,jsc:jec,nz) )
198  allocate( rdata(isc:iec,jsc:jec,nz) )
199  do k = 1,nz
200  do j = jsc, jec
201  do i = isc, iec
202  data(i,j,k) = k + i*1e-3 + j*1e-6
203  enddo
204  enddo
205  enddo
206 
207  ! open with netcdf distribute write if ntiles = 1, otherwise single-thread write
208  output_file = type
209  select case(type)
210  case("Single_tile_single_file_R4")
211  call mpp_open( unit, output_file, action=mpp_overwr, form=mpp_netcdf, threading=mpp_single, fileset=mpp_single )
212  case("Single_tile_mult_file_R4")
213  call mpp_open( unit, output_file, action=mpp_overwr, form=mpp_netcdf, threading=mpp_multi, fileset=mpp_multi )
214  case("Mult_tile_R4")
215  write(output_file, '(a,I4.4)') type//'.tile', my_tile
216  call mpp_open( unit, output_file, action=mpp_overwr, form=mpp_netcdf, threading=mpp_single, is_root_pe=is_root_pe )
217  case("Single_tile_with_group_R4")
218  call mpp_define_io_domain(domain, io_layout)
219  call mpp_open( unit, output_file, action=mpp_overwr, form=mpp_netcdf, threading=mpp_multi, fileset=mpp_multi, domain=domain)
220  case("Mult_tile_with_group_R4")
221  write(output_file, '(a,I4.4)') type//'.tile', my_tile
222  call mpp_open( unit, output_file, action=mpp_overwr, form=mpp_netcdf, threading=mpp_multi, fileset=mpp_multi, domain=domain)
223 
224  case default
225  call mpp_error(fatal, "program test_io_mosaic_R4_R8: invaid value of type="//type)
226  end select
227  ! write data
228  call mpp_write_meta( unit, x, 'X', 'km', 'X distance', 'X', domain=xdom, data=(/(i-1.,i=1,nlon)/) )
229  call mpp_write_meta( unit, y, 'Y', 'km', 'Y distance', 'Y', domain=ydom, data=(/(i-1.,i=1,nlat)/) )
230  call mpp_write_meta( unit, z, 'Z', 'km', 'Z distance', 'Z', data=(/(i-1.,i=1,nz)/) )
231  call mpp_write_meta( unit, t, 'T', 'sec', 'Time', 'T' )
232  call mpp_write_meta( unit, f, (/x,y,z,t/), 'Data', 'metres', 'Random data', pack=pack_size )
233  call mpp_write( unit, x )
234  call mpp_write( unit, y )
235  call mpp_write( unit, z )
236  do i = 0,nt-1
237  time4 = i*10.
238  call mpp_write( unit, f, domain, data, time4 )
239  end do
240  call mpp_close(unit)
241 
242  call mpp_sync() !wait for previous write to complete
243  ! reopen file and check results
244  select case(type)
245  case("Single_tile_single_file_R4")
246  call mpp_open( unit, output_file, action=mpp_rdonly, form=mpp_netcdf, threading=mpp_multi, fileset=mpp_single )
247  case("Single_tile_mult_file_R4")
248  call mpp_open( unit, output_file, action=mpp_rdonly, form=mpp_netcdf, threading=mpp_multi, fileset=mpp_multi )
249  case("Mult_tile_R4")
250  call mpp_open( unit, output_file, action=mpp_rdonly, form=mpp_netcdf, threading=mpp_multi, &
251  fileset=mpp_single, is_root_pe=is_root_pe )
252  case("Single_tile_with_group_R4", "Mult_tile_with_group_R4")
253  call mpp_open( unit, output_file, action=mpp_rdonly, form=mpp_netcdf, threading=mpp_multi, fileset=mpp_multi, domain=domain)
254  case default
255  call mpp_error(fatal, "program test_io_mosaic_R4_R8: invaid value of type="//type)
256  end select
257 
258  call mpp_get_info( unit, ndim, nvar, natt, ntime )
259  call mpp_get_fields ( unit, vars(:) )
260  call mpp_get_atts(vars(1),name=varname)
261 
262  if( varname.NE.'Data' )call mpp_error( fatal, 'File being read is not the expected one.' )
263  do i = 0,nt-1
264  call mpp_read( unit, vars(1), domain, rdata, 1 )
265  enddo
266  ! compare read and stored data to validate successful write/read
267  rchk = mpp_chksum(rdata)
268  chk = mpp_chksum( data)
269  if( pe.EQ.mpp_root_pe() )print '(a,2z18)', trim(type)//' checksum=', rchk, chk
270  if( rchk == chk ) then
271  if( pe.EQ.mpp_root_pe() )call mpp_error( note, trim(type)//': data comparison are OK.' )
272  else
273  call mpp_error( fatal, 'Checksum error on netCDF read for type ' &
274  //trim(type) )
275  end if
276 
277  deallocate( rdata, data)
278  call mpp_deallocate_domain(domain)
279 
280  end subroutine test_netcdf_io_mosaic_r4
