24 !> @defgroup drifters_mod drifters_mod
25 !> @ingroup drifters
26 !! @brief <TT>Drifters_mod</TT>is a module designed to advect a set of particles, in parallel or
27 !! sequentially, given an prescribed velocity field.
28 !! @author Alexander Pletzer
29 !!
30 !> Drifters are idealized point particles with positions that evolve in time according
31 !! to a prescribed velocity field, starting from some initial conditions. Drifters have
32 !! no mass, no energy, no size, and no friction and therefore have no impact on the
33 !! dynamics of the underlying system. The only feature that distinguishes a drifter
34 !! from another is its trajectory. This makes drifters ideal for tracking pollution
35 !! clouds and probing fields (e.g. temperature, salinity) along ocean currents, to name
36 !! a few applications.
37 !! Drifters can mimic real experiments such as the Argo floats
38 !! http://www.metoffice.com/research/ocean/argo/ukfloats.html.
39 !!
40 !! When run in parallel, on a 2d decomposed domain, <TT>drifters_mod</TT> will handle all the
41 !! bookkeeping and communication transparently for the user. This involves adding/removing
42 !! drifters as they enter/leave a processor element (PE) domain. Note that the number of drifters
43 !! can vary greatly both between PE domains and within a PE domain in the course of a simulation; the drifters'
44 !! module will also manage dynamically the memory for the user.
45 !!
46 !! There are a number of basic assumptions which could make the drifters' module
47 !! ill-suited for some tasks. First and foremost, it is assumed that the motion of
48 !! drifters is not erratic but follows deterministic trajectories. Furthermore,
49 !! drifters should not cross both compute and data domain boundaries within less
50 !! than a time step. This limitation is imposed by the Runge-Kutta integration
51 !! scheme, which must be able to complete, within a time step, a trajectory
52 !! calculation that starts inside the compute domain and ends inside the data domain. Therefore, the drifters,
53 !! as they are presently modelled, are unlikely to work for very fast objects.
54 !! This constraint also puts a upper limit to the domain decomposition, although
55 !! it can often be remedied by increasing the number of ghost nodes.
56 !!
57 !! Another fundamental assumption is that the (e.g. velocity) fields are structured,
58 !! on a per PE domain basis. There is no support for locally nested or unstrucured
59 !! meshes. Meshes need not be smooth and continuous across PE domains, however.

771  !============================================================================
772  !> @brief Interpolates positions onto longitude/latitude grid.
773  !! @details In many cases, the integrated positions will not be longitudes or latitudes. This call
774  !! can be ionvoked to recover the longitude/latitude positions from the "logical" positions.
775  subroutine drifters_positions2lonlat(self, positions, &
776  & x1, y1, geolon1, &
777  & x2, y2, geolat2, &
778  & lons, lats, &
779  & ermesg)
780 
781  type(drifters_type) :: self !< Opaque data structure.
782  ! Input positions
783  real, intent(in) :: positions(:,:) !< Logical positions.
784  ! Input mesh
785  real, intent(in) :: x1(:) !< X-axis of "geolon1" field.
786  real, intent(in) :: y1(:) !< Y-axis of "geolon1" field.
787  real, intent(in) :: geolon1(:,:) !< Y-axis of "geolon1" field.
788  real, intent(in) :: x2(:) !< X-axis of "geolat2" field.
789  real, intent(in) :: y2(:) !< Y-axis of "geolat2" field.
790  real, intent(in) :: geolat2(:,:) !< Latitude field as an array of (x2, y2)
791  ! Output lon/lat
792  real, intent(out) :: lons(:) !< Returned longitudes.
793  real, intent(out) :: lats(:) !< Returned latitudes.
794  character(len=*), intent(out) :: ermesg !< Error message (if any).
795 
796  real fvals(2**self%core%nd), ts(self%core%nd)
797  integer np, ij(2), ip, ier, n1s(2), n2s(2), i, j, iertot
798  character(len=10) :: n1_str, n2_str, np_str, iertot_str
799 
800  ermesg = ''
801  lons = -huge(1.)
802  lats = -huge(1.)
803 
804  ! check dimensions
805  n1s = (/size(x1), size(y1)/)
806  n2s = (/size(x2), size(y2)/)
807  if(n1s(1) /= size(geolon1, 1) .or. n1s(2) /= size(geolon1, 2)) then
808  ermesg = 'drifters_positions2geolonlat: ERROR incompatibles dims between (x1, y1, geolon1)'
809  return
810  endif
811  if(n2s(1) /= size(geolat2, 1) .or. n2s(2) /= size(geolat2, 2)) then
812  ermesg = 'drifters_positions2geolonlat: ERROR incompatibles dims between (x2, y2, geolat2)'
813  return
814  endif
815 
816  np = size(positions, 2)
817  if(size(lons) < np .or. size(lats) < np) then
818  write(np_str, '(i10)') np
819  write(n1_str, '(i10)') size(lons)
820  write(n2_str, '(i10)') size(lats)
821  ermesg = 'drifters_positions2geolonlat: ERROR size of "lons" ('//trim(n1_str)// &
822  & ') or "lats" ('//trim(n2_str)//') < '//trim(np_str)
823  return
824  endif
825 
826  ! Interpolate
827  iertot = 0
828  do ip = 1, np
829 
830  ! get longitude
831  call cld_ntrp_locate_cell(x1, positions(1,ip), i, ier)
832  iertot = iertot + ier
833  call cld_ntrp_locate_cell(y1, positions(2,ip), j, ier)
834  iertot = iertot + ier
835  ij(1) = i; ij(2) = j;
836  call cld_ntrp_get_cell_values(n1s, _flatten(geolon1), ij, fvals, ier)
837  iertot = iertot + ier
838  ts(1) = (positions(1,ip) - x1(i))/(x1(i+1) - x1(i))
839  ts(2) = (positions(2,ip) - y1(j))/(y1(j+1) - y1(j))
840  call cld_ntrp_linear_cell_interp(fvals, ts, lons(ip), ier)
841  iertot = iertot + ier
842 
843  ! get latitude
844  call cld_ntrp_locate_cell(x2, positions(1,ip), i, ier)
845  iertot = iertot + ier
846  call cld_ntrp_locate_cell(y2, positions(2,ip), j, ier)
847  iertot = iertot + ier
848  ij(1) = i; ij(2) = j;
849  call cld_ntrp_get_cell_values(n2s, _flatten(geolat2), ij, fvals, ier)
850  iertot = iertot + ier
851  ts(1) = (positions(1,ip) - x2(i))/(x2(i+1) - x2(i))
852  ts(2) = (positions(2,ip) - y2(j))/(y2(j+1) - y2(j))
853  call cld_ntrp_linear_cell_interp(fvals, ts, lats(ip), ier)
854  iertot = iertot + ier
855 
856  enddo
857 
858  if(iertot /= 0) then
859  write(iertot_str, '(i10)') iertot
860  ermesg = 'drifters_positions2geolonlat: ERROR '//trim(iertot_str)// &
861  & ' interpolation errors (domain out of bounds?)'
862  endif
863 
864  end subroutine drifters_positions2lonlat
