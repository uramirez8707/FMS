104  subroutine mpp_get_field_atts(field, name, units, longname, min, max, missing, ndim, siz, axes, atts, &
105  valid, scale, add, checksum)
106 
107  type(fieldtype), intent(in) :: field
108  character(len=*), intent(out), optional :: name, units
109  character(len=*), intent(out), optional :: longname
110  real, intent(out), optional :: min,max,missing
111  integer, intent(out), optional :: ndim
112  integer, intent(out), dimension(:), optional :: siz
113  type(validtype), intent(out), optional :: valid
114  real, intent(out), optional :: scale
115  real, intent(out), optional :: add
116  integer(i8_kind), intent(out), dimension(:), optional :: checksum
117 
118  type(atttype), intent(inout), dimension(:), optional :: atts
119  type(axistype), intent(inout), dimension(:), optional :: axes
120 
121  integer :: n,m, check_exist
122 
123  if (PRESENT(name)) name = field%name
124  if (PRESENT(units)) units = field%units
125  if (PRESENT(longname)) longname = field%longname
126  if (PRESENT(min)) min = field%min
127  if (PRESENT(max)) max = field%max
128  if (PRESENT(missing)) missing = field%missing
129  if (PRESENT(ndim)) ndim = field%ndim
130  if (PRESENT(atts)) then
131  atts = default_att
132  n = size(atts(:));m=size(field%Att(:))
133  if (n.LT.m)&
134  call mpp_error(fatal,'attribute array not large enough in mpp_get_field_atts, field '//&
135  trim(field%name))
136  do n=1,m
137  atts(n) = field%Att(n)
138  end do
139  end if
140  if (PRESENT(axes)) then
141  axes = default_axis
142  n = size(axes(:));m=field%ndim
143  if (n.LT.m) &
144  call mpp_error(fatal,'axis array not large enough in mpp_get_field_atts, field '//&
145  trim(field%name))
146  do n=1,m
147  axes(n) = field%axes(n)
148  end do
149  end if
150  if (PRESENT(siz)) then
151  siz = -1
152  n = size(siz(:));m=field%ndim
153  if (n.LT.m) &
154  call mpp_error(fatal,'size array not large enough in mpp_get_field_atts, field '//&
155  trim(field%name))
156  do n=1,m
157  siz(n) = field%size(n)
158  end do
159  end if
160 
161  if(PRESENT(valid)) then
162  call mpp_get_valid(field,valid)
163  endif
164 
165  if(PRESENT(scale)) scale = field%scale
166  if(present(add)) add = field%add
167  if(present(checksum)) then
168  checksum = 0
169  check_exist = mpp_find_att(field%Att(:),"checksum")
170  if ( check_exist >= 0 ) then
171  if(size(checksum(:)) >size(field%checksum(:))) call mpp_error(fatal,"size(checksum(:)) >size(field%checksum(:))")
172  checksum = field%checksum(1:size(checksum(:)))
173  endif
174  endif
175 
176  return
177  end subroutine mpp_get_field_atts

576  !#####################################################################
577  !> Based on presence/absence of attributes, defines valid range or missing
578  ! value. For details, see section 8.1 of NetCDF User Guide
579  subroutine mpp_get_valid(f,v)
580  type(fieldtype),intent(in) :: f ! field
581  type(validtype),intent(out) :: v ! validator
582 
583  integer :: irange,imin,imax,ifill,imissing,iscale
584  integer :: valid_T, scale_T ! types of attributes
585 
586  v%is_range = .true.
587  v%min = -huge(v%min); v%max = huge(v%max)
588  if (f%natt == 0) return
589  ! find indices of relevant attributes
590  irange = mpp_find_att(f%att,'valid_range')
591  imin = mpp_find_att(f%att,'valid_min')
592  imax = mpp_find_att(f%att,'valid_max')
593  ifill = mpp_find_att(f%att,'_FillValue')
594  imissing = mpp_find_att(f%att,'missing_value')
595 
596  ! find the widest type of scale and offset; note that the code
597  ! uses assumption that NetCDF types are arranged in th order of rank,
598  ! that is NF_BYTE < NF_SHORT < NF_INT < NF_FLOAT < NF_DOUBLE
599  scale_t = 0
600  iscale = mpp_find_att(f%att,'scale_factor')
601  if(iscale>0) scale_t = f%att(iscale)%type
602  iscale = mpp_find_att(f%att,'add_offset')
603  if(iscale>0) scale_t = max(scale_t,f%att(iscale)%type)
604 
605 
606  ! examine possible range attributes
607  valid_t = 0
608  if (irange>0) then
609  v%min = f%att(irange)%fatt(1)
610  v%max = f%att(irange)%fatt(2)
611  valid_t = f%att(irange)%type
612  else if (imax>0.or.imin>0) then
613  if(imax>0) then
614  v%max = f%att(imax)%fatt(1)
615  valid_t = max(valid_t,f%att(imax)%type)
616  endif
617  if(imin>0) then
618  v%min = f%att(imin)%fatt(1)
619  valid_t = max(valid_t,f%att(imin)%type)
620  endif
621  else if (imissing > 0) then
622  v%is_range = .false.
623  ! here we always scale, since missing_value is supposed to be in
624  ! external representation
625  v%min = f%att(imissing)%fatt(1)*f%scale + f%add
626  else if (ifill>0) then
627  !z1l ifdef is added in to be able to compile without using use_netCDF.
628 #ifdef use_netCDF
629  ! define min and max according to _FillValue
630  if(f%att(ifill)%fatt(1)>0) then
631  ! if _FillValue is positive, then it defines valid maximum
632  v%max = f%att(ifill)%fatt(1)
633  select case(f%type)
634  case (nf_byte,nf_short,nf_int)
635  v%max = v%max-1
636  case (nf_float)
637  v%max = nearest(nearest(real(v%max,4),-1.0),-1.0)
638  case (nf_double)
639  v%max = nearest(nearest(real(v%max,8),-1.0),-1.0)
640  end select
641  ! always do the scaling, as the _FillValue is in external
642  ! representation
643  v%max = v%max*f%scale + f%add
644  else
645  ! if _FillValue is negative or zero, then it defines valid minimum
646  v%min = f%att(ifill)%fatt(1)
647  select case(f%type)
648  case (nf_byte,nf_short,nf_int)
649  v%min = v%min+1
650  case (nf_float)
651  v%min = nearest(nearest(real(v%min,4),+1.0),+1.0)
652  case (nf_double)
653  v%min = nearest(nearest(real(v%min,8),+1.0),+1.0)
654  end select
655  ! always do the scaling, as the _FillValue is in external
656  ! representation
657  v%min = v%min*f%scale + f%add
658  endif
659 #endif
660  endif
661  ! If valid_range is the same type as scale_factor (actually the wider of
662  ! scale_factor and add_offset) and this is wider than the external data, then it
663  ! will be interpreted as being in the units of the internal (unpacked) data.
664  ! Otherwise it is in the units of the external (packed) data.
665  ! Note that it is not relevant if we went through the missing_data of _FillValue
666  ! brances, because in this case all irange, imin, and imax are less then 0
667  if(.not.((valid_t == scale_t).and.(scale_t>f%type))) then
668  if(irange>0 .or. imin>0) then
669  v%min = v%min*f%scale + f%add
670  endif
671  if(irange>0 .or. imax>0) then
672  v%max = v%max*f%scale + f%add
673  endif
674  endif
675 
676  end subroutine mpp_get_valid
