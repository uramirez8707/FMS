41 subroutine fill_coarse_data_r8(data, rotate, iadd, jadd, is_c, ie_c, js_c, je_c, nz, isd, jsd, nx, ny, &
42  ishift, jshift, x_add, y_add, sign1, sign2, x_cyclic, y_cyclic, ieg, jeg)
43  integer, intent(in) :: rotate, is_c, ie_c, js_c, je_c, nz, isd, jsd, iadd, jadd, nx, ny, ishift, jshift
44  integer, intent(in) :: sign1, sign2
45  real(kind=r8_kind), intent(inout) :: data(isd:, jsd:, :)
46  real(kind=r8_kind), intent(in) :: x_add, y_add
47  logical, intent(in) :: x_cyclic, y_cyclic
48  integer, intent(in) :: ieg, jeg
49  integer :: i, j, k
50 
51  select case (rotate)
52  case (zero)
53  ! convert the index to be consistent with the fine grid.
54  do k = 1, nz
55  do j = js_c, je_c+jshift
56  do i = is_c, ie_c+ishift
57  data(i,j,k) = dble(i+iadd)*1.d+6 + dble(j+jadd)*1.d+3 + dble(k) + x_add
58  enddo
59  enddo
60  enddo
61  case (ninety)
62  ! convert the index to be consistent with the fine grid.
63  do k = 1, nz
64  do j = js_c, je_c+jshift
65  do i = is_c, ie_c+ishift
66  data(i,j,k) = sign1*( dble(nx-j+1+iadd+jshift)*1.d+6 + dble(i+jadd)*1.d+3 + dble(k) + y_add)
67  enddo
68  enddo
69  enddo
70  case (minus_ninety)
71  ! convert the index to be consistent with the fine grid.
72  do k = 1, nz
73  do j = js_c, je_c+jshift
74  do i = is_c, ie_c+ishift
75  data(i,j,k) = sign2*( dble(j+iadd)*1.d+6 + dble(ny-i+1+jadd+ishift)*1.d+3 + dble(k) + y_add)
76  enddo
77  enddo
78  enddo
79  case default
80  call mpp_error(fatal,"fill_coarse_data_r8: rotate_coarse must be ZERO, NINETY, MINUS_NINETY")
81  end select
82 
83  !---handle cyclic condition
84  if(x_cyclic) then
85  if(ie_c+ishift+iadd == ieg) then
86  i = ie_c+ishift
87  do k = 1, nz
88  do j = js_c, je_c+jshift
89  data(i,j,k) = dble(i)*1.d+6 + dble(j+jadd)*1.d+3 + dble(k) + x_add
90  enddo
91  enddo
92  endif
93  endif
94 
95 
96  if(y_cyclic) then
97  if(je_c+jshift+jadd == jeg) then
98  j = je_c+jshift
99  do k = 1, nz
100  do j = js_c, je_c+jshift
101  data(i,j,k) = dble(i+iadd)*1.d+6 + j*1.d+3 + dble(k) + x_add
102  enddo
103  enddo
104  endif
105  endif
106 
107  end subroutine fill_coarse_data_r8

155  i = ie_c+ishift

158  data(i,j,k) = i*1.e+6 + (j+jadd)*1.e+3 + k + x_add

166  if(je_c+jshift+jadd == jeg) then
167  j = je_c+jshift
168  do k = 1, nz
169  do j = js_c, je_c+jshift
170  data(i,j,k) = (i+iadd)*1.e+6 + j*1.e+3 + k + x_add
171  enddo
172  enddo
173  endif
174  endif

217  do i = isl(n), iel(n)+ioff

280  !--- define mosaic domain for cubic grid
281  subroutine define_cubic_mosaic(type, domain, ni, nj, global_indices, layout, pe_start, pe_end, &
282  use_memsize, halo)
283  character(len=*), intent(in) :: type
284  type(domain2d), intent(inout) :: domain
285  integer, intent(in) :: global_indices(:,:), layout(:,:)
286  integer, intent(in) :: ni(:), nj(:)
287  integer, intent(in) :: pe_start(:), pe_end(:)
288  logical, optional, intent(in) :: use_memsize
289  integer, optional, intent(in) :: halo
290  integer, dimension(12) :: istart1, iend1, jstart1, jend1, tile1
291  integer, dimension(12) :: istart2, iend2, jstart2, jend2, tile2
292  integer :: ntiles, num_contact, msize(2)
293  logical :: use_memsize_local
294  integer :: whalo = 2, shalo = 2, ehalo = 2, nhalo = 2
295 
296  use_memsize_local = .true.
297  if(present(use_memsize)) use_memsize_local = use_memsize
298  if(present(halo)) then
299  nhalo = halo; shalo = halo; whalo = halo; shalo = halo
300  endif
301 
302  ntiles = 6
303  num_contact = 12
304  if(size(pe_start(:)) .NE. 6 .OR. size(pe_end(:)) .NE. 6 ) call mpp_error(fatal, &
305  "define_cubic_mosaic: size of pe_start and pe_end should be 6")
306  if(size(global_indices,1) .NE. 4) call mpp_error(fatal, &
307  "define_cubic_mosaic: size of first dimension of global_indices should be 4")
308  if(size(global_indices,2) .NE. 6) call mpp_error(fatal, &
309  "define_cubic_mosaic: size of second dimension of global_indices should be 6")
310  if(size(layout,1) .NE. 2) call mpp_error(fatal, &
311  "define_cubic_mosaic: size of first dimension of layout should be 2")
312  if(size(layout,2) .NE. 6) call mpp_error(fatal, &
313  "define_cubic_mosaic: size of second dimension of layout should be 6")
314  if(size(ni(:)) .NE. 6 .OR. size(nj(:)) .NE. 6) call mpp_error(fatal, &
315  "define_cubic_mosaic: size of ni and nj should be 6")
316 
317  !--- Contact line 1, between tile 1 (EAST) and tile 2 (WEST)
318  tile1(1) = 1; tile2(1) = 2
319  istart1(1) = ni(1); iend1(1) = ni(1); jstart1(1) = 1; jend1(1) = nj(1)
320  istart2(1) = 1; iend2(1) = 1; jstart2(1) = 1; jend2(1) = nj(2)
321  !--- Contact line 2, between tile 1 (NORTH) and tile 3 (WEST)
322  tile1(2) = 1; tile2(2) = 3
323  istart1(2) = 1; iend1(2) = ni(1); jstart1(2) = nj(1); jend1(2) = nj(1)
324  istart2(2) = 1; iend2(2) = 1; jstart2(2) = nj(3); jend2(2) = 1
325  !--- Contact line 3, between tile 1 (WEST) and tile 5 (NORTH)
326  tile1(3) = 1; tile2(3) = 5
327  istart1(3) = 1; iend1(3) = 1; jstart1(3) = 1; jend1(3) = nj(1)
328  istart2(3) = ni(5); iend2(3) = 1; jstart2(3) = nj(5); jend2(3) = nj(5)
329  !--- Contact line 4, between tile 1 (SOUTH) and tile 6 (NORTH)
330  tile1(4) = 1; tile2(4) = 6
331  istart1(4) = 1; iend1(4) = ni(1); jstart1(4) = 1; jend1(4) = 1
332  istart2(4) = 1; iend2(4) = ni(6); jstart2(4) = nj(6); jend2(4) = nj(6)
333  !--- Contact line 5, between tile 2 (NORTH) and tile 3 (SOUTH)
334  tile1(5) = 2; tile2(5) = 3
335  istart1(5) = 1; iend1(5) = ni(2); jstart1(5) = nj(2); jend1(5) = nj(2)
336  istart2(5) = 1; iend2(5) = ni(3); jstart2(5) = 1; jend2(5) = 1
337  !--- Contact line 6, between tile 2 (EAST) and tile 4 (SOUTH)
338  tile1(6) = 2; tile2(6) = 4
339  istart1(6) = ni(2); iend1(6) = ni(2); jstart1(6) = 1; jend1(6) = nj(2)
340  istart2(6) = ni(4); iend2(6) = 1; jstart2(6) = 1; jend2(6) = 1
341  !--- Contact line 7, between tile 2 (SOUTH) and tile 6 (EAST)
342  tile1(7) = 2; tile2(7) = 6
343  istart1(7) = 1; iend1(7) = ni(2); jstart1(7) = 1; jend1(7) = 1
344  istart2(7) = ni(6); iend2(7) = ni(6); jstart2(7) = nj(6); jend2(7) = 1
345  !--- Contact line 8, between tile 3 (EAST) and tile 4 (WEST)
346  tile1(8) = 3; tile2(8) = 4
347  istart1(8) = ni(3); iend1(8) = ni(3); jstart1(8) = 1; jend1(8) = nj(3)
348  istart2(8) = 1; iend2(8) = 1; jstart2(8) = 1; jend2(8) = nj(4)
349  !--- Contact line 9, between tile 3 (NORTH) and tile 5 (WEST)
350  tile1(9) = 3; tile2(9) = 5
351  istart1(9) = 1; iend1(9) = ni(3); jstart1(9) = nj(3); jend1(9) = nj(3)
352  istart2(9) = 1; iend2(9) = 1; jstart2(9) = nj(5); jend2(9) = 1
353  !--- Contact line 10, between tile 4 (NORTH) and tile 5 (SOUTH)
354  tile1(10) = 4; tile2(10) = 5
355  istart1(10) = 1; iend1(10) = ni(4); jstart1(10) = nj(4); jend1(10) = nj(4)
356  istart2(10) = 1; iend2(10) = ni(5); jstart2(10) = 1; jend2(10) = 1
357  !--- Contact line 11, between tile 4 (EAST) and tile 6 (SOUTH)
358  tile1(11) = 4; tile2(11) = 6
359  istart1(11) = ni(4); iend1(11) = ni(4); jstart1(11) = 1; jend1(11) = nj(4)
360  istart2(11) = ni(6); iend2(11) = 1; jstart2(11) = 1; jend2(11) = 1
361  !--- Contact line 12, between tile 5 (EAST) and tile 6 (WEST)
362  tile1(12) = 5; tile2(12) = 6
363  istart1(12) = ni(5); iend1(12) = ni(5); jstart1(12) = 1; jend1(12) = nj(5)
364  istart2(12) = 1; iend2(12) = 1; jstart2(12) = 1; jend2(12) = nj(6)
365  msize(1) = maxval(ni(:)/layout(1,:)) + whalo + ehalo + 1 ! make sure memory domain size is no smaller than
366  msize(2) = maxval(nj(:)/layout(2,:)) + shalo + nhalo + 1 ! data domain size
367 
368  if(use_memsize_local) then
369  call mpp_define_mosaic(global_indices, layout, domain, ntiles, num_contact, tile1, tile2, &
370  istart1, iend1, jstart1, jend1, istart2, iend2, jstart2, jend2, &
371  pe_start, pe_end, symmetry = .true., whalo=whalo, ehalo=ehalo, &
372  shalo=shalo, nhalo=nhalo, name = trim(type), memory_size = msize )
373  else
374  call mpp_define_mosaic(global_indices, layout, domain, ntiles, num_contact, tile1, tile2, &
375  istart1, iend1, jstart1, jend1, istart2, iend2, jstart2, jend2, &
376  pe_start, pe_end, symmetry = .true., whalo=whalo, ehalo=ehalo, &
377  shalo=shalo, nhalo=nhalo, name = trim(type) )
378  endif
379 
380  return
381 
382  end subroutine define_cubic_mosaic
