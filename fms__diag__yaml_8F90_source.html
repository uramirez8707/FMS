333 !> @brief Uses the yaml_parser_mod to read in the diag_table and fill in the
334 !! diag_yaml object
335 subroutine diag_yaml_object_init(diag_subset_output)
336  integer, intent(in) :: diag_subset_output !< DIAG_ALL - Current PE is in the one and only pelist
337  !! DIAG_OTHER - Current PE is not in the ocean pelist
338  !! and there are multiple pelists
339  !! DIAG_OCEAN - Current PE is in the ocean pelist
340  !! and there are multiple pelists
341  integer :: diag_yaml_id !< Id for the diag_table yaml
342  integer :: nfiles !< Number of files in the diag_table yaml
343  integer, allocatable :: diag_file_ids(:) !< Ids of the files in the diag_table yaml
344  integer :: i, j !< For do loops
345  integer :: total_nvars !< The total number of variables in the diag_table yaml
346  integer :: var_count !< The current number of variables added to the diag_yaml obj
347  integer :: file_var_count !< The current number of variables added in the diag_file
348  integer :: nvars !< The number of variables in the current file
349  integer, allocatable :: var_ids(:) !< Ids of the variables in diag_table yaml
350  logical :: is_ocean !< Flag indicating if it is an ocean file
351  logical, allocatable :: ignore(:) !< Flag indicating if the diag_file is going to be ignored
352  integer :: actual_num_files !< The actual number of files that were saved
353  integer :: file_count !! The current number of files added to the diag_yaml obj
354  logical :: write_file !< Flag indicating if the user wants the file to be written
355  logical :: write_var !< Flag indicating if the user wants the variable to be written
356 
357  if (diag_yaml_module_initialized) return
358 
359  diag_yaml_id = open_and_parse_file("diag_table.yaml")
360 
361  call diag_get_value_from_key(diag_yaml_id, 0, "title", diag_yaml%diag_title)
362  call get_value_from_key(diag_yaml_id, 0, "base_date", diag_yaml%diag_basedate)
363  call set_base_time(diag_yaml%diag_basedate)
364 
365  nfiles = get_num_blocks(diag_yaml_id, "diag_files")
366  allocate(diag_file_ids(nfiles))
367  allocate(ignore(nfiles))
368 
369  call get_block_ids(diag_yaml_id, "diag_files", diag_file_ids)
370 
371  ignore = .false.
372  total_nvars = 0
373  !< If you are on two seperate pelists
374  if(diag_subset_output .ne. diag_all) then
375  do i = 1, nfiles
376  is_ocean = .false.
377  call get_value_from_key(diag_yaml_id, diag_file_ids(i), "is_ocean", is_ocean, is_optional=.true.)
378  !< If you are on the ocean pelist and the file is not an ocean file, skip the file
379  if (diag_subset_output .eq. diag_ocean .and. .not. is_ocean) ignore(i) = .true.
380 
381  !< If you are not on the ocean pelist and the file is ocean, skip the file
382  if(diag_subset_output .eq. diag_other .and. is_ocean) ignore(i) = .true.
383  enddo
384  endif
385 
386  !< Determine how many files are in the diag_yaml, ignoring those with write_file = False
387  actual_num_files = 0
388  do i = 1, nfiles
389  write_file = .true.
390  call get_value_from_key(diag_yaml_id, diag_file_ids(i), "write_file", write_file, is_optional=.true.)
391  if(.not. write_file) ignore(i) = .true.
392 
393  if (.not. ignore(i)) then
394  actual_num_files = actual_num_files + 1
395  !< If ignoring the file, ignore the fields in that file too!
396  total_nvars = total_nvars + get_total_num_vars(diag_yaml_id, diag_file_ids(i))
397  endif
398  enddo
399 
400  allocate(diag_yaml%diag_files(actual_num_files))
401  allocate(diag_yaml%diag_fields(total_nvars))
402  allocate(variable_list%var_name(total_nvars))
403  allocate(variable_list%diag_field_indices(total_nvars))
404  allocate(file_list%file_name(actual_num_files))
405  allocate(file_list%diag_file_indices(actual_num_files))
406 
407  var_count = 0
408  file_count = 0
409  !> Loop through the number of nfiles and fill in the diag_yaml obj
410  nfiles_loop: do i = 1, nfiles
411  if(ignore(i)) cycle
412  file_count = file_count + 1
413  call diag_yaml_files_obj_init(diag_yaml%diag_files(file_count))
414  call fill_in_diag_files(diag_yaml_id, diag_file_ids(i), diag_yaml%diag_files(file_count))
415 
416  !> Save the file name in the file_list
417  !! The diag_table is not case sensitive (so we are saving it as lowercase)
418  file_list%file_name(file_count) = lowercase(trim(diag_yaml%diag_files(file_count)%file_fname)//c_null_char)
419  file_list%diag_file_indices(file_count) = file_count
420 
421  nvars = 0
422  nvars = get_num_blocks(diag_yaml_id, "varlist", parent_block_id=diag_file_ids(i))
423  allocate(var_ids(nvars))
424  call get_block_ids(diag_yaml_id, "varlist", var_ids, parent_block_id=diag_file_ids(i))
425  file_var_count = 0
426  allocate(diag_yaml%diag_files(file_count)%file_varlist(get_total_num_vars(diag_yaml_id, diag_file_ids(i))))
427  nvars_loop: do j = 1, nvars
428  write_var = .true.
429  call get_value_from_key(diag_yaml_id, var_ids(j), "write_var", write_var, is_optional=.true.)
430  if (.not. write_var) cycle
431 
432  var_count = var_count + 1
433  file_var_count = file_var_count + 1
434 
435  !> Save the filename in the diag_field type
436  diag_yaml%diag_fields(var_count)%var_fname = diag_yaml%diag_files(file_count)%file_fname
437 
438  call fill_in_diag_fields(diag_yaml_id, var_ids(j), diag_yaml%diag_fields(var_count))
439 
440  !> Save the variable name in the diag_file type
441  diag_yaml%diag_files(file_count)%file_varlist(file_var_count) = diag_yaml%diag_fields(var_count)%var_varname
442 
443  !> Save the variable name and the module name in the variable_list
444  variable_list%var_name(var_count) = trim(diag_yaml%diag_fields(var_count)%var_varname)//&
445  ":"//trim(diag_yaml%diag_fields(var_count)%var_module)//c_null_char
446  !! The diag_table is not case sensitive (so we are saving it as lowercase)
447  variable_list%var_name(var_count) = lowercase(variable_list%var_name(var_count))
448  variable_list%diag_field_indices(var_count) = var_count
449  enddo nvars_loop
450  deallocate(var_ids)
451  enddo nfiles_loop
452 
453  !> Sort the file list in alphabetical order
454  file_list%file_pointer = fms_array_to_pointer(file_list%file_name)
455  call fms_sort_this(file_list%file_pointer, actual_num_files, file_list%diag_file_indices)
456 
457  variable_list%var_pointer = fms_array_to_pointer(variable_list%var_name)
458  call fms_sort_this(variable_list%var_pointer, total_nvars, variable_list%diag_field_indices)
459 
460  deallocate(diag_file_ids)
461  diag_yaml_module_initialized = .true.
462 end subroutine
