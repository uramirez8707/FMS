87 !> @brief Uses the yaml_parser_mod to read in the diag_table and fill in the
88 !! diag_yaml object
89 subroutine diag_yaml_object_init
90  integer :: diag_yaml_id !< Id for the diag_table yaml
91  integer :: nfiles !< Number of files in the diag_table yaml
92  integer, allocatable :: file_ids(:) !< Ids of the files in the diag_table yaml
93  integer :: i, j !< For do loops
94  integer :: total_nvars !< The total number of variables in the diag_table yaml
95  integer :: var_count !< The current number of variables added to the diag_yaml obj
96  integer :: nvars !< The number of variables in the current file
97  integer, allocatable :: var_ids(:) !< Ids of the variables in diag_table yaml
98 
99  diag_yaml_id = open_and_parse_file("diag_table.yaml")
100 
101  call diag_get_value_from_key(diag_yaml_id, 0, "title", diag_yaml%diag_title)
102  call get_value_from_key(diag_yaml_id, 0, "base_date", diag_yaml%diag_basedate)
103 
104  nfiles = get_num_blocks(diag_yaml_id, "diag_files")
105  allocate(diag_yaml%diag_files(nfiles))
106  allocate(file_ids(nfiles))
107  call get_block_ids(diag_yaml_id, "diag_files", file_ids)
108 
109  total_nvars = get_total_num_vars(diag_yaml_id, file_ids)
110  allocate(diag_yaml%diag_fields(total_nvars))
111 
112  var_count = 0
113  do i = 1, nfiles
114  call diag_yaml_files_obj_init(diag_yaml%diag_files(i))
115  call fill_in_diag_files(diag_yaml_id, file_ids(i), diag_yaml%diag_files(i))
116 
117  nvars = 0
118  nvars = get_num_blocks(diag_yaml_id, "varlist", parent_block_id=file_ids(i))
119  allocate(var_ids(nvars))
120  call get_block_ids(diag_yaml_id, "varlist", var_ids, parent_block_id=file_ids(i))
121  do j = 1, nvars
122  var_count = var_count + 1
123  call fill_in_diag_fields(diag_yaml_id, var_ids(j), diag_yaml%diag_fields(var_count))
124  enddo
125  deallocate(var_ids)
126  enddo
127 
128  call dump_diag_yaml()
129 
130  deallocate(file_ids)
131 end subroutine
132 
133 !> @brief Destroys the diag_yaml object
134 subroutine diag_yaml_object_end()
135  integer :: i !< For do loops
136 
137  do i = 1, size(diag_yaml%diag_files, 1)
138  if(allocated(diag_yaml%diag_files(i)%file_global_meta)) deallocate(diag_yaml%diag_files(i)%file_global_meta)
139  enddo
140  if(allocated(diag_yaml%diag_files)) deallocate(diag_yaml%diag_files)
141 
142  do i = 1, size(diag_yaml%diag_fields, 1)

144  enddo
145  if(allocated(diag_yaml%diag_fields)) deallocate(diag_yaml%diag_fields)
146 
147 end subroutine diag_yaml_object_end

149 !> @brief Fills in a diag_yaml_files_type with the contents of a file block in diag_table.yaml
150 subroutine fill_in_diag_files(diag_yaml_id, file_id, fileobj)
151  integer, intent(in) :: diag_yaml_id !< Id of the diag_table.yaml
152  integer, intent(in) :: file_id !< Id of the file block to read
153  type(diag_yaml_files_type), intent(inout) :: fileobj !< diag_yaml_files_type obj to read the contents into
154 
155  integer :: nregion !< Flag indicating of there any regions (0 or 1)
156  integer :: region_id(1) !< Id of the region clock
157  integer :: natt !< Number of global attributes in the current file
158  integer :: global_att_id(1) !< Id of the global attributes block
159  integer :: nkeys !< Number of key/value global attributes pair
160  integer :: j !< For do loops
161 
162  integer, allocatable :: key_ids(:) !< Id of the gloabl atttributes key/value pairs
163 
164  call diag_get_value_from_key(diag_yaml_id, file_id, "file_name", fileobj%file_fname)
165  call diag_get_value_from_key(diag_yaml_id, file_id, "freq_units", fileobj%file_frequnit)
166  call get_value_from_key(diag_yaml_id, file_id, "freq", fileobj%file_freq)
167  call diag_get_value_from_key(diag_yaml_id, file_id, "unlimdim", fileobj%file_unlimdim)
168  call diag_get_value_from_key(diag_yaml_id, file_id, "time_units", fileobj%file_timeunit)
169  call diag_get_value_from_key(diag_yaml_id, file_id, "write_file", fileobj%string_file_write, is_optional=.true.)
170  if (fileobj%string_file_write .eq. "false") fileobj%file_write = .false.
171  !< Modeling realm?
172  call get_value_from_key(diag_yaml_id, file_id, "new_file_freq", fileobj%file_new_file_freq, is_optional=.true.)
173  call diag_get_value_from_key(diag_yaml_id, file_id, "new_file_freq_units", fileobj%file_new_file_freq_units, &
174  is_optional=.true.)
175  call diag_get_value_from_key(diag_yaml_id, file_id, "start_time", fileobj%file_start_time, is_optional=.true.)
176  call get_value_from_key(diag_yaml_id, file_id, "file_duration", fileobj%file_duration, is_optional=.true.)
177  call diag_get_value_from_key(diag_yaml_id, file_id, "file_duration_units", fileobj%file_duration_units, &
178  is_optional=.true.)
179 
180  nregion = 0
181  nregion = get_num_blocks(diag_yaml_id, "sub_region", parent_block_id=file_id)
182  if (nregion .ne. 0) then
183  call get_block_ids(diag_yaml_id, "sub_region", region_id, parent_block_id=file_id)
184  call diag_get_value_from_key(diag_yaml_id, region_id(1), "grid_type", fileobj%file_sub_region%grid_type)
185  if (trim(fileobj%file_sub_region%grid_type) .eq. "latlon") then
186  call get_region(diag_yaml_id, region_id(1), fileobj%file_sub_region%lat_lon_region)
187  elseif (trim(fileobj%file_sub_region%grid_type) .eq. "index") then
188  call get_region(diag_yaml_id, region_id(1), fileobj%file_sub_region%index_region)
189  endif
190  endif
191 
192  natt = 0
193  natt = get_num_blocks(diag_yaml_id, "global_meta", parent_block_id=file_id)
194  if (natt .ne. 0) then
195  call get_block_ids(diag_yaml_id, "global_meta", global_att_id, parent_block_id=file_id)
196  nkeys = get_nkeys(diag_yaml_id, global_att_id(1))
197  allocate(key_ids(nkeys))
198  call get_key_ids(diag_yaml_id, global_att_id(1), key_ids)
199 
200  allocate(fileobj%file_global_meta(nkeys, 2))
201  do j = 1, nkeys
202  call get_key_name(diag_yaml_id, key_ids(j), fileobj%file_global_meta(j, 1))
203  call get_key_value(diag_yaml_id, key_ids(j), fileobj%file_global_meta(j, 2))
204  enddo
205  deallocate(key_ids)
206  endif
207 
208 end subroutine

210 !> @brief Fills in a diag_yaml_files_var_type with the contents of a variable block in
211 !! diag_table.yaml
212 subroutine fill_in_diag_fields(file_id, var_id, field)
213  integer, intent(in) :: file_id !< Id of diag_table.yaml file
214  integer, intent(in) :: var_id !< Id of the variable block in the yaml file
215  type(diag_yaml_files_var_type), intent(out) :: field !< diag_yaml_files_var_type obj to read the contents into
216 
217 
218  integer :: natt !< Number of attributes in variable
219  integer :: var_att_id(1) !< Id of the variable attribute block
220  integer :: nkeys !< Number of key/value pairs of attributes
221  integer :: j !< For do loops
222 
223  integer, allocatable :: key_ids(:) !< Id of each attribute key/value pair
224 
225  field%var_write = .true.
226  call diag_get_value_from_key(file_id, var_id, "var_name", field%var_varname)
227  call diag_get_value_from_key(file_id, var_id, "reduction", field%var_reduction)
228  call diag_get_value_from_key(file_id, var_id, "module", field%var_module)
229  call diag_get_value_from_key(file_id, var_id, "kind", field%var_skind)
230  call diag_get_value_from_key(file_id, var_id, "write_var", field%string_var_write, is_optional=.true.)
231  if (trim(field%string_var_write) .eq. "false") field%var_write = .false.
232 
233  call diag_get_value_from_key(file_id, var_id, "output_name", field%var_outname)
234  call diag_get_value_from_key(file_id, var_id, "long_name", field%var_longname, is_optional=.true.)
235  !! VAR_UNITS !!
236 
237  natt = 0
238  natt = get_num_blocks(file_id, "attributes", parent_block_id=var_id)
239  if (natt .ne. 0) then
240  call get_block_ids(file_id, "attributes", var_att_id, parent_block_id=var_id)
241  nkeys = get_nkeys(file_id, var_att_id(1))
242  allocate(key_ids(nkeys))
243  call get_key_ids(file_id, var_att_id(1), key_ids)
244 
245  allocate(field%var_attributes(nkeys, 2))
246  do j = 1, nkeys
247  call get_key_name(file_id, key_ids(j), field%var_attributes(j, 1))
248  call get_key_value(file_id, key_ids(j), field%var_attributes(j, 2))
249  enddo
250  deallocate(key_ids)
251  endif
252 
253 end subroutine
