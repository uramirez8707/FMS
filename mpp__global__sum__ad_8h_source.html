23 subroutine MPP_GLOBAL_SUM_AD_( domain, field, gsum_, flags, position, tile_count, overflow_check)
24  MPP_TYPE_ :: MPP_GLOBAL_SUM_
25  type(domain2D), intent(in) :: domain
26  MPP_TYPE_, intent(inout) :: field(:,: MPP_EXTRA_INDICES_ )
27  MPP_TYPE_, intent(in) :: gsum_
28  integer, intent(in), optional :: flags
29  integer, intent(in), optional :: position
30  integer, intent(in), optional :: tile_count
31  logical, intent(in), optional :: overflow_check
32 
33  MPP_TYPE_, dimension(:,:), allocatable :: field2D
34  MPP_TYPE_, dimension(:,:), allocatable :: global2D
35  MPP_TYPE_, dimension(MAX_TILES), save :: gsum, nbrgsum, mygsum
36 
37  integer :: i,j, ioff,joff, isc, iec, jsc, jec, is, ie, js, je, ishift, jshift, ioffset, joffset
38  integer :: gxsize, gysize
39  integer :: global_flag, tile, ntile, nlist, n, list, m
40 
41  if( domain%max_ntile_pe > MAX_TILES ) call mpp_error(FATAL, "MPP_GLOBAL_SUM: number of tiles is exceed MAX_TILES")
42  ntile = size(domain%x(:))
43  nlist = size(domain%list(:))
44  tile = 1
45  if(present(tile_count)) tile = tile_count
46  global_flag = NON_BITWISE_EXACT_SUM
47  if(present(flags)) global_flag = flags
48 
49  call mpp_get_domain_shift(domain, ishift, jshift, position)
50 
51  if( size(field,1).EQ.domain%x(tile)%compute%size+ishift .AND. &
52  & size(field,2).EQ.domain%y(tile)%compute%size+jshift )then
53 !field is on compute domain
54  ioff = -domain%x(tile)%compute%begin + 1
55  joff = -domain%y(tile)%compute%begin + 1
56  else if( size(field,1).EQ.domain%x(tile)%memory%size+ishift .AND. &
57  & size(field,2).EQ.domain%y(tile)%memory%size+jshift )then
58 !field is on data domain
59  ioff = -domain%x(tile)%data%begin + 1
60  joff = -domain%y(tile)%data%begin + 1
61  else
62  call mpp_error( FATAL, &
63  & 'MPP_GLOBAL_SUM_: incoming field array must match either compute domain or data domain.' )
64  end if
65 
66  if(domain%ntiles > MAX_TILES) call mpp_error( FATAL, &
67  'MPP_GLOBAL_SUM_: number of tiles on this mosaic is greater than MAXTILES')
68 
69  call mpp_get_compute_domain( domain, is, ie, js, je, tile_count = tile_count )
70  isc = is; iec = ie + ishift; jsc = js; jec = je + jshift
71 
72  call mpp_get_global_domain(domain, xsize = gxsize, ysize = gysize )
73  MPP_GLOBAL_SUM_ = 0
74  if( global_flag == BITWISE_EXACT_SUM )then
75  !this is bitwise exact across different PE counts.
76 
77  allocate( field2D (isc:iec,jsc:jec) )
78  allocate( global2D( gxsize+ishift, gysize+jshift ) )
79  field2D = 0.
80  ioffset = domain%x(tile)%goffset*ishift; joffset = domain%y(tile)%goffset*jshift
81  if( tile == ntile) then
82  if(domain%ntiles == 1 ) then
83  MPP_GLOBAL_SUM_ = gsum_
84  mygsum(tile) = MPP_GLOBAL_SUM_
85  else if( nlist == 1) then
86  MPP_GLOBAL_SUM_ = gsum_
87  mygsum(1:ntile) = MPP_GLOBAL_SUM_
88  else ! need to sum by the order of tile_count
89  MPP_GLOBAL_SUM_ = gsum_
90  gsum(1:domain%ntiles) = MPP_GLOBAL_SUM_
91  !--- receive data from root_pe of each tile
92  do list = 1, nlist - 1
93  m = mod( domain%pos+nlist-list, nlist )
94  if( domain%list(m)%pe == domain%list(m)%tile_root_pe ) then
95  do n = 1, size(domain%list(m)%x(:))
96  nbrgsum(n) = gsum(domain%list(m)%tile_id(n))
97  end do
98  call mpp_send( nbrgsum(1), plen=size(domain%list(m)%x(:)), to_pe=domain%list(m)%pe, tag=COMM_TAG_1)
99  end if
100  end do
101  !--- send the data to other pe if the current pe is the root pe of any tile
102  if( mpp_domain_is_tile_root_pe(domain) ) then
103  do list = 1, nlist - 1
104  m = mod( domain%pos+list, nlist )
105  call mpp_recv( mygsum(1), glen=ntile, from_pe=domain%list(m)%pe, tag=COMM_TAG_1 )
106  end do
107  end if
108  call mpp_sync_self()
109  ! first fill the global sum on current pe.
110  do n = 1, ntile
111  mygsum(n) = gsum(domain%tile_id(n))
112  end do
113 
114  end if
115  end if
116  global2D(1:gxsize+ioffset,1:gysize+joffset) = mygsum(tile)
117  if ( present( tile_count ) ) then
118  call mpp_global_field_ad( domain, field2D, global2D, position=position, tile_count=tile_count )
119  else
120  call mpp_global_field_ad( domain, field2D, global2D, position=position )
121  endif
122 
123  do j = jsc, jec
124  do i = isc, iec
125  field(i+ioff:i+ioff,j+joff:j+joff MPP_EXTRA_INDICES_) = field2D(i,j)
126  end do
127  end do
128 
129  deallocate(global2D, field2d)
130  else if ( global_flag == BITWISE_EFP_SUM )then
131 #ifdef DO_EFP_SUM_
132  !this is bitwise across different PE counts using EFP sum
133  if( ntile > 1 ) then
134  call mpp_error( FATAL, 'MPP_GLOBAL_SUM_: multiple tile per pe is not supported for BITWISE_EFP_SUM')
135  endif
136 
137  do j = jsc, jec
138  do i = isc, iec
139  field(i+ioff:i+ioff,j+joff:j+joff MPP_EXTRA_INDICES_) = gsum_
140  end do
141  end do
142 #else
143  call mpp_error( FATAL, &
144  & 'MPP_GLOBAL_SUM_: BITWISE_EFP_SUM is only implemented for real number, contact developer')
145 #endif
146  else !this is not bitwise-exact across different PE counts
147  ioffset = domain%x(tile)%loffset*ishift; joffset = domain%y(tile)%loffset*jshift
148  mygsum(1:ntile) = 0
149  if(tile == ntile) then
150  MPP_GLOBAL_SUM_ = gsum_
151  call mpp_sum_ad( MPP_GLOBAL_SUM_, domain%list(:)%pe )
152  mygsum(1:ntile) = MPP_GLOBAL_SUM_
153  end if
154  field(is+ioff:ie+ioff+ioffset, js+joff:je+joff+joffset MPP_EXTRA_INDICES_) = mygsum(tile)
155  end if
156 
157  return
158  end subroutine MPP_GLOBAL_SUM_AD_
