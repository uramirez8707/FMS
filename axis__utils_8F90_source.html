143  !> @brief Return axis_bound either from an array of available axes, or defined based on axis mid-points
144  subroutine get_axis_bounds(axis,axis_bound,axes,bnd_name,err_msg)
145 
146  type(axistype), intent(in) :: axis
147  type(axistype), intent(inout) :: axis_bound
148  type(axistype), intent(in), dimension(:) :: axes
149  character(len=*), intent(inout), optional :: bnd_name
150  character(len=*), intent(out), optional :: err_msg
151 
152  real, dimension(:), allocatable :: data, tmp
153 
154  integer :: i, len
155  character(len=128) :: name, units
156  character(len=256) :: longname
157  character(len=1) :: cartesian
158  logical :: bounds_found
159 
160  if(present(err_msg)) then
161  err_msg = ''
162  endif
163  axis_bound = default_axis
164  call mpp_get_atts(axis,units=units,longname=longname,&
165  cartesian=cartesian, len=len)
166  if(len .LE. 0) return
167  allocate(data(len+1))
168 
169  bounds_found = mpp_get_axis_bounds(axis, data, name=name)
170  longname = trim(longname)//' bounds'
171 
172  if(.not.bounds_found .and. len>1 ) then
173  ! The following calculation can not be done for len=1
174  call mpp_get_atts(axis,name=name)
175  name = trim(name)//'_bnds'
176  allocate(tmp(len))
177  call mpp_get_axis_data(axis,tmp)
178  do i=2,len
179  data(i)= tmp(i-1)+fp5*(tmp(i)-tmp(i-1))
180  enddo
181  data(1)= tmp(1)- fp5*(tmp(2)-tmp(1))
182  if (abs(data(1)) < epsln) data(1) = 0.0
183  data(len+1)= tmp(len)+ fp5*(tmp(len)-tmp(len-1))
184  if (data(1) == 0.0) then
185  if (abs(data(len+1)-360.) > epsln) data(len+1)=360.0
186  endif
187  endif
188  if(bounds_found .OR. len>1) then
189  call mpp_modify_meta(axis_bound,name=name,units=units,longname=&
190  longname,cartesian=cartesian,data=data)
191  endif
192  if(allocated(tmp)) deallocate(tmp)
193  deallocate(data)
194 
195  return
196  end subroutine get_axis_bounds

292  !> @brief Returns lon_strt <= longitude <= lon_strt+360
293  !! @return real lon_in_range
294  function lon_in_range(lon, l_strt)
295  real :: lon, l_strt, lon_in_range, l_end
296 
297  lon_in_range = lon
298  l_end = l_strt+360.
299 
300  if (abs(lon_in_range - l_strt) < 1.e-4) then
301  lon_in_range = l_strt
302  return
303  endif
304 
305  if (abs(lon_in_range - l_end) < 1.e-4) then
306  lon_in_range = l_strt
307  return
308  endif
309 
310  do
311  if (lon_in_range < l_strt) then
312  lon_in_range = lon_in_range + f360;
313  else if (lon_in_range > l_end) then
314  lon_in_range = lon_in_range - f360;
315  else
316  exit
317  end if
318  end do
319 
320  end function lon_in_range

322  !> @brief Returns monotonic array of longitudes s.t., lon_strt <= lon(:) <= lon_strt+360.
323  subroutine tranlon(lon, lon_start, istrt)
324 
325  ! returns array of longitudes s.t. lon_strt <= lon < lon_strt+360.
326  ! also, the first istrt-1 entries are moved to the end of the array
327  !
328  ! e.g.
329  ! lon = 0 1 2 3 4 5 ... 358 359; lon_strt = 3 ==>
330  ! tranlon = 3 4 5 6 7 8 ... 359 360 361 362; istrt = 4
331 
332  real, intent(inout), dimension(:) :: lon
333  real, intent(in) :: lon_start
334  integer, intent(out) :: istrt
335 
336 
337  integer :: len, i
338  real :: lon_strt, tmp(size(lon(:))-1)
339 
340  len = size(lon(:))
341 
342  do i=1,len
343  lon(i) = lon_in_range(lon(i),lon_start)
344  enddo
345 
346  istrt=0
347  do i=1,len-1
348  if (lon(i+1) < lon(i)) then
349  istrt=i+1
350  exit
351  endif
352  enddo
353 
354  if (istrt>1) then ! grid is not monotonic
355  if (abs(lon(len)-lon(1)) < epsln) then
356  tmp = cshift(lon(1:len-1),istrt-1)
357  lon(1:len-1) = tmp
358  lon(len) = lon(1)
359  else
360  lon = cshift(lon,istrt-1)
361  endif
362  lon_strt = lon(1)
363  do i=2,len+1
364  lon(i) = lon_in_range(lon(i),lon_strt)
365  lon_strt = lon(i)
366  enddo
367  endif
368 
369  return
370  end subroutine tranlon

372  !> nearest_index = index of nearest data point within "array" corresponding to
373  !! "value".
374  !!
375  !! inputs:
376  !!
377  !! value = arbitrary data...same units as elements in "array"
378  !! array = array of data points (must be monotonically increasing)
379  !!
380  !! output:
381  !!
382  !! nearest_index = index of nearest data point to "value"
383  !! if "value" is outside the domain of "array" then nearest_index = 1
384  !! or "ia" depending on whether array(1) or array(ia) is
385  !! closest to "value"
386  !!
387  !! note: if "array" is dimensioned array(0:ia) in the calling
388  !! program, then the returned index should be reduced
389  !! by one to account for the zero base.
390  !!
391  !! example:
392  !!
393  !! let model depths be defined by the following:
394  !! parameter (km=5)
395  !! dimension z(km)
396  !! data z /5.0, 10.0, 50.0, 100.0, 250.0/
397  !!
398  !! k1 = nearest_index (12.5, z, km)
399  !! k2 = nearest_index (0.0, z, km)
400  !!
401  !! k1 would be set to 2, and k2 would be set to 1 so that
402  !! z(k1) would be the nearest data point to 12.5 and z(k2) would
403  !! be the nearest data point to 0.0
404  !!
405  !! @return real frac_index
406  function frac_index (value, array)
407  integer :: ia, i, ii, unit
408  real :: value !< arbitrary data...same units as elements in "array"
409  real :: frac_index
410  real, dimension(:) :: array !< array of data points (must be monotonically increasing)
411  logical keep_going
412 
413  ia = size(array(:))
414 
415  do i=2,ia
416  if (array(i) < array(i-1)) then
417  unit = stdout()
418  write (unit,*) &
419  '=> Error: "frac_index" array must be monotonically increasing when searching for nearest value to ', value
420  write (unit,*) ' array(i) < array(i-1) for i=',i
421  write (unit,*) ' array(i) for i=1..ia follows:'
422  do ii=1,ia
423  write (unit,*) 'i=',ii, ' array(i)=',array(ii)
424  enddo
425  call mpp_error(fatal,' "frac_index" array must be monotonically increasing.')
426  endif
427  enddo
428  if (value < array(1) .or. value > array(ia)) then
429 ! if (value < array(1)) frac_index = 1.
430 ! if (value > array(ia)) frac_index = float(ia)
431  frac_index = -1.0
432  else
433  i=1
434  keep_going = .true.
435  do while (i <= ia .and. keep_going)
436  i = i+1
437  if (value <= array(i)) then
438  frac_index = float(i-1) + (value-array(i-1))/(array(i)-array(i-1))
439  keep_going = .false.
440  endif
441  enddo
442  endif
443  end function frac_index

445  !> @brief Return index of nearest point along axis
446  !!
447  !> nearest_index = index of nearest data point within "array" corresponding to
448  !! "value".
449  !!
450  !! inputs:
451  !!
452  !! value = arbitrary data...same units as elements in "array"
453  !! array = array of data points (must be monotonically increasing)
454  !! ia = dimension of "array"
455  !!
456  !! output:
457  !!
458  !! nearest_index = index of nearest data point to "value"
459  !! if "value" is outside the domain of "array" then nearest_index = 1
460  !! or "ia" depending on whether array(1) or array(ia) is
461  !! closest to "value"
462  !!
463  !! note: if "array" is dimensioned array(0:ia) in the calling
464  !! program, then the returned index should be reduced
465  !! by one to account for the zero base.
466  !!
467  !! example:
468  !!
469  !! let model depths be defined by the following:
470  !! parameter (km=5)
471  !! dimension z(km)
472  !! data z /5.0, 10.0, 50.0, 100.0, 250.0/
473  !!
474  !! k1 = nearest_index (12.5, z, km)
475  !! k2 = nearest_index (0.0, z, km)
476  !!
477  !! k1 would be set to 2, and k2 would be set to 1 so that
478  !! z(k1) would be the nearest data point to 12.5 and z(k2) would
479  !! be the nearest data point to 0.0
480  !! @return integer nearest_index
481  function nearest_index (value, array)
482  !=======================================================================
483  !
484  !
485  !=======================================================================
486 
487  integer :: nearest_index !< index of nearest data point to "value"
488  !! if "value" is outside the domain of "array" then nearest_index = 1
489  !! or "ia" depending on whether array(1) or array(ia) is
490  !! closest to "value"
491  Integer :: i, ii, unit
492  integer :: ia !< dimension of "array"
493  real :: value !< arbitrary data...same units as elements in "array"
494  real, dimension(:) :: array !< array of data points (must be monotonically increasing)
495  logical keep_going
496 
497  ia = size(array(:))
498 
499  do i=2,ia
500  if (array(i) < array(i-1)) then
501  unit = stdout()
502  write (unit,*) '=> Error: "nearest_index" array must be monotonically increasing &
503  &when searching for nearest value to ',value
504  write (unit,*) ' array(i) < array(i-1) for i=',i
505  write (unit,*) ' array(i) for i=1..ia follows:'
506  do ii=1,ia
507  write (unit,*) 'i=',ii, ' array(i)=',array(ii)
508  enddo
509  call mpp_error(fatal,' "nearest_index" array must be monotonically increasing.')
510  endif
511  enddo
512  if (value < array(1) .or. value > array(ia)) then
513  if (value < array(1)) nearest_index = 1
514  if (value > array(ia)) nearest_index = ia
515  else
516  i=1
517  keep_going = .true.
518  do while (i <= ia .and. keep_going)
519  i = i+1
520  if (value <= array(i)) then
521  nearest_index = i
522  if (array(i)-value > value-array(i-1)) nearest_index = i-1
523  keep_going = .false.
524  endif
525  enddo
526  endif
527  end function nearest_index

531  subroutine interp_1d_linear(grid1,grid2,data1,data2)
532 
533  real, dimension(:), intent(in) :: grid1, data1, grid2
534  real, dimension(:), intent(inout) :: data2
535 
536  integer :: n1, n2, i, n
537  real :: w
538 
539  n1 = size(grid1(:))
540  n2 = size(grid2(:))
541 
542 
543  do i=2,n1
544  if (grid1(i) <= grid1(i-1)) call mpp_error(fatal, 'grid1 not monotonic')
545  enddo
546 
547  do i=2,n2
548  if (grid2(i) <= grid2(i-1)) call mpp_error(fatal, 'grid2 not monotonic')
549  enddo
550 
551  if (grid1(1) > grid2(1) ) call mpp_error(fatal, 'grid2 lies outside grid1')
552  if (grid1(n1) < grid2(n2) ) call mpp_error(fatal, 'grid2 lies outside grid1')
553 
554  do i=1,n2
555  n = nearest_index(grid2(i),grid1)
556 
557  if (grid1(n) < grid2(i)) then
558  w = (grid2(i)-grid1(n))/(grid1(n+1)-grid1(n))
559  data2(i) = (1.-w)*data1(n) + w*data1(n+1)
560  else
561  if(n==1) then
562  data2(i) = data1(n)
563  else
564  w = (grid2(i)-grid1(n-1))/(grid1(n)-grid1(n-1))
565  data2(i) = (1.-w)*data1(n-1) + w*data1(n)
566  endif
567  endif
568  enddo
569 
570 
571  return
572 
573  end subroutine interp_1d_linear

576  subroutine interp_1d_cubic_spline(grid1, grid2, data1, data2, yp1, ypn)
577 
578  real, dimension(:), intent(in) :: grid1, grid2, data1
579  real, dimension(:), intent(inout) :: data2
580  real, intent(in) :: yp1, ypn
581 
582  real, dimension(size(grid1)) :: y2, u
583  real :: sig, p, qn, un, h, a ,b
584  integer :: n, m, i, k, klo, khi
585 
586  n = size(grid1(:))
587  m = size(grid2(:))
588 
589  do i=2,n
590  if (grid1(i) <= grid1(i-1)) call mpp_error(fatal, 'grid1 not monotonic')
591  enddo
592 
593  do i=2,m
594  if (grid2(i) <= grid2(i-1)) call mpp_error(fatal, 'grid2 not monotonic')
595  enddo
596 
597  if (grid1(1) > grid2(1) ) call mpp_error(fatal, 'grid2 lies outside grid1')
598  if (grid1(n) < grid2(m) ) call mpp_error(fatal, 'grid2 lies outside grid1')
599 
600  if (yp1 >.99e30) then
601  y2(1)=0.
602  u(1)=0.
603  else
604  y2(1)=-0.5
605  u(1)=(3./(grid1(2)-grid1(1)))*((data1(2)-data1(1))/(grid1(2)-grid1(1))-yp1)
606  endif
607 
608  do i=2,n-1
609  sig=(grid1(i)-grid1(i-1))/(grid1(i+1)-grid1(i-1))
610  p=sig*y2(i-1)+2.
611  y2(i)=(sig-1.)/p
612  u(i)=(6.*((data1(i+1)-data1(i))/(grid1(i+1)-grid1(i))-(data1(i)-data1(i-1)) &
613  /(grid1(i)-grid1(i-1)))/(grid1(i+1)-grid1(i-1))-sig*u(i-1))/p
614  enddo
615 
616  if (ypn > .99e30) then
617  qn=0.
618  un=0.
619  else
620  qn=0.5
621  un=(3./(grid1(n)-grid1(n-1)))*(ypn-(data1(n)-data1(n-1))/(grid1(n)-grid1(n-1)))
622  endif
623 
624  y2(n)=(un-qn*u(n-1))/(qn*y2(n-1)+1.)
625 
626  do k=n-1,1,-1
627  y2(k)=y2(k)*y2(k+1)+u(k)
628  enddo
629 
630  do k = 1, m
631  n = nearest_index(grid2(k),grid1)
632  if (grid1(n) < grid2(k)) then
633  klo = n
634  else
635  if(n==1) then
636  klo = n
637  else
638  klo = n -1
639  endif
640  endif
641  khi = klo+1
642  h = grid1(khi)-grid1(klo)
643  a = (grid1(khi) - grid2(k))/h
644  b = (grid2(k) - grid1(klo))/h
645  data2(k) = a*data1(klo) + b*data1(khi)+ ((a**3-a)*y2(klo) + (b**3-b)*y2(khi))*(h**2)/6.
646  enddo
647 
648  end subroutine interp_1d_cubic_spline
