95 !> @brief Add an argument to the parser object.
96 subroutine add_argument(parser,short_name,description,requires_val, long_name)
97 
98  type(parser_t), intent(inout) :: parser !< Parser object.
99  character(len=*), intent(in) :: short_name !< Short name.
100  character(len=*), intent(in) :: description !< What the option does.
101  logical, intent(in), optional :: requires_val !< Does the option require
102  !! a value?
103  character(len=*), intent(in), optional :: long_name !< Long name.
104  type(argument_t), pointer :: arg
105  type(argument_t), pointer :: a
106 
107  !Create a new arg linked list node.
108  allocate(arg)
109  arg%head => null()
110  call string_copy(arg%short_name, short_name)
111  call string_copy(arg%description, description)
112  parser%argc = parser%argc + 1
113  arg%long_name = ""
114  arg%val = ""
115 
116  !Optional arguments must have short names that begin with a -, or
117  !else they are assumed to be positional arguments.
118  if (short_name(1:1) .eq. "-") then
119  arg%positional = .false.
120  if (present(long_name)) then
121 
122  !Long names are required to start with --.
123  if (long_name(1:2) .ne. "--") then
124  call error("argument long names must start with --")
125  endif
126  call string_copy(arg%long_name, long_name)
127  endif
128  if (.not. present(requires_val)) then
129  call error("you must specify whether or not a value is" &
130  //" required for optional arguments.")
131  endif
132  arg%requires_val = requires_val
133  else
134  arg%positional = .true.
135  arg%requires_val = .false.
136  parser%num_pos_args = parser%num_pos_args + 1
137  endif
138 
139  !Add the new argument to the end of the linked list. Trap any
140  !duplicate arguments (this results in an error).
141  !To do: switch from a linked list to a dictionary to increase
142  !efficiency.
143  if (associated(parser%args)) then
144  a => parser%args
145  do while (.true.)
146  call catch_duplicate_args(a, arg)
147  if (associated(a%head)) then
148  a => a%head
149  else
150  exit
151  endif
152  enddo
153  a%head => arg
154  else
155  parser%args => arg
156  endif
157 end subroutine add_argument

161 subroutine help_mesg(parser)
162 
163  type(parser_t), intent(inout) :: parser !< Parser object.
164 
165  type(argument_t), pointer :: arg
166 
167  !Write out positional arguments first.
168  if (parser%num_pos_args .gt. 0) then
169  write(error_unit,*) "Positional arguments:"
170  arg => parser%args
171  do while (associated(arg))
172  if (arg%positional) then
173  call print_argument(arg)
174  endif
175  arg => arg%head
176  enddo
177  write(error_unit,*)
178  endif
179 
180  !Write out optional arguments.
181  if (parser%argc .gt. parser%num_pos_args) then
182  write(error_unit,*) "Optional arguments:"
183  arg => parser%args
184  do while (associated(arg))
185  if (.not. arg%positional) then
186  call print_argument(arg)
187  endif
188  arg => arg%head
189  enddo
190  write(error_unit,*)
191  endif

196 subroutine usage_mesg(parser)
197 
198  type(parser_t),intent(inout) :: parser !< Parser object.
199 
200  type(argument_t), pointer :: arg
201  character(len=512) :: buf
202  character(len=512) :: b
203 
204  call get_command_argument(0, buf)
205  call string_copy(buf,"Usage: "//trim(buf))
206  if (parser%argc .gt. 0) then
207  arg => parser%args
208  do while (associated(arg))
209  if (arg%positional) then
210  call string_copy(b, arg%short_name)
211  else
212  !Print optional arguments as [-shortname|--longname <value>].
213  call string_copy(b, "["//trim(arg%short_name))
214  if (len_trim(arg%long_name) .gt. 0) then
215  call string_copy(b, trim(b)//"|"//trim(arg%long_name))
216  endif
217  if (arg%requires_val) then
218  call string_copy(b, trim(b)//" <value>")
219  endif
220  call string_copy(b, trim(b)//"]")
221  endif
222  call string_copy(buf, trim(buf)//" "//trim(b))
223  arg => arg%head
224  enddo
225  endif
226  write(error_unit, '(a)') trim(buf)

230 !> @brief Loop through the supplied command line arguments and store
231 !! them in the parser object's arguments linked list.
232 subroutine parse_args(parser)
233 
234  type(parser_t), intent(inout) :: parser !< Parser object.
235 
236  character(len=128), dimension(:), allocatable :: args
237  type(argument_t), pointer :: arg
238  integer :: argc
239  integer :: i
240  logical :: arg_found
241  logical :: skip
242  integer :: cl_pos_arg_counter
243  integer :: cl_opt_arg_counter
244 
245  cl_pos_arg_counter = 0
246  cl_opt_arg_counter = parser%num_pos_args
247 
248  argc = command_argument_count()
249  allocate(args(argc))
250  do i = 1, argc
251  call get_command_argument(i, args(i))
252  enddo
253 
254  arg_found = .false.
255  skip = .false.
256  do i = 1, argc
257  if (skip) then
258  arg_found = .false.
259  skip = .false.
260  cycle
261  endif
262  if (trim(args(i)) .eq. "-h" .or. trim(args(i)) .eq. "--help") then
263  call usage_mesg(parser)
264  call help_mesg(parser)
265  stop
266  endif
267  arg => parser%args
268  do while (associated(arg))
269  if (.not. arg%positional) then
270  if (trim(arg%long_name) .eq. trim(args(i))) then
271  call string_copy(args(i), arg%short_name)
272  endif
273  if (trim(arg%short_name) .eq. trim(args(i))) then
274  cl_opt_arg_counter = cl_opt_arg_counter + 1
275  if (cl_opt_arg_counter .gt. parser%argc) then
276  call usage_mesg(parser)
277  call error("too many input arguments.")
278  endif
279  if (trim(arg%val) .ne. "") then
280  call error("duplicate arguments given.")
281  endif
282  if (arg%requires_val) then
283  if (i .eq. argc) then
284  call usage_mesg(parser)
285  call error("too few input arguments.")
286  endif
287  skip = .true.
288  call string_copy(arg%val, args(i+1))
289  else
290  call string_copy(arg%val, "present")
291  endif
292  arg_found = .true.
293  exit
294  endif
295  endif
296  arg => arg%head
297  enddo
298  if (arg_found) then
299  arg_found = .false.
300  cycle
301  endif
302  cl_pos_arg_counter = cl_pos_arg_counter + 1
303  if (cl_pos_arg_counter .gt. parser%num_pos_args) then
304  call error("too many positional arguments.")
305  endif
306  arg => parser%args
307  do while (associated(arg))
308  if (arg%positional .and. trim(arg%val) .eq. "") then
309  call string_copy(arg%val,args(i))
310  exit
311  endif
312  arg => arg%head
313  enddo
314  enddo
315  if (cl_pos_arg_counter .ne. parser%num_pos_args) then
316  call usage_mesg(parser)
317  call error("incorrect number of positional arguments.")
318  endif
319  deallocate(args)
320 end subroutine parse_args

323 !> @brief Given an argument's shortname, return the value that was passed on
324 !! the command line. If the argument is optional and not,
325 !! not supplied, return the string "not present". Note that this
326 !! routine must be called after parse_args.
327 subroutine get_argument(parser, short_name, val)
328 
329  type(parser_t), intent(in) :: parser
330  character(len=*), intent(in) :: short_name
331  character(len=*), intent(inout) :: val
332 
333  type(argument_t), pointer :: arg
334 
335  arg => parser%args
336  do while (associated(arg))
337  if (trim(arg%short_name) .eq. trim(short_name)) then
338  if (trim(arg%val) .eq. "") then
339  call string_copy(val, "not present")
340  else
341  call string_copy(val, arg%val)
342  endif
343  return
344  endif
345  arg => arg%head
346  enddo
347  call error("unrecognized argument "//trim(short_name)//".")
348 end subroutine get_argument

351 !> @brief Print an argument name and its description to stderr.
352 subroutine print_argument(arg)
353 
354  type(argument_t), intent(in) :: arg !< Argument.
355 
356  character(len=32) :: buf
357 
358  call string_copy(buf, arg%short_name)
359  if (len_trim(arg%long_name) .gt. 0) then
360  call string_copy(buf, trim(buf)//","//arg%long_name)
361  endif
362  if (arg%requires_val) then
363  write(error_unit, '(a32,1x,a5,7x,a)') trim(buf),"value", trim(arg%description)

370 !> @brief Throw an error if the two input arguments have the same
371 !! short and/or long names.
372 subroutine catch_duplicate_args(arg1, arg2)
373 
374  type(argument_t), intent(in) :: arg1 !< First argument.
375  type(argument_t), intent(in) :: arg2 !< Second argument.
376 
377  if (trim(arg1%short_name) .eq. trim(arg2%short_name)) then
378  call error("argument "//trim(arg1%short_name)//" already exists.")
379  endif
380  if (len_trim(arg1%long_name) .ge. 1 .and. trim(arg1%long_name) .eq. &
381  trim(arg2%long_name)) then
382  call error("argument "//trim(arg1%long_name)//" already exists.")

398 !> @brief Safely copy a string from one buffer to another.
399 subroutine string_copy(dest, source)
400 
401  character(len=*), intent(inout) :: dest !< Destination string.
402  character(len=*), intent(in) :: source !< Source string.
403
