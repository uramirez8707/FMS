76  !> Perform simple addition on arrays of 64-bit integersin different domain configurations and update the domains
77  subroutine test_halo_update_i8( domain_type )
78  character(len=*), intent(in) :: domain_type !< the domain type that will be tested
79  integer(kind=i8_kind), allocatable, dimension(:,:,:) :: xi8, x1i8, x2i8, x3i8, x4i8
80  type(domain2d) :: domain
81  integer(kind=i8_kind), allocatable :: globali8(:,:,:)
82  integer :: shift, xhalo, yhalo
83  logical :: is_symmetry
84  integer :: is, ie, js, je, isd, ied, jsd, jed
85  integer(kind=i8_kind) :: i, j, k ! kind specified because i,j,k define the x#i8 and globali8 values
86  integer :: pe, npes
87 
88  pe = mpp_pe()
89  npes = mpp_npes()
90 
91  allocate(globali8(1-whalo:nx+ehalo,1-shalo:ny+nhalo,nz) )
92  ! populate the global array
93  globali8 = 0.0
94  do k = 1,nz
95  do j = 1,ny
96  do i = 1,nx
97  globali8(i,j,k) = k + i*1e-3 + j*1e-6
98  end do
99  end do
100  end do
101 
102  if(index(domain_type, 'symmetry') == 0) then
103  is_symmetry = .false.
104  else
105  is_symmetry = .true.
106  end if
107  select case(trim(domain_type))
108  case( 'Simple', 'Simple symmetry' )
109  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
110  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
111  shalo=shalo, nhalo=nhalo, name=trim(domain_type), symmetry = is_symmetry )
112  case( 'Cyclic', 'Cyclic symmetry' )
113  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
114  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
115  shalo=shalo, nhalo=nhalo, xflags=cyclic_global_domain, yflags=cyclic_global_domain, &
116  name=trim(domain_type), symmetry = is_symmetry )
117  globali8(1-whalo:0, 1:ny,:) = globali8(nx-whalo+1:nx, 1:ny,:)
118  globali8(nx+1:nx+ehalo, 1:ny,:) = globali8(1:ehalo, 1:ny,:)
119  globali8(1-whalo:nx+ehalo, 1-shalo:0,:) = globali8(1-whalo:nx+ehalo, ny-shalo+1:ny,:)
120  globali8(1-whalo:nx+ehalo, ny+1:ny+nhalo,:) = globali8(1-whalo:nx+ehalo, 1:nhalo,:)
121  case default
122  call mpp_error( fatal, 'test_halo_update_i8: no such test: '//domain_type )
123  end select
124 
125  ! define the arrays
126  call mpp_get_compute_domain( domain, is, ie, js, je )
127  call mpp_get_data_domain( domain, isd, ied, jsd, jed )
128  allocate(xi8(isd:ied,jsd:jed,nz) )
129  allocate(x1i8(isd:ied,jsd:jed,nz) )
130  allocate(x2i8(isd:ied,jsd:jed,nz) )
131  allocate(x3i8(isd:ied,jsd:jed,nz) )
132  allocate(x4i8(isd:ied,jsd:jed,nz) )
133  xi8(:,:,:) = 0.0
134  xi8(is:ie,js:je,:) = globali8(is:ie,js:je,:)
135  x1i8 = xi8; x2i8 = xi8; x3i8 = xi8; x4i8 = xi8
136 
137  ! update the halo region
138  id = mpp_clock_id( domain_type, flags=mpp_clock_sync+mpp_clock_detailed )
139  call mpp_clock_begin(id)
140  call mpp_update_domains( xi8, domain )
141  call mpp_clock_end (id)
142  call compare_checksums( xi8, globali8(isd:ied,jsd:jed,:), domain_type )
143 
144  ! update part of the halo region
145  id = mpp_clock_id( domain_type//' partial', flags=mpp_clock_sync+mpp_clock_detailed )
146  call mpp_clock_begin(id)
147  call mpp_update_domains( x1i8, domain, nupdate+eupdate, complete=.false. )
148  call mpp_update_domains( x2i8, domain, nupdate+eupdate, complete=.false. )
149  call mpp_update_domains( x3i8, domain, nupdate+eupdate, complete=.false. )
150  call mpp_update_domains( x4i8, domain, nupdate+eupdate, complete=.true. )
151  call mpp_clock_end (id)
152  call compare_checksums( x1i8(is:ied,js:jed,:), globali8(is:ied,js:jed,:), domain_type//' partial x1i8' )
153  call compare_checksums( x2i8(is:ied,js:jed,:), globali8(is:ied,js:jed,:), domain_type//' partial x2i8' )
154  call compare_checksums( x3i8(is:ied,js:jed,:), globali8(is:ied,js:jed,:), domain_type//' partial x3i8' )
155  call compare_checksums( x4i8(is:ied,js:jed,:), globali8(is:ied,js:jed,:), domain_type//' partial x4i8' )
156 
157  deallocate(globali8, xi8, x1i8, x2i8, x3i8, x4i8)
158 
159  end subroutine test_halo_update_i8
