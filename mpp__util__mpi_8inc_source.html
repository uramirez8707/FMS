31 !> A very basic error handler
32 !! uses ABORT and FLUSH calls, may need to use cpp to rename
33 subroutine mpp_error_basic( errortype, errormsg )
34 #ifdef __INTEL_COMPILER
35  ! Intel module containing tracebackQQ
36  use ifcore
37 #endif
38  integer, intent(in) :: errortype
39  character(len=*), intent(in), optional :: errormsg
40  character(len=512) :: text
41  integer :: errunit
42 
43  if( .NOT.module_is_initialized )call abort()
44 
45  select case( errortype )
46  case(note)
47  text = 'NOTE' !just FYI
48  case(warning)
49  text = 'WARNING' !probable error
50  case(fatal)
51  text = 'FATAL' !fatal error
52  case default
53  text = 'WARNING: non-existent errortype (must be NOTE|WARNING|FATAL)'
54  end select
55 
56  if( npes.GT.1 )write( text,'(a,i6)' )trim(text)//' from PE', pe !this is the mpp part
57  if( PRESENT(errormsg) )text = trim(text)//': '//trim(errormsg)
58 !$OMP CRITICAL (MPP_ERROR_CRITICAL)
59  select case( errortype )
60  case(note)
61  if(pe==root_pe)write( out_unit,'(a)' )trim(text)
62  case default
63  errunit = stderr()
64  write( errunit, '(/a/)' )trim(text)
65  if(pe==root_pe)write( out_unit,'(/a/)' )trim(text)
66  if( errortype.EQ.fatal .OR. warnings_are_fatal )then
67  FLUSH(out_unit)
68 #ifdef __INTEL_COMPILER
69  ! Get traceback and return quietly for correct abort
70  call tracebackqq(user_exit_code=-1)
71 #elif __GFORTRAN__
72  call backtrace
73 #endif
74  call mpi_abort( mpi_comm_world, 1, error )
75  end if
76  end select
77 
78  error_state = errortype
79 !$OMP END CRITICAL (MPP_ERROR_CRITICAL)
80 
81 
82 end subroutine mpp_error_basic

84 !#####################################################################
85 !> Makes a PE set out of a PE list. A PE list is an ordered list of PEs
86 !! a PE set is a triad (start,log2stride,size) for SHMEM, an a communicator for MPI
87 !! if stride is non-uniform or not a power of 2,
88 !! will return error (not required for MPI but enforced for uniformity)
89 function get_peset(pelist)
90  integer :: get_peset
91  integer, intent(in), optional :: pelist(:)
92  integer :: errunit
93  integer :: i, n
94  integer, allocatable :: sorted(:)
95 
96  if( .NOT.PRESENT(pelist) )then !set it to current_peset_num
97  get_peset = current_peset_num; return
98  end if
99 
100  !--- first make sure pelist is monotonically increasing.
101  if (size(pelist(:)) .GT. 1) then
102  do n = 2, size(pelist(:))
103  if(pelist(n) <= pelist(n-1)) call mpp_error(fatal, "GET_PESET: pelist is not monotonically increasing")
104  enddo
105  endif
106 
107  allocate( sorted(size(pelist(:))) )
108  sorted = pelist
109  errunit = stderr()
110  if( debug )write( errunit,* )'pelist=', pelist
111 
112  !find if this array matches any existing peset
113  do i = 1,peset_num
114  if( debug )write( errunit,'(a,3i6)' )'pe, i, peset_num=', pe, i, peset_num
115  if( size(sorted(:)).EQ.size(peset(i)%list(:)) )then
116  if( all(sorted.EQ.peset(i)%list) )then
117  deallocate(sorted)
118  get_peset = i; return
119  end if
120  end if
121  end do
122  !not found, so create new peset
123  peset_num = peset_num + 1
124  if( peset_num > current_peset_max ) call expand_peset()
125  i = peset_num !shorthand
126  !create list
127  allocate( peset(i)%list(size(sorted(:))) )
128  peset(i)%list(:) = sorted(:)
129  peset(i)%count = size(sorted(:))
130 
131  call mpi_group_incl( peset(current_peset_num)%group, size(sorted(:)), sorted-mpp_root_pe(), peset(i)%group, error )
132  call mpi_comm_create_group(peset(current_peset_num)%id, peset(i)%group, &
133  default_tag, peset(i)%id, error )
134  deallocate(sorted)
135  get_peset = i
136 
137  return
138 
139 end function get_peset

162 !#######################################################################
163 !> This is to check if current PE's outstanding puts are complete
164 !! but we can't use shmem_fence because we are actually waiting for
165 !! a remote PE to complete its get
166 subroutine mpp_sync_self( pelist, check, request, msg_size, msg_type)
167  integer, intent(in), optional :: pelist(:)
168  integer, intent(in), optional :: check
169  integer, intent(inout), optional :: request(:)
170  integer, intent(in ), optional :: msg_size(:)
171  integer, intent(in ), optional :: msg_type(:)
172 
173  integer :: m, my_check, rsize
174 
175  if( debug .and. (current_clock.NE.0) )call system_clock(start_tick)
176  my_check = event_send
177  if(present(check)) my_check = check
178  if( my_check .NE. event_send .AND. my_check .NE. event_recv ) then
179  call mpp_error( fatal, 'mpp_sync_self: The value of optional argument check should be EVENT_SEND or EVENT_RECV')
180  endif
181 
182  if(PRESENT(request)) then
183  if( .not. present(check) ) then
184  call mpp_error(fatal, 'mpp_sync_self: check is not present when request is present')
185  endif
186  if( my_check == event_recv ) then
187  if( .not. present(msg_size) ) then
188  call mpp_error(fatal, 'mpp_sync_self: msg_size is not present when request is present and it is EVENT_RECV')
189  endif
190  if( .not. present(msg_type) ) then
191  call mpp_error(fatal, 'mpp_sync_self: msg_type is not present when request is present and it is EVENT_RECV')
192  endif
193  if(size(msg_size) .NE. size(request)) then
194  call mpp_error(fatal, 'mpp_sync_self: dimension mismatch between msg_size and request')
195  endif
196  if(size(msg_type) .NE. size(request)) then
197  call mpp_error(fatal, 'mpp_sync_self: dimension mismatch between msg_type and request')
198  endif
199 
200  do m = 1, size(request(:))
201  if( request(m) == mpi_request_null ) cycle
202  call mpi_wait(request(m), stat, error )
203  call mpi_get_count(stat, msg_type(m), rsize, error)
204  if(msg_size(m) .NE. rsize) then
205  call mpp_error(fatal, "mpp_sync_self: msg_size does not match size of data received")
206  endif
207  enddo
208  else
209  do m = 1, size(request(:))
210  if(request(m) .NE.mpi_request_null )call mpi_wait(request(m), stat, error )
211  enddo
212  endif
213  else
214  select case(my_check)
215  case(event_send)
216  do m = 1,cur_send_request
217  if( request_send(m).NE.mpi_request_null )call mpi_wait( request_send(m), stat, error )
218  end do
219  cur_send_request = 0
220  case(event_recv)
221  do m = 1,cur_recv_request
222  call mpi_wait( request_recv(m), stat, error )
223  call mpi_get_count(stat, type_recv(m), rsize, error)
224  if(size_recv(m) .NE. rsize) then
225  call mpp_error(fatal, "mpp_sync_self: size_recv does not match of data received")
226  endif
227  size_recv(m) = 0
228  end do
229  cur_recv_request = 0
230  end select
231  endif
232  if( debug .and. (current_clock.NE.0) )call increment_current_clock(event_wait)
233  return
234 end subroutine mpp_sync_self
