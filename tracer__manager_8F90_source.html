19 !> @defgroup tracer_manager_mod tracer_manager_mod
20 !> @ingroup tracer_manager
21 !> @brief Code to manage the simple addition of tracers to the FMS code.
22 !! This code keeps track of the numbers and names of tracers included
23 !! in a tracer table.
24 !!
25 !> @author William Cooke
26 !!
27 !> This code is a grouping of calls which will allow the simple
28 !! introduction of tracers into the FMS framework. It is designed to
29 !! allow users of a variety of component models interact easily with
30 !! the dynamical core of the model.
31 !!
32 !! In calling the tracer manager routines the user must provide a
33 !! parameter identifying the model that the user is working with. This
34 !! parameter is defined within field_manager as MODEL_X
35 !! where X is one of [ATMOS, OCEAN, LAND, ICE].
36 !!
37 !! In many of these calls the argument list includes model and tracer_index. These
38 !! are the parameter corresponding to the component model and the tracer_index N is
39 !! the Nth tracer within the component model. Therefore a call with MODEL_ATMOS and 5
40 !! is different from a call with MODEL_OCEAN and 5.

101 !> @brief Function which returns the number assigned to the tracer name.

103 !> This is a function which returns the index, as implied within the component model.

106 !! @param model A integer parameter to identify which model is being used

117 !!
118 !! <br>Example usage:
119 !! @code{.F90}
120 !! integer:
121 !! index = get_tracer_index(model, name, indices, verbose)

203 !> @brief Read in tracer table and store tracer information associated with "model" in "tracers"
204 !! array.
205 subroutine get_tracer_meta_data(model, num_tracers,num_prog,num_diag)
206 
207 integer, intent(in) :: model !< model being used
208 integer, intent(out) :: num_tracers, num_prog, num_diag
209 character(len=256) :: warnmesg
210 
211 character(len=32) :: name_type, type, name
212 integer :: n, m, mod, num_tracer_methods, nfields, swop
213 integer :: j, log_unit, num_methods
214 logical :: flag_type
215 type(method_type), dimension(MAX_TRACER_METHOD) :: methods
216 integer :: instances, siz_inst,i
217 character(len = 32) :: digit,suffnam
218 
219 character(len=128) :: list_name , control
220 integer :: index_list_name
221 logical :: fm_success
222 
223 ! <ERROR MSG="invalid model type" STATUS="FATAL">
224 ! The index for the model type is invalid.
225 ! </ERROR>
226 if (model .ne. model_atmos .and. model .ne. model_land .and. &
227  model .ne. model_ocean .and. model .ne. model_ice .and. &
228  model .ne. model_coupler) call mpp_error(fatal,'tracer_manager_init : invalid model type')
229 
230 ! One should only call get_tracer_meta_data once for each model type
231 ! Therefore need to set up an array to stop the subroutine being
232 ! unnecssarily called multiple times.
233 
234 if ( model_registered(model) ) then
235 ! This routine has already been called for the component model.
236 ! Fill in the values from the previous registration and return.
237  num_tracers = total_tracers(model)
238  num_prog = prog_tracers(model)
239  num_diag = diag_tracers(model)
240  return
241 endif
242 
243 ! Initialize the number of tracers to zero.
244 num_tracers = 0; num_prog = 0; num_diag = 0
245 
246 call field_manager_init(nfields=nfields)
247 
248 ! <ERROR MSG="No tracers are available to be registered." STATUS="NOTE">
249 ! No tracers are available to be registered. This means that the field
250 ! table does not exist or is empty.
251 ! </ERROR>
252 if (nfields == 0 ) then
253 if (mpp_pe() == mpp_root_pe()) &
254  call mpp_error(note,'tracer_manager_init : No tracers are available to be registered.')
255  return
256 endif
257 
258 ! search through field entries for model tracers
259 total_tracers(model) = 0
260 
261 do n=1,nfields
262  call get_field_info(n,type,name,mod,num_methods)
263 
264  if (mod == model .and. type == 'tracer') then
265  num_tracer_fields = num_tracer_fields + 1
266  total_tracers(model) = total_tracers(model) + 1
267 ! <ERROR MSG="MAX_TRACER_FIELDS exceeded" STATUS="FATAL">
268 ! The maximum number of tracer fields has been exceeded.
269 ! </ERROR>
270  if(num_tracer_fields > max_tracer_fields) call mpp_error(fatal,'tracer_manager_init: MAX_TRACER_FIELDS exceeded')
271  tracer_array(model,total_tracers(model)) = num_tracer_fields
272  tracers(num_tracer_fields)%model = model
273  tracers(num_tracer_fields)%tracer_name = name
274  tracers(num_tracer_fields)%tracer_units = 'none'
275  tracers(num_tracer_fields)%tracer_longname = tracers(num_tracer_fields)%tracer_name
276  tracers(num_tracer_fields)%instances_set = .false.
277 ! By default, tracers need mass and positive definite adjustments.
278 ! We hardwire exceptions for compatibility with existing field_tables
279 ! This should ideally be cleaned up.
280  tracers(num_tracer_fields)%needs_mass_adjust = .true.
281  tracers(num_tracer_fields)%needs_positive_adjust = .true.
282  if (name == 'cld_amt') then
283  tracers(num_tracer_fields)%needs_mass_adjust = .false.
284  endif
285  if (name == 'cld_amt' .or. name == 'liq_wat' .or. name == 'ice_wat') then
286  tracers(num_tracer_fields)%needs_positive_adjust = .false.
287  endif
288 
289  num_tracer_methods = 0
290  methods = default_method ! initialize methods array
291  call get_field_methods(n,methods)
292  do j=1,num_methods
293  select case (methods(j)%method_type)
294  case ('units')
295  tracers(num_tracer_fields)%tracer_units = methods(j)%method_name
296  case ('longname')
297  tracers(num_tracer_fields)%tracer_longname = methods(j)%method_name
298  case ('instances')
299 ! tracers(num_tracer_fields)%instances = methods(j)%method_name
300  siz_inst = parse(methods(j)%method_name,"",instances)
301  tracers(num_tracer_fields)%instances = instances
302  tracers(num_tracer_fields)%instances_set = .true.
303  case ('adjust_mass')
304  if (methods(j)%method_name == "false") then
305  tracers(num_tracer_fields)%needs_mass_adjust = .false.
306  endif
307  case ('adjust_positive_def')
308  if (methods(j)%method_name == "false") then
309  tracers(num_tracer_fields)%needs_positive_adjust = .false.
310  endif
311  case default
312  num_tracer_methods = num_tracer_methods+1
313 ! tracers(num_tracer_fields)%methods(num_tracer_methods) = methods(j)
314  end select
315  enddo
316  tracers(num_tracer_fields)%num_methods = num_tracer_methods
317  tracers(num_tracer_fields)%needs_init = .false.
318  flag_type = query_method('tracer_type',model,total_tracers(model),name_type)
319  if (flag_type .and. name_type == 'diagnostic') then
320  tracers(num_tracer_fields)%is_prognostic = .false.
321  else
322  tracers(num_tracer_fields)%is_prognostic = .true.
323  endif
324  if (tracers(num_tracer_fields)%is_prognostic) then
325  num_prog = num_prog+1
326  else
327  num_diag = num_diag+1
328  endif
329  endif
330 enddo
331 
332 ! Now cycle through the tracers and add additional instances of the tracers.
333 
334 do n = 1, num_tracer_fields !{
335 ! call get_field_info(n,type,name,mod,num_methods)
336 
337  if ( model == tracers(n)%model .and. tracers(n)%instances_set ) then !{ We have multiple instances of this tracer
338 
339  if ( num_tracer_fields + tracers(n)%instances > max_tracer_fields ) then
340  write(warnmesg, '("tracer_manager_init: Number of tracers will exceed MAX_TRACER_FIELDS with &
341  &multiple (",I3," instances) setup of tracer ",A)') tracers(n)%instances,tracers(n)%tracer_name
342  call mpp_error(fatal, warnmesg)
343  endif
344 
345  do i = 2, tracers(n)%instances !{
346  num_tracer_fields = num_tracer_fields + 1
347  total_tracers(model) = total_tracers(model) + 1
348  tracer_array(model,total_tracers(model)) = num_tracer_fields
349  ! Copy the original tracer type to the multiple instances.
350  tracers(num_tracer_fields) = tracers(n)
351  if ( query_method('instances', model,model_tracer_number(model,n),name, control)) then !{
352 
353  if (i .lt. 10) then !{
354  write (suffnam,'(''suffix'',i1)') i
355  siz_inst = parse(control, suffnam,digit)
356  if (siz_inst == 0 ) then
357  write (digit,'(''_'',i1)') i
358  else
359  digit = "_"//trim(digit)
360  endif
361  elseif (i .lt. 100) then !}{
362  write (suffnam,'(''suffix'',i2)') i
363  siz_inst = parse(control, suffnam,digit)
364  if (siz_inst == 0 ) then
365  write (digit,'(''_'',i2)') i
366  else
367  digit = "_"//trim(digit)
368  endif
369  else !}{
370  call mpp_error(fatal, 'tracer_manager_init: MULTIPLE_TRACER_SET_UP exceeds 100 for '//tracers(n)%tracer_name )
371  endif !}
372 
373  select case(model)
374  case (model_coupler)
375  list_name = "/coupler_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
376  case (model_atmos)
377  list_name = "/atmos_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
378  case (model_ocean)
379  list_name = "/ocean_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
380  case (model_ice )
381  list_name = "/ice_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
382  case (model_land )
383  list_name = "/land_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
384  case default
385  list_name = "/default/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
386  end select
387 
388  if (mpp_pe() == mpp_root_pe() ) write (*,*) "Creating list name = ",trim(list_name)//trim(digit)
389 
390  index_list_name = fm_copy_list(trim(list_name),digit, create = .true.)
391  tracers(num_tracer_fields)%tracer_name = trim(tracers(num_tracer_fields)%tracer_name)//trim(digit)
392  endif !}
393 
394  if (tracers(num_tracer_fields)%is_prognostic) then !{
395  num_prog = num_prog+1
396  else !}{
397  num_diag = num_diag+1
398  endif !}
399  enddo !}
400  ! Multiple instances of tracers were found so need to rename the original tracer.
401  digit = "_1"
402  siz_inst = parse(control, "suffix1",digit)
403  if (siz_inst > 0 ) then !{
404  digit = "_"//trim(digit)
405  endif !}
406  fm_success = fm_modify_name(trim(list_name), trim(tracers(n)%tracer_name)//trim(digit))
407  tracers(n)%tracer_name = trim(tracers(n)%tracer_name)//trim(digit)
408  endif !}
409 enddo !}
410 
411 ! Find any field entries with the instances keyword.
412 do n=1,nfields
413  call get_field_info(n,type,name,mod,num_methods)
414 
415  if ( mod == model .and. type == 'instances' ) then
416  call get_field_methods(n,methods)
417  do j=1,num_methods
418 
419  if (.not.get_tracer_index(mod,methods(j)%method_type,m)) then
420  call mpp_error(fatal,'tracer_manager_init: The instances keyword was found for undefined tracer '&
421  //trim(methods(j)%method_type))
422  else
423  if ( tracers(m)%instances_set ) &
424  call mpp_error(fatal,'tracer_manager_init: The instances keyword was found for '&
425  //trim(methods(j)%method_type)//' but has previously been defined in the tracer entry')
426  siz_inst = parse(methods(j)%method_name,"",instances)
427  tracers(m)%instances = instances
428  call mpp_error(note,'tracer_manager_init: '//trim(instantiations(j)%name)// &
429  ' will have '//trim(methods(j)%method_name)//' instances')
430  endif
431  if ( num_tracer_fields + instances > max_tracer_fields ) then
432  write(warnmesg, '("tracer_manager_init: Number of tracers will exceed MAX_TRACER_FIELDS with &
433  &multiple (",I3," instances) setup of tracer ",A)') tracers(m)%instances,tracers(m)%tracer_name
434  call mpp_error(fatal, warnmesg)
435  endif
436 ! We have found a valid tracer that has more than one instantiation.
437 ! We need to modify that tracer name to tracer_1 and add extra tracers for the extra instantiations.
438  if (instances .eq. 1) then
439  siz_inst = parse(methods(j)%method_control, 'suffix1',digit)
440  if (siz_inst == 0 ) then
441  digit = '_1'
442  else
443  digit = "_"//trim(digit)
444  endif
445  endif
446  do i = 2, instances
447  num_tracer_fields = num_tracer_fields + 1
448  total_tracers(model) = total_tracers(model) + 1
449  tracer_array(model,total_tracers(model)) = num_tracer_fields
450  tracers(num_tracer_fields) = tracers(m)
451 
452  if (i .lt. 10) then !{
453  write (suffnam,'(''suffix'',i1)') i
454  siz_inst = parse(methods(j)%method_control, suffnam,digit)
455  if (siz_inst == 0 ) then
456  write (digit,'(''_'',i1)') i
457  else
458  digit = "_"//trim(digit)
459  endif
460  elseif (i .lt. 100) then !}{
461  write (suffnam,'(''suffix'',i2)') i
462  siz_inst = parse(methods(j)%method_control, suffnam,digit)
463  if (siz_inst == 0 ) then
464  write (digit,'(''_'',i2)') i
465  else
466  digit = "_"//trim(digit)
467  endif
468  else !}{
469  call mpp_error(fatal, 'tracer_manager_init: MULTIPLE_TRACER_SET_UP exceeds 100 for '&
470  //tracers(num_tracer_fields)%tracer_name )
471  endif !}
472 
473  select case(model)
474  case (model_coupler)
475  list_name = "/coupler_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
476  case (model_atmos)
477  list_name = "/atmos_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
478  case (model_ocean)
479  list_name = "/ocean_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
480  case (model_ice )
481  list_name = "/ice_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
482  case (model_land )
483  list_name = "/land_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
484  case default
485  list_name = "/default/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
486  end select
487 
488  if (mpp_pe() == mpp_root_pe() ) write (*,*) "Creating list name = ",trim(list_name)
489 
490  index_list_name = fm_copy_list(trim(list_name),digit, create = .true.)
491 
492  tracers(num_tracer_fields)%tracer_name = trim(tracers(num_tracer_fields)%tracer_name)//digit
493  if (tracers(num_tracer_fields)%is_prognostic) then
494  num_prog = num_prog+1
495  else
496  num_diag = num_diag+1
497  endif
498  enddo
499 !Now rename the original tracer to tracer_1 (or if suffix1 present to tracer_'value_of_suffix1')
500  siz_inst = parse(methods(j)%method_control, 'suffix1',digit)
501  if (siz_inst == 0 ) then
502  digit = '_1'
503  else
504  digit = "_"//trim(digit)
505  endif
506  fm_success = fm_modify_name(trim(list_name), trim(tracers(m)%tracer_name)//trim(digit))
507  tracers(m)%tracer_name = trim(tracers(m)%tracer_name)//trim(digit)
508  enddo
509  endif
510 enddo
511 
512 num_tracers = num_prog + num_diag
513 ! Make the number of tracers available publicly.
514 total_tracers(model) = num_tracers
515 prog_tracers(model) = num_prog
516 diag_tracers(model) = num_diag
517 model_registered(model) = .true.
518 
519 ! Now sort through the tracer fields and sort them so that the
520 ! prognostic tracers are first.
521 
522 do n=1, num_tracers
523  if (.not.check_if_prognostic(model,n) .and. n.le.num_prog) then
524  ! This is a diagnostic tracer so find a prognostic tracer to swop with
525  do m = n, num_tracers
526  if (check_if_prognostic(model,m) .and. .not.check_if_prognostic(model,n)) then
527  swop = tracer_array(model,n)
528  tracer_array(model,n) = tracer_array(model,m)
529  tracer_array(model,m) = swop
530  cycle
531  endif
532  enddo
533  endif
534 enddo
535 
536 do n=1, num_tracer_fields
537  call print_tracer_info(model,n)
538 enddo
539 
540 log_unit = stdlog()
541 if ( mpp_pe() == mpp_root_pe() ) then
542  write (log_unit,15) trim(model_names(model)),total_tracers(model)
543 endif
544 
545 15 format ('Number of tracers in field table for ',a,' model = ',i4)
546 
547 end subroutine get_tracer_meta_data

551 integer model_tracer_number
552 
553 integer :: i
554

1012 !> @brief Subroutine to set the tracer field to the wanted profile.
1013 !!
1014 !> If the profile type is 'fixed' then the tracer field values are set
1015 !! equal to the surface value.
1016 !! If the profile type is 'profile' then the top/bottom of model and
1017 !! surface values are read and an exponential profile is calculated,
1018 !! with the profile being dependent on the number of levels in the
1019 !! component model. This should be called from the part of the dynamical
1020 !! core where tracer restarts are called in the event that a tracer
1021 !! restart file does not exist.
1022 !!
1023 !! This can be activated by adding a method to the field_table
1024 !! e.g.
1025 !! @verbose "profile_type","fixed","surface_value = 1e-12" @endverbose
1026 !! would return values of surf_value = 1e-12 and a multiplier of 1.0
1027 !! One can use these to initialize the entire field with a value of 1e-12.
1028 !!
1029 !! "profile_type","profile","surface_value = 1e-12, top_value = 1e-15"
1030 !! In a 15 layer model this would return values of surf_value = 1e-12 and
1031 !! multiplier = 0.6309573 i.e 1e-15 = 1e-12*(0.6309573^15)
1032 !! In this case the model should be MODEL_ATMOS as you have a "top" value.
1033 !!
1034 !! If you wish to initialize the ocean model, one can use bottom_value instead
1035 !! of top_value.
1036 subroutine set_tracer_profile(model, n, tracer, err_msg)
1037 
1038 integer, intent(in) :: model !< Parameter representing component model in use
1039 integer, intent(in) :: n !< Tracer number
1040 real, intent(inout), dimension(:,:,:) :: tracer !< Initialized tracer array
1041 character(len=*), intent(out), optional :: err_msg
1042 
1043 real :: surf_value, multiplier
1044 integer :: numlevels, k, n1, flag
1045 real :: top_value, bottom_value
1046 character(len=80) :: scheme, control,profile_type
1047 character(len=128) :: err_msg_local
1048 character(len=11) :: chn
1049 
1050 if(.not.module_is_initialized) call tracer_manager_init
1051 
1052 if (n < 1 .or. n > total_tracers(model)) then
1053  write(chn, '(i11)') n
1054  err_msg_local = ' Invalid tracer index. Model name = '//trim(model_names(model))//', Index='//trim(chn)
1055  if(error_handler('set_tracer_profile', err_msg_local, err_msg)) return
1056 endif
1057 n1 = tracer_array(model,n)
1058 
1059 !default values
1060 profile_type = 'Fixed'
1061 surf_value = 0.0e+00
1062 top_value = surf_value
1063 bottom_value = surf_value
1064 multiplier = 1.0
1065 
1066 tracer = surf_value
1067 
1068 if ( query_method( 'profile_type',model,n,scheme,control)) then
1069 !Change the tracer_number to the tracer_manager version
1070 
1071  if(lowercase(trim(scheme(1:5))).eq.'fixed') then
1072  profile_type = 'Fixed'
1073  flag =parse(control,'surface_value',surf_value)
1074  multiplier = 1.0
1075  tracer = surf_value
1076  endif
1077 
1078  if(lowercase(trim(scheme(1:7))).eq.'profile') then
1079  profile_type = 'Profile'
1080  flag=parse(control,'surface_value',surf_value)
1081  if (surf_value .eq. 0.0) &
1082  call mpp_error(fatal,'set_tracer_profile : Cannot have a zero surface value for an exponential profile. Tracer '&
1083  //tracers(n1)%tracer_name//" "//control//" "//scheme)
1084  select case (tracers(n1)%model)
1085  case (model_atmos)
1086  flag=parse(control,'top_value',top_value)
1087  if(mpp_pe()==mpp_root_pe() .and. flag == 0) &
1088  call mpp_error(note,'set_tracer_profile : Parameter top_value needs to be defined for the tracer profile.')
1089  case (model_ocean)
1090  flag =parse(control,'bottom_value',bottom_value)
1091  if(mpp_pe() == mpp_root_pe() .and. flag == 0) &
1092  call mpp_error(note,'set_tracer_profile : Parameter bottom_value needs to be defined for the tracer profile.')
1093  case default
1094 ! Should there be a NOTE or WARNING message here?
1095  end select
1096 
1097 ! If profile type is profile then set the surface value to the input
1098 ! value and calculate the vertical multiplier.
1099 !
1100 ! Assume an exponential decay/increase from the surface to the top level
1101 ! C = C0 exp ( -multiplier* level_number)
