20 program horiz_interp_test
21 
22 use mpp_mod, only : mpp_init, mpp_exit, mpp_error, fatal, stdout, mpp_npes
23 use mpp_mod, only : mpp_clock_id, mpp_clock_begin, mpp_clock_end
24 use mpp_mod, only : mpp_pe, mpp_root_pe, note, mpp_clock_sync, mpp_clock_detailed
25 use mpp_mod, only : input_nml_file
26 use mpp_domains_mod, only : mpp_define_layout, mpp_define_domains, mpp_get_compute_domain
27 use mpp_domains_mod, only : mpp_domains_init, domain2d
28 use fms_mod, only : check_nml_error, fms_init
29 use horiz_interp_mod, only : horiz_interp_init, horiz_interp_new, horiz_interp_del
30 use horiz_interp_mod, only : horiz_interp, horiz_interp_type
31 use constants_mod, only : constants_init, pi
32 
33 implicit none
34 
35  integer :: ni_src = 360, nj_src = 180
36  integer :: ni_dst = 144, nj_dst = 72
37 
38  namelist /test_horiz_interp_nml/ ni_src, nj_src, ni_dst, nj_dst
39 
40  real :: lon_src_beg = 0, lon_src_end = 360
41  real :: lat_src_beg = -90, lat_src_end = 90
42  real :: lon_dst_beg = -280, lon_dst_end = 80
43  real :: lat_dst_beg = -90, lat_dst_end = 90
44  real :: D2R = pi/180.
45  real, parameter :: SMALL = 1.0e-10
46 
47  type(domain2d) :: domain
48  type(horiz_interp_type) :: Interp
49  integer :: id1, id2, id3, id4
50  integer :: isc, iec, jsc, jec, i, j
51  integer :: io, ierr, layout(2)
52  real :: dlon_src, dlat_src, dlon_dst, dlat_dst
53  real, allocatable, dimension(:) :: lon1D_src, lat1D_src, lon1D_dst, lat1D_dst
54  real, allocatable, dimension(:,:) :: lon2D_src, lat2D_src, lon2D_dst, lat2D_dst
55  real, allocatable, dimension(:,:) :: data_src, data1_dst, data2_dst, data3_dst, data4_dst
56 
57  call fms_init
58  call constants_init
59  call horiz_interp_init
60 
61  !--- read namelist
62  read (input_nml_file, test_horiz_interp_nml, iostat=io)
63  ierr = check_nml_error(io, 'test_horiz_interp_nml')
64 
65  !--- define domains
66  call mpp_define_layout( (/1, ni_dst, 1, nj_dst/), mpp_npes(), layout)
67  call mpp_define_domains((/1, ni_dst, 1, nj_dst/), layout, domain)
68  call mpp_get_compute_domain(domain,isc,iec,jsc,jec)
69 
70  !--- test conservative horiz_interp with a simple test. the source grid is the region
71  ! (0:360,-90:90) with grid size ni_src, nj_src ( default 360X180). and the destination
72  ! is the region (-280:80, -90:90) with grid size ni_dstXnj_dst( default 144X72).
73  ! integer checksum and global sum will be printed out for both the 1D and 2D version.
74 
75  allocate(lon2d_src(ni_src+1, nj_src+1), lat2d_src(ni_src+1, nj_src+1) )
76  allocate(lon1d_src(ni_src+1), lat1d_src(nj_src+1), data_src(ni_src, nj_src) )
77 
78  allocate(lon2d_dst(isc:iec+1, jsc:jec+1), lat2d_dst(isc:iec+1, jsc:jec+1) )
79  allocate(lon1d_dst(isc:iec+1), lat1d_dst(jsc:jec+1) )
80  allocate(data1_dst(isc:iec, jsc:jec), data2_dst(isc:iec, jsc:jec) )
81  allocate(data3_dst(isc:iec, jsc:jec), data4_dst(isc:iec, jsc:jec) )
82 
83  ! set up longitude and latitude of source/destination grid.
84  dlon_src = (lon_src_end-lon_src_beg)/ni_src
85  dlat_src = (lat_src_end-lat_src_beg)/nj_src
86  dlon_dst = (lon_dst_end-lon_dst_beg)/ni_dst
87  dlat_dst = (lat_dst_end-lat_dst_beg)/nj_dst
88 
89  do i = 1, ni_src+1
90  lon1d_src(i) = lon_src_beg + (i-1)*dlon_src
91  end do
92 
93  do j = 1, nj_src+1
94  lat1d_src(j) = lat_src_beg + (j-1)*dlat_src
95  end do
96 
97  do i = isc, iec+1
98  lon1d_dst(i) = lon_dst_beg + (i-1)*dlon_dst
99  end do
100 
101  do j = jsc, jec+1
102  lat1d_dst(j) = lat_dst_beg + (j-1)*dlat_dst
103  end do
104 
105  ! scale grid to radians.
106  lon1d_src = lon1d_src * d2r
107  lat1d_src = lat1d_src * d2r
108  lon1d_dst = lon1d_dst * d2r
109  lat1d_dst = lat1d_dst * d2r
110 
111  do i = 1, ni_src+1
112  lon2d_src(i,:) = lon1d_src(i)
113  end do
114 
115  do j = 1, nj_src+1
116  lat2d_src(:,j) = lat1d_src(j)
117  end do
118 
119  do i = isc, iec+1
120  lon2d_dst(i,:) = lon1d_dst(i)
121  end do
122 
123  do j = jsc, jec+1
124  lat2d_dst(:,j) = lat1d_dst(j)
125  end do
126 
127  !--- set up the source data
128  do j = 1, nj_src
129  do i = 1, ni_src
130  data_src(i,j) = i + j*0.001
131  end do
132  end do
133 
134  id1 = mpp_clock_id( 'horiz_interp_1dx1d', flags=mpp_clock_sync+mpp_clock_detailed )
135  id2 = mpp_clock_id( 'horiz_interp_1dx2d', flags=mpp_clock_sync+mpp_clock_detailed )
136  id3 = mpp_clock_id( 'horiz_interp_2dx1d', flags=mpp_clock_sync+mpp_clock_detailed )
137  id4 = mpp_clock_id( 'horiz_interp_2dx2d', flags=mpp_clock_sync+mpp_clock_detailed )
138 
139  ! --- 1dx1d version conservative interpolation
140  call mpp_clock_begin(id1)
141  call horiz_interp_new(interp, lon1d_src, lat1d_src, lon1d_dst, lat1d_dst, interp_method = "conservative")
142  call horiz_interp(interp, data_src, data1_dst)
143  call horiz_interp_del(interp)
144  call mpp_clock_end(id1)
145 
146  ! --- 1dx2d version conservative interpolation
147  call mpp_clock_begin(id2)
148  call horiz_interp_new(interp, lon1d_src, lat1d_src, lon2d_dst, lat2d_dst, interp_method = "conservative")
149  call horiz_interp(interp, data_src, data2_dst)
150  call horiz_interp_del(interp)
151  call mpp_clock_end(id2)
152 
153  ! --- 2dx1d version conservative interpolation
154  call mpp_clock_begin(id3)
155  call horiz_interp_new(interp, lon2d_src, lat2d_src, lon1d_dst, lat1d_dst, interp_method = "conservative")
156  call horiz_interp(interp, data_src, data3_dst)
157  call horiz_interp_del(interp)
158  call mpp_clock_end(id3)
159 
160  ! --- 2dx2d version conservative interpolation
161  call mpp_clock_begin(id4)
162  call horiz_interp_new(interp, lon2d_src, lat2d_src, lon2d_dst, lat2d_dst, interp_method = "conservative")
163  call horiz_interp(interp, data_src, data4_dst)
164  call horiz_interp_del(interp)
165  call mpp_clock_end(id4)
166 
167  !--- compare the data after interpolation between 1-D and 2-D version interpolation
168  do j = jsc, jsc
169  do i = isc, iec
170 
171  if( abs(data1_dst(i,j)-data2_dst(i,j)) > small ) then
172  print*, "After interpolation At point (i,j) = (", i, ",", j, "), data1 = ", data1_dst(i,j), &
173  ", data2 = ", data2_dst(i,j), ", data1-data2 = ", data1_dst(i,j) - data2_dst(i,j)
174  call mpp_error(fatal,"horiz_interp_test: data1_dst does not approxiamate data2_dst")
175  end if
176  end do
177  end do
178 
179  if(mpp_pe() == mpp_root_pe()) call mpp_error(note, &
180  "The test that verify 1dx2d version horiz_interp can reproduce 1dx1d version of horiz_interp is succesful")
181 
182  do j = jsc, jsc
183  do i = isc, iec
184 
185  if( abs(data1_dst(i,j)-data3_dst(i,j)) > small ) then
186  print*, "After interpolation At point (i,j) = (", i, ",", j, "), data1 = ", data1_dst(i,j), &
187  ", data2 = ", data3_dst(i,j), ", data1-data2 = ", data1_dst(i,j) - data3_dst(i,j)
188  call mpp_error(fatal,"horiz_interp_test: data1_dst does not approxiamate data3_dst")
189  end if
190  end do
191  end do
192 
193  if(mpp_pe() == mpp_root_pe()) call mpp_error(note, &
194  "The test that verify 2dx1d version horiz_interp can reproduce 1dx1d version of horiz_interp is succesful")
195 
196  do j = jsc, jsc
197  do i = isc, iec
198 
199  if( abs(data1_dst(i,j)-data4_dst(i,j)) > small ) then
200  print*, "After interpolation At point (i,j) = (", i, ",", j, "), data1 = ", data1_dst(i,j), &
201  ", data2 = ", data4_dst(i,j), ", data1-data2 = ", data1_dst(i,j) - data4_dst(i,j)
202  call mpp_error(fatal,"horiz_interp_test: data1_dst does not approxiamate data4_dst")
203  end if
204  end do
205  end do
206 
207  if(mpp_pe() == mpp_root_pe()) call mpp_error(note, &
208  "The test that verify 2dx2d version horiz_interp can reproduce 1dx1d version of horiz_interp is succesful")
209 
210  call mpp_exit
211 
212 end program horiz_interp_test
