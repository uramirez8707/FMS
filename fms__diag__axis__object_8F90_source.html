191  !> @brief Initialize the axis
192  subroutine register_diag_axis_obj(this, axis_name, axis_data, units, cart_name, long_name, direction,&
193  & set_name, Domain, Domain2, DomainU, aux, req, tile_count, domain_position, axis_length )
194  class(fmsdiagfullaxis_type),INTENT(inout):: this !< Diag_axis obj
195  CHARACTER(len=*), INTENT(in) :: axis_name !< Name of the axis
196  class(*), INTENT(in) :: axis_data(:) !< Array of coordinate values
197  CHARACTER(len=*), INTENT(in) :: units !< Units for the axis
198  CHARACTER(len=1), INTENT(in) :: cart_name !< Cartesian axis ("X", "Y", "Z", "T", "U", "N")
199  CHARACTER(len=*), INTENT(in), OPTIONAL :: long_name !< Long name for the axis.
200  CHARACTER(len=*), INTENT(in), OPTIONAL :: set_name !< Name of the parent axis, if it is a subaxis
201  INTEGER, INTENT(in), OPTIONAL :: direction !< Indicates the direction of the axis
202  TYPE(domain1d), INTENT(in), OPTIONAL :: Domain !< 1D domain
203  TYPE(domain2d), INTENT(in), OPTIONAL :: Domain2 !< 2D domain
204  TYPE(domainug), INTENT(in), OPTIONAL :: DomainU !< Unstructured domain
205  CHARACTER(len=*), INTENT(in), OPTIONAL :: aux !< Auxiliary name, can only be <TT>geolon_t</TT>
206  !! or <TT>geolat_t</TT>
207  CHARACTER(len=*), INTENT(in), OPTIONAL :: req !< Required field names.
208  INTEGER, INTENT(in), OPTIONAL :: tile_count !< Number of tiles
209  INTEGER, INTENT(in), OPTIONAL :: domain_position !< Domain position, "NORTH" or "EAST"
210  integer, intent(in), optional :: axis_length !< The length of the axis size(axis_data(:))
211 
212  this%axis_name = trim(axis_name)
213  this%units = trim(units)
214  this%cart_name = uppercase(cart_name)
215  call check_if_valid_cart_name(this%cart_name)
216 
217  if (present(long_name)) this%long_name = trim(long_name)
218 
219  select type (axis_data)
220  type is (real(kind=r8_kind))
221  allocate(real(kind=r8_kind) :: this%axis_data(axis_length))
222  this%axis_data = axis_data
223  this%length = axis_length
224  this%type_of_data = "double" !< This is what fms2_io expects in the register_field call
225  type is (real(kind=r4_kind))
226  allocate(real(kind=r4_kind) :: this%axis_data(axis_length))
227  this%axis_data = axis_data
228  this%length = axis_length
229  this%type_of_data = "float" !< This is what fms2_io expects in the register_field call
230  class default
231  call mpp_error(fatal, "The axis_data in your diag_axis_init call is not a supported type. &
232  & Currently only r4 and r8 data is supported.")
233  end select
234 
235  this%type_of_domain = no_domain
236  if (present(domain)) then
237  if (present(domain2) .or. present(domainu)) call mpp_error(fatal, &
238  "The presence of Domain with any other domain type is prohibited. "//&
239  "Check you diag_axis_init call for axis_name:"//trim(axis_name))
240  allocate(diagdomain1d_t :: this%axis_domain)
241  call this%axis_domain%set(domain=domain)
242  else if (present(domain2)) then
243  if (present(domainu)) call mpp_error(fatal, &
244  "The presence of Domain2 with any other domain type is prohibited. "//&
245  "Check you diag_axis_init call for axis_name:"//trim(axis_name))
246  allocate(diagdomain2d_t :: this%axis_domain)
247  call this%axis_domain%set(domain2=domain2)
248  this%type_of_domain = two_d_domain
249  else if (present(domainu)) then
250  allocate(diagdomainug_t :: this%axis_domain)
251  call this%axis_domain%set(domainu=domainu)
252  this%type_of_domain = ug_domain
253  endif
254 
255  this%tile_count = 1
256  if (present(tile_count)) this%tile_count = tile_count
257 
258  this%domain_position = center
259  if (present(domain_position)) this%domain_position = domain_position
260  call check_if_valid_domain_position(this%domain_position)
261 
262  this%direction = 0
263  if (present(direction)) this%direction = direction
264  call check_if_valid_direction(this%direction)
265 
266  if (present(aux)) this%aux = trim(aux)
267  if (present(req)) this%req = trim(req)
268 
269  this%nsubaxis = 0
270  this%num_attributes = 0
271  end subroutine register_diag_axis_obj

290  !> @brief Write the axis meta data to an open fileobj
291  subroutine write_axis_metadata(this, fileobj, parent_axis)
292  class(fmsdiagaxis_type), target, INTENT(IN) :: this !< diag_axis obj
293  class(fmsnetcdffile_t), INTENT(INOUT) :: fileobj !< Fms2_io fileobj to write the data to
294  class(fmsdiagaxis_type), OPTIONAL, target, INTENT(IN) :: parent_axis !< If the axis is a subaxis, axis object
295  !! for the parent axis (this will be used
296  !! to get some of the metadata info)
297 
298  character(len=:), ALLOCATABLE :: axis_edges_name !< Name of the edges, if it exist
299  character(len=:), pointer :: axis_name !< Name of the axis
300  integer :: axis_length !< Size of the axis
301  integer :: i !< For do loops
302  type(fmsdiagfullaxis_type), pointer :: diag_axis !< Local pointer to the diag_axis
303 
304  integer :: type_of_domain !< The type of domain the current axis is in
305  logical :: is_subaxis !< .true. if the axis is a subaxis
306 
307  is_subaxis = .false.
308 
309  select type(this)
310  type is (fmsdiagfullaxis_type)
311  axis_name => this%axis_name
312  axis_length = this%length
313  diag_axis => this
314  type_of_domain = this%type_of_domain
315  type is (fmsdiagsubaxis_type)
316  is_subaxis = .true.
317  axis_name => this%subaxis_name
318  axis_length = this%ending_index - this%starting_index + 1
319  !< Get all the other information from the parent axis (i.e the cart_name, units, etc)
320  if (present(parent_axis)) then
321  select type(parent_axis)
322  type is (fmsdiagfullaxis_type)
323  diag_axis => parent_axis
324  end select
325  endif
326  type_of_domain = no_domain !< All subaxes are treated as non-domain decomposed (each rank writes it own file)
327  type is (fmsdiagdiurnalaxis_type)
328  call this%write_diurnal_metadata(fileobj)
329  return
330  end select
331 
332  !< Add the axis as a dimension in the netcdf file based on the type of axis_domain and the fileobj type
333  select type (fileobj)
334  !< The register_field calls need to be inside the select type block so that it can go inside the correct
335  !! register_field interface
336  type is (fmsnetcdffile_t)
337  !< Here the axis is not domain decomposed (i.e z_axis)
338  call register_axis(fileobj, axis_name, axis_length)
339  call register_field(fileobj, axis_name, diag_axis%type_of_data, (/axis_name/))
340  type is (fmsnetcdfdomainfile_t)
341  select case (type_of_domain)
342  case (no_domain)
343  !< Here the fileobj is domain decomposed, but the axis is not
344  !! Domain decomposed fileobjs can have axis that are not domain decomposed (i.e "Z" axis)
345  call register_axis(fileobj, axis_name, axis_length)
346  call register_field(fileobj, axis_name, diag_axis%type_of_data, (/axis_name/))
347  case (two_d_domain)
348  !< Here the axis is domain decomposed
349  call register_axis(fileobj, axis_name, diag_axis%cart_name, domain_position=diag_axis%domain_position)
350  call register_field(fileobj, axis_name, diag_axis%type_of_data, (/axis_name/))
351  end select
352  type is (fmsnetcdfunstructureddomainfile_t)
353  select case (type_of_domain)
354  case (ug_domain)
355  !< Here the axis is in a unstructured domain
356  call register_axis(fileobj, axis_name)
357  call register_field(fileobj, axis_name, diag_axis%type_of_data, (/axis_name/))
358  case default
359  !< Here the fileobj is in the unstructured domain, but the axis is not
360  !< Unstructured domain fileobjs can have axis that are not domain decomposed (i.e "Z" axis)
361  call register_axis(fileobj, axis_name, axis_length)
362  call register_field(fileobj, axis_name, diag_axis%type_of_data, (/axis_name/))
363  end select
364  end select
365 
366  !< Write its metadata
367  call register_variable_attribute(fileobj, axis_name, "long_name", diag_axis%long_name, &
368  str_len=len_trim(diag_axis%long_name))
369 
370  if (diag_axis%cart_name .NE. "N") &
371  call register_variable_attribute(fileobj, axis_name, "axis", diag_axis%cart_name, str_len=1)
372 
373  if (trim(diag_axis%units) .NE. "none") &
374  call register_variable_attribute(fileobj, axis_name, "units", diag_axis%units, str_len=len_trim(diag_axis%units))
375 
376  select case (diag_axis%direction)
377  case (direction_up)
378  call register_variable_attribute(fileobj, axis_name, "positive", "up", str_len=2)
379  case (direction_down)
380  call register_variable_attribute(fileobj, axis_name, "positive", "down", str_len=4)
381  end select
382 
383  if (allocated(diag_axis%edges_name) .and. .not. is_subaxis) then
384  call register_variable_attribute(fileobj, axis_name, "edges", diag_axis%edges_name, &
385  str_len=len_trim(diag_axis%edges_name))
386  endif
387 
388  if(allocated(diag_axis%attributes)) then
389  do i = 1, diag_axis%num_attributes
390  select type (att_value => diag_axis%attributes(i)%att_value)
391  type is (character(len=*))
392  call register_variable_attribute(fileobj, axis_name, diag_axis%attributes(i)%att_name, trim(att_value(1)), &
393  str_len=len_trim(att_value(1)))
394  class default
395  call register_variable_attribute(fileobj, axis_name, diag_axis%attributes(i)%att_name, att_value)
396  end select
397  enddo
398  endif
399 
400  end subroutine write_axis_metadata

434  integer, intent(inout) :: naxis !< Number of axis that have
