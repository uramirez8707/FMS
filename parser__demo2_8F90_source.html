20 program parser_demo
21 !> @brief This programs demostrates how to use the parser
22 
23 #ifdef use_yaml
24 use fms_mod, only: fms_init, fms_end
25 use yaml_parser_mod
26 use platform_mod
27 
28 implicit none
29 
30 integer :: diag_yaml_id !< Id for the diag_table yaml
31 integer :: nfiles !< Number of files in the diag_table yaml
32 integer, allocatable :: file_ids(:) !< Ids of the files in the diag_table yaml
33 integer :: nvariables !< Number of variables in the diag_table yaml
34 integer, allocatable :: var_ids(:) !< Ids of the variables in the diag_table yaml
35 integer :: i, j, k !< For do loops
36 integer :: nkeys !< Number of keys
37 integer, allocatable :: key_ids(:) !< Ids of keys in the diag_table_yaml
38 character(len=255) :: key_value !< The value of a key
39 character(len=255) :: key_name !< The name of a key
40 
41 call fms_init
42 call fms_end
43 
44 diag_yaml_id = open_and_parse_file("diag_table.yaml")
45 print *, ""
46 
47 nkeys = get_nkeys(diag_yaml_id, 0)
48 allocate(key_ids(nkeys))
49 call get_key_ids(diag_yaml_id, 0, key_ids)
50 
51 do i = 1, nkeys
52  call get_key_name(diag_yaml_id, key_ids(i), key_name)
53  call get_key_value(diag_yaml_id, key_ids(i), key_value)
54  print *, "Key:", trim(key_name), " Value:", trim(key_value)
55 enddo
56 
57 deallocate(key_ids)
58 
59 nfiles = get_num_blocks(diag_yaml_id, "diag_files")
60 allocate(file_ids(nfiles))
61 call get_block_ids(diag_yaml_id, "diag_files", file_ids)
62 print *, ""
63 
64 do i = 1, nfiles
65  print *, "File number:", i
66 
67  nkeys = get_nkeys(diag_yaml_id, file_ids(i))
68  allocate(key_ids(nkeys))
69  call get_key_ids(diag_yaml_id, file_ids(i), key_ids)
70 
71  do j = 1, nkeys
72  call get_key_name(diag_yaml_id, key_ids(j), key_name)
73  call get_key_value(diag_yaml_id, key_ids(j), key_value)
74  print *, " Key:", trim(key_name), " Value:", trim(key_value)
75  enddo
76 
77  deallocate(key_ids)
78  print *, ""
79  !< The number of variables that are part of the current file
80  nvariables = get_num_blocks(diag_yaml_id, "varlist", parent_block_id=file_ids(i))
81  allocate(var_ids(nvariables))
82  call get_block_ids(diag_yaml_id, "varlist", var_ids, parent_block_id=file_ids(i))
83 
84  do j = 1, nvariables
85  print *, " Variable number:", j
86 
87  nkeys = get_nkeys(diag_yaml_id, var_ids(j))
88  allocate(key_ids(nkeys))
89  call get_key_ids(diag_yaml_id, var_ids(j), key_ids)
90 
91  do k = 1, nkeys
92  call get_key_name(diag_yaml_id, key_ids(k), key_name)
93  call get_key_value(diag_yaml_id, key_ids(k), key_value)
94  print *, " Key:", trim(key_name), " Value:", trim(key_value)
95  enddo
96 
97  deallocate(key_ids)
98  print *, ""
99  end do
100 
101  deallocate(var_ids)
102  print *, ""
103 enddo
104 deallocate(file_ids)
105 
106 #endif
107 
108 end program parser_demo
