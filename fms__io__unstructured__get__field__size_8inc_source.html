24 !>Get the size of the dimensions of a field from a file associated with an
25 !!unstructured mpp domain.
26 subroutine fms_io_unstructured_get_field_size(filename, &
27  fieldname, &
28  field_dimension_sizes, &
29  domain, &
30  field_found)
31 
32  !Inputs/Outputs
33  character(len=*),intent(in) :: filename !<The name of a file.
34  character(len=*),intent(in) :: fieldname !<The name of a field in the input file.
35  integer,dimension(:),intent(inout) :: field_dimension_sizes !<Array of dimension sizes for the inputted field.
36  type(domainug),intent(in) :: domain !<An unstructured mpp domain
37  !! associated with the input file.
38  logical,intent(out),optional :: field_found !<Flag telling if the inputted
39  !! field was found in the inputted file.
40 
41  !Local variables
42  type(domainug),pointer :: io_domain !<Pointer to the unstructured I/O domain.
43  integer(INT_KIND) :: io_domain_npes !<The total number of ranks in an I/O domain pelist.
44  integer(INT_KIND),dimension(:),allocatable :: pelist !<A pelist.
45  integer(INT_KIND) :: funit !<File unit for the inputted file.
46  integer(INT_KIND) :: num_axes !<The total number of axes
47  !! contained in the inputted file.
48  integer(INT_KIND) :: num_fields !<The total number of fields
49  !! contained in the inputted file.
50  integer(INT_KIND) :: num_atts !<The total number of global
51  !! attributes contained in the inputted file.
52  integer(INT_KIND) :: num_time_levels !<The total number of time
53  !! levels contained in the inputted file.
54  type(fieldtype),dimension(max_fields) :: file_fields !<An array of all fields
55  !! contained in the inputted file (max_fields is a module variable).
56  logical(INT_KIND) :: found !<Flag telling if the field was found in the file.
57  character(len=128) :: file_field_name !<Name of a field from the inputted file.
58  integer(INT_KIND) :: file_field_ndim !<Number of dimensions of
59  !! a field from the inputted file.
60  type(axistype),dimension(max_fields) :: file_field_axes !<An array of all axes of
61  !! a field contained in the inputted file (max_fields is a module variable).
62  character(len=128) :: file_axis_name !<Name of an axis from the inputted file.
63  integer(INT_KIND) :: file_axis_size !<Size of an axis from the inputted file.
64  integer(INT_KIND) :: i !<Loop variable.
65  integer(INT_KIND) :: j !<Loop variable.
66 
67  !Point to the I/O domain associated with the inputted unstructured mpp
68  !domain.
69  io_domain => null()
70  io_domain => mpp_get_ug_io_domain(domain)
71 
72  !Get the pelist associated with the I/O domain.
73  io_domain_npes = mpp_get_ug_domain_npes(io_domain)
74  allocate(pelist(io_domain_npes))
75  call mpp_get_ug_domain_pelist(io_domain, &
76  pelist)
77  io_domain => null()
78 
79  !Get the file unit for the inputted file.
80  call fms_io_unstructured_file_unit(filename, &
81  funit, &
82  domain)
83 
84  !Have the root rank of the I/O domain pelist get the size of the dimensions
85  !of the inputted fields from the inputted file.
86  if (mpp_pe() .eq. pelist(1)) then
87 
88  !Get the number of fields and axes contained in the inputted file.
89  call mpp_get_info(funit, &
90  num_axes, &
91  num_fields, &
92  num_atts, &
93  num_time_levels)
94 
95  !Make sure that the number of fields in the file does not exceed the
96  !maximum number allowed per file.
97  !max_fields is a module variable.
98  if (num_fields .gt. max_fields) then
99  call mpp_error(fatal, &
100  "fms_io_unstructured_get_field_size:" &
101  //" the number of fields in the file " &
102  //trim(filename)//" exceeds the maximum number" &
103  //" of fields allowed per file (max_fields)")
104  endif
105 
106  !Read in all fields contained in the inputted file.
107  call mpp_get_fields(funit, &
108  file_fields(1:num_fields))
109 
110  !Check if the inputted field matches one the fields contained in
111  !the inputted file. If it matches, get the size of the field
112  !dimensions.
113  found = .false.
114  field_dimension_sizes = -1
115  do i = 1,num_fields
116  call mpp_get_atts(file_fields(i), &
117  name=file_field_name)
118  if (lowercase(trim(file_field_name)) .eq. &
119  lowercase(trim(fieldname))) then
120  call mpp_get_atts(file_fields(i), &
121  ndim=file_field_ndim)
122  call mpp_get_atts(file_fields(i), &
123  axes=file_field_axes(1:file_field_ndim))
124  do j = 1,file_field_ndim
125  call mpp_get_atts(file_field_axes(j), &
126  len=field_dimension_sizes(j))
127  enddo
128  found = .true.
129  exit
130  endif
131  enddo
132 
133  !If the inputted field does not match any of the fields contained
134  !in the inputted file, then check if it matches any of the axes
135  !contained in the file.
136  if (.not. found) then
137  call mpp_get_axes(funit, &
138  file_field_axes(1:num_axes))
139  do i = 1,num_axes
140  call mpp_get_atts(file_field_axes(i), &
141  name=file_axis_name, &
142  len=file_axis_size)
143  if (lowercase(trim(file_axis_name)) .eq. &
144  lowercase(trim(fieldname))) then
145  field_dimension_sizes(1) = file_axis_size
146  found = .true.
147  exit
148  endif
149  enddo
150  endif
151  endif
152 
153  !Broadcast the flag telling if the inputted field was found in the inputted
154  !file and the field dimension sizes array to all non-root ranks on the
155  !I/O domain pelist.
156  if (mpp_pe() .eq. pelist(1)) then
157  do i = 2,io_domain_npes
158  call mpp_send(found, &
159  pelist(i), &
160  tag=comm_tag_1)
161  call mpp_send(field_dimension_sizes, &
162  size(field_dimension_sizes), &
163  pelist(i), &
164  tag=comm_tag_2)
165  enddo
166  call mpp_sync_self()
167  else
168  call mpp_recv(found, &
169  pelist(1), &
170  block = .false., &
171  tag=comm_tag_1)
172  call mpp_recv(field_dimension_sizes, &
173  size(field_dimension_sizes), &
174  pelist(1), &
175  block = .false., &
176  tag=comm_tag_2)
177  call mpp_sync_self(check=event_recv)
178  endif
179 
180  !If the field_found flag is present, then return the value of the found
181  !flag. It is assumed that this value will be checked by the calling
182  !routine. If the field_found flag is not present and the field was not
183  !found in the file, then throw a fatal error.
184  if (present(field_found)) then
185  field_found = found
186  elseif (.not. found) then
187  call mpp_error(fatal, &
188  "fms_io_unstructured_get_field_size:" &
189  //" the inputted field "//trim(fieldname) &
190  //" was not found in the file "//trim(filename))
191  endif
192 
193  !Deallocate local allocatables.
194  deallocate(pelist)
195 
196  return
197 end subroutine fms_io_unstructured_get_field_size
