175 subroutine compare_data(Domain, actual_result, expected_result)
176 type(domain2d), intent(in) :: Domain !< Domain with mask table
177 real, intent(in) :: expected_result !< Expected result from data_override
178 real, dimension(:,:), intent(in) :: actual_result !< Result from data_override
179 
180 integer :: xsizec, ysizec !< Size of the compute domain
181 integer :: xsized, ysized !< Size of the data domain
182 integer :: nx, ny !< Size of acual_result
183 integer :: nhalox, nhaloy !< Size of the halos
184 integer :: i, j !< Helper indices
185 
186 !< Data is only expected to be overriden for the compute domain -not at the halos.
187 call mpp_get_compute_domain(domain, xsize=xsizec, ysize=ysizec)
188 call mpp_get_data_domain(domain, xsize=xsized, ysize=ysized)
189 
190 !< Note that actual_result has indices at (1:nx,1:ny) not (is:ie,js:je)
191 nhalox= (xsized-xsizec)/2
192 nhaloy = (ysized-ysizec)/2
193 nx = size(actual_result, 1)
194 ny = size(actual_result, 2)
195 
196 do i = 1, nx
197  do j = 1, ny
198  if (i <= nhalox .or. i > (nx-nhalox) .or. j <= nhaloy .or. j > (ny-nhaloy)) then
199  !< This is the result at the halos it should 999.
200  if (actual_result(i,j) .ne. 999.) then
201  print *, "for i=", i, " and j=", j, " result=", actual_result(i,j)
202  call mpp_error(fatal, "test_data_override_ongrid: Data was overriden in the halos!!")
203  endif
204  else
205  if (actual_result(i,j) .ne. expected_result) then
206  print *, "for i=", i, " and j=", j, " result=", actual_result(i,j)
207  call mpp_error(fatal, "test_data_override_ongrid: Result is different from expected answer!")
208  endif
209  endif
210  enddo
211 enddo
212 
213 end subroutine
