73  !> Perform simple addition on 64-bit real arrays in different domain configurations and update the domains
74  subroutine test_halo_update_r8( domain_type )
75  character(len=*), intent(in) :: domain_type !< the domain type that will be tested
76  real(kind=r8_kind), allocatable, dimension(:,:,:) :: xr8, x1r8, x2r8, x3r8, x4r8
77  real(kind=r8_kind), allocatable, dimension(:,:,:) :: yr8, y1r8, y2r8, y3r8, y4r8
78  type(domain2d) :: domain
79  real(kind=r8_kind), allocatable :: global1r8(:,:,:), global2r8(:,:,:), globalr8(:,:,:)
80  logical, allocatable :: maskmap(:,:)
81  integer :: shift, i, xhalo, yhalo
82  logical :: is_symmetry, folded_south, folded_west, folded_east
83  integer :: is, ie, js, je, isd, ied, jsd, jed
84  integer :: pe, npes
85 
86  pe = mpp_pe()
87  npes = mpp_npes()
88  ! when testing maskmap option, nx*ny should be able to be divided by both npes and npes+1
89  if ((domain_type == 'Masked') .OR. (domain_type == 'Masked symmetry')) then
90  if((mod(nx*ny, npes) .NE. 0) .OR. (mod(nx*ny, npes+1) .NE. 0)) then
91  call mpp_error(note,'test_halo_update_r8: nx*ny can not be divided by both npes and npes+1, '//&
92  'Masked test_halo_update_r8 will not be tested')
93  return
94  end if
95  end if
96 
97  if(trim(domain_type) == 'Folded xy_halo' ) then
98  xhalo = max(whalo, ehalo); yhalo = max(shalo, nhalo)
99  allocate(globalr8(1-xhalo:nx+xhalo,1-yhalo:ny+yhalo,nz) )
100  else
101  allocate(globalr8(1-whalo:nx+ehalo,1-shalo:ny+nhalo,nz) )
102  end if
103  ! populate the global array
104  globalr8 = 0.0
105  do k = 1,nz
106  do j = 1,ny
107  do i = 1,nx
108  globalr8(i,j,k) = k + i*1e-3 + j*1e-6
109  end do
110  end do
111  end do
112 
113  if(index(domain_type, 'symmetry') == 0) then
114  is_symmetry = .false.
115  else
116  is_symmetry = .true.
117  end if
118  select case(trim(domain_type))
119  case( 'Simple', 'Simple symmetry' )
120  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
121  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
122  shalo=shalo, nhalo=nhalo, name=trim(domain_type), symmetry = is_symmetry )
123  case( 'Cyclic', 'Cyclic symmetry' )
124  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
125  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
126  shalo=shalo, nhalo=nhalo, xflags=cyclic_global_domain, yflags=cyclic_global_domain, &
127  name=trim(domain_type), symmetry = is_symmetry )
128  globalr8(1-whalo:0, 1:ny,:) = globalr8(nx-whalo+1:nx, 1:ny,:)
129  globalr8(nx+1:nx+ehalo, 1:ny,:) = globalr8(1:ehalo, 1:ny,:)
130  globalr8(1-whalo:nx+ehalo, 1-shalo:0,:) = globalr8(1-whalo:nx+ehalo, ny-shalo+1:ny,:)
131  globalr8(1-whalo:nx+ehalo, ny+1:ny+nhalo,:) = globalr8(1-whalo:nx+ehalo, 1:nhalo,:)
132  case( 'Folded-north', 'Folded-north symmetry' )
133  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
134  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
135  shalo=shalo, nhalo=nhalo, xflags=cyclic_global_domain, yflags=fold_north_edge, &
136  name=domain_type, symmetry = is_symmetry )
137  call fill_folded_north_halo(globalr8, 0, 0, 0, 0, 1)
138  case( 'Folded-south symmetry' )
139  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
140  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
141  shalo=shalo, nhalo=nhalo, xflags=cyclic_global_domain, yflags=fold_south_edge, &
142  name=domain_type, symmetry = is_symmetry )
143  call fill_folded_south_halo(globalr8, 0, 0, 0, 0, 1)
144  case( 'Folded-west symmetry' )
145  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
146  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
147  shalo=shalo, nhalo=nhalo, xflags=fold_west_edge, yflags=cyclic_global_domain, &
148  name=domain_type, symmetry = is_symmetry )
149  call fill_folded_west_halo(globalr8, 0, 0, 0, 0, 1)
150  case( 'Folded-east symmetry' )
151  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
152  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
153  shalo=shalo, nhalo=nhalo, xflags=fold_east_edge, yflags=cyclic_global_domain, &
154  name=domain_type, symmetry = is_symmetry )
155  call fill_folded_east_halo(globalr8, 0, 0, 0, 0, 1)
156  case( 'Folded xy_halo' )
157  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
158  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, xhalo=xhalo, yhalo=yhalo, &
159  xflags=cyclic_global_domain, yflags=fold_north_edge, name=domain_type, symmetry = is_symmetry )
160  globalr8(1-xhalo:0, 1:ny,:) = globalr8(nx-xhalo+1:nx, 1:ny,:)
161  globalr8(nx+1:nx+xhalo, 1:ny,:) = globalr8(1:xhalo, 1:ny,:)
162  globalr8(1-xhalo:nx+xhalo,ny+1:ny+yhalo,:) = globalr8(nx+xhalo:1-xhalo:-1, ny:ny-yhalo+1:-1,:)
163  case( 'Masked', 'Masked symmetry' )
164  !with fold and cyclic, assign to npes+1 and mask out the top-rightdomain
165  call mpp_define_layout( (/1,nx,1,ny/), npes+1, layout )
166  allocate( maskmap(layout(1),layout(2)) )
167  maskmap(:,:) = .true.; maskmap(layout(1),layout(2)) = .false.
168  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
169  shalo=shalo, nhalo=nhalo, xflags=cyclic_global_domain, yflags=fold_north_edge, &
170  maskmap=maskmap, name=domain_type, symmetry = is_symmetry )
171  deallocate(maskmap)
172  !we need to zero out the globalr8 data on the missing domain.
173  !this logic assumes top-right, in an even division
174  if( mod(nx,layout(1)).NE.0 .OR. mod(ny,layout(2)).NE.0 )call mpp_error( fatal, &
175  'TEST_MPP_DOMAINS: test for masked domains needs (nx,ny) to divide evenly on npes+1 PEs.' )
176  globalr8(nx-nx/layout(1)+1:nx,ny-ny/layout(2)+1:ny,:) = 0
177  call fill_folded_north_halo(globalr8, 0, 0, 0, 0, 1)
178  case default
179  call mpp_error( fatal, 'test_halo_update_r8: no such test: '//domain_type )
180  end select
181 
182  ! define the arrays
183  call mpp_get_compute_domain( domain, is, ie, js, je )
184  call mpp_get_data_domain( domain, isd, ied, jsd, jed )
185  allocate(xr8(isd:ied,jsd:jed,nz) )
186  allocate(x1r8(isd:ied,jsd:jed,nz) )
187  allocate(x2r8(isd:ied,jsd:jed,nz) )
188  allocate(x3r8(isd:ied,jsd:jed,nz) )
189  allocate(x4r8(isd:ied,jsd:jed,nz) )
190  xr8(:,:,:) = 0.0
191  xr8(is:ie,js:je,:) = globalr8(is:ie,js:je,:)
192  x1r8 = xr8; x2r8 = xr8; x3r8 = xr8; x4r8 = xr8
193 
194  ! update the halo region
195  id = mpp_clock_id( domain_type, flags=mpp_clock_sync+mpp_clock_detailed )
196  call mpp_clock_begin(id)
197  call mpp_update_domains( xr8, domain )
198  call mpp_clock_end (id)
199  call compare_checksums( xr8, globalr8(isd:ied,jsd:jed,:), domain_type )
200 
201  ! update part of the halo region
202  id = mpp_clock_id( domain_type//' partial', flags=mpp_clock_sync+mpp_clock_detailed )
203  call mpp_clock_begin(id)
204  call mpp_update_domains( x1r8, domain, nupdate+eupdate, complete=.false. )
205  call mpp_update_domains( x2r8, domain, nupdate+eupdate, complete=.false. )
206  call mpp_update_domains( x3r8, domain, nupdate+eupdate, complete=.false. )
207  call mpp_update_domains( x4r8, domain, nupdate+eupdate, complete=.true. )
208  call mpp_clock_end (id)
209  call compare_checksums( x1r8(is:ied,js:jed,:), globalr8(is:ied,js:jed,:), domain_type//' partial x1r8' )
210  call compare_checksums( x2r8(is:ied,js:jed,:), globalr8(is:ied,js:jed,:), domain_type//' partial x2r8' )
211  call compare_checksums( x3r8(is:ied,js:jed,:), globalr8(is:ied,js:jed,:), domain_type//' partial x3r8' )
212  call compare_checksums( x4r8(is:ied,js:jed,:), globalr8(is:ied,js:jed,:), domain_type//' partial x4r8' )
213 
214  !--- test vector update for FOLDED and MASKED case.
215  if(domain_type == 'Simple' .or. domain_type == 'Simple symmetry' .or. domain_type == 'Cyclic' .or. domain_type == 'Cyclic symmetry') then
216  deallocate(xr8,x1r8,x2r8,x3r8,x4r8)
217  return
218  end if
219 
220  !------------------------------------------------------------------
221  ! vector update : BGRID_NE
222  !------------------------------------------------------------------
223  shift = 0
224  if(is_symmetry) then
225  shift = 1
226  deallocate(globalr8)
227  allocate(globalr8(1-whalo:nx+ehalo+shift,1-shalo:ny+nhalo+shift,nz) )
228  globalr8(:,:,:) = 0.0
229  do k = 1,nz
230  do j = 1,ny+1
231  do i = 1,nx+1
232  globalr8(i,j,k) = k + i*1e-3 + j*1e-6
233  end do
234  end do
235  end do
236  if(domain_type == 'Masked symmetry') then
237  globalr8(nx-nx/layout(1)+1:nx+1,ny-ny/layout(2)+1:ny+1,:) = 0.0
238  endif
239  deallocate(xr8, x1r8, x2r8, x3r8, x4r8)
240  allocate( xr8(isd:ied+1,jsd:jed+1,nz) )
241  allocate( x1r8(isd:ied+1,jsd:jed+1,nz) )
242  allocate( x2r8(isd:ied+1,jsd:jed+1,nz) )
243  allocate( x3r8(isd:ied+1,jsd:jed+1,nz) )
244  allocate( x4r8(isd:ied+1,jsd:jed+1,nz) )
245  endif
246 
247  folded_south = .false.
248  folded_west = .false.
249  folded_east = .false.
250  select case (domain_type)
251  case ('Folded-north', 'Masked')
252  ! fill in folded north edge, cyclic east and west edge
253  call fill_folded_north_halo(globalr8, 1, 1, 0, 0, -1)
254  case ('Folded xy_halo')
255  ! fill in folded north edge, cyclic east and west edge
256  globalr8(1-xhalo:0, 1:ny,:) = globalr8(nx-xhalo+1:nx, 1:ny,:)
257  globalr8(nx+1:nx+xhalo, 1:ny,:) = globalr8(1:xhalo, 1:ny,:)
258  globalr8(1-xhalo:nx+xhalo-1,ny+1:ny+yhalo,:) = -globalr8(nx+xhalo-1:1-xhalo:-1,ny-1:ny-yhalo:-1,:)
259  globalr8(nx+xhalo, ny+1:ny+yhalo,:) = -globalr8(nx-xhalo, ny-1:ny-yhalo:-1,:)
260  case ('Folded-north symmetry', 'Masked symmetry' )
261  call fill_folded_north_halo(globalr8, 1, 1, 1, 1, -1)
262  case ('Folded-south symmetry' )
263  folded_south = .true.
264  call fill_folded_south_halo(globalr8, 1, 1, 1, 1, -1)
265  case ('Folded-west symmetry' )
266  folded_west = .true.
267  call fill_folded_west_halo(globalr8, 1, 1, 1, 1, -1)
268  case ('Folded-east symmetry' )
269  folded_east = .true.
270  call fill_folded_east_halo(globalr8, 1, 1, 1, 1, -1)
271  case default
272  call mpp_error( fatal, 'TEST_MPP_DOMAINS: no such test: '//domain_type )
273  end select
274 
275  xr8(:,:,:) = 0.0
276  xr8(is:ie+shift,js:je+shift,:) = globalr8(is:ie+shift,js:je+shift,:)
277  !set up yr8 array
278  allocate( yr8(isd:ied+shift,jsd:jed+shift,nz) )
279  allocate( y1r8(isd:ied+shift,jsd:jed+shift,nz) )
280  allocate( y2r8(isd:ied+shift,jsd:jed+shift,nz) )
281  allocate( y3r8(isd:ied+shift,jsd:jed+shift,nz) )
282  allocate( y4r8(isd:ied+shift,jsd:jed+shift,nz) )
283  yr8 = xr8; x1r8 = xr8; x2r8 = xr8; x3r8 = xr8; x4r8 = xr8
284  yr8 = xr8; y1r8 = xr8; y2r8 = xr8; y3r8 = xr8; y4r8 = xr8
285 
286  id = mpp_clock_id( domain_type//' vector BGRID_NE', flags=mpp_clock_sync+mpp_clock_detailed )
287  call mpp_clock_begin(id)
288  call mpp_update_domains( xr8, yr8, domain, gridtype=bgrid_ne)
289  call mpp_update_domains( x1r8, y1r8, domain, gridtype=bgrid_ne, complete=.false. )
290  call mpp_update_domains( x2r8, y2r8, domain, gridtype=bgrid_ne, complete=.false. )
291  call mpp_update_domains( x3r8, y3r8, domain, gridtype=bgrid_ne, complete=.false. )
292  call mpp_update_domains( x4r8, y4r8, domain, gridtype=bgrid_ne, complete=.true. )
293  call mpp_clock_end (id)
294 
295  ! redundant points must be equal and opposite
296 
297  if(folded_south) then
298  globalr8(nx/2+shift, 1,:) = 0.0 !pole points must have 0 velocity
299  globalr8(nx+shift , 1,:) = 0.0 !pole points must have 0 velocity
300  globalr8(nx/2+1+shift:nx-1+shift, 1,:) = -globalr8(nx/2-1+shift:1+shift:-1, 1,:)
301  globalr8(1-whalo:shift, 1,:) = -globalr8(nx-whalo+1:nx+shift, 1,:)
302  globalr8(nx+1+shift:nx+ehalo+shift, 1,:) = -globalr8(1+shift:ehalo+shift, 1,:)
303  !--- the following will fix the +0/-0 problem on altix
304  if(shalo >0) globalr8(shift,1,:) = 0.0 !pole points must have 0 velocity
305  else if(folded_west) then
306  globalr8(1, ny/2+shift, :) = 0.0 !pole points must have 0 velocity
307  globalr8(1, ny+shift, :) = 0.0 !pole points must have 0 velocity
308  globalr8(1, ny/2+1+shift:ny-1+shift, :) = -globalr8(1, ny/2-1+shift:1+shift:-1, :)
309  globalr8(1, 1-shalo:shift, :) = -globalr8(1, ny-shalo+1:ny+shift, :)
310  globalr8(1, ny+1+shift:ny+nhalo+shift, :) = -globalr8(1, 1+shift:nhalo+shift, :)
311  !--- the following will fix the +0/-0 problem on altix
312  if(whalo>0) globalr8(1, shift, :) = 0.0 !pole points must have 0 velocity
313  else if(folded_east) then
314  globalr8(nx+shift, ny/2+shift, :) = 0.0 !pole points must have 0 velocity
315  globalr8(nx+shift, ny+shift, :) = 0.0 !pole points must have 0 velocity
316  globalr8(nx+shift, ny/2+1+shift:ny-1+shift, :) = -globalr8(nx+shift, ny/2-1+shift:1+shift:-1, :)
317  globalr8(nx+shift, 1-shalo:shift, :) = -globalr8(nx+shift, ny-shalo+1:ny+shift, :)
318  globalr8(nx+shift, ny+1+shift:ny+nhalo+shift, :) = -globalr8(nx+shift, 1+shift:nhalo+shift, :)
319  if(ehalo >0) globalr8(nx+shift, shift, :) = 0.0 !pole points must have 0 velocity
320  else
321  globalr8(nx/2+shift, ny+shift,:) = 0.0 !pole points must have 0 velocity
322  globalr8(nx+shift , ny+shift,:) = 0.0 !pole points must have 0 velocity
323  globalr8(nx/2+1+shift:nx-1+shift, ny+shift,:) = -globalr8(nx/2-1+shift:1+shift:-1, ny+shift,:)
324  if (domain_type == 'Folded xy_halo') then
325  globalr8(1-xhalo:shift, ny+shift,:) = -globalr8(nx-xhalo+1:nx+shift, ny+shift,:)
326  globalr8(nx+1+shift:nx+xhalo+shift, ny+shift,:) = -globalr8(1+shift:xhalo+shift, ny+shift,:)
327  else
328  globalr8(1-whalo:shift, ny+shift,:) = -globalr8(nx-whalo+1:nx+shift, ny+shift,:)
329  globalr8(nx+1+shift:nx+ehalo+shift, ny+shift,:) = -globalr8(1+shift:ehalo+shift, ny+shift,:)
330  end if
331  !--- the following will fix the +0/-0 problem on altix
332  if(nhalo >0) globalr8(shift,ny+shift,:) = 0.0 !pole points must have 0 velocity
333  endif
334 
335  call compare_checksums( xr8, globalr8(isd:ied+shift,jsd:jed+shift,:), domain_type//' BGRID_NE xr8' )
336  call compare_checksums( yr8, globalr8(isd:ied+shift,jsd:jed+shift,:), domain_type//' BGRID_NE yr8' )
337  call compare_checksums( x1r8, globalr8(isd:ied+shift,jsd:jed+shift,:), domain_type//' BGRID_NE x1r8' )
338  call compare_checksums( x2r8, globalr8(isd:ied+shift,jsd:jed+shift,:), domain_type//' BGRID_NE x2r8' )
339  call compare_checksums( x3r8, globalr8(isd:ied+shift,jsd:jed+shift,:), domain_type//' BGRID_NE x3r8' )
340  call compare_checksums( x4r8, globalr8(isd:ied+shift,jsd:jed+shift,:), domain_type//' BGRID_NE x4r8' )
341  call compare_checksums( y1r8, globalr8(isd:ied+shift,jsd:jed+shift,:), domain_type//' BGRID_NE y1r8' )
342  call compare_checksums( y2r8, globalr8(isd:ied+shift,jsd:jed+shift,:), domain_type//' BGRID_NE y2r8' )

394  global2r8(nx+1:nx+xhalo, 1:ny,:) = global2r8(1:xhalo, 1:ny,:)
395  global1r8(1-xhalo:nx+xhalo-1, ny+1:ny+yhalo,:) = -global1r8(nx+xhalo-1:1-xhalo:-1, ny:ny-yhalo+1:-1,:)

409  call fill_folded_east_halo(global2r8, 0, 1, 0, 1, -1)

433  global2r8(nx+1:nx+ehalo, 1,:) = -global2r8(1:ehalo, 1, :)
434  else if(folded_west) then
435  global1r8(1, ny/2+1:ny, :) = -global1r8(1, ny/2:1:-1, :)
436  global1r8(1, 1-shalo:0, :) = -global1r8(1, ny-shalo+1:ny, :)
437  global1r8(1, ny+1:ny+nhalo, :) = -global1r8(1, 1:nhalo, :)
438  else if(folded_east) then
439  global1r8(nx+shift, ny/2+1:ny, :) = -global1r8(nx+shift, ny/2:1:-1, :)
440  global1r8(nx+shift, 1-shalo:0, :) = -global1r8(nx+shift, ny-shalo+1:ny, :)
441  global1r8(nx+shift, ny+1:ny+nhalo, :) = -global1r8(nx+shift, 1:nhalo, :)
442  else
443  global2r8(nx/2+1:nx, ny+shift,:) = -global2r8(nx/2:1:-1, ny+shift,:)
444  if(domain_type == 'Folded xy_halo') then
445  global2r8(1-xhalo:0, ny+shift,:) = -global2r8(nx-xhalo+1:nx, ny+shift,:)
446  global2r8(nx+1:nx+xhalo, ny+shift,:) = -global2r8(1:xhalo, ny+shift,:)
447  else
448  global2r8(1-whalo:0, ny+shift,:) = -global2r8(nx-whalo+1:nx, ny+shift,:)
449  global2r8(nx+1:nx+ehalo, ny+shift,:) = -global2r8(1:ehalo, ny+shift,:)
450  end if
451  endif
452 
453  call compare_checksums( xr8, global1r8(isd:ied+shift,jsd:jed, :), domain_type//' CGRID_NE xr8' )
454  call compare_checksums( yr8, global2r8(isd:ied, jsd:jed+shift,:), domain_type//' CGRID_NE yr8' )
455  call compare_checksums( x1r8, global1r8(isd:ied+shift,jsd:jed, :), domain_type//' CGRID_NE x1r8' )
456  call compare_checksums( x2r8, global1r8(isd:ied+shift,jsd:jed, :), domain_type//' CGRID_NE x2r8' )
457  call compare_checksums( x3r8, global1r8(isd:ied+shift,jsd:jed, :), domain_type//' CGRID_NE x3r8' )
458  call compare_checksums( x4r8, global1r8(isd:ied+shift,jsd:jed, :), domain_type//' CGRID_NE x4r8' )
459  call compare_checksums( y1r8, global2r8(isd:ied, jsd:jed+shift,:), domain_type//' CGRID_NE y1r8' )
460  call compare_checksums( y2r8, global2r8(isd:ied, jsd:jed+shift,:), domain_type//' CGRID_NE y2r8' )
461  call compare_checksums( y3r8, global2r8(isd:ied, jsd:jed+shift,:), domain_type//' CGRID_NE y3r8' )
462  call compare_checksums( y4r8, global2r8(isd:ied, jsd:jed+shift,:), domain_type//' CGRID_NE y4r8' )
463

466  end subroutine test_halo_update_r8

468  !> Perform simple addition on 32-bit real arrays in different domain configurations and update the domains
469  subroutine test_halo_update_r4( domain_type )
470  character(len=*), intent(in) :: domain_type !< the domain type that will be tested
471  real(kind=r4_kind), allocatable, dimension(:,:,:) :: xr4, x1r4, x2r4, x3r4, x4r4
472  real(kind=r4_kind), allocatable, dimension(:,:,:) :: yr4, y1r4, y2r4, y3r4, y4r4
473  type(domain2d) :: domain
474  real(kind=r4_kind), allocatable :: global1r4(:,:,:), global2r4(:,:,:), globalr4(:,:,:)
475  logical, allocatable :: maskmap(:,:)
476  integer :: shift, i, xhalo, yhalo
477  logical :: is_symmetry, folded_south, folded_west, folded_east
478  integer :: is, ie, js, je, isd, ied, jsd, jed
479  integer :: pe, npes
480 
481  pe = mpp_pe()
482  npes = mpp_npes()
483 
484  ! when testing maskmap option, nx*ny should be able to be divided by both npes and npes+1
485  if((domain_type == 'Masked') .OR. (domain_type == 'Masked symmetry')) then
486  if((mod(nx*ny, npes) .NE. 0) .OR. (mod(nx*ny, npes+1) .NE. 0)) then
487  call mpp_error(note,'test_halo_update_r4: nx*ny can not be divided by both npes and npes+1, '//&
488  'Masked test_halo_update_r4 will not be tested')
489  return
490  end if
491  end if
492 
493  if(trim(domain_type) == 'Folded xy_halo' ) then
494  xhalo = max(whalo, ehalo); yhalo = max(shalo, nhalo)
495  allocate(globalr4(1-xhalo:nx+xhalo,1-yhalo:ny+yhalo,nz) )
496  else
497  allocate(globalr4(1-whalo:nx+ehalo,1-shalo:ny+nhalo,nz) )
498  end if

500  globalr4 = 0
501  do k = 1,nz
502  do j = 1,ny
503  do i = 1,nx
504  globalr4(i,j,k) = k + i*1e-3 + j*1e-6
505  end do
506  end do
507  end do
508 
509  if(index(domain_type, 'symmetry') == 0) then
510  is_symmetry = .false.
511  else
512  is_symmetry = .true.
513  end if
514  select case(trim(domain_type))
515  case( 'Simple', 'Simple symmetry' )
516  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
517  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
518  shalo=shalo, nhalo=nhalo, name=trim(domain_type), symmetry = is_symmetry )
519  case( 'Cyclic', 'Cyclic symmetry' )
520  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
521  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
522  shalo=shalo, nhalo=nhalo, xflags=cyclic_global_domain, yflags=cyclic_global_domain, &
523  name=trim(domain_type), symmetry = is_symmetry )
524  globalr4(1-whalo:0, 1:ny,:) = globalr4(nx-whalo+1:nx, 1:ny,:)
525  globalr4(nx+1:nx+ehalo, 1:ny,:) = globalr4(1:ehalo, 1:ny,:)
526  globalr4(1-whalo:nx+ehalo, 1-shalo:0,:) = globalr4(1-whalo:nx+ehalo, ny-shalo+1:ny,:)
527  globalr4(1-whalo:nx+ehalo, ny+1:ny+nhalo,:) = globalr4(1-whalo:nx+ehalo, 1:nhalo,:)
528  case( 'Folded-north', 'Folded-north symmetry' )
529  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
530  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
531  shalo=shalo, nhalo=nhalo, xflags=cyclic_global_domain, yflags=fold_north_edge, &
532  name=domain_type, symmetry = is_symmetry )
533  call fill_folded_north_halo(globalr4, 0, 0, 0, 0, 1)
534  case( 'Folded-south symmetry' )
535  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
536  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
537  shalo=shalo, nhalo=nhalo, xflags=cyclic_global_domain, yflags=fold_south_edge, &
538  name=domain_type, symmetry = is_symmetry )
539  call fill_folded_south_halo(globalr4, 0, 0, 0, 0, 1)
540  case( 'Folded-west symmetry' )
541  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
542  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
543  shalo=shalo, nhalo=nhalo, xflags=fold_west_edge, yflags=cyclic_global_domain, &
544  name=domain_type, symmetry = is_symmetry )
545  call fill_folded_west_halo(globalr4, 0, 0, 0, 0, 1)
546  case( 'Folded-east symmetry' )
547  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
548  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
549  shalo=shalo, nhalo=nhalo, xflags=fold_east_edge, yflags=cyclic_global_domain, &
550  name=domain_type, symmetry = is_symmetry )
551  call fill_folded_east_halo(globalr4, 0, 0, 0, 0, 1)
552  case( 'Folded xy_halo' )
553  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
554  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, xhalo=xhalo, yhalo=yhalo, &
555  xflags=cyclic_global_domain, yflags=fold_north_edge, name=domain_type, symmetry = is_symmetry)
556  globalr4(1-xhalo:0, 1:ny,:) = globalr4(nx-xhalo+1:nx, 1:ny,:)
557  globalr4(nx+1:nx+xhalo, 1:ny,:) = globalr4(1:xhalo, 1:ny,:)

559  case( 'Masked', 'Masked symmetry' )
560  ! with fold and cyclic, assign to npes+1 and mask out the top-rightdomain
561  call mpp_define_layout( (/1,nx,1,ny/), npes+1, layout )
562  allocate( maskmap(layout(1),layout(2)) )
563  maskmap(:,:) = .true.; maskmap(layout(1),layout(2)) = .false.
564  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
565  shalo=shalo, nhalo=nhalo, xflags=cyclic_global_domain, yflags=fold_north_edge, &
566  maskmap=maskmap, name=domain_type, symmetry = is_symmetry )
567  deallocate(maskmap)

569  ! this logic assumes top-right, in an even division
570  if ( mod(nx,layout(1)).NE.0 .OR. mod(ny,layout(2)).NE.0) call mpp_error( fatal, &

573  call fill_folded_north_halo(globalr4, 0, 0, 0, 0, 1)
574  case default
575  call mpp_error( fatal, 'test_halo_update_r4: '//domain_type//' is not a valid test.')
576  end select

578  call mpp_get_compute_domain( domain, is, ie, js, je )
579  call mpp_get_data_domain( domain, isd, ied, jsd, jed )
580  allocate(xr4(isd:ied,jsd:jed,nz) )
581  allocate(x1r4(isd:ied,jsd:jed,nz) )
582  allocate(x2r4(isd:ied,jsd:jed,nz) )
583  allocate(x3r4(isd:ied,jsd:jed,nz) )
584  allocate(x4r4(isd:ied,jsd:jed,nz) )

589  id = mpp_clock_id( domain_type, flags=mpp_clock_sync+mpp_clock_detailed )
590  call mpp_clock_begin(id)
591  call mpp_update_domains( xr4, domain )
592  call mpp_clock_end (id)
593  call compare_checksums( xr4, globalr4(isd:ied,jsd:jed,:), domain_type )
594  ! update part of the halo region
595  id = mpp_clock_id( domain_type//' partial', flags=mpp_clock_sync+mpp_clock_detailed )
596  call mpp_clock_begin(id)
597  call mpp_update_domains( x1r4, domain, nupdate+eupdate, complete=.false. )
598  call mpp_update_domains( x2r4, domain, nupdate+eupdate, complete=.false. )
599  call mpp_update_domains( x3r4, domain, nupdate+eupdate, complete=.false. )
600  call mpp_update_domains( x4r4, domain, nupdate+eupdate, complete=.true. )
601  call mpp_clock_end (id)
602  call compare_checksums( x1r4(is:ied,js:jed,:), globalr4(is:ied,js:jed,:), domain_type//' partial x1r4' )
603  call compare_checksums( x2r4(is:ied,js:jed,:), globalr4(is:ied,js:jed,:), domain_type//' partial x2r4' )
604  call compare_checksums( x3r4(is:ied,js:jed,:), globalr4(is:ied,js:jed,:), domain_type//' partial x3r4' )
605  call compare_checksums( x4r4(is:ied,js:jed,:), globalr4(is:ied,js:jed,:), domain_type//' partial x4r4' )
606 
607  !--- test vector update for FOLDED and MASKED case.
608  if(domain_type == 'Simple' .or. domain_type == 'Simple symmetry' .or. domain_type == 'Cyclic' .or. domain_type == 'Cyclic symmetry') then
609  deallocate(xr4,x1r4,x2r4,x3r4,x4r4)
610  return
611  end if
612 
613  !------------------------------------------------------------------
614  ! vector update : BGRID_NE
615  !------------------------------------------------------------------
616  shift = 0
617  if (is_symmetry) then
618  shift = 1
619  deallocate(globalr4)
620  allocate(globalr4(1-whalo:nx+ehalo+shift,1-shalo:ny+nhalo+shift,nz) )
621  globalr4 = 0.0
622  do k = 1,nz
623  do j = 1,ny+1
624  do i = 1,nx+1
625  globalr4(i,j,k) = k + i*1e-3 + j*1e-6
626  end do
627  end do
628  end do
629  if (domain_type == 'Masked symmetry') then
630  globalr4(nx-nx/layout(1)+1:nx+1,ny-ny/layout(2)+1:ny+1,:) = 0
631  endif
632  deallocate(xr4, x1r4, x2r4, x3r4, x4r4)
633  allocate( xr4(isd:ied+1,jsd:jed+1,nz) )
634  allocate( x1r4(isd:ied+1,jsd:jed+1,nz) )
635  allocate( x2r4(isd:ied+1,jsd:jed+1,nz) )
636  allocate( x3r4(isd:ied+1,jsd:jed+1,nz) )
637  allocate( x4r4(isd:ied+1,jsd:jed+1,nz) )
638  endif
639 
640  folded_south = .false.
641  folded_west = .false.
642  folded_east = .false.
643  select case (domain_type)
644  case ('Folded-north', 'Masked')
645  ! fill in folded north edge, cyclic east and west edge
646  call fill_folded_north_halo(globalr4, 1, 1, 0, 0, -1)
647  case ('Folded xy_halo')

652  globalr4(nx+xhalo, ny+1:ny+yhalo,:) = -globalr4(nx-xhalo, ny-1:ny-yhalo:-1,:)
653  case ('Folded-north symmetry', 'Masked symmetry' )
654  call fill_folded_north_halo(globalr4, 1, 1, 1, 1, -1)
655  case ('Folded-south symmetry' )
656  folded_south = .true.
657  call fill_folded_south_halo(globalr4, 1, 1, 1, 1, -1)
658  case ('Folded-west symmetry' )
659  folded_west = .true.
660  call fill_folded_west_halo(globalr4, 1, 1, 1, 1, -1)
661  case ('Folded-east symmetry' )
662  folded_east = .true.
663  call fill_folded_east_halo(globalr4, 1, 1, 1, 1, -1)
664  case default
665  call mpp_error( fatal, 'TEST_MPP_DOMAINS: no such test: '//domain_type )
666  end select
667 
668  xr4 = 0.0
669  xr4(is:ie+shift,js:je+shift,:) = globalr4(is:ie+shift,js:je+shift,:)
