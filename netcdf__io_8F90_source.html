522 function netcdf_file_open(fileobj, path, mode, nc_format, pelist, is_restart, dont_add_res_to_filename) &
523  result(success)
524 
525  class(fmsnetcdffile_t), intent(inout) :: fileobj !< File object.
526  character(len=*), intent(in) :: path !< File path.
527  character(len=*), intent(in) :: mode !< File mode. Allowed values are:
528  !! "read", "append", "write", or
529  !! "overwrite".
530  character(len=*), intent(in), optional :: nc_format !< Netcdf format that
531  !! new files are written
532  !! as. Allowed values
533  !! are: "64bit", "classic",
534  !! or "netcdf4". Defaults to
535  !! "64bit". This overwrites
536  !! the value set in the fms2io
537  !! namelist
538  integer, dimension(:), intent(in), optional :: pelist !< List of ranks associated
539  !! with this file. If not
540  !! provided, only the current
541  !! rank will be able to
542  !! act on the file.
543  logical, intent(in), optional :: is_restart !< Flag telling if this file
544  !! is a restart file. Defaults
545  !! to false.
546  logical, intent(in), optional :: dont_add_res_to_filename !< Flag indicating not to add
547  !! ".res" to the filename
548  logical :: success
549 
550  integer :: nc_format_param
551  integer :: err
552  character(len=256) :: buf !< Filename with .res in the filename if it is a restart
553  character(len=256) :: buf2 !< Filename with the filename appendix if there is one
554  logical :: is_res
555  logical :: dont_add_res !< flag indicated to not add ".res" to the filename
556 
557  if (allocated(fileobj%is_open)) then
558  if (fileobj%is_open) then
559  success = .true.
560  return
561  endif
562  endif
563  !< Only add ".res" to the file path if is_restart is set to true
564  !! and dont_add_res_to_filename is set to false.
565  is_res = .false.
566  if (present(is_restart)) then
567  is_res = is_restart
568  endif
569  dont_add_res = .false.
570  if (present(dont_add_res_to_filename)) then
571  dont_add_res = dont_add_res_to_filename
572  endif
573 
574  if (is_res .and. .not. dont_add_res) then
575  call restart_filepath_mangle(buf, trim(path))
576  else
577  call string_copy(buf, trim(path))
578  endif
579
