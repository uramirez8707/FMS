112  !> @brief This statistics is for bilinear interpolation and spherical regrid.
113  subroutine stats ( dat, low, high, avg, miss, missing_value, mask )
114  real, intent(in) :: dat(:,:)
115  real, intent(out) :: low, high, avg
116  integer, intent(out) :: miss
117  real, intent(in), optional :: missing_value
118  real, intent(in), optional :: mask(:,:)
119 
120  real :: dsum, npts, buffer_real(3)
121  integer :: pe, root_pe, npes, p, buffer_int(2)
122 
123  pe = mpp_pe()
124  root_pe = mpp_root_pe()
125  npes = mpp_npes()
126 
127  dsum = 0.0
128  miss = 0
129 
130  if (present(missing_value)) then
131  miss = count(dat(:,:) == missing_value)
132  low = minval(dat(:,:), dat(:,:) /= missing_value)
133  high = maxval(dat(:,:), dat(:,:) /= missing_value)
134  dsum = sum(dat(:,:), dat(:,:) /= missing_value)
135  else if(present(mask)) then
136  miss = count(mask(:,:) <= 0.5)
137  low = minval(dat(:,:),mask=mask(:,:) > 0.5)
138  high = maxval(dat(:,:),mask=mask(:,:) > 0.5)
139  dsum = sum(dat(:,:), mask=mask(:,:) > 0.5)
140  else
141  miss = 0
142  low = minval(dat(:,:))
143  high = maxval(dat(:,:))
144  dsum = sum(dat(:,:))
145  endif
146  avg = 0.0
147 
148  npts = size(dat(:,:)) - miss
149  if(pe == root_pe) then
150  do p = 1, npes - 1 ! root_pe receive data from other pe
151  ! Force use of "scalar", integer pointer mpp interface
152  call mpp_recv(buffer_real(1),glen=3, from_pe=p+root_pe, tag=comm_tag_1)
153  dsum = dsum + buffer_real(1)
154  low = min(low, buffer_real(2))
155  high = max(high, buffer_real(3))
156  call mpp_recv(buffer_int(1), glen=2, from_pe=p+root_pe, tag=comm_tag_2)
157  miss = miss + buffer_int(1)
158  npts = npts + buffer_int(2)
159  enddo
160  if(npts == 0.) then
161  print*, 'Warning: no points is valid'
162  else
163  avg = dsum/real(npts)
164  endif
165  else ! other pe send data to the root_pe.
166  buffer_real(1) = dsum
167  buffer_real(2) = low
168  buffer_real(3) = high
169  ! Force use of "scalar", integer pointer mpp interface
170  call mpp_send(buffer_real(1),plen=3,to_pe=root_pe, tag=comm_tag_1)
171  buffer_int(1) = miss
172  buffer_int(2) = npts
173  call mpp_send(buffer_int(1), plen=2, to_pe=root_pe, tag=comm_tag_2)
174  endif
175 
176  call mpp_sync_self()
177 
178  return
179 
180  end subroutine stats
