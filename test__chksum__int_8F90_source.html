21 !> @brief Test mpp_chksum with mixed precision integers
22 !> @description Tests mpp_chksum with 8 and 4 byte integer arrays with
23 !> normal and distributed checksums
24 program test_chksum_int
25 
26  use platform_mod
27  use mpp_mod, only : mpp_init, mpp_exit, mpp_pe, mpp_npes, mpp_root_pe, stdout
28  use mpp_mod, only : mpp_set_stack_size, mpp_sync, mpp_init_test_init_true_only
29  use mpp_mod, only : mpp_transmit, mpp_chksum, all_pes
30  use mpp_mod, only : mpp_error, fatal, mpp_sync_self, note
31  use mpp_io_mod, only: mpp_io_init
32 
33  implicit none
34 
35  integer :: pe, npes, root, out_unit, ierr
36  integer(i8_kind), allocatable :: data8(:), distData(:),temp(:)
37  integer(i8_kind) :: res4, res8, resDist
38  integer(i4_kind), allocatable :: data4(:)
39  real, allocatable :: rands(:)
40  integer :: i, length
41 
42  call mpp_init(mpp_init_test_init_true_only)
43  call mpp_io_init()
44  call mpp_set_stack_size(3145746)
45  pe = mpp_pe()
46  npes = mpp_npes()
47  root = mpp_root_pe()
48  out_unit = stdout()
49 
50  !> generate random arrays
51  length = 1024
52  allocate(rands(length), data8(length), data4(length), distdata(length))
53  call random_number(rands)
54  do i = 1, length
55  data8(i) = rands(i) * huge(data4(1))
56  data4(i) = rands(i) * huge(data4(1))
57  distdata(i) = rands(i) * huge(distdata(1))
58  end do
59  !>test mixed precision int checksums
60  res4 = mpp_chksum(data4)
61  res8 = mpp_chksum(data8)
62  if(res4.NE.res8) then
63  call mpp_error(fatal, 'Test mpp_chksum_int: mixed precision checksums do not match')
64  else
65  call mpp_error(note, 'Test mpp_chksum_int: mixed precision checksums match')
66  endif
67  !>test distributed int checksums
68  call mpp_sync()
69  call mpp_transmit( put_data=distdata(1), plen=length, to_pe=all_pes, &
70  get_data=distdata(1),glen=length, from_pe=root)
71  call mpp_sync_self()
72  allocate(temp(length/npes))
73  temp = distdata( pe*(length/npes)+1 : (pe+1)*(length/npes))!> distribute data for pelist
74  resdist = mpp_chksum(distdata(1:length), (/pe/))
75  if(resdist.NE.mpp_chksum(temp)) then
76  call mpp_error(fatal, 'Test mpp_chksum_int: distributed checksums do not match')
77  else
78  call mpp_error(note, 'Test mpp_chksum_int: distributed checksums match')
79  endif
80  deallocate(rands, data8, data4, distdata, temp)
81 
82  call mpi_finalize(ierr)
83 
84 end program test_chksum_int
