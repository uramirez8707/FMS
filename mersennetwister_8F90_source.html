103  ! Period parameters
104  integer, parameter :: blocksize = 624, &
105  m = 397, &
106  matrix_a = -1727483681, & !< constant vector a (0x9908b0dfUL)
107  umask = -2147483648_8, & !< most significant w-r bits (0x80000000UL)
108  lmask = 2147483647 !< least significant r bits (0x7fffffffUL)

137  function mixbits(u, v)
138  integer, intent( in) :: u, v
139  integer :: mixbits
140 
141  mixbits = ior(iand(u, umask), iand(v, lmask))
142  end function mixbits
143  ! ---------------------------
144  function twist(u, v)
145  integer, intent( in) :: u, v
146  integer :: twist
147 
148  ! Local variable
149  integer, parameter, dimension(0:1) :: t_matrix = (/ 0, matrix_a /)
150 
151  twist = ieor(ishft(mixbits(u, v), -1), t_matrix(iand(v, 1)))
152  twist = ieor(ishft(mixbits(u, v), -1), t_matrix(iand(v, 1)))
153  end function twist
154  ! ---------------------------
155  subroutine nextstate(twister)
156  type(randomNumberSequence), intent(inout) :: twister
157 
158  ! Local variables
159  integer :: k
160 
161  do k = 0, blocksize - m - 1
162  twister%state(k) = ieor(twister%state(k + m), &
163  twist(twister%state(k), twister%state(k + 1)))
164  end do
165  do k = blocksize - m, blocksize - 2
166  twister%state(k) = ieor(twister%state(k + m - blocksize), &
167  twist(twister%state(k), twister%state(k + 1)))
168  end do
169  twister%state(blocksize - 1) = ieor(twister%state(m - 1), &
170  twist(twister%state(blocksize - 1), twister%state(0)))
171  twister%currentElement = 0
172 
173  end subroutine nextstate
174  ! ---------------------------
175  elemental function temper(y)
176  integer, intent(in) :: y
177  integer :: temper
178 
179  integer :: x
180 
181  ! Tempering
182  x = ieor(y, ishft(y, -11))
183  x = ieor(x, iand(ishft(x, 7), tmaskb))
184  x = ieor(x, iand(ishft(x, 15), tmaskc))
185  temper = ieor(x, ishft(x, -18))
186  end function temper

190  function initialize_scalar(seed) result(twister)
191  integer, intent(in ) :: seed
192  type(randomNumberSequence) :: twister
193 
194  integer :: i
195  ! See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. In the previous versions,
196  ! MSBs of the seed affect only MSBs of the array state[].
197  ! 2002/01/09 modified by Makoto Matsumoto
198 
199  twister%state(0) = iand(seed, -1)
200  do i = 1, blocksize - 1 ! ubound(twister%state)
201  twister%state(i) = 1812433253 * ieor(twister%state(i-1), &
202  ishft(twister%state(i-1), -30)) + i
203  twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
204  end do
205  twister%currentElement = blocksize
206  end function initialize_scalar
207  ! -------------------------------------------------------------
208  function initialize_vector(seed) result(twister)
209  integer, dimension(0:), intent(in) :: seed
210  type(randomNumberSequence) :: twister
211 
212  integer :: i, j, k, nFirstLoop, nWraps
213 
214  nwraps = 0
215  twister = initialize_scalar(19650218)
216 
217  nfirstloop = max(blocksize, size(seed))
218  do k = 1, nfirstloop
219  i = mod(k + nwraps, blocksize)
220  j = mod(k - 1, size(seed))
221  if(i == 0) then
222  twister%state(i) = twister%state(blocksize - 1)
223  twister%state(1) = ieor(twister%state(1), &
224  ieor(twister%state(1-1), &
225  ishft(twister%state(1-1), -30)) * 1664525) + &
226  seed(j) + j ! Non-linear
227  twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
228  nwraps = nwraps + 1
229  else
230  twister%state(i) = ieor(twister%state(i), &
231  ieor(twister%state(i-1), &
232  ishft(twister%state(i-1), -30)) * 1664525) + &
233  seed(j) + j ! Non-linear
234  twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
235  end if
236  end do
237 
238  !
239  ! Walk through the state array, beginning where we left off in the block above
240  !
241  do i = mod(nfirstloop, blocksize) + nwraps + 1, blocksize - 1
242  twister%state(i) = ieor(twister%state(i), &
243  ieor(twister%state(i-1), &
244  ishft(twister%state(i-1), -30)) * 1566083941) - i ! Non-linear
245  twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
246  end do
247 
248  twister%state(0) = twister%state(blocksize - 1)
249 
250  do i = 1, mod(nfirstloop, blocksize) + nwraps
251  twister%state(i) = ieor(twister%state(i), &
252  ieor(twister%state(i-1), &
253  ishft(twister%state(i-1), -30)) * 1566083941) - i ! Non-linear
254  twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
255  end do
256 
257  twister%state(0) = umask
258  twister%currentElement = blocksize
259 
260  end function initialize_vector

264  !> Generate a random integer on the interval [0,0xffffffff]
265  !!
266  !> Equivalent to genrand_int32 in the C code.
267  !! Fortran doesn't have a type that's unsigned like C does,
268  !! so this is integers in the range -2**31 - 2**31
269  !! All functions for getting random numbers call this one,
270  !! then manipulate the result
271  function getrandomint(twister)
272  type(randomnumbersequence), intent(inout) :: twister
273  integer :: getrandomint
274 
275  if(twister%currentElement >= blocksize) call nextstate(twister)
276 
277  getrandomint = temper(twister%state(twister%currentElement))
278  twister%currentElement = twister%currentElement + 1
279 
280  end function getrandomint
281  ! --------------------

296  ! --------------------
297  !> Generate a random number on [0,1]
298  !! Equivalent to genrand_real1 in the C code.
299  !! The result is stored as double precision but has 32 bit resolution
300  function getrandomreal(twister)
301  type(randomnumbersequence), intent(inout) :: twister
302  double precision :: getrandomreal
303 
304  integer :: localint
305 
306  localint = getrandomint(twister)
307  if(localint < 0) then
308  getrandomreal = dble(localint + 2.0d0**32)/(2.0d0**32 - 1.0d0)
309  else
310  getrandomreal = dble(localint )/(2.0d0**32 - 1.0d0)
311  end if
312  end function getrandomreal
