19 !> @defgroup topography_mod topography_mod
20 !> @ingroup topography
21 !> @brief Routines for creating land surface topography fields and land-water masks
22 !! for latitude-longitude grids.
23 !> @author Bruce Wyman
24 !!
25 !! This module generates realistic mountains and land-water masks
26 !! on a specified latitude-longitude grid by interpolating from the
27 !! 1/6 degree Navy mean topography and percent water data sets.
28 !! The fields that can be generated are mean and standard deviation
29 !! of topography within the specified grid boxes; and land-ocean (or
30 !! water) mask and land-ocean (or water) fractional area.
31 !!
32 !! The interpolation scheme conserves the area-weighted average
33 !! of the input data by using module horiz_interp.
34 !!
35 !! The interfaces get_gaussian_topog and gaussian_topog_init are documented
36 !! in @ref gaussian_topog_mod
37

203 !-----------------------------------------------------------------------
204 ! --- resolution of the topography data set ---
205 ! <DATASET NAME="">
206 ! This module uses the 1/6 degree U.S. Navy mean topography
207 ! and percent water data sets.
208 !
209 ! These data sets have been re-formatted to separate 32-bit IEEE files.
210 ! The names of these files is specified by the <LINK SRC="#NAMELIST">namelist</LINK> input.
211 !
212 !The format for both files is as follows:
213 ! <PRE>
214 ! record = 1 nlon, nlat
215 ! record = 2 blon, blat
216 ! record = 3 data
217 ! </PRE>
218 !where:
219 ! <PRE>
220 ! nlon, nlat = The number of longitude and latitude points
221 ! in the horizontal grid. For the 1/6 degree
222 ! data sets this is 2160 x 1080. [integer]
223 ! blon, blat = The longitude and latitude grid box boundaries in degrees.
224 ! [real :: blon(nlon+1), blat(nlat+1)]
225 !
226 ! data = The topography or percent water data.
227 ! [real :: data(nlon,nlat)]
228 ! </PRE>
229 ! </DATASET>
230  integer :: ipts, jpts
231  integer, parameter :: compute_stdev = 123 ! use this flag to
232  ! compute st dev

260  !> @brief Returns a "realistic" mean surface height field.
261  !!
262  !> Returns realistic mountains on a latitude-longtude grid.
263  !! The returned field is the mean topography for the given grid boxes.
264  !! Computed using a conserving area-weighted interpolation.
265  !! The current input data set is the 1/6 degree Navy mean topography.
266  !!
267  !! @returns A logical value of true is returned if the surface height field was successfully
268  !! created. A value of false may be returned if the input topography data set was not readable.
269  !!
270  !! @throws FATAL, shape(zmean) is not equal to (/size(blon)-1,size(blat)-1/)
271  !! Check the input grid size and output field size.
272  function get_topog_mean_1d(blon, blat, zmean)
273 
274  real, intent(in), dimension(:) :: blon !< Longitude (radians) at grid box boundaries
275  real, intent(in), dimension(:) :: blat !< Latitude (radians) at grid box boundaries
276  real, intent(out), dimension(:,:) :: zmean !< Mean surface height(meters). Size must be
277  !! size(blon)-1 by size(blat)-1
278  logical :: get_topog_mean_1d
279 
280 !-----------------------------------------------------------------------
281  if (.not. module_is_initialized) call topography_init()
282 
283  if ( any(shape(zmean(:,:)) /= (/size(blon(:))-1,size(blat(:))-1/)) ) &
284  call error_mesg('get_topog_mean_1d','shape(zmean) is not&
285  & equal to (/size(blon)-1,size(blat)-1/))', fatal)
286 
287  get_topog_mean_1d = open_topog_file()
288 
289  if ( get_topog_mean_1d ) call interp_topog_1d ( blon, blat, zmean)
290 
291 !-----------------------------------------------------------------------
292 
293  end function get_topog_mean_1d

803  subroutine determine_ocean_points ( pctwater )
804  real, intent(inout) :: pctwater(:,:)
805  logical :: ocean(size(pctwater,1),size(pctwater,2))
806  integer :: i, j, m, n, im, ip, jm, jp, new
807 
808  real :: ocean_pct_crit = .500
809 
810  ! resolution of the grid
811  m = size(pctwater,1)
812  n = size(pctwater,2)
813 
814  ! the 1/6 degree navy percent water data set
815  ! designates ocean grid boxes as 100 percent water
816  ! all other grid boxes have <= 99 percent water
817 
818  ! set a mask for ocean grid boxes
819  ocean = (pctwater > .999)
820  new = count(ocean)
821 
822  ! set land grid boxes that have sufficient amount of water
823  ! to ocean grid boxes when they are adjacent to ocean points
824  ! iterate until there are no new ocean points
825  do
826  if (new == 0) exit
827  new = 0
828 
829  do j = 1, n
830  do i = 1, m
831  if (.not.ocean(i,j) .and. pctwater(i,j) > ocean_pct_crit) then
832  im = i-1; ip = i+1; jm = j-1; jp = j+1
833  if (im == 0) im = m
834  if (ip == m+1) ip = 1
835  if (jm == 0) jm = 1
836  if (jp == n+1) jp = n
837  ! check the 8 grid boxes that surround this grid box
838  if (ocean(im,j ) .or. ocean(ip,j ) .or. ocean(i ,jm) .or. ocean(i ,jp) .or. &
839  ocean(im,jm) .or. ocean(ip,jm) .or. ocean(ip,jp) .or. ocean(im,jp)) then
840  ocean(i,j) = .true.
841  new = new + 1
842  endif
843  endif
844  enddo
845  enddo
846  !print *, 'new=',new
847 
848  enddo
849 
850  ! final step is to elimate water percentage if land
851  where (.not.ocean) pctwater = 0.
852 
853  end subroutine determine_ocean_points
