47  subroutine mpp_memuse_begin
48  if(memuse_started) then

56  !#######################################################################
57  !> End the memory collection, and report on total memory used during the
58  !! execution of the model run.
59  subroutine mpp_memuse_end(text, unit)
60  character(len=*), intent(in) :: text !< Text to include in memory use statement
61  integer, intent(in), optional :: unit !< Fortran unit number where memory report should go.
62  !! Default is stderr.
63  real :: m, mmin, mmax, mavg, mstd, end_memuse
64  integer :: mu
65 
66  if(.NOT.memuse_started) then
67  call mpp_error(fatal, "mpp_memutils_mod: mpp_memuse_begin must be called before calling mpp_memuse_being")
68  endif
69  memuse_started = .false.
70 
71  call mpp_mem_dump(end_memuse)
72 
73  mu = stderr(); if( PRESENT(unit) )mu = unit
74  m = end_memuse - begin_memuse
75  mmin = m; call mpp_min(mmin)
76  mmax = m; call mpp_max(mmax)
77  mavg = m; call mpp_sum(mavg); mavg = mavg/mpp_npes()
78  mstd = (m-mavg)**2; call mpp_sum(mstd); mstd = sqrt( mstd/mpp_npes() )
79  if( mpp_pe().EQ.mpp_root_pe() )write( mu,'(a64,4es11.3)' ) &
80  'Memory(MB) used in '//trim(text)//'=', mmin, mmax, mstd, mavg
81 
82  return
83  end subroutine mpp_memuse_end
