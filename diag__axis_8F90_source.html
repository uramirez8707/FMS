106  !> @brief Initialize the axis, and return the axis ID.
107  !!
108  !> <TT>diag_axis_init</TT> initializes an axis and returns the axis ID that
109  !! is to be used with <TT>register_diag_field</TT>. This function also
110  !! increments the axis counter and fills in the axes
111  !!
112  !! @return integer axis ID
113  INTEGER FUNCTION diag_axis_init(name, DATA, units, cart_name, long_name, direction,&
114  & set_name, edges, Domain, Domain2, DomainU, aux, req, tile_count, domain_position )
115  CHARACTER(len=*), INTENT(in) :: name !< Short name for axis
116  REAL, DIMENSION(:), INTENT(in) :: data !< Array of coordinate values
117  CHARACTER(len=*), INTENT(in) :: units !< Units for the axis
118  CHARACTER(len=*), INTENT(in) :: cart_name !< Cartesian axis ("X", "Y", "Z", "T")
119  CHARACTER(len=*), INTENT(in), OPTIONAL :: long_name !< Long name for the axis.
120  CHARACTER(len=*), INTENT(in), OPTIONAL :: set_name
121  INTEGER, INTENT(in), OPTIONAL :: direction !< Indicates the direction of the axis
122  INTEGER, INTENT(in), OPTIONAL :: edges !< Axis ID for the previously defined "edges axis"
123  TYPE(domain1d), INTENT(in), OPTIONAL :: domain
124  TYPE(domain2d), INTENT(in), OPTIONAL :: domain2
125  TYPE(domainug), INTENT(in), OPTIONAL :: domainu
126  CHARACTER(len=*), INTENT(in), OPTIONAL :: aux !< Auxiliary name, can only be <TT>geolon_t</TT> or <TT>geolat_t</TT>
127  CHARACTER(len=*), INTENT(in), OPTIONAL :: req !< Required field names.
128  INTEGER, INTENT(in), OPTIONAL :: tile_count
129  INTEGER, INTENT(in), OPTIONAL :: domain_position
130 
131 
132  TYPE(domain1d) :: domain_x, domain_y
133  INTEGER :: ierr, axlen
134  INTEGER :: i, set, tile
135  INTEGER :: isc, iec, isg, ieg
136  CHARACTER(len=128) :: emsg
137 
138  IF ( .NOT.module_is_initialized ) THEN
139  CALL write_version_number("DIAG_AXIS_MOD", version)
140  ENDIF
141 
142  IF ( PRESENT(tile_count)) THEN
143  tile = tile_count
144  ELSE
145  tile = 1
146  END IF
147 
148  ! Allocate the axes
149  IF (.NOT. ALLOCATED(axis_sets)) ALLOCATE(axis_sets(max_num_axis_sets))
150  IF (.NOT. ALLOCATED(axes)) ALLOCATE(axes(max_axes))
151  IF (.NOT. ALLOCATED(num_subaxes)) THEN
152  ALLOCATE(num_subaxes(max_axes))
153  num_subaxes = 0
154  END IF
155 
156  !---- is there an axis set? ----
157  IF ( PRESENT(set_name) ) THEN
158  set = get_axis_set_num(set_name)
159  !---- add new set name ----
160  IF (set == 0) THEN
161  num_axis_sets = num_axis_sets + 1
162  IF ( num_axis_sets > max_num_axis_sets ) THEN
163  WRITE (emsg, fmt='("num_axis_sets (",I2,") exceeds max_num_axis_sets (",I2,"). ")')&
164  & num_axis_sets, max_num_axis_sets
165  ! <ERROR STATUS="FATAL">
166  ! num_axis_sets (<num_axis_sets>) exceeds max_num_axis_sets(<num_axis_sets>).
167  ! Increase max_num_axis_sets via diag_manager_nml.
168  ! </ERROR>
169  CALL error_mesg('diag_axis_mod::diag_axis_init',&
170  & trim(emsg)//' Increase max_num_axis_sets via diag_manager_nml.', fatal)
171  END IF
172  set = num_axis_sets
173  axis_sets(set) = set_name
174  END IF
175  ELSE
176  set = 0
177  END IF
178 
179  !---- see if axis already exists --
180  ! if this is time axis, return the ID of a previously defined
181  ! if this is spatial axis, FATAL error
182  DO i = 1, num_def_axes
183  IF ( trim(name) == axes(i)%name ) THEN
184  IF ( trim(name) == 'Stations' .OR. trim(name) == 'Levels') THEN
185  diag_axis_init = i
186  RETURN
187  ELSE IF ( set == axes(i)%set ) THEN
188  IF ( trim(lowercase(name)) == 'time' .OR.&
189  & trim(lowercase(cart_name)) == 't' .OR.&
190  & trim(lowercase(name)) == 'nv' .OR.&
191  & trim(lowercase(cart_name)) == 'n' ) THEN
192  diag_axis_init = i
193  RETURN
194  ELSE IF ( (lowercase(cart_name) /= 'x' .AND. lowercase(cart_name) /= 'y')&
195  & .OR. tile /= axes(i)%tile_count) THEN
196  ! <ERROR STATUS="FATAL">axis_name <NAME> and axis_set already exist.</ERROR>
197  CALL error_mesg('diag_axis_mod::diag_axis_init',&
198  & 'axis_name '//trim(name)//' and axis_set already exist.', fatal)
199  END IF
200  END IF
201  END IF
202  END DO
203 
204  !---- register axis ----
205  num_def_axes = num_def_axes + 1
206  ! <ERROR STATUS="FATAL">max_axes exceeded, increase it via diag_manager_nml</ERROR>
207  IF (num_def_axes > max_axes) CALL error_mesg ('diag_axis_mod::diag_axis_init',&
208  & 'max_axes exceeded, increase via diag_manager_nml', fatal)
209  diag_axis_init = num_def_axes
210 
211  !---- check and then save cart_name name ----
212  IF ( trim(uppercase(cart_name)) == 'X' .OR.&
213  & trim(uppercase(cart_name)) == 'Y' .OR.&
214  & trim(uppercase(cart_name)) == 'Z' .OR.&
215  & trim(uppercase(cart_name)) == 'T' .OR.&
216  & trim(uppercase(cart_name)) == 'U' .OR.&
217  & trim(uppercase(cart_name)) == 'N' ) THEN
218  axes(diag_axis_init)%cart_name = trim(uppercase(cart_name))
219  ELSE
220  ! <ERROR STATUS="FATAL">Invalid cart_name name.</ERROR>
221  CALL error_mesg('diag_axis_mod::diag_axis_init', 'Invalid cart_name name. '//trim(uppercase(cart_name)), fatal)
222  END IF
223 
224  !---- allocate storage for coordinate values of axis ----
225  IF ( axes(diag_axis_init)%cart_name == 'T' ) THEN
226  axlen = 0
227  ELSE
228  axlen = SIZE(DATA(:))
229  END IF
230  ALLOCATE ( axes(diag_axis_init)%data(1:axlen) )
231 
232  ! Initialize Axes(diag_axis_init)
233  axes(diag_axis_init)%name = trim(name)
234  axes(diag_axis_init)%data = DATA(1:axlen)
235  axes(diag_axis_init)%units = units
236  axes(diag_axis_init)%length = axlen
237  axes(diag_axis_init)%set = set
238  ! start and end are used in subaxes information only
239  axes(diag_axis_init)%start = -1
240  axes(diag_axis_init)%end = -1
241  axes(diag_axis_init)%subaxis_name = ""
242  axes(diag_axis_init)%shift = 0
243  axes(diag_axis_init)%num_attributes = 0
244 
245  IF ( PRESENT(long_name) ) THEN
246  axes(diag_axis_init)%long_name = long_name
247  ELSE
248  axes(diag_axis_init)%long_name = name
249  END IF
250 
251  IF ( PRESENT(aux) ) THEN
252  axes(diag_axis_init)%aux = trim(aux)
253  ELSE
254  axes(diag_axis_init)%aux = 'none'
255  END IF
256 
257  IF ( PRESENT(req) ) THEN
258  axes(diag_axis_init)%req = trim(req)
259  ELSE
260  axes(diag_axis_init)%req = 'none'
261  END IF
262  IF ( PRESENT(domain_position) ) THEN
263  if (domain_position == north .or. domain_position == east .or. domain_position == center) then
264  axes(diag_axis_init)%domain_position = domain_position
265  else
266  CALL error_mesg('diag_axis_mod::diag_axis_init', "Position must be NORTH, EAST, or CENTER" ,&
267  fatal)
268  endif
269  ELSE
270  axes(diag_axis_init)%domain_position = center
271  END IF
272 
273  !---- axis direction (-1, 0, or +1) ----
274  IF ( PRESENT(direction) )THEN
275  IF ( abs(direction) /= 1 .AND. direction /= 0 )&
276  ! <ERROR STATUS="FATAL">direction must be 0, +1, or -1</ERROR>
277  & CALL error_mesg('diag_axis_mod::diag_axis_init', 'direction must be 0, +1 or -1', fatal)
278  axes(diag_axis_init)%direction = direction
279  ELSE
280  axes(diag_axis_init)%direction = 0
281  END IF
282 
283  !---- Handle the DomainU check
284  IF (present(domainu) .AND. (PRESENT(domain2) .OR. PRESENT(domain)) ) THEN
285  ! <ERROR STATUS="FATAL">Presence of DomainU and another Domain at the same time is prohibited</ERROR>
286  CALL error_mesg('diag_axis_mod::diag_axis_init',&
287  & 'Presence of DomainU and another Domain at the same time is prohibited', fatal)
288  !---- domain2d type ----
289  ELSE IF ( PRESENT(domain2) .AND. PRESENT(domain)) THEN
290  ! <ERROR STATUS="FATAL">Presence of both Domain and Domain2 at the same time is prohibited</ERROR>
291  CALL error_mesg('diag_axis_mod::diag_axis_init',&
292  & 'Presence of both Domain and Domain2 at the same time is prohibited', fatal)
293  ELSE IF ( PRESENT(domain2) .OR. PRESENT(domain)) THEN
294  IF ( axes(diag_axis_init)%cart_name /= 'X' .AND. axes(diag_axis_init)%cart_name /= 'Y') THEN
295  ! <ERROR STATUS="FATAL">Domain must not be present for an axis which is not in the X or Y direction.</ERROR>
296  CALL error_mesg('diag_axis_mod::diag_axis_init',&
297  & 'A Structured Domain must not be present for an axis which is not in the X or Y direction', fatal)
298  END IF
299  ELSE IF (present(domainu) .AND. axes(diag_axis_init)%cart_name /= 'U') THEN
300  CALL error_mesg('diag_axis_mod::diag_axis_init',&
301  & 'In the unstructured domain, the axis cart_name must be U', fatal)
302  END IF
303 
304  axes(diag_axis_init)%tile_count = tile
305 
306  IF ( PRESENT(domain2) ) THEN
307  axes(diag_axis_init)%Domain2 = domain2
308  CALL mpp_get_domain_components(domain2, domain_x, domain_y, tile_count=tile_count)
309  IF ( axes(diag_axis_init)%cart_name == 'X' ) axes(diag_axis_init)%Domain = domain_x
310  IF ( axes(diag_axis_init)%cart_name == 'Y' ) axes(diag_axis_init)%Domain = domain_y
311  axes(diag_axis_init)%DomainUG = null_domainug
312  ELSE IF ( PRESENT(domain)) THEN
313  !---- domain1d type ----
314  axes(diag_axis_init)%Domain2 = null_domain2d ! needed since not 2-D domain
315  axes(diag_axis_init)%Domain = domain
316  axes(diag_axis_init)%DomainUG = null_domainug
317  ELSE IF (present(domainu)) THEN
318  axes(diag_axis_init)%Domain2 = null_domain2d
319  axes(diag_axis_init)%Domain = null_domain1d
320  axes(diag_axis_init)%DomainUG = domainu
321  ELSE
322  axes(diag_axis_init)%Domain2 = null_domain2d
323  axes(diag_axis_init)%Domain = null_domain1d
324  axes(diag_axis_init)%DomainUG = null_domainug
325  END IF
326 
327  !--- set up the shift value for x-y axis
328  IF ( axes(diag_axis_init)%Domain .NE. null_domain1d ) THEN
329  CALL mpp_get_compute_domain(axes(diag_axis_init)%Domain, isc, iec)
330  CALL mpp_get_global_domain(axes(diag_axis_init)%Domain, isg, ieg)
331  IF ( axes(diag_axis_init)%length == ieg - isg + 2 ) THEN
332  axes(diag_axis_init)%shift = 1
333  END IF
334  END IF
335 
336  !---- have axis edges been defined ? ----
337  axes(diag_axis_init)%edges = 0
338  IF (PRESENT(edges) ) THEN
339  IF ( edges > 0 .AND. edges < num_def_axes ) THEN
340  ierr=0
341  IF ( axes(edges)%cart_name /= axes(diag_axis_init)%cart_name) ierr=1
342  IF ( axes(edges)%length /= axes(diag_axis_init)%length+1 ) ierr=ierr+2
343  IF ( axes(edges)%set /= axes(diag_axis_init)%set ) ierr=ierr+4
344  IF ( ierr > 0 ) THEN
345  ! <ERROR STATUS="FATAL">Edges axis does not match axis (code <CODE>).</ERROR>
346  WRITE (emsg,'("Edges axis does not match axis (code ",I1,").")') ierr
347  CALL error_mesg('diag_axis_mod::diag_axis_init', emsg, fatal)
348  END IF
349  axes(diag_axis_init)%edges = edges
350  ELSE
351  ! <ERROR STATUS="FATAL">Edges axis is not defined.</ERROR>
352  CALL error_mesg('diag_axis_mod::diag_axis_init', 'Edges axis is not defined', fatal)
353  END IF
354  END IF
355 
356  ! Module is now initialized
357  module_is_initialized = .true.
358 
359  END FUNCTION diag_axis_init

361  !> @brief Create a subaxis on a parent axis.
362  !!
363  !> Given the ID of a parent axis, create a subaxis and fill it with data,
364  !! and return the ID of the corresponding subaxis.
365  !!
366  !! The subaxis is defined on the parent axis from <TT>start_indx</TT>
367  !! to <TT>end_indx</TT>.
368  !!
369  !! @return Integer ID of the corresponding subaxis.
370  INTEGER FUNCTION diag_subaxes_init(axis, subdata, start_indx, end_indx, domain_2d)
371  INTEGER, INTENT(in) :: axis !< ID of the parent axis
372  REAL, DIMENSION(:), INTENT(in) :: subdata !< Data of the subaxis
373  INTEGER, INTENT(in) :: start_indx !< Start index of the subaxis
374  INTEGER, INTENT(in) :: end_indx !< End index of the subaxis
375  TYPE(domain2d), INTENT(in), OPTIONAL :: domain_2d
376 
377  INTEGER :: i, nsub_axis, direction
378  INTEGER :: xbegin, xend, ybegin, yend
379  INTEGER :: ad_xbegin, ad_xend, ad_ybegin, ad_yend
380  CHARACTER(len=128) :: name, nsub_name
381  CHARACTER(len=128) :: units
382  CHARACTER(len=128) :: cart_name
383  CHARACTER(len=128) :: long_name
384  CHARACTER(len=128) :: emsg
385  LOGICAL :: subaxis_set, hasdomain
386 
387  ! there may be more than 1 subaxis on a parent axis, check for redundancy
388  nsub_axis = 0
389  subaxis_set = .false.
390 
391  IF ( PRESENT(domain_2d) ) THEN
392  hasdomain = .true.
393  CALL mpp_get_compute_domain(domain_2d, xbegin, xend, ybegin, yend)
394  ELSE
395  hasdomain = .false.
396  END IF
397  sa_search: DO i = 1, num_subaxes(axis)
398  IF ( start_indx == axes(axis)%start(i) .AND. end_indx == axes(axis)%end(i) ) THEN
399  IF ( hasdomain ) THEN
400  CALL mpp_get_compute_domain(axes(axis)%subaxis_domain2(i), ad_xbegin, ad_xend, ad_ybegin, ad_yend)
401  IF ( .NOT.((xbegin == ad_xbegin .AND. xend == ad_xend) .AND.&
402  & (ybegin == ad_ybegin .AND. yend == ad_yend)) ) THEN
403  cycle sa_search
404  END IF
405  END IF
406  nsub_axis = i
407  subaxis_set = .true. !subaxis already exists
408  name = trim(axes(axis)%subaxis_name(nsub_axis))
409  EXIT sa_search
410  END IF
411  END DO sa_search
412 
413  IF ( nsub_axis == 0 ) THEN ! create new subaxis
414  num_subaxes(axis) = num_subaxes(axis) + 1
415  IF (num_subaxes(axis) > max_subaxes) THEN
416  ! <ERROR STATUS="FATAL">max_subaxes (value <VALUE>) is too small. Consider increasing max_subaxes.</ERROR>
417  WRITE (emsg,'("max_subaxes (value ",I4,") is too small. Consider increasing max_subaxes.")') max_subaxes
418  CALL error_mesg('diag_axis_mod::diag_subaxes_init', emsg, fatal)
419  END IF
420  nsub_axis = num_subaxes(axis)
421  axes(axis)%start(nsub_axis) = start_indx
422  axes(axis)%end(nsub_axis) = end_indx
423  if ( hasdomain ) axes(axis)%subaxis_domain2(nsub_axis) = domain_2d
424  END IF
425 
426  ! Create new name for the subaxis from name of parent axis
427  ! If subaxis already exists, get the index and return
428  IF(subaxis_set) THEN
429  IF ( axes(axis)%set > 0 ) THEN
430  diag_subaxes_init = get_axis_num(name, set_name=trim(axis_sets(axes(axis)%set)))
431  ELSE
432  diag_subaxes_init = get_axis_num(name)
433  END IF
434  ELSE
435  ! get a new index for subaxis
436  !::sdu:: Need a check to allow larger numbers in the index number.
437  WRITE (nsub_name,'(I2.2)') nsub_axis
438  name = trim(axes(axis)%name)//'_sub'//trim(nsub_name)
439  axes(axis)%subaxis_name(nsub_axis) = name
440  long_name = trim(axes(axis)%long_name)
441  units = trim(axes(axis)%units)
442  cart_name = trim(axes(axis)%cart_name)
443  direction = axes(axis)%direction
444  IF (axes(axis)%set > 0) THEN
445  diag_subaxes_init = diag_axis_init(trim(name), subdata, trim(units), trim(cart_name), trim(long_name),&
446  & set_name=trim(axis_sets(axes(axis)%set)), direction=direction, domain2=domain_2d)
447  ELSE
448  diag_subaxes_init = diag_axis_init(trim(name), subdata, trim(units), trim(cart_name), trim(long_name),&
449  & direction=direction, domain2=domain_2d)
450  END IF
451  END IF
452  END FUNCTION diag_subaxes_init

454  SUBROUTINE get_diag_axis(id, name, units, long_name, cart_name,&
455  & direction, edges, Domain, DomainU, DATA, num_attributes, attributes, domain_position)
456  CHARACTER(len=*), INTENT(out) :: name, units, long_name, cart_name
457  INTEGER, INTENT(in) :: id !< Axis ID
458  TYPE(domain1d), INTENT(out) :: domain
459  TYPE(domainug), INTENT(out) :: domainu
460  INTEGER, INTENT(out) :: direction !< Direction of data. (See <TT>@ref diag_axis_init</TT> for a description of
461  !! allowed values)
462  INTEGER, INTENT(out) :: edges !< Axis ID for the previously defined "edges axis".
463  REAL, DIMENSION(:), INTENT(out) :: data !< Array of coordinate values for this axis.
464  INTEGER, INTENT(out), OPTIONAL :: num_attributes
465  TYPE(diag_atttype), ALLOCATABLE, DIMENSION(:), INTENT(out), OPTIONAL :: attributes
466  INTEGER, INTENT(out), OPTIONAL :: domain_position
467 
468  INTEGER :: i, j, istat
469 
470  CALL valid_id_check(id, 'get_diag_axis')
471  name = axes(id)%name
472  units = axes(id)%units
473  long_name = axes(id)%long_name
474  cart_name = axes(id)%cart_name
475  direction = axes(id)%direction
476  edges = axes(id)%edges
477  domain = axes(id)%Domain
478  domainu = axes(id)%DomainUG
479  if (present(domain_position)) domain_position = axes(id)%domain_position
480  IF ( axes(id)%length > SIZE(DATA(:)) ) THEN
481  ! <ERROR STATUS="FATAL">array data is too small.</ERROR>
482  CALL error_mesg('diag_axis_mod::get_diag_axis', 'array data is too small', fatal)
483  ELSE
484  DATA(1:axes(id)%length) = axes(id)%data(1:axes(id)%length)
485  END IF
486  IF ( PRESENT(num_attributes) ) THEN
487  num_attributes = axes(id)%num_attributes
488  END IF
489  IF ( PRESENT(attributes) ) THEN
490  IF ( allocated(axes(id)%attributes) ) THEN
