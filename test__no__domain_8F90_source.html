74 if ( id_var1ds > 0 ) used = send_data(id_var1ds, var1d)

86  if ( id_var1d > 0 ) used = send_data(id_var1d, var1d, time)
87  if ( id_varscalar > 0 ) used = send_data(id_varscalar, varscalar, time)
88 
89 end do
