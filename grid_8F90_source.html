216 !> Return grid cell area for the specified model component and tile
217 subroutine get_grid_cell_area_sg(component, tile, cellarea, domain)
218  character(len=*), intent(in) :: component
219  integer , intent(in) :: tile
220  real , intent(inout) :: cellarea(:,:)
221  type(domain2d) , intent(in), optional :: domain
222 
223  ! local vars
224  integer :: nlon, nlat
225  real, allocatable :: glonb(:,:), glatb(:,:)
226 
227  select case(get_grid_version())
228  case(version_0,version_1)
229  select case(trim(component))
230  case('LND')
231  call read_data(grid_file, 'AREA_LND_CELL', cellarea, &
232  no_domain=.not.present(domain), domain=domain)
233  case('ATM','OCN')
234  call read_data(grid_file, 'AREA_'//trim(uppercase(component)),cellarea,&
235  no_domain=.not.present(domain),domain=domain)
236  case default
237  call mpp_error(fatal, module_name//'/get_grid_cell_area: Illegal component name "'//trim(component)//'": must be one of ATM, LND, or OCN')
238  end select
239  ! convert area to m2
240  cellarea = cellarea*4.*pi*radius**2
241  case(version_2)
242  if (present(domain)) then
243  call mpp_get_compute_domain(domain,xsize=nlon,ysize=nlat)
244  else
245  call get_grid_size(component,tile,nlon,nlat)
246  endif
247  allocate(glonb(nlon+1,nlat+1),glatb(nlon+1,nlat+1))
248  call get_grid_cell_vertices(component, tile, glonb, glatb, domain)
249  if (great_circle_algorithm) then
250  call calc_mosaic_grid_great_circle_area(glonb*pi/180.0, glatb*pi/180.0, cellarea)
251  else
252  call calc_mosaic_grid_area(glonb*pi/180.0, glatb*pi/180.0, cellarea)
253  end if
254  deallocate(glonb,glatb)
255  end select
256 
257 end subroutine get_grid_cell_area_sg

259 ! ============================================================================
260 ! ============================================================================
261 !> Get the area of the component per grid cell
262 subroutine get_grid_comp_area_sg(component,tile,area,domain)
263  character(len=*) :: component
264  integer, intent(in) :: tile
265  real, intent(inout) :: area(:,:)
266  type(domain2d), intent(in), optional :: domain
267  ! local vars
268  integer :: n_xgrid_files ! number of exchange grid files in the mosaic
269  integer :: siz(4), nxgrid
270  integer :: i,j,m,n
271  integer, allocatable :: i1(:), j1(:), i2(:), j2(:)
272  real, allocatable :: xgrid_area(:)
273  real, allocatable :: rmask(:,:)
274  character(len=MAX_NAME) :: &
275  xgrid_name, & ! name of the variable holding xgrid names
276  tile_name, & ! name of the tile
277  xgrid_file, & ! name of the current xgrid file
278  mosaic_name,& ! name of the mosaic
279  mosaic_file,&
280  tilefile
281  character(len=4096) :: attvalue
282  character(len=MAX_NAME), allocatable :: nest_tile_name(:)
283  character(len=MAX_NAME) :: varname1, varname2
284  integer :: is,ie,js,je ! boundaries of our domain
285  integer :: i0, j0 ! offsets for x and y, respectively
286  integer :: num_nest_tile, ntiles
287  logical :: is_nest
288  integer :: found_xgrid_files ! how many xgrid files we actually found in the grid spec
289  integer :: ibegin, iend, bsize, l
290 
291  select case (get_grid_version())
292  case(version_0,version_1)
293  select case(component)
294  case('ATM')
295  call read_data(grid_file,'AREA_ATM',area, no_domain=.not.present(domain),domain=domain)
296  case('OCN')
297  allocate(rmask(size(area,1),size(area,2)))
298  call read_data(grid_file,'AREA_OCN',area, no_domain=.not.present(domain),domain=domain)
299  call read_data(grid_file,'wet', rmask,no_domain=.not.present(domain),domain=domain)
300  area = area*rmask
301  deallocate(rmask)
302  case('LND')
303  call read_data(grid_file,'AREA_LND',area,no_domain=.not.present(domain),domain=domain)
304  case default
305  call mpp_error(fatal, module_name//'/get_grid_comp_area: Illegal component name "'//trim(component)//'": must be one of ATM, LND, or OCN')
306  end select
307  case(version_2) ! mosaic gridspec
308  select case (component)
309  case ('ATM')
310  ! just read the grid cell area and return
311  call get_grid_cell_area(component,tile,area)
312  return
313  case ('LND')
314  xgrid_name = 'aXl_file'
315  call read_data(grid_file, 'lnd_mosaic', mosaic_name)
316  tile_name = trim(mosaic_name)//'_tile'//char(tile+ichar('0'))
317  case ('OCN')
318  xgrid_name = 'aXo_file'
319  call read_data(grid_file, 'ocn_mosaic', mosaic_name)
320  tile_name = trim(mosaic_name)//'_tile'//char(tile+ichar('0'))
321  case default
322  call mpp_error(fatal, module_name//'/get_grid_comp_area: Illegal component name "'//trim(component)//'": must be one of ATM, LND, or OCN')
323  end select
324  ! get the boundaries of the requested domain
325  if(present(domain)) then
326  call mpp_get_compute_domain(domain,is,ie,js,je)
327  i0 = 1-is ; j0=1-js
328  else
329  call get_grid_size(component,tile,ie,je)
330  is = 1 ; i0 = 0
331  js = 1 ; j0 = 0
332  endif
333  if (size(area,1)/=ie-is+1.or.size(area,2)/=je-js+1) &
334  call mpp_error(fatal, module_name//'/get_grid_comp_area: size of the output argument "area" is not consistent with the domain')
335 
336  ! find the nest tile
337  call read_data(grid_file, 'atm_mosaic', mosaic_name)
338  call read_data(grid_file,'atm_mosaic_file',mosaic_file)
339  mosaic_file = grid_dir//trim(mosaic_file)
340  ntiles = get_mosaic_ntiles(trim(mosaic_file))
341  allocate(nest_tile_name(ntiles))
342  num_nest_tile = 0
343  do n = 1, ntiles
344  call read_data(mosaic_file, 'gridfiles', tilefile, level=n)
345  tilefile = grid_dir//trim(tilefile)
346  if( get_global_att_value(tilefile, "nest_grid", attvalue) ) then
347  if(trim(attvalue) == "TRUE") then
348  num_nest_tile = num_nest_tile + 1
349  nest_tile_name(num_nest_tile) = trim(mosaic_name)//'_tile'//char(n+ichar('0'))
350  else if(trim(attvalue) .NE. "FALSE") then
351  call mpp_error(fatal, module_name//'/get_grid_comp_area: value of global attribute nest_grid in file'// trim(tilefile)//' should be TRUE of FALSE')
352  endif
353  end if
354  end do
355  area(:,:) = 0.
356  if(field_exist(grid_file,xgrid_name)) then
357  ! get the number of the exchange-grid files
358  call field_size(grid_file,xgrid_name,siz)
359  n_xgrid_files = siz(2)
360  found_xgrid_files = 0
361  ! loop through all exchange grid files
362  do n = 1, n_xgrid_files
363  ! get the name of the current exchange grid file
364  call read_data(grid_file,xgrid_name,xgrid_file,level=n)
365  ! skip the rest of the loop if the name of the current tile isn't found
366  ! in the file name, but check this only if there is more than 1 tile
367  if(n_xgrid_files>1) then
368  if(index(xgrid_file,trim(tile_name))==0) cycle
369  endif
370  found_xgrid_files = found_xgrid_files + 1
371  !---make sure the atmosphere grid is not a nested grid
372  is_nest = .false.
373  do m = 1, num_nest_tile
374  if(index(xgrid_file, trim(nest_tile_name(m))) .NE. 0) then
375  is_nest = .true.
376  exit
377  end if
378  end do
379  if(is_nest) cycle
380 
381  ! finally read the exchange grid
382  nxgrid = get_mosaic_xgrid_size(grid_dir//xgrid_file)
383  if(nxgrid < bufsize) then
384  allocate(i1(nxgrid), j1(nxgrid), i2(nxgrid), j2(nxgrid), xgrid_area(nxgrid))
385  else
386  allocate(i1(bufsize), j1(bufsize), i2(bufsize), j2(bufsize), xgrid_area(bufsize))
387  endif
388  ibegin = 1
389  do l = 1,nxgrid,bufsize
390  bsize = min(bufsize, nxgrid-l+1)
391  iend = ibegin + bsize - 1
392  call get_mosaic_xgrid(grid_dir//xgrid_file, i1(1:bsize), j1(1:bsize), i2(1:bsize), j2(1:bsize), &
393  xgrid_area(1:bsize), ibegin, iend)
394  ! and sum the exchange grid areas
395  do m = 1, bsize
396  i = i2(m); j = j2(m)
397  if (i<is.or.i>ie) cycle
398  if (j<js.or.j>je) cycle
399  area(i+i0,j+j0) = area(i+i0,j+j0) + xgrid_area(m)
400  end do
401  ibegin = iend + 1
402  enddo
403  deallocate(i1, j1, i2, j2, xgrid_area)
404  enddo
405  if (found_xgrid_files == 0) &
406  call mpp_error(fatal, 'get_grid_comp_area: no xgrid files were found for component '//trim(component)//' (mosaic name is '//trim(mosaic_name)//')')
407 
408  endif
409  deallocate(nest_tile_name)
410  end select ! version
411  ! convert area to m2
412  area = area*4.*pi*radius**2
413 end subroutine get_grid_comp_area_sg

451 ! ============================================================================
452 ! ============================================================================
453 !> Returns arrays of global grid cell boundaries for given model component and
454 !! mosaic tile number.
455 !!
456 !> @note In the case of non-lat-lon grid the returned coordinates may have be not so
457 !! meaningful, by the very nature of such grids. But presumably these 1D coordinate
458 !! arrays are good enough for diag axis and such.
459 subroutine get_grid_cell_vertices_1d(component, tile, glonb, glatb)
460  character(len=*), intent(in) :: component
461  integer, intent(in) :: tile
462  real, intent(inout) :: glonb(:),glatb(:)
463 
464  integer :: nlon, nlat
465  integer :: start(4), nread(4)
466  real, allocatable :: tmp(:,:), x_vert_t(:,:,:), y_vert_t(:,:,:)
467  character(len=MAX_FILE) :: filename1, filename2
468 
469  call get_grid_size_for_one_tile(component, tile, nlon, nlat)
470  if (size(glonb(:))/=nlon+1) &
471  call mpp_error (fatal, module_name//'/get_grid_cell_vertices_1D: Size of argument "glonb" is not consistent with the grid size')
472  if (size(glatb(:))/=nlat+1) &
473  call mpp_error (fatal, module_name//'/get_grid_cell_vertices_1D: Size of argument "glatb" is not consistent with the grid size')
474  if(trim(component) .NE. 'ATM' .AND. component .NE. 'LND' .AND. component .NE. 'OCN') then
475  call mpp_error(fatal, module_name//'/get_grid_cell_vertices_1D: Illegal component name "'//trim(component)//'": must be one of ATM, LND, or OCN')
476  endif
477 
478  select case(get_grid_version())
479  case(version_0)
480  select case(trim(component))
481  case('ATM','LND')
482  call read_data(grid_file, 'xb'//lowercase(component(1:1)), glonb, no_domain=.true.)
483  call read_data(grid_file, 'yb'//lowercase(component(1:1)), glatb, no_domain=.true.)
484  case('OCN')
485  call read_data(grid_file, "gridlon_vert_t", glonb, no_domain=.true.)
486  call read_data(grid_file, "gridlat_vert_t", glatb, no_domain=.true.)
487  end select
488  case(version_1)
489  select case(trim(component))
490  case('ATM','LND')
491  call read_data(grid_file, 'xb'//lowercase(component(1:1)), glonb, no_domain=.true.)
492  call read_data(grid_file, 'yb'//lowercase(component(1:1)), glatb, no_domain=.true.)
493  case('OCN')
494  allocate (x_vert_t(nlon,1,2), y_vert_t(1,nlat,2) )
495  start = 1; nread = 1
496  nread(1) = nlon; nread(2) = 1; start(3) = 1
497  call read_data(grid_file, "x_vert_T", x_vert_t(:,:,1), start, nread, no_domain=.true.)
498  nread(1) = nlon; nread(2) = 1; start(3) = 2
499  call read_data(grid_file, "x_vert_T", x_vert_t(:,:,2), start, nread, no_domain=.true.)
500 
501  nread(1) = 1; nread(2) = nlat; start(3) = 1
502  call read_data(grid_file, "y_vert_T", y_vert_t(:,:,1), start, nread, no_domain=.true.)
503  nread(1) = 1; nread(2) = nlat; start(3) = 4
504  call read_data(grid_file, "y_vert_T", y_vert_t(:,:,2), start, nread, no_domain=.true.)
505  glonb(1:nlon) = x_vert_t(1:nlon,1,1)
506  glonb(nlon+1) = x_vert_t(nlon,1,2)
507  glatb(1:nlat) = y_vert_t(1,1:nlat,1)
508  glatb(nlat+1) = y_vert_t(1,nlat,2)
509  deallocate(x_vert_t, y_vert_t)
510  end select
511  case(version_2)
512  ! get the name of the mosaic file for the component
513  call read_data(grid_file, trim(lowercase(component))//'_mosaic_file', filename1)
514  filename1=grid_dir//trim(filename1)
515  ! get the name of the grid file for the component and tile
516  call read_data(filename1, 'gridfiles', filename2, level=tile)
517  filename2 = grid_dir//trim(filename2)
518 
519  start = 1; nread = 1
