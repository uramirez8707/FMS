261 !> @brief diag_integral_init is the constructor for diag_integral_mod.
262 !!
263 !! <b> Template: </b>
264 !!
265 !! @code{.f90}
266 !! call diag_integral_init (Time_init, Time, blon, blat)
267 !! @endcode
268 !!
269 !! <b> Parameters: </b>
270 !!
271 !! @code{.f90}
272 !! type (time_type), intent(in), optional :: Time_init, Time
273 !! real,dimension(:,:), intent(in), optional :: blon, blat, area_in
274 !! @endcode
275 !!
276 !! @param [in] <Time_init> Initial time to start the integral
277 !! @param [in] <Time> current time
278 !! @param [in] <latb> array of model latitudes at cell boundaries [radians]
279 !! @param [in] <lonb> array of model longitudes at cell boundaries [radians]
280 !!
281 subroutine diag_integral_init (Time_init, Time, blon, blat, area_in)
282 
283 type (time_type), intent(in), optional :: time_init !< Initial time to start the integral
284 type (time_type), intent(in), optional :: time !< current time
285 real,dimension(:,:), intent(in), optional :: blon !< array of model latitudes at cell boundaries [radians]
286 real,dimension(:,:), intent(in), optional :: blat !< array of model longitudes at cell boundaries [radians]
287 real,dimension(:,:), intent(in), optional :: area_in
288 
289 !-------------------------------------------------------------------------------
290 ! local variables:
291 ! r2
292 ! rsize
293 ! unit
294 ! io
295 ! ierr
296 ! seconds
297 ! nc
298 ! i,j
299 !-------------------------------------------------------------------------------
300  real :: rsize
301  integer :: io, ierr, nc, logunit
302  integer :: field_size_local
303  real :: sum_area_local
304  integer :: ensemble_size(6)
305 !-------------------------------------------------------------------------------
306 ! if routine has already been executed, exit.
307 !-------------------------------------------------------------------------------
308  if (module_is_initialized) return
309 
310 !-------------------------------------------------------------------------------
311 ! verify that modules used by this module that are not called later
312 ! have already been initialized.
313 !-------------------------------------------------------------------------------
314  call fms_init
315  call mpp_init
316  call constants_init
317  call time_manager_init
318 
319 !-------------------------------------------------------------------------------
320 ! if this is the initialization call, proceed. if this was simply
321 ! a verification of previous initialization, return.
322 !-------------------------------------------------------------------------------
323  if (present(time_init) .and. present(time) .and. &
324  present(blon) .and. present(blat) ) then
325 
326 !-------------------------------------------------------------------------------
327 ! read namelist.
328 !-------------------------------------------------------------------------------
329  read (input_nml_file, nml=diag_integral_nml, iostat=io)
330  ierr = check_nml_error(io,'diag_integral_nml')
331 
332 !-------------------------------------------------------------------------------
333 ! write version number and namelist to logfile.
334 !-------------------------------------------------------------------------------
335  call write_version_number ('DIAG_INTEGRAL_MOD', version)
336  logunit = stdlog()
337  if (mpp_pe() == mpp_root_pe() ) &
338  write (logunit, nml=diag_integral_nml)
339 
340 !-------------------------------------------------------------------------------
341 ! save the initial time to time-stamp the integrals which will be
342 ! calculated.
343 !-------------------------------------------------------------------------------
344  time_init_save = time_init
345 
346 !-------------------------------------------------------------------------------
347 ! define the model grid sizes and the total number of columns on
348 ! the processor. sum over all processors and store the global
349 ! number of columns in field_size.
350 !-------------------------------------------------------------------------------
351  idim = size(blon,1) - 1
352  jdim = size(blon,2) - 1
353  field_size_local = idim*jdim
354  rsize = real(field_size_local)
355  call mpp_sum (rsize)
356  field_size = nint(rsize)
357 
358 !-------------------------------------------------------------------------------
359 ! define an array to hold the surface area of each grid column
360 ! so that the integrals may be weighted properly. sum over the
361 ! processor, and then over all processors, storing the total
362 ! global surface area in sum_area.
363 !-------------------------------------------------------------------------------
364  allocate (area(idim,jdim))
365 
366  area = area_in
367 
368  sum_area_local = sum(area)
369  sum_area = sum_area_local
370  call mpp_sum (sum_area)
371 
372 !-------------------------------------------------------------------------------
373 ! if integral output is to go to a file, open the file on unit
374 ! diag_unit.
375 !-------------------------------------------------------------------------------
376  if (file_name(1:1) /= ' ' ) then
377  ensemble_size = get_ensemble_size()
378  if (ensemble_size(1) > 1) then
379  file_name = ensemble_file_name(file_name)
380  endif
381  nc = len_trim(file_name)
382  diag_unit = open_file(file_name(1:nc), action='write')
383  endif
384 
385 !-------------------------------------------------------------------------------
386 ! define the variables needed to control the time interval of
387 ! output. Zero time is a flag indicating that the alarm is not set,
388 ! i.e., integrals are not desired. otherwise set the next time to
389 ! output integrals to be at the value of nml variable
390 ! output_interval from now.
391 !-------------------------------------------------------------------------------
392  zero_time = set_time(0,0)
393  if (output_interval >= -0.01) then
394  alarm_interval = set_axis_time(output_interval, time_units)
395  next_alarm_time = time + alarm_interval
396  else
397  alarm_interval = zero_time
398  endif
399  next_alarm_time = time + alarm_interval
400 
401 !-------------------------------------------------------------------------------
402 ! deallocate the local array and mark the module as initialized.
403 !-------------------------------------------------------------------------------
404  module_is_initialized = .true.
405  endif ! (present optional arguments)
406 
407 end subroutine diag_integral_init

486 !###############################################################################
487 !> @brief Perform a 2 dimensional summation of named field
488 !!
489 !! @implements sum_diag_integral_field
490 !!
491 !! <b> Template: </b>
492 !!
493 !! @code{.f90}
494 !! call sum_field_2d (name, data, is, js)
495 !! @endcode
496 !!
497 !! <b> Parameters: </b>
498 !!
499 !! @code{.f90}
500 !! character(len=*), intent(in) :: name
501 !! real, intent(in) :: data(:,:)
502 !! integer, optional, intent(in) :: is, js
503 !! @endcode
504 !!
505 !! @param [in] <name> Name of the field to be integrated
506 !! @param [in] <data> field of integrands to be summed over
507 !! @param [in] <is, js> starting i,j indices over which summation is to occur
508 !!
509 subroutine sum_field_2d (name, data, is, js)
510 
511 character(len=*), intent(in) :: name !< Name of the field to be integrated
512 real, intent(in) :: data(:,:) !< field of integrands to be summed over
513 integer, optional, intent(in) :: is !< starting i indices over which summation is to occur
514 integer, optional, intent(in) :: js !< starting j indices over which summation is to occur
515 
516 !-------------------------------------------------------------------------------
517 ! local variables:
518 !-------------------------------------------------------------------------------
519  integer :: field !< index of desired integral
520  integer :: i1 !< location indices of current data in
521  !! processor-global coordinates
522  integer :: j1 !< location indices of current data in
523  !! processor-global coordinates
524  integer :: i2 !< location indices of current data in
525  !! processor-global coordinates
526  integer :: j2 !< location indices of current data in
527  !! processor-global coordinates
528 
529 
530 !-------------------------------------------------------------------------------
531 ! be sure module has been initialized.
532 !-------------------------------------------------------------------------------
533  if (.not. module_is_initialized ) then
534  call error_mesg ('diag_integral_mod', &
535  'module has not been initialized', fatal )
536  endif
537 
538 !-------------------------------------------------------------------------------
539 ! obtain the index of the current integral. make certain it is valid.
540 !-------------------------------------------------------------------------------
541  field = get_field_index(name)
542  if (field == 0) then
543  call error_mesg ('diag_integral_mod', &
544  'field does not exist', fatal)
545  endif
546 
547 !-------------------------------------------------------------------------------
548 ! define the processor-global indices of the current data. use the
549 ! value 1 for the initial grid points, if is and js are not input.
550 !-------------------------------------------------------------------------------
551  i1 = 1; if (present(is)) i1 = is
552  j1 = 1; if (present(js)) j1 = js
553  i2 = i1 + size(data,1) - 1
554  j2 = j1 + size(data,2) - 1
555 
556 !-------------------------------------------------------------------------------
557 ! increment the count of points toward this integral and add the
558 ! values at this set of grid points to the accumulation array.
559 !-------------------------------------------------------------------------------
560 !$OMP CRITICAL
561  field_count(field) = field_count(field) + &
562  size(data,1)*size(data,2)
563  field_sum(field) = field_sum(field) + &
564  sum(data * area(i1:i2,j1:j2))
565 
566 !$OMP END CRITICAL
567 
568 end subroutine sum_field_2d

628  endif

631 ! obtain the index of the current integral. make certain it is valid.

634  if (field == 0) then
635  call error_mesg ('diag_integral_mod', &
636  'field does not exist', fatal)
637  endif
