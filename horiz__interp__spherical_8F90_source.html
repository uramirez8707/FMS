19 !> @defgroup horiz_interp_spherical_mod horiz_interp_spherical_mod

21 !> @brief Performs spatial interpolation between grids using inverse-distance-weighted scheme.
22 !> This module can interpolate data from rectangular/tripolar grid
23 !! to rectangular/tripolar grid. The interpolation scheme is inverse-distance-weighted
24 !! scheme. There is an optional mask field for missing input data.
25 !! An optional output mask field may be used in conjunction with
26 !! the input mask to show where output data exists.

61  !! nearest neighbor points. Its value can be "radial_search" and "full_search",

65  !! it will be always accurate, but will be slower comparing to "radial_search".
66  !! Normally these two search algorithm will produce same results other than
67  !! order of operation. "radial_search" are recommended to use. The purpose to
68  !! add "full_search" is in case you think you interpolation results is
69  !! not right, you have other option to verify.

101  !> Initialization routine.
102  !!
103  !> Allocates space and initializes a derived-type variable
104  !! that contains pre-computed interpolation indices and weights.
105  subroutine horiz_interp_spherical_new(Interp, lon_in,lat_in,lon_out,lat_out, &
106  num_nbrs, max_dist, src_modulo)
107 
108  type(horiz_interp_type), intent(inout) :: interp !< A derived type variable containing indices
109  !! and weights for subsequent interpolations. To
110  !! reinitialize for different grid-to-grid interpolation
111  !! @ref horiz_interp_del must be used first.
112  real, intent(in), dimension(:,:) :: lon_in !< Latitude (radians) for source data grid
113  real, intent(in), dimension(:,:) :: lat_in !< Longitude (radians) for source data grid
114  real, intent(in), dimension(:,:) :: lon_out !< Longitude (radians) for output data grid
115  real, intent(in), dimension(:,:) :: lat_out !< Latitude (radians) for output data grid
116  logical, intent(in), optional :: src_modulo !< indicates if the boundary condition
117  !! along zonal boundary is cyclic or not. Cyclic when true
118  integer, intent(in), optional :: num_nbrs !< Number of nearest neighbors for regridding
119  !! When number of neighbors within the radius max_dist
120  !! is less than num_nbrs, All the neighbors will be used
121  !! to interpolate onto destination grid. when number of
122  !! neighbors within the radius max_dist is greater than
123  !! num_nbrs, at least "num_nbrs"
124  ! neighbors will be used to remap onto destination grid
125  real, optional, intent(in) :: max_dist !< Maximum region of influence around
126  !! destination grid points
127 
128  !------local variables ---------------------------------------
129  integer :: i, j, n
130  integer :: map_dst_xsize, map_dst_ysize, map_src_xsize, map_src_ysize
131  integer :: map_src_size, num_neighbors
132  real :: max_src_dist, tpi, hpi
133  logical :: src_is_modulo
134  real :: min_theta_dst, max_theta_dst, min_phi_dst, max_phi_dst
135  real :: min_theta_src, max_theta_src, min_phi_src, max_phi_src
136  integer :: map_src_add(size(lon_out,1),size(lon_out,2),max_neighbors)
137  real :: map_src_dist(size(lon_out,1),size(lon_out,2),max_neighbors)
138  integer :: num_found(size(lon_out,1),size(lon_out,2))
139  integer :: ilon(max_neighbors), jlat(max_neighbors)
140  real, dimension(size(lon_out,1),size(lon_out,2)) :: theta_dst, phi_dst
141  real, dimension(size(lon_in,1)*size(lon_in,2)) :: theta_src, phi_src
142 
143  !--------------------------------------------------------------
144 
145  pe = mpp_pe()
146  root_pe = mpp_root_pe()
147 
148  tpi = 2.0*pi; hpi = 0.5*pi
149 
150  num_neighbors = num_nbrs_default
151  if(present(num_nbrs)) num_neighbors = num_nbrs
152  if (num_neighbors <= 0) call mpp_error(fatal,'horiz_interp_spherical_mod: num_neighbors must be > 0')
153 
154  max_src_dist = max_dist_default
155  if (PRESENT(max_dist)) max_src_dist = max_dist
156  interp%max_src_dist = max_src_dist
157 
158  src_is_modulo = .true.
159  if (PRESENT(src_modulo)) src_is_modulo = src_modulo
160 
161  !--- check the grid size comformable
162  map_dst_xsize=size(lon_out,1);map_dst_ysize=size(lon_out,2)
163  map_src_xsize=size(lon_in,1); map_src_ysize=size(lon_in,2)
164  map_src_size = map_src_xsize*map_src_ysize
165 
166  if (map_dst_xsize /= size(lat_out,1) .or. map_dst_ysize /= size(lat_out,2)) &
167  call mpp_error(fatal,'horiz_interp_spherical_mod: destination grids not conformable')
168  if (map_src_xsize /= size(lat_in,1) .or. map_src_ysize /= size(lat_in,2)) &
169  call mpp_error(fatal,'horiz_interp_spherical_mod: source grids not conformable')
170 
171  theta_src = reshape(lon_in,(/map_src_size/))
172  phi_src = reshape(lat_in,(/map_src_size/))
173  theta_dst(:,:) = lon_out(:,:)
174  phi_dst(:,:) = lat_out(:,:)
175 
176  min_theta_dst=tpi;max_theta_dst=0.0;min_phi_dst=pi;max_phi_dst=-pi
177  min_theta_src=tpi;max_theta_src=0.0;min_phi_src=pi;max_phi_src=-pi
178 
179  where(theta_dst<0.0) theta_dst = theta_dst+tpi
180  where(theta_dst>tpi) theta_dst = theta_dst-tpi
181  where(theta_src<0.0) theta_src = theta_src+tpi
182  where(theta_src>tpi) theta_src = theta_src-tpi
183 
184  where(phi_dst < -hpi) phi_dst = -hpi
185  where(phi_dst > hpi) phi_dst = hpi
186  where(phi_src < -hpi) phi_src = -hpi
187  where(phi_src > hpi) phi_src = hpi
188 
189  do j=1,map_dst_ysize
190  do i=1,map_dst_xsize
191  min_theta_dst = min(min_theta_dst,theta_dst(i,j))
192  max_theta_dst = max(max_theta_dst,theta_dst(i,j))
193  min_phi_dst = min(min_phi_dst,phi_dst(i,j))
194  max_phi_dst = max(max_phi_dst,phi_dst(i,j))
195  enddo
196  enddo
197 
198  do i=1,map_src_size
199  min_theta_src = min(min_theta_src,theta_src(i))
200  max_theta_src = max(max_theta_src,theta_src(i))
201  min_phi_src = min(min_phi_src,phi_src(i))
202  max_phi_src = max(max_phi_src,phi_src(i))
203  enddo
204 
205  if (min_phi_dst < min_phi_src) print *, '=> WARNING: latitute of dest grid exceeds src'
206  if (max_phi_dst > max_phi_src) print *, '=> WARNING: latitute of dest grid exceeds src'
207  ! when src is cyclic, no need to print out the following warning.
208  if(.not. src_is_modulo) then
209  if (min_theta_dst < min_theta_src) print *, '=> WARNING : longitude of dest grid exceeds src'
210  if (max_theta_dst > max_theta_src) print *, '=> WARNING : longitude of dest grid exceeds src'
211  endif
212 
213  ! allocate memory to data type
214  if(ASSOCIATED(interp%i_lon)) then
215  if(size(interp%i_lon,1) .NE. map_dst_xsize .OR. &
216  size(interp%i_lon,2) .NE. map_dst_ysize ) call mpp_error(fatal, &
217  .NE..OR.'horiz_interp_spherical_mod: size(Interp%i_lon(:),1) map_dst_xsize '// &
218  .NE.'size(Interp%i_lon(:),2) map_dst_ysize')
219  else
220  allocate(interp%i_lon(map_dst_xsize,map_dst_ysize,max_neighbors), &
221  interp%j_lat(map_dst_xsize,map_dst_ysize,max_neighbors), &
222  interp%src_dist(map_dst_xsize,map_dst_ysize,max_neighbors), &
223  interp%num_found(map_dst_xsize,map_dst_ysize) )
224  endif
225 
226  map_src_add = 0
227  map_src_dist = large
228  num_found = 0
229 
230  !using radial_search to find the nearest points and corresponding distance.
231 
232  select case(trim(search_method))
233  case ("radial_search") ! will be efficient, but may be not so accurate for some cases
234  call radial_search(theta_src, phi_src, theta_dst, phi_dst, map_src_xsize, map_src_ysize, &
235  map_src_add, map_src_dist, num_found, num_neighbors,max_src_dist,src_is_modulo)
236  case ("full_search") ! always accurate, but less efficient.
237  call full_search(theta_src, phi_src, theta_dst, phi_dst, map_src_add, map_src_dist, &
238  num_found, num_neighbors,max_src_dist )
239  case default
240  call mpp_error(fatal,"horiz_interp_spherical_new: nml search_method = "// &
241  trim(search_method)//" is not a valid namelist option")
242  end select
243 
244  do j=1,map_dst_ysize
245  do i=1,map_dst_xsize
246  do n=1,num_found(i,j)
247  if(map_src_add(i,j,n) == 0) then
248  jlat(n) = 0; ilon(n) = 0
249  else
250  jlat(n) = map_src_add(i,j,n)/map_src_xsize + 1
251  ilon(n) = map_src_add(i,j,n) - (jlat(n)-1)*map_src_xsize
252  if(ilon(n) == 0) then
253  jlat(n) = jlat(n) - 1
254  ilon(n) = map_src_xsize
255  endif
256  endif
257  enddo
258  interp%i_lon(i,j,:) = ilon(:)
259  interp%j_lat(i,j,:) = jlat(:)
260  interp%num_found(i,j) = num_found(i,j)
261  interp%src_dist(i,j,:) = map_src_dist(i,j,:)
262  enddo
263  enddo
264 
265  interp%nlon_src = map_src_xsize; interp%nlat_src = map_src_ysize
266  interp%nlon_dst = map_dst_xsize; interp%nlat_dst = map_dst_ysize
267 
268  return
269 
270  end subroutine horiz_interp_spherical_new

274  !> Subroutine for performing the horizontal interpolation between two grids.
275  !! horiz_interp_spherical_new must be called before calling this routine.
276  subroutine horiz_interp_spherical( Interp, data_in, data_out, verbose, mask_in, mask_out, missing_value)
277  type(horiz_interp_type), intent(in) :: interp !< A derived type variable containing indices
278  !! and weights for subsequent interpolations. Returned
279  !! by a previous call to horiz_interp_spherical_new
280  real, intent(in), dimension(:,:) :: data_in !< Input data on source grid
281  real, intent(out), dimension(:,:) :: data_out !< Output data on destination grid
282  integer, intent(in), optional :: verbose !< 0 = no output; 1 = min,max,means; 2 = most output
283  real, intent(in), dimension(:,:), optional :: mask_in !< Input mask, must be the same size as
284  !! the input data. The real value of mask_in must be
285  !! in the range (0.,1.). Set mask_in=0.0 for data points
286  !! that should not be used or have missing data
287  real, intent(out), dimension(:,:), optional :: mask_out !< Output mask that specifies whether data was computed.
288  real, intent(in), optional :: missing_value !< Used to indicate missing data
289 
290  !--- some local variables ----------------------------------------
291  real, dimension(Interp%nlon_dst, Interp%nlat_dst,size(Interp%src_dist,3)) :: wt
292  real, dimension(Interp%nlon_src, Interp%nlat_src) :: mask_src
293  real, dimension(Interp%nlon_dst, Interp%nlat_dst) :: mask_dst
294  integer :: nlon_in, nlat_in, nlon_out, nlat_out, num_found
295  integer :: m, n, i, j, k, miss_in, miss_out, i1, i2, j1, j2, iverbose
296  real :: min_in, max_in, avg_in, min_out, max_out, avg_out, sum
297  !-----------------------------------------------------------------
298 
299  iverbose = 0; if (present(verbose)) iverbose = verbose
300 
301  nlon_in = interp%nlon_src; nlat_in = interp%nlat_src
302  nlon_out = interp%nlon_dst; nlat_out = interp%nlat_dst
303 
304  if(size(data_in,1) .ne. nlon_in .or. size(data_in,2) .ne. nlat_in ) &
305  call mpp_error(fatal,'horiz_interp_spherical_mod: size of input array incorrect')
306 
307  if(size(data_out,1) .ne. nlon_out .or. size(data_out,2) .ne. nlat_out ) &
308  call mpp_error(fatal,'horiz_interp_spherical_mod: size of output array incorrect')
309 
310  mask_src = 1.0; mask_dst = 1.0
311  if(present(mask_in)) mask_src = mask_in
312 
313  do n=1,nlat_out
314  do m=1,nlon_out
315  ! neighbors are sorted nearest to farthest
316  ! check nearest to see if it is a land point
317  num_found = interp%num_found(m,n)
318  if(num_found == 0 ) then
319  mask_dst(m,n) = 0.0
320  else
321  i1 = interp%i_lon(m,n,1); j1 = interp%j_lat(m,n,1)
322  if (mask_src(i1,j1) .lt. 0.5) then
323  mask_dst(m,n) = 0.0
324  endif
325 
326  if(num_found .gt. 1 ) then
327  i2 = interp%i_lon(m,n,2); j2 = interp%j_lat(m,n,2)
328  ! compare first 2 nearest neighbors -- if they are nearly
329  ! equidistant then use this mask for robustness
330  if(abs(interp%src_dist(m,n,2)-interp%src_dist(m,n,1)) .lt. epsln) then
331  if((mask_src(i1,j1) .lt. 0.5)) mask_dst(m,n) = 0.0
332  endif
333  endif
334 
335  sum=0.0
336  do k=1, num_found
337  if(mask_src(interp%i_lon(m,n,k),interp%j_lat(m,n,k)) .lt. 0.5 ) then
338  wt(m,n,k) = 0.0
339  else
340  if (interp%src_dist(m,n,k) <= epsln) then
341  wt(m,n,k) = large
342  sum = sum + large
343  else if(interp%src_dist(m,n,k) <= interp%max_src_dist ) then
344  wt(m,n,k) = 1.0/interp%src_dist(m,n,k)
345  sum = sum+wt(m,n,k)
346  else
347  wt(m,n,k) = 0.0
348  endif
349  endif
350  enddo
351  if (sum > epsln) then
352  do k = 1, num_found
353  wt(m,n,k) = wt(m,n,k)/sum
354  enddo
355  else
356  mask_dst(m,n) = 0.0
357  endif
358  endif
359  enddo
360  enddo
361 
362  data_out = 0.0
363  do n=1,nlat_out
364  do m=1,nlon_out
365  if(mask_dst(m,n) .gt. 0.5) then
366  do k=1, interp%num_found(m,n)
367  i = interp%i_lon(m,n,k)
368  j = interp%j_lat(m,n,k)
369  data_out(m,n) = data_out(m,n)+data_in(i,j)*wt(m,n,k)
370  enddo
371  else
372  if(present(missing_value)) then
373  data_out(m,n) = missing_value
374  else
375  data_out(m,n) = 0.0
376  endif
377  endif
378  enddo
379  enddo
380 
381  if(present(mask_out)) mask_out = mask_dst
382 
383  !***********************************************************************
384  ! compute statistics: minimum, maximum, and mean
385  !-----------------------------------------------------------------------
386 
387  if (iverbose > 0) then
388 
389  ! compute statistics of input data
390 
391  call stats (data_in, min_in, max_in, avg_in, miss_in, missing_value, mask=mask_src)
392 
393  ! compute statistics of output data
394  call stats (data_out, min_out, max_out, avg_out, miss_out, missing_value, mask=mask_dst)
395 
396  !---- output statistics ----
397  ! root_pe have the information of global mean, min and max
398  if(pe == root_pe) then
399  write (*,900)
400  write (*,901) min_in ,max_in, avg_in
401  if (present(mask_in)) write (*,903) miss_in
402  write (*,902) min_out,max_out,avg_out
403  if (present(mask_out)) write (*,903) miss_out
404  endif
405 900 format (/,1x,10('-'),' output from horiz_interp ',10('-'))
406 901 format (' input: min=',f16.9,' max=',f16.9,' avg=',f22.15)
407 902 format (' output: min=',f16.9,' max=',f16.9,' avg=',f22.15)
408 903 format (' number of missing points = ',i6)
409 
410  endif
411 
412  return
413  end subroutine horiz_interp_spherical

415  !#######################################################################
416  subroutine horiz_interp_spherical_wght( Interp, wt, verbose, mask_in, mask_out, missing_value)
417  type (horiz_interp_type), intent(in) :: interp
418  real, intent(out), dimension(:,:,:) :: wt
419  integer, intent(in), optional :: verbose
420  real, intent(in), dimension(:,:), optional :: mask_in
421  real, intent(inout), dimension(:,:), optional :: mask_out
422  real, intent(in), optional :: missing_value
423 
424  !--- some local variables ----------------------------------------
425  real, dimension(Interp%nlon_src, Interp%nlat_src) :: mask_src
426  real, dimension(Interp%nlon_dst, Interp%nlat_dst) :: mask_dst
427  integer :: nlon_in, nlat_in, nlon_out, nlat_out, num_found
428  integer :: m, n, i, j, k, miss_in, miss_out, i1, i2, j1, j2, iverbose
429  real :: min_in, max_in, avg_in, min_out, max_out, avg_out, sum
430  !-----------------------------------------------------------------
431 
432  iverbose = 0; if (present(verbose)) iverbose = verbose
433 
434  nlon_in = interp%nlon_src; nlat_in = interp%nlat_src
435  nlon_out = interp%nlon_dst; nlat_out = interp%nlat_dst
436 
437  mask_src = 1.0; mask_dst = 1.0
438  if(present(mask_in)) mask_src = mask_in
439 
440  do n=1,nlat_out
441  do m=1,nlon_out
442  ! neighbors are sorted nearest to farthest
443  ! check nearest to see if it is a land point
444  num_found = interp%num_found(m,n)
445 
446  if (num_found > num_nbrs_default) then
447  print*,'pe=',mpp_pe(),'num_found=',num_found
448  num_found = num_nbrs_default
449  end if
450 
451  if(num_found == 0 ) then
452  mask_dst(m,n) = 0.0
453  else
454  i1 = interp%i_lon(m,n,1); j1 = interp%j_lat(m,n,1)
455  if (mask_src(i1,j1) .lt. 0.5) then
456  mask_dst(m,n) = 0.0
457  endif
458 
459  if(num_found .gt. 1 ) then
460  i2 = interp%i_lon(m,n,2); j2 = interp%j_lat(m,n,2)
461  ! compare first 2 nearest neighbors -- if they are nearly
462  ! equidistant then use this mask for robustness
463  if(abs(interp%src_dist(m,n,2)-interp%src_dist(m,n,1)) .lt. epsln) then
464  if((mask_src(i1,j1) .lt. 0.5)) mask_dst(m,n) = 0.0
465  endif
466  endif
467 
468  sum=0.0
469  do k=1, num_found
470  if(mask_src(interp%i_lon(m,n,k),interp%j_lat(m,n,k)) .lt. 0.5 ) then
471  wt(m,n,k) = 0.0
472  else
473  if (interp%src_dist(m,n,k) <= epsln) then
474  wt(m,n,k) = large
475  sum = sum + large
476  else if(interp%src_dist(m,n,k) <= interp%max_src_dist ) then
477  wt(m,n,k) = 1.0/interp%src_dist(m,n,k)
478  sum = sum+wt(m,n,k)
479  else
480  wt(m,n,k) = 0.0
481  endif
482  endif
483  enddo
484  if (sum > epsln) then
485  do k = 1, num_found
486  wt(m,n,k) = wt(m,n,k)/sum
487  enddo
488  else
489  mask_dst(m,n) = 0.0
490  endif
491  endif
492  enddo
493  enddo
494 
495  return
496  end subroutine horiz_interp_spherical_wght

499

848  function spherical_distance(theta1,phi1,theta2,phi2)
849 
850  real, intent(in) :: theta1, phi1, theta2, phi2
851  real :: spherical_distance, dot
852

854  spherical_distance = 0.0
