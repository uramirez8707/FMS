118  !> @brief Creates a new @ref horiz_interp_type
119  !!
120  !> Allocates space and initializes a derived-type variable
121  !! that contains pre-computed interpolation indices and weights.
122  subroutine horiz_interp_bicubic_new_1d_s ( Interp, lon_in, lat_in, lon_out, lat_out, &
123  verbose, src_modulo )
124 
125  !-----------------------------------------------------------------------
126  type(horiz_interp_type), intent(inout) :: Interp !< A derived-type variable containing indices
127  !! and weights used for subsequent interpolations. To
128  !! reinitialize this variable for a different grid-to-grid
129  !! interpolation you must first use the
130  !! @ref horiz_interp_bicubic_del interface.
131  real, intent(in), dimension(:) :: lon_in !< Longitude (radians) for source data grid
132  real, intent(in), dimension(:) :: lat_in !< Latitude (radians) for source data grid
133  real, intent(in), dimension(:,:) :: lon_out !< Longitude (radians) for output data grid
134  real, intent(in), dimension(:,:) :: lat_out !< Latitude (radians) for output data grid
135  integer, intent(in), optional :: verbose !< flag for print output amount
136  logical, intent(in), optional :: src_modulo !< indicates if the boundary condition along
137  !! zonal boundary is cyclic or not. Zonal boundary condition
138  !!is cyclic when true
139  integer :: i, j, ip1, im1, jp1, jm1
140  logical :: src_is_modulo
141  integer :: nlon_in, nlat_in, nlon_out, nlat_out
142  integer :: jcl, jcu, icl, icu, jj
143  real :: xz, yz
144  integer :: unit
145 
146  if(present(verbose)) verbose_bicubic = verbose
147  src_is_modulo = .false.
148  if (present(src_modulo)) src_is_modulo = src_modulo
149 
150  if(size(lon_out,1) /= size(lat_out,1) .or. size(lon_out,2) /= size(lat_out,2) ) &
151  call mpp_error(fatal,'horiz_interp_bilinear_mod: when using bilinear ' // &
152  'interplation, the output grids should be geographical grids')
153 
154  !--- get the grid size
155  nlon_in = size(lon_in) ; nlat_in = size(lat_in)
156  nlon_out = size(lon_out,1); nlat_out = size(lat_out,2)
157  interp%nlon_src = nlon_in; interp%nlat_src = nlat_in
158  interp%nlon_dst = nlon_out; interp%nlat_dst = nlat_out
159 ! use wti(:,:,1) for x-derivative, wti(:,:,2) for y-derivative, wti(:,:,3) for xy-derivative
160  allocate ( interp%wti (nlon_in, nlat_in, 3) )
161  allocate ( interp%lon_in (nlon_in) )
162  allocate ( interp%lat_in (nlat_in) )
163  allocate ( interp%rat_x (nlon_out, nlat_out) )
164  allocate ( interp%rat_y (nlon_out, nlat_out) )
165  allocate ( interp%i_lon (nlon_out, nlat_out, 2) )
166  allocate ( interp%j_lat (nlon_out, nlat_out, 2) )
167 
168  interp%lon_in = lon_in
169  interp%lat_in = lat_in
170 
171  if ( verbose_bicubic > 0 ) then
172  unit = stdout()
173  write (unit,'(/,"Initialising bicubic interpolation, interface horiz_interp_bicubic_new_1d_s")')
174  write (unit,'(/," Longitude of coarse grid points (radian): xc(i) i=1, ",i4)') interp%nlon_src
175  write (unit,'(1x,10f10.4)') (interp%lon_in(jj),jj=1,interp%nlon_src)
176  write (unit,'(/," Latitude of coarse grid points (radian): yc(j) j=1, ",i4)') interp%nlat_src
177  write (unit,'(1x,10f10.4)') (interp%lat_in(jj),jj=1,interp%nlat_src)
178  do i=1, interp%nlat_dst
179  write (unit,*)
180  write (unit,'(/," Longitude of fine grid points (radian): xf(i) i=1, ",i4)') interp%nlat_dst
181  write (unit,'(1x,10f10.4)') (lon_out(jj,i),jj=1,interp%nlon_dst)
182  enddo
183  do i=1, interp%nlon_dst
184  write (unit,*)
185  write (unit,'(/," Latitude of fine grid points (radian): yf(j) j=1, ",i4)') interp%nlon_dst
186  write (unit,'(1x,10f10.4)') (lat_out(i,jj),jj=1,interp%nlat_dst)
187  enddo
188  endif
189 
190 
191 !---------------------------------------------------------------------------
192 ! Find the x-derivative. Use central differences and forward or
193 ! backward steps at the boundaries
194 
195  do j=1,nlat_in
196  do i=1,nlon_in
197  ip1=min(i+1,nlon_in)
198  im1=max(i-1,1)
199  interp%wti(i,j,1) = 1./(interp%lon_in(ip1)-interp%lon_in(im1))
200  enddo
201  enddo
202 
203 
204 !---------------------------------------------------------------------------
205 
206 ! Find the y-derivative. Use central differences and forward or
207 ! backward steps at the boundaries
208  do j=1,nlat_in
209  jp1=min(j+1,nlat_in)
210  jm1=max(j-1,1)
211  do i=1,nlon_in
212  interp%wti(i,j,2) = 1./(interp%lat_in(jp1)-interp%lat_in(jm1))
213  enddo
214  enddo
215 
216 !---------------------------------------------------------------------------
217 
218 ! Find the xy-derivative. Use central differences and forward or
219 ! backward steps at the boundaries
220  do j=1,nlat_in
221  jp1=min(j+1,nlat_in)
222  jm1=max(j-1,1)
223  do i=1,nlon_in
224  ip1=min(i+1,nlon_in)
225  im1=max(i-1,1)
226  interp%wti(i,j,3) = 1./((interp%lon_in(ip1)-interp%lon_in(im1))*(interp%lat_in(jp1)-interp%lat_in(jm1)))
227  enddo
228  enddo
229 !---------------------------------------------------------------------------
230 ! Now for each point at the dest-grid find the boundary points of
231 ! the source grid
232  do j=1, nlat_out
233  do i=1,nlon_out
234  yz = lat_out(i,j)
235  xz = lon_out(i,j)
236 
237  jcl = 0
238  jcu = 0
239  if( yz .le. interp%lat_in(1) ) then
240  jcl = 1
241  jcu = 1
242  else if( yz .ge. interp%lat_in(nlat_in) ) then
243  jcl = nlat_in
244  jcu = nlat_in
245  else
246  jcl = indl(interp%lat_in, yz)
247  jcu = indu(interp%lat_in, yz)
248  endif
249 
250  icl = 0
251  icu = 0
252  !--- cyclic condition, do we need to use do while
253  if( xz .gt. interp%lon_in(nlon_in) ) xz = xz - tpi
254  if( xz .le. interp%lon_in(1) ) xz = xz + tpi
255  if( xz .ge. interp%lon_in(nlon_in) ) then
256  icl = nlon_in
257  icu = 1
258  interp%rat_x(i,j) = (xz - interp%lon_in(icl))/(interp%lon_in(icu) - interp%lon_in(icl) + tpi)
259  else
260  icl = indl(interp%lon_in, xz)
261  icu = indu(interp%lon_in, xz)
262  interp%rat_x(i,j) = (xz - interp%lon_in(icl))/(interp%lon_in(icu) - interp%lon_in(icl))
263  endif
264  interp%j_lat(i,j,1) = jcl
265  interp%j_lat(i,j,2) = jcu
266  interp%i_lon(i,j,1) = icl
267  interp%i_lon(i,j,2) = icu
268  if(jcl == jcu) then
269  interp%rat_y(i,j) = 0.0
270  else
271  interp%rat_y(i,j) = (yz - interp%lat_in(jcl))/(interp%lat_in(jcu) - interp%lat_in(jcl))
272  endif
273 ! if(yz.gt.Interp%lat_in(jcu)) call mpp_error(FATAL, ' horiz_interp_bicubic_new_1d_s: yf < ycl, no valid boundary point')
274 ! if(yz.lt.Interp%lat_in(jcl)) call mpp_error(FATAL, ' horiz_interp_bicubic_new_1d_s: yf > ycu, no valid boundary point')
275 ! if(xz.gt.Interp%lon_in(icu)) call mpp_error(FATAL, ' horiz_interp_bicubic_new_1d_s: xf < xcl, no valid boundary point')
276 ! if(xz.lt.Interp%lon_in(icl)) call mpp_error(FATAL, ' horiz_interp_bicubic_new_1d_s: xf > xcu, no valid boundary point')
277  enddo
278  enddo
279  end subroutine horiz_interp_bicubic_new_1d_s

427  !> @brief Perform bicubic horizontal interpolation
428  subroutine horiz_interp_bicubic( Interp, data_in, data_out, verbose, mask_in, mask_out, missing_value, missing_permit)
429  type (horiz_interp_type), intent(in) :: interp
430  real, intent(in), dimension(:,:) :: data_in
431  real, intent(out), dimension(:,:) :: data_out
432  integer, intent(in), optional :: verbose
433  real, intent(in), dimension(:,:), optional :: mask_in
434  real, intent(out), dimension(:,:), optional :: mask_out
435  real, intent(in), optional :: missing_value
436  integer, intent(in), optional :: missing_permit
437  real :: yz, ycu, ycl
438  real :: xz, xcu, xcl
439  real :: val, val1, val2
440  real, dimension(4) :: y, y1, y2, y12
441  integer :: icl, icu, jcl, jcu
442  integer :: iclp1, icup1, jclp1, jcup1
443  integer :: iclm1, icum1, jclm1, jcum1
444  integer :: i,j
445 
446  if ( present(verbose) ) verbose_bicubic = verbose
447 ! fill_in = .false.
448 ! if ( present(fill) ) fill_in = fill
449 ! use dfc_x and dfc_y as workspace
450 ! if ( fill_in ) call fill_xy(fc(ics:ice,jcs:jce), ics, ice, jcs, jce, maxpass=2)
451 ! where ( data_in .le. missing ) data_in(:,:) = 0.
452 !!
453  do j=1, interp%nlat_dst
454  do i=1, interp%nlon_dst
455  yz = interp%rat_y(i,j)
456  xz = interp%rat_x(i,j)
457  jcl = interp%j_lat(i,j,1)
458  jcu = interp%j_lat(i,j,2)
459  icl = interp%i_lon(i,j,1)
460  icu = interp%i_lon(i,j,2)
461  if( icl > icu ) then
462  iclp1 = icu
463  icum1 = icl
464  xcl = interp%lon_in(icl)
465  xcu = interp%lon_in(icu)+tpi
466  else
467  iclp1 = min(icl+1,interp%nlon_src)
468  icum1 = max(icu-1,1)
469  xcl = interp%lon_in(icl)
470  xcu = interp%lon_in(icu)
471  endif
472  iclm1 = max(icl-1,1)
473  icup1 = min(icu+1,interp%nlon_src)
474  jclp1 = min(jcl+1,interp%nlat_src)
475  jclm1 = max(jcl-1,1)
476  jcup1 = min(jcu+1,interp%nlat_src)
477  jcum1 = max(jcu-1,1)
478  ycl = interp%lat_in(jcl)
479  ycu = interp%lat_in(jcu)
480 ! xcl = Interp%lon_in(icl)
481 ! xcu = Interp%lon_in(icu)
482  y(1) = data_in(icl,jcl)
483  y(2) = data_in(icu,jcl)
484  y(3) = data_in(icu,jcu)
485  y(4) = data_in(icl,jcu)
486  y1(1) = ( data_in(iclp1,jcl) - data_in(iclm1,jcl) ) * interp%wti(icl,jcl,1)
