224 function string_compare(string1, string2, ignore_case) &
225  result(same)
226 
227  character(len=*), intent(in) :: string1
228  character(len=*), intent(in) :: string2
229  logical,intent(in), optional :: ignore_case
230  logical :: same
231 
232  if (len_trim(string1) .ne. len_trim(string2)) then
233  same = .false.
234  return
235  endif
236  if (present(ignore_case)) then
237  if (ignore_case) then
238  same = trim(lowercase(string1)) .eq. trim(lowercase(string2))
239  return
240  endif
241  endif
242  same = trim(string1) .eq. trim(string2)

246 !> @brief Determine if a string exists in a character linked list.
247 !! @return Flag telling if the string is found in the list.
248 function is_in_list(list, string, ignore_case) &
249  result(in_list)
250 
251  type(char_linked_list), pointer, intent(in) :: list !< Linked list.
252  character(len=*), intent(in) :: string !< Input string.
253  logical, intent(in), optional :: ignore_case !< Flag to ignore case.
254  logical :: in_list
255 
256  type(char_linked_list), pointer :: p
257 
258  in_list = .false.
259  p => list
260  do while(associated(p))
261  if (string_compare(string, p%string, ignore_case)) then
262  in_list = .true.
263  return
264  else
265  p => p%head
266  endif
267  enddo
268 end function is_in_list

311 !> @brief Determine if the "domain tile string" (.tilex.) exists in the input filename.
312 !! @internal
313 function has_domain_tile_string(string) &
314  result(has_string)
315 
316  character(len=*), intent(in) :: string !< Input string.
317  logical :: has_string
318 
319  integer :: l
320  integer :: i, j
321 
322  has_string = .false.
323 ! Assigns i to the index where ".tile" starts
324  i = index(trim(string), ".tile", back=.true.)
325  if (i .ne. 0) then
326  l = len_trim(string)
327 ! Sets i to the index after .tile
328  i = i + 5
329  j = i
330  do while (i .le. l)
331 ! If the ith characters is a dot but i not equal to the index after .tile set has_string to true
332  if (verify(string(i:i), ".") .eq. 0 .and. j .ne. i) then
333  has_string = .true.
334  exit
335 ! If the ith characters is NOT a number exit function and has_string will stay as false
336  elseif (verify(string(i:i), "0123456789") .ne. 0) then
337  exit
338  endif
339  i = i + 1
340  enddo
341  endif
342 end function has_domain_tile_string

476 !> @brief Populate 2D maskmap from mask_table given a model
477 subroutine parse_mask_table_2d(mask_table, maskmap, modelname)
478 
479  character(len=*), intent(in) :: mask_table !< Mask table to be read in
480  logical, intent(out) :: maskmap(:,:) !< 2D Mask output
481  character(len=*), intent(in) :: modelname !< Model to which this applies
482 
483  integer :: nmask, layout(2)
484  integer, allocatable :: mask_list(:,:)
485  character(len=:), dimension(:), allocatable :: mask_table_contents
486  integer :: iocheck, n, stdoutunit, offset
487  character(len=128) :: record
488 
489  maskmap = .true.
490  nmask = 0
491  stdoutunit = stdout()
492  call ascii_read(mask_table, mask_table_contents)
493  if( mpp_pe() == mpp_root_pe() ) then
494  read(mask_table_contents(1), fmt=*, iostat=iocheck) nmask
495  if (iocheck > 0) then
496  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error in reading nmask from file variable")
497  elseif (iocheck < 0) then
498  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error: nmask not completely read from file variable")
499  endif
500  write(stdoutunit,*)"parse_mask_table: Number of domain regions masked in ", trim(modelname), " = ", nmask
501  if( nmask > 0 ) then
502  !--- read layout from mask_table and confirm it matches the shape of maskmap
503  read(mask_table_contents(2), fmt=*, iostat=iocheck) layout
504  if (iocheck > 0) then
505  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error in reading layout from file variable")
506  elseif (iocheck < 0) then
507  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error: layout not completely read from file variable")
508  endif
509  if( (layout(1) .NE. size(maskmap,1)) .OR. (layout(2) .NE. size(maskmap,2)) )then
510  write(stdoutunit,*)"layout=", layout, ", size(maskmap) = ", size(maskmap,1), size(maskmap,2)
511  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): layout in file "//trim(mask_table)// &
512  "does not match size of maskmap for "//trim(modelname))
513  endif
514  !--- make sure mpp_npes() == layout(1)*layout(2) - nmask
515  if( mpp_npes() .NE. layout(1)*layout(2) - nmask ) call mpp_error(fatal, &
516  .NE."fms2_io(parse_mask_table_2d): mpp_npes() layout(1)*layout(2) - nmask for "//trim(modelname))
517  endif
518  endif
519 
520  call mpp_broadcast(nmask, mpp_root_pe())
521 
522  if(nmask==0) return
523 
524  allocate(mask_list(nmask,2))
525 
526  if( mpp_pe() == mpp_root_pe() ) then
527  n = 0
528  offset = 3
529  do while (offset + n < size(mask_table_contents)+1)
530  read(mask_table_contents(n+offset),'(a)',iostat=iocheck) record
531  if (iocheck > 0) then
532  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error in reading record from file variable")
533  elseif (iocheck < 0) then
534  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error: record not completely read from file variable")
535  endif
536  if (record(1:1) == '#') then
537  offset = offset + 1
538  cycle
539  elseif (record(1:10) == ' ') then
540  offset = offset + 1
541  cycle
542  endif
543  n = n + 1
544  if( n > nmask ) then
545  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): number of mask_list entry "// &
546  "is greater than nmask in file "//trim(mask_table) )
547  endif
548  read(record,*,iostat=iocheck) mask_list(n,1), mask_list(n,2)
549  if (iocheck > 0) then
550  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error in reading mask_list from record variable")
551  elseif (iocheck < 0) then
552  call mpp_error(fatal, &
553  & "fms2_io(parse_mask_table_2d): Error: mask_list not completely read from record variable")
554  endif
555  enddo
556 
557  !--- make sure the number of entry for mask_list is nmask
558  if( n .NE. nmask) call mpp_error(fatal, &
559  "fms2_io(parse_mask_table_2d): number of mask_list entry does not match nmask in file "//trim(mask_table))
560  endif
561 
562  call mpp_broadcast(mask_list, 2*nmask, mpp_root_pe())
563  do n = 1, nmask
564  maskmap(mask_list(n,1),mask_list(n,2)) = .false.
565  enddo
566 
567  deallocate(mask_list, mask_table_contents)
568 
569 end subroutine parse_mask_table_2d

572 !> @brief Populate 3D maskmap from mask_table given a model
573 subroutine parse_mask_table_3d(mask_table, maskmap, modelname)
574 
575  character(len=*), intent(in) :: mask_table !< Mask table to be read in
576  logical, intent(out) :: maskmap(:,:,:) !< 2D Mask output
577  character(len=*), intent(in) :: modelname !< Model to which this applies
578

580  integer, allocatable :: mask_list(:,:)
581  character(len=:), dimension(:), allocatable :: mask_table_contents
582  integer :: iocheck, n, stdoutunit, ntiles, offset
583  character(len=128) :: record
584
