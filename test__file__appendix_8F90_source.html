20 !> @brief This programs tests calls to
21 !! set_filename_appendix(string_in) -> sets string_in as a module variable, filename_appendix, in fms2io
22 !! nullify_filename_appendix -> Resets the module variable, filename_appendix, in fms2io
23 !! get_instance_filename(name_in,name_out) -> Adds the filename_appendix to name_in and sets it as name_out
24 !! get_filename_appendix(string_out) -> gets the filename_appendix and sets it as string_out
25 
26 program test_file_appendix
27 
28 use fms2_io_mod , only: fms2_io_init, set_filename_appendix, get_filename_appendix, &
29  get_instance_filename, open_file, fmsnetcdffile_t, close_file, &
30  nullify_filename_appendix
31 use mpp_mod , only: mpp_init, mpp_exit, mpp_error, fatal
32 
33 implicit none
34 
35 logical :: test_passed !> Flag indicating if the test_passed
36 type(FmsNetcdfFile_t) :: fileobj !> Fms2io file obj
37 character(len=50) :: buf !> String buffer
38 
39 test_passed = .false.
40 
41 call mpp_init()
42 call fms2_io_init()
43 
44 call set_filename_appendix("nest01")
45 
46 !< Call get_filename_appendix to check if the filename appendix was set up
47 buf = ""
48 call get_filename_appendix(buf)
49 if (trim(buf) .ne. "nest01") call mpp_error(fatal, "get_filename_appendix is not set up")
50 
51 !< Call get_instance_filename to check if it adds the filename_appendix to the original string
52 call get_instance_filename("nestfile.nc", buf)
53 if (trim(buf) .ne. "nestfile.nest01.nc") &
54  call mpp_error(fatal, "get_instance_filename does not add the filename appendix")
55 
56 !< Call get instance_filename where the string send in has tile in the name
57 call get_instance_filename("nestfile.tile1.nc", buf)
58 if (trim(buf) .ne. "nestfile.nest01.tile1.nc") &
59  call mpp_error(fatal, "get_instance_filename does not add the filename appendix before the .tile")
60 
61 !< Call get_instance_filename where the string send in has the pe number in the end
62 call get_instance_filename("nestfile.nc.0001", buf)
63 if (trim(buf) .ne. "nestfile.nest01.nc.0001") &
64  call mpp_error(fatal, trim(buf)//": get_instance_filename does not add the filename appendix before the .nc")
65 
66 !< Call get_instance_filename where the string send in has the pe number in the
67 !end
68 call get_instance_filename("nestfile.tile1.nc.0001", buf)
69 if (trim(buf) .ne. "nestfile.nest01.tile1.nc.0001") &
70  call mpp_error(fatal, trim(buf)//": get_instance_filename does not add the filename appendix before the tile")
71 
72 !< Call open_file to check if the filename_appendix was appended to the file created
73 if (open_file(fileobj, "nestfile.nc", "overwrite", is_restart=.true.)) then
74  call close_file(fileobj)
75 else
76  call mpp_error(fatal, "Error opening nestfile.nc")
77 end if
78 
79 inquire(file="nestfile.res.nest01.nc", exist=test_passed)
80 if (.not. test_passed) call mpp_error(fatal, "Error creating nestfile.res.nest01.nc")
81 
82 !< Call nullify_filename_appendix and check if the filename_appendix was removed
83 call nullify_filename_appendix()
84 call get_filename_appendix(buf)
85 if (trim(buf) .ne. "") call mpp_error(fatal, "get_filename_appendix was not removed")
86 
87 !< Call open file again and check if the expected file: "nestfile.res.nc" was created
88 if (open_file(fileobj, "nestfile.nc", "overwrite", is_restart=.true.)) then
89  call close_file(fileobj)
90 else
91  call mpp_error(fatal, "Error opening nestfile.nc")
92 end if
93 
94 test_passed = .false.
95 inquire(file="nestfile.res.nc", exist=test_passed)
96 if (.not. test_passed) call mpp_error(fatal, "Error creating nestfile.res.nc")
97 
98 call mpp_exit()
99 
100 end program test_file_appendix
