62  !> @brief Determine the next time data/file is to be written based on the frequency and units using the clock.
63  !! For example, if doing daily averages and the input time is day1_hour3, the output time will be day2_hour0.
64  !! @return the next time data/file is to be written
65  TYPE(time_type) function diag_clock_time_inc(time, output_freq, output_units, err_msg)
66  TYPE(time_type), INTENT(in) :: time !< Current model time.
67  INTEGER, INTENT(in) :: output_freq !< Output frequency number value.
68  INTEGER, INTENT(in) :: output_units !< Output frequency unit.
69  CHARACTER(len=*), INTENT(out), OPTIONAL :: err_msg !< Function error message.
70  !! An empty string indicates the next output
71  !! time was found successfully.
72  CHARACTER(len=128) :: error_message_local !< Local variable to store the error_message
73  integer :: cyear !< The current year stored in the time type
74  integer :: cmonth !< The current month stored in the time type
75  integer :: cday !< The current day stored in the time type
76  integer :: chour !< The current hour stored in the time type
77  integer :: cmin !< The current minute stored in the time type
78  integer :: csecond !< The current second stored in the time type
79  type(time_type) :: my_time !< Time set at the begining of the <output_freq>
80 
81  IF ( PRESENT(err_msg) ) err_msg = ''
82  error_message_local = ''
83 
84  IF ( get_calendar_type() == no_calendar) then
85  error_message_local = 'If using use_clock_average =.TRUE., your calendar must be set.'
86  IF ( fms_error_handler('diag_clock_time_inc',error_message_local,err_msg) ) RETURN
87  endif
88 
89  ! special values for output frequency are -1 for output at end of run
90  ! and 0 for every timestep. Need to check for these here?
91  ! Return zero time increment, hopefully this value is never used
92  IF ( output_freq == end_of_run .OR. output_freq == every_time ) THEN
93  diag_clock_time_inc = time
94  RETURN
95  END IF
96 
97  call get_date(time, cyear, cmonth, cday, chour, cmin, csecond)
98 
99  select case (output_units)
100  case (diag_seconds)
101  my_time = set_date(cyear, cmonth, cday, chour, cmin, csecond) !< set my_time to the begining of the hour
102  diag_clock_time_inc = increment_date(my_time, 0, 0, 0, 0, 0, output_freq, err_msg=error_message_local)
103  case (diag_minutes)
104  my_time = set_date(cyear, cmonth, cday, chour, cmin, 0) !< set my_time to the begining of the hour
105  diag_clock_time_inc = increment_date(my_time, 0, 0, 0, 0, output_freq, 0, err_msg=error_message_local)
106  case (diag_hours)
107  my_time = set_date(cyear, cmonth, cday, chour, 0, 0) !< set my_time to the begining of the hour
108  diag_clock_time_inc = increment_date(my_time, 0, 0, 0, output_freq, 0, 0, err_msg=error_message_local)
109  case (diag_days)
110  my_time = set_date(cyear, cmonth, cday, 0, 0, 0) !< set my_time to the begining of the day
111  diag_clock_time_inc = increment_date(my_time, 0, 0, output_freq, 0, 0, 0, err_msg=error_message_local)
112  case (diag_months)
113  my_time = set_date(cyear, cmonth, 1, 0, 0, 0) !< set my_time to the begining of the month
114  diag_clock_time_inc = increment_date(my_time, 0, output_freq, 0, 0, 0, 0, err_msg=error_message_local)
115  case (diag_years)
116  my_time = set_date(cyear, 1, 1, 0, 0, 0) !< set my_time to the begining of the year
117  diag_clock_time_inc = increment_date(my_time, output_freq, 0, 0, 0, 0, 0, err_msg=error_message_local)
118  end select
119 
120  end function diag_clock_time_inc

122  !> @brief Determine the next time data/file is to be written based on the frequency and units using forecast time.
123  !! For example, if doing daily averages and the input time is day1_hour3, the output time will be day2_hour3.
124  !! @return the next time data/file is to be written
125  TYPE(time_type) function diag_forecast_time_inc(time, output_freq, output_units, err_msg)
126  TYPE(time_type), INTENT(in) :: time !< Current model time.
127  INTEGER, INTENT(in) :: output_freq !< Output frequency number value.
128  INTEGER, INTENT(in) :: output_units !< Output frequency unit.
129  CHARACTER(len=*), INTENT(out), OPTIONAL :: err_msg !< Function error message.
130  !! An empty string indicates the next output
131  !! time was found successfully.
132 
133  CHARACTER(len=128) :: error_message_local !< Local variable to store the error_message
134 
135  IF ( PRESENT(err_msg) ) err_msg = ''
136  error_message_local = ''
137 
138  ! special values for output frequency are -1 for output at end of run
139  ! and 0 for every timestep. Need to check for these here?
140  ! Return zero time increment, hopefully this value is never used
141  IF ( output_freq == end_of_run .OR. output_freq == every_time ) THEN
142  diag_forecast_time_inc = time
143  RETURN
144  END IF
145 
146  ! Make sure calendar was not set after initialization
147  IF ( output_units == diag_seconds ) THEN
148  IF ( get_calendar_type() == no_calendar ) THEN
149  diag_forecast_time_inc = increment_time(time, output_freq, 0, err_msg=error_message_local)
150  ELSE
151  diag_forecast_time_inc = increment_date(time, 0, 0, 0, 0, 0, output_freq, err_msg=error_message_local)
152  END IF
153  ELSE IF ( output_units == diag_minutes ) THEN
154  IF ( get_calendar_type() == no_calendar ) THEN
155  diag_forecast_time_inc = increment_time(time, nint(output_freq*seconds_per_minute), 0, &
156  &err_msg=error_message_local)
157  ELSE
158  diag_forecast_time_inc = increment_date(time, 0, 0, 0, 0, output_freq, 0, err_msg=error_message_local)
159  END IF
160  ELSE IF ( output_units == diag_hours ) THEN
161  IF ( get_calendar_type() == no_calendar ) THEN
162  diag_forecast_time_inc = increment_time(time, nint(output_freq*seconds_per_hour), 0, &
163  &err_msg=error_message_local)
164  ELSE
165  diag_forecast_time_inc = increment_date(time, 0, 0, 0, output_freq, 0, 0, err_msg=error_message_local)
166  END IF
167  ELSE IF ( output_units == diag_days ) THEN
168  IF (get_calendar_type() == no_calendar) THEN
169  diag_forecast_time_inc = increment_time(time, 0, output_freq, err_msg=error_message_local)
170  ELSE
171  diag_forecast_time_inc = increment_date(time, 0, 0, output_freq, 0, 0, 0, err_msg=error_message_local)
172  END IF
173  ELSE IF ( output_units == diag_months ) THEN
174  IF (get_calendar_type() == no_calendar) THEN
175  error_message_local = 'output units of months NOT allowed with no calendar'
176  ELSE
177  diag_forecast_time_inc = increment_date(time, 0, output_freq, 0, 0, 0, 0, err_msg=error_message_local)
178  END IF
179  ELSE IF ( output_units == diag_years ) THEN
180  IF ( get_calendar_type() == no_calendar ) THEN
181  error_message_local = 'output units of years NOT allowed with no calendar'
182  ELSE
183  diag_forecast_time_inc = increment_date(time, output_freq, 0, 0, 0, 0, 0, err_msg=error_message_local)
184  END IF
185  ELSE
186  error_message_local = 'illegal output units'
187  END IF
188 
189  IF ( error_message_local /= '' ) THEN
190  IF ( fms_error_handler('diag_forecast_time_inc',error_message_local,err_msg) ) RETURN
191  END IF
192  END FUNCTION diag_forecast_time_inc

194  !> @brief This function determines a string based on current time.
195  !! This string is used as suffix in output file name
196  !! @return Character(len=128) get_time_string
197  CHARACTER(len=128) FUNCTION get_time_string(filename, current_time)
198  CHARACTER(len=*), INTENT(in) :: filename !< File name.
199  TYPE(time_type), INTENT(in) :: current_time !< Current model time.
200 
201  INTEGER :: yr1 !< get from current time
202  INTEGER :: mo1 !< get from current time
203  INTEGER :: dy1 !< get from current time
204  INTEGER :: hr1 !< get from current time
205  INTEGER :: mi1 !< get from current time
206  INTEGER :: sc1 !< get from current time
207  INTEGER :: yr2 !< for computing next_level time unit
208  INTEGER :: dy2 !< for computing next_level time unit
209  INTEGER :: hr2 !< for computing next_level time unit
210  INTEGER :: mi2 !< for computing next_level time unit
211  INTEGER :: yr1_s !< actual values to write string
212  INTEGER :: mo1_s !< actual values to write string
213  INTEGER :: dy1_s !< actual values to write string
214  INTEGER :: hr1_s !< actual values to write string
215  INTEGER :: mi1_s !< actual values to write string
216  INTEGER :: sc1_s !< actual values to write string
217  INTEGER :: abs_day !< component of current_time
218  INTEGER :: abs_sec !< component of current_time
219  INTEGER :: days_per_month(12) = (/31,28,31,30,31,30,31,31,30,31,30,31/)
220  INTEGER :: julian_day, i, position, len, first_percent
221  CHARACTER(len=1) :: width !< width of the field in format write
222  CHARACTER(len=10) :: format
223  CHARACTER(len=20) :: yr !< string of current time (output)
224  CHARACTER(len=20) :: mo !< string of current time (output)
225  CHARACTER(len=20) :: dy !< string of current time (output)
226  CHARACTER(len=20) :: hr !< string of current time (output)
227  CHARACTER(len=20) :: mi !< string of current time (output)
228  CHARACTER(len=20) :: sc !< string of current time (output)
229  CHARACTER(len=128) :: filetail
230 
231  format = '("_",i*.*)'
232  CALL get_date(current_time, yr1, mo1, dy1, hr1, mi1, sc1)
233  len = len_trim(filename)
234  first_percent = index(filename, '%')
235  filetail = filename(first_percent:len)
236  ! compute year string
237  position = index(filetail, 'yr')
238  IF ( position > 0 ) THEN
239  width = filetail(position-1:position-1)
240  yr1_s = yr1
241  format(7:9) = width//'.'//width
242  WRITE(yr, format) yr1_s
243  yr2 = 0
244  ELSE
245  yr = ' '
246  yr2 = yr1 - 1
247  END IF
248  ! compute month string
249  position = index(filetail, 'mo')
250  IF ( position > 0 ) THEN
251  width = filetail(position-1:position-1)
252  mo1_s = yr2*12 + mo1
253  format(7:9) = width//'.'//width
254  WRITE(mo, format) mo1_s
255  ELSE
256  mo = ' '
257  END IF
258  ! compute day string
259  IF ( len_trim(mo) > 0 ) THEN ! month present
260  dy1_s = dy1
261  dy2 = dy1_s - 1
262  ELSE IF ( len_trim(yr) >0 ) THEN ! no month, year present
263  ! compute julian day
264  IF ( mo1 == 1 ) THEN
265  dy1_s = dy1
266  ELSE
267  julian_day = 0
268  DO i = 1, mo1-1
269  julian_day = julian_day + days_per_month(i)
270  END DO
271  IF ( leap_year(current_time) .AND. mo1 > 2 ) julian_day = julian_day + 1
272  julian_day = julian_day + dy1
273  dy1_s = julian_day
274  END IF
275  dy2 = dy1_s - 1
276  ELSE ! no month, no year
277  CALL get_time(current_time, abs_sec, abs_day)
278  dy1_s = abs_day
279  dy2 = dy1_s
280  END IF
281  position = index(filetail, 'dy')
282  IF ( position > 0 ) THEN
283  width = filetail(position-1:position-1)
284  FORMAT(7:9) = width//'.'//width
285  WRITE(dy, format) dy1_s
286  ELSE
287  dy = ' '
288  END IF
289  ! compute hour string
290  IF ( len_trim(dy) > 0 ) THEN
291  hr1_s = hr1
292  ELSE
293  hr1_s = dy2*24 + hr1
294  END IF
295  hr2 = hr1_s
296  position = index(filetail, 'hr')
297  IF ( position > 0 ) THEN
298  width = filetail(position-1:position-1)
299  format(7:9) = width//'.'//width
300  WRITE(hr, format) hr1_s
301  ELSE
302  hr = ' '
303  END IF
304  ! compute minute string
305  IF ( len_trim(hr) > 0 ) THEN
306  mi1_s = mi1
307  ELSE
308  mi1_s = hr2*60 + mi1
309  END IF
310  mi2 = mi1_s
311  position = index(filetail, 'mi')
312  IF(position>0) THEN
313  width = filetail(position-1:position-1)
314  format(7:9) = width//'.'//width
315  WRITE(mi, format) mi1_s
316  ELSE
317  mi = ' '
318  END IF
319  ! compute second string
320  IF ( len_trim(mi) > 0 ) THEN
321  sc1_s = sc1
322  ELSE
323  sc1_s = nint(mi2*seconds_per_minute) + sc1
324  END IF
325  position = index(filetail, 'sc')
326  IF ( position > 0 ) THEN
327  width = filetail(position-1:position-1)
328  format(7:9) = width//'.'//width
329  WRITE(sc, format) sc1_s
330  ELSE
331  sc = ' '
332  ENDIF
333  get_time_string = trim(yr)//trim(mo)//trim(dy)//trim(hr)//trim(mi)//trim(sc)
334  END FUNCTION get_time_string

336  !> @brief Return the difference between two times in units.
337  !! @return Real get_data_dif
338  REAL function get_date_dif(t2, t1, units)
339  TYPE(time_type), INTENT(in) :: t2 !< Most recent time.
340  TYPE(time_type), INTENT(in) :: t1 !< Most distant time.
341  INTEGER, INTENT(in) :: units !< Unit of return value.
342 
343  INTEGER :: dif_seconds, dif_days
344  TYPE(time_type) :: dif_time
345 
346  IF ( t2 < t1 ) CALL mpp_error(fatal, 'diag_util_mod::get_date_dif '//&
347  &'in variable t2 is less than in variable t1')
348 
349  dif_time = t2 - t1
350 
351  CALL get_time(dif_time, dif_seconds, dif_days)
352 
353  IF ( units == diag_seconds ) THEN
354  get_date_dif = dif_seconds + seconds_per_day * dif_days
355  ELSE IF ( units == diag_minutes ) THEN
356  get_date_dif = 1440 * dif_days + dif_seconds / seconds_per_minute
357  ELSE IF ( units == diag_hours ) THEN
358  get_date_dif = 24 * dif_days + dif_seconds / seconds_per_hour
359  ELSE IF ( units == diag_days ) THEN
360  get_date_dif = dif_days + dif_seconds / seconds_per_day
361  ELSE IF ( units == diag_months ) THEN
362  CALL mpp_error(fatal, 'diag_util_mod::get_date_dif months not supported as output units')
363  ELSE IF ( units == diag_years ) THEN
364  CALL mpp_error(fatal, 'diag_util_mod::get_date_dif years not supported as output units')
365  ELSE
366  CALL mpp_error(fatal, 'diag_util_mod::diag_date_dif illegal time units')
367  END IF
368  END FUNCTION get_date_dif
