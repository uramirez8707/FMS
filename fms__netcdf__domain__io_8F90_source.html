182 !> @brief Given a variable, get the index of the "x" or "y" domain decomposed
183 !! dimension.
184 !! @return Index of the domain decomposed dimension or else
185 !! no_domain_decomposed_dimension.
186 function get_domain_decomposed_dimension_index(fileobj, variable_name, &
187  xory, broadcast) &
188  result(index_)
189 
190  type(fmsnetcdfdomainfile_t), intent(in), target :: fileobj !< File object.
191  character(len=*), intent(in) :: variable_name !< Variable name.
192  character(len=*), intent(in) :: xory !< String telling which dimension to
193  !! look for. Valid values are "x"
194  !! or "y".
195  logical, intent(in), optional :: broadcast !< Flag controlling whether or
196  !! not the index will be
197  !! broadcast to non "I/O root"
198  !! ranks. The broadcast will
199  !! be done by default.
200  integer :: index_
201 
202  integer :: ndims
203  character(len=nf90_max_name), dimension(:), allocatable :: dim_names
204  type(domaindimension_t), dimension(:), pointer :: p
205  integer :: n
206  integer :: i
207 
208  index_ = no_domain_decomposed_dimension
209  if (fileobj%is_root) then
210  ndims = get_variable_num_dimensions(fileobj, variable_name, broadcast=.false.)
211  allocate(dim_names(ndims))
212  dim_names(:) = ""
213  call get_variable_dimension_names(fileobj, variable_name, dim_names, broadcast=.false.)
214  if (string_compare(xory, x, .true.)) then
215  p => fileobj%xdims
216  n = fileobj%nx
217  elseif (string_compare(xory, y, .true.)) then
218  p => fileobj%ydims
219  n = fileobj%ny
220  else
221  call error("unrecognized xory flag value.")
222  endif
223  do i = 1, size(dim_names)
224  if (get_domain_decomposed_index(dim_names(i), p, n) .ne. variable_not_found) then
225  index_ = i
226  exit
227  endif
228  enddo
229  deallocate(dim_names)
230  endif
231  if (present(broadcast)) then
232  if (.not. broadcast) then
233  return
234  endif
235  endif
236  call mpp_broadcast(index_, fileobj%io_root, pelist=fileobj%pelist)
237 end function get_domain_decomposed_dimension_index

240 !> @brief Determine if a variable is "domain decomposed."
241 !! @return Flag telling if the variable is "domain decomposed."
242 function is_variable_domain_decomposed(fileobj, variable_name, broadcast, &
243  xindex, yindex, xpos, ypos) &
244  result(is_decomposed)
245 
246  type(fmsnetcdfdomainfile_t), intent(in) :: fileobj !< File object.
247  character(len=*), intent(in) :: variable_name !< Variable name.
248  logical, intent(in), optional :: broadcast !< Flag controlling whether or
249  !! not the index will be
250  !! broadcast to non "I/O root"
251  !! ranks. The broadcast will
252  !! be done by default.
253  integer, intent(out), optional :: xindex !< The index of the domain
254  !! x dimension.
255  integer, intent(out), optional :: yindex !< The index of the domain
256  !! y dimension.
257  integer, intent(out), optional :: xpos !< Domain position of the x dimension.
258  integer, intent(out), optional :: ypos !< Domain position of the y dimension.
259  logical :: is_decomposed
260 
261  integer, dimension(2) :: indices
262  integer :: ndims
263  character(len=nf90_max_name), dimension(:), allocatable :: dim_names
264 
265  indices(1) = get_domain_decomposed_dimension_index(fileobj, variable_name, x, broadcast)
266  if (present(xindex)) then
267  xindex = indices(1)
268  endif
269  indices(2) = get_domain_decomposed_dimension_index(fileobj, variable_name, y, broadcast)
270  if (present(yindex)) then
271  yindex = indices(2)
272  endif
273  is_decomposed = (indices(1) .ne. no_domain_decomposed_dimension) .and. &
274  (indices(2) .ne. no_domain_decomposed_dimension)
275  if (is_decomposed) then
276  if (.not. present(xpos) .and. .not. present(ypos)) then
277  return
278  endif
279  ndims = get_variable_num_dimensions(fileobj, variable_name, broadcast)
280  allocate(dim_names(ndims))
281  dim_names(:) = ""
282  call get_variable_dimension_names(fileobj, variable_name, dim_names, broadcast)
283  if (present(xpos)) then
284  xpos = get_domain_position(dim_names(indices(1)), fileobj%xdims, fileobj%nx)
285  endif
286  if (present(ypos)) then
287  ypos = get_domain_position(dim_names(indices(2)), fileobj%ydims, fileobj%ny)
288  endif
289  deallocate(dim_names)
290  else
291  if (present(xpos)) then
292  xpos = -1
293  endif
294  if (present(ypos)) then
295  ypos = -1
296  endif
297  endif
298 end function is_variable_domain_decomposed

326 !> @brief Open a domain netcdf file.
327 !! @return Flag telling if the open completed successfully.
328 function open_domain_file(fileobj, path, mode, domain, nc_format, is_restart, dont_add_res_to_filename) &
329  result(success)
330 
331  type(fmsnetcdfdomainfile_t),intent(inout) :: fileobj !< File object.
332  character(len=*), intent(in) :: path !< File path.
333  character(len=*), intent(in) :: mode !< File mode. Allowed values
334  !! are "read", "append", "write", or
335  !! "overwrite".
336  type(domain2d), intent(in) :: domain !< Two-dimensional domain.
337  character(len=*), intent(in), optional :: nc_format !< Netcdf format that
338  !! new files are written
339  !! as. Allowed values
340  !! are: "64bit", "classic",
341  !! or "netcdf4". Defaults to
342  !! "64bit".
343  logical, intent(in), optional :: is_restart !< Flag telling if this file
344  !! is a restart file. Defaults
345  !! to false.
346  logical, intent(in), optional :: dont_add_res_to_filename !< Flag indicating not to add
347  !! ".res" to the filename
348  logical :: success
349 
350  integer, dimension(2) :: io_layout
351  integer, dimension(1) :: tile_id
352  character(len=256) :: combined_filepath
353  type(domain2d), pointer :: io_domain
354  character(len=256) :: distributed_filepath
355  integer :: pelist_size
356  integer, dimension(:), allocatable :: pelist
357  logical :: success2
358  type(fmsnetcdfdomainfile_t) :: fileobj2
359 
360  !Get the path of a "combined" file.
361  io_layout = mpp_get_io_domain_layout(domain)
362  tile_id = mpp_get_tile_id(domain)
363 
364  !< If the number of tiles is greater than 1 or if the current tile is greater
365  !than 1 add .tileX. to the filename
366  if (mpp_get_ntile_count(domain) .gt. 1 .or. tile_id(1) > 1) then
367  call domain_tile_filepath_mangle(combined_filepath, path, tile_id(1))
368  else
369  call string_copy(combined_filepath, path)
370  endif
371 
372  !Get the path of a "distributed" file.
373  io_domain => mpp_get_io_domain(domain)
374  if (.not. associated(io_domain)) then
375  call error("The domain associated with the file:"//trim(fileobj%path)//" does not have an io_domain.")
376  endif
377  if (io_layout(1)*io_layout(2) .gt. 1) then
378  tile_id = mpp_get_tile_id(io_domain)
379  call io_domain_tile_filepath_mangle(distributed_filepath, combined_filepath, tile_id(1))
380  else
381  call string_copy(distributed_filepath, combined_filepath)
382  endif
383 
384  !Make sure the input domain has an I/O domain and get its pelist.
385  pelist_size = mpp_get_domain_npes(io_domain)
386  allocate(pelist(pelist_size))
387  call mpp_get_pelist(io_domain, pelist)
388  fileobj%adjust_indices = .true. !Set the default to true
389 
390  !Open the distibuted files.
391  success = netcdf_file_open(fileobj, distributed_filepath, mode, nc_format, pelist, &
392  is_restart, dont_add_res_to_filename)
393  if (string_compare(mode, "read", .true.) .or. string_compare(mode, "append", .true.)) then
394  if (success) then
395  if (.not. string_compare(distributed_filepath, combined_filepath)) then
396  success2 = netcdf_file_open(fileobj2, combined_filepath, mode, nc_format, pelist, &
397  is_restart, dont_add_res_to_filename)
398  if (success2) then
399  call error("The domain decomposed file:"//trim(fileobj%path)// &
400  & " contains both combined (*.nc) and distributed files (*.nc.XXXX).")
401  endif
402  endif
403  else
404  success = netcdf_file_open(fileobj, combined_filepath, mode, nc_format, pelist, &
405  is_restart, dont_add_res_to_filename)
406  !If the file is combined and the layout is not (1,1) set the adjust_indices flag to false
407  if (success .and. (io_layout(1)*io_layout(2) .gt. 1)) fileobj%adjust_indices = .false.
408  endif
409  endif
410  if (.not. success) then
411  deallocate(pelist)
412  return
413  endif
414 
415  !Store/initialize necessary properties.
416  call string_copy(fileobj%non_mangled_path, path)
417  fileobj%domain = domain
418  allocate(fileobj%xdims(max_num_domain_decomposed_dims))
419  fileobj%nx = 0
420  allocate(fileobj%ydims(max_num_domain_decomposed_dims))
421  fileobj%ny = 0
422  call string_copy(fileobj%non_mangled_path, path)
423 
424  if (string_compare(mode, "write", .true.) .or. string_compare(mode, "overwrite", .true.)) then
425  !Add global attribute needed by mppnccombine.
426  call register_global_attribute(fileobj, "NumFilesInSet", io_layout(1)*io_layout(2))
427  endif
428 end function open_domain_file

444 !> @brief Add a dimension to a file associated with a two-dimensional domain.
445 subroutine register_domain_decomposed_dimension(fileobj, dim_name, xory, domain_position)
446 
447  type(fmsnetcdfdomainfile_t), intent(inout) :: fileobj !< File object.
448  character(len=*), intent(in) :: dim_name !< Dimension name.
449  character(len=*), intent(in) :: xory !< Flag telling if the dimension
450  !! is associated with the "x" or "y"
451  !! axis of the 2d domain. Allowed
452  !! values are "x" or "y".
453  integer, intent(in), optional :: domain_position !< Domain position.
454 
455  integer :: dpos
456  type(domain2d), pointer :: io_domain
457  integer :: domain_size
458  integer :: dim_size
459 
460  dpos = default_domain_position
461  if (mpp_domain_is_symmetry(fileobj%domain) .and. present(domain_position)) then
462  dpos = domain_position
463  endif
464  io_domain => mpp_get_io_domain(fileobj%domain)
465  if (string_compare(xory, x, .true.)) then
466  if (dpos .ne. center .and. dpos .ne. east) then
467  call error("Only domain_position=center or domain_position=EAST is supported for x dimensions."// &
468  & " Fix your register_axis call for file:"&
469  &//trim(fileobj%path)//" and dimension:"//trim(dim_name))
470  endif
471  call mpp_get_global_domain(io_domain, xsize=domain_size, position=dpos)
472  call append_domain_decomposed_dimension(dim_name, dpos, fileobj%xdims, fileobj%nx)
473  elseif (string_compare(xory, y, .true.)) then
474  if (dpos .ne. center .and. dpos .ne. north) then
475  call error("Only domain_position=center or domain_position=NORTH is supported for y dimensions."// &
476  & " Fix your register_axis call for file:"&
477  &//trim(fileobj%path)//" and dimension:"//trim(dim_name))
478  endif
479  call mpp_get_global_domain(io_domain, ysize=domain_size, position=dpos)
480  call append_domain_decomposed_dimension(dim_name, dpos, fileobj%ydims, fileobj%ny)
481  else
482  call error("The register_axis call for file:"//trim(fileobj%path)//" and dimension:"//trim(dim_name)// &
483  & " has an unrecognized xory flag value:"&
484  &//trim(xory)//" only 'x' and 'y' are allowed.")
485  endif
486  if (fileobj%is_readonly .or. (fileobj%mode_is_append .and. dimension_exists(fileobj, dim_name))) then
487  call get_dimension_size(fileobj, dim_name, dim_size, broadcast=.true.)
488  if (dim_size .lt. domain_size) then
489  call error("dimension "//trim(dim_name)//" in the file "//trim(fileobj%path)//" is smaller than the size of" &
490  //" the associated domain "//trim(xory)//" axis.")
491  endif
492  else
493  call netcdf_add_dimension(fileobj, dim_name, domain_size, is_compressed=.false.)
494  endif
495 end subroutine register_domain_decomposed_dimension

498 !> @brief Add a "domain_decomposed" attribute to the axis variables because it is
499 !! required by mppnccombine.
500 !! @internal
501 subroutine add_domain_attribute(fileobj, variable_name)
502 
503  type(fmsnetcdfdomainfile_t), intent(inout) :: fileobj !< File object.
504  character(len=*), intent(in) :: variable_name !< Variable name.
505 
506  type(domain2d), pointer :: io_domain
507  integer :: dpos
508  integer :: sg
509  integer :: eg
510  integer :: s
511  integer :: e
512  integer, dimension(2) :: io_layout !< Io_layout in the fileobj's domain
513 
514  !< Don't add the "domain_decomposition" variable attribute if the io_layout is
515  !! 1,1, to avoid frecheck "failures"
516  io_layout = mpp_get_io_domain_layout(fileobj%domain)
517  if (io_layout(1) .eq. 1 .and. io_layout(2) .eq. 1) return
518 
519  io_domain => mpp_get_io_domain(fileobj%domain)
520  dpos = get_domain_decomposed_index(variable_name, fileobj%xdims, fileobj%nx)
521  if (dpos .ne. variable_not_found) then
522  dpos = fileobj%xdims(dpos)%pos
523  call mpp_get_global_domain(fileobj%domain, xbegin=sg, xend=eg, position=dpos)
524  call mpp_get_global_domain(io_domain, xbegin=s, xend=e, position=dpos)
525  call register_variable_attribute(fileobj, variable_name, "domain_decomposition", &
526  (/sg, eg, s, e/))
527  else
528  dpos = get_domain_decomposed_index(variable_name, fileobj%ydims, fileobj%ny)
529  if (dpos .ne. variable_not_found) then
530  dpos = fileobj%ydims(dpos)%pos
531  call mpp_get_global_domain(fileobj%domain, ybegin=sg, yend=eg, position=dpos)
532  call mpp_get_global_domain(io_domain, ybegin=s, yend=e, position=dpos)
533  call register_variable_attribute(fileobj, variable_name, "domain_decomposition", &
534  (/sg, eg, s, e/))
535  endif
536  endif
537 end subroutine add_domain_attribute
