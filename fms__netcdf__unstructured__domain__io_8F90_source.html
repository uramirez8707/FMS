72 !> @brief Open a netcdf file that is associated with an unstructured domain.
73 !! @return Flag telling if the open completed successfully.
74 function open_unstructured_domain_file(fileobj, path, mode, domain, nc_format, &
75  is_restart, dont_add_res_to_filename) &
76  result(success)
77 
78  type(fmsnetcdfunstructureddomainfile_t), intent(inout) :: fileobj !< File object.
79  character(len=*), intent(in) :: path !< File path.
80  character(len=*), intent(in) :: mode !< File mode. Allowed values
81  !! are "read", "append", "write", or
82  !! "overwrite".
83  type(domainug), intent(in) :: domain !< Unstructured domain.
84  character(len=*), intent(in), optional :: nc_format !< Netcdf format that
85  !! new files are written
86  !! as. Allowed values
87  !! are: "64bit", "classic",
88  !! or "netcdf4". Defaults to
89  !! "64bit".
90  logical, intent(in), optional :: is_restart !< Flag telling if this file
91  !! is a restart file. Defaults
92  !! to false.
93  logical, intent(in), optional :: dont_add_res_to_filename !< Flag indicating not to add
94  !! ".res" to the filename
95  logical :: success
96 
97  type(domainug), pointer :: io_domain
98  integer :: pelist_size
99  integer, dimension(:), allocatable :: pelist
100  character(len=256) :: buf
101  character(len=256) :: buf2
102  integer :: tile_id
103 
104  !Get the input domain's I/O domain pelist.
105  io_domain => mpp_get_ug_io_domain(domain)
106  if (.not. associated(io_domain)) then
107  call error("The input domain associated with the file:"//trim(fileobj%path)//" does not have an io_domain.")
108  endif
109  pelist_size = mpp_get_ug_domain_npes(io_domain)
110  allocate(pelist(pelist_size))
111  call mpp_get_ug_domain_pelist(io_domain, pelist)
112 
113  !Add the domain tile id to the file name (if necessary).
114  call string_copy(buf, path)
115  if (mpp_get_ug_domain_ntiles(domain) .gt. 1) then
116  tile_id = mpp_get_ug_domain_tile_id(domain)
117  call domain_tile_filepath_mangle(buf, path, tile_id)
118  endif
119 
120  success = .false.
121  if (string_compare(mode, "read", .true.) .or. string_compare(mode, "append", .true.)) then
122  !Only for reading: attempt to open non-distributed files.
123  success = netcdf_file_open(fileobj, buf, mode, nc_format, pelist, is_restart, dont_add_res_to_filename)
124  endif
125  if (.not. success) then
126  !Add the domain tile id to the file name (if necessary).
127  if (mpp_get_io_domain_ug_layout(domain) .gt. 1) then
128  tile_id = mpp_get_ug_domain_tile_id(io_domain)
129  call string_copy(buf2, buf)
130  call io_domain_tile_filepath_mangle(buf, buf2, tile_id)
131  endif
132 
133  !Open distributed files.
134  success = netcdf_file_open(fileobj, buf, mode, nc_format, pelist, is_restart, dont_add_res_to_filename)
135  endif
136  deallocate(pelist)
137 
138  if (.not. success) then
139  !This branch should only be entered if the file attempting to be read
140  !does not exist.
141  return
142  endif
143 
144  !Store/initialize necessary properties.
145  fileobj%domain = domain
146  call string_copy(fileobj%non_mangled_path, path)
147 end function open_unstructured_domain_file

159 !> @brief Add an unstructured dimension.
160 subroutine register_unstructured_dimension(fileobj, dim_name)
161 
162  type(fmsnetcdfunstructureddomainfile_t), intent(inout) :: fileobj !< File object.
163  character(len=*), intent(in) :: dim_name !< Dimension name.
164 
165  type(domainug),pointer :: io_domain
166  integer, dimension(:), allocatable :: c
167  integer, dimension(:), allocatable :: e
168 
169  allocate(c(size(fileobj%pelist)))
170  allocate(e(size(fileobj%pelist)))
171  io_domain => mpp_get_ug_io_domain(fileobj%domain)
172  call mpp_get_ug_compute_domains(io_domain, begin=c, size=e)
173  if (c(1) .ne. 1) then
174  c(:) = c(:) - c(1) + 1
175  endif
176  call register_compressed_dimension(fileobj, dim_name, c, e)
177  deallocate(c)
178  deallocate(e)
179 end subroutine register_unstructured_dimension
