120  !> Routine to set affinity for a component
121  subroutine fms_affinity_set (component, use_hyper_thread, nthreads)
122  !--- interface variables
123  character(len=*), intent(in):: component !< Component name
124  logical, intent(in):: use_hyper_thread !< .TRUE. if using hyperthreads
125  integer, intent(in):: nthreads !< Number of threads
126 
127  !--- local declarations for Fortran/C affinity interoperability
128  integer(c_int):: cpuset_sz
129  integer(c_int), dimension(:), allocatable:: cpu_set
130  integer(c_int):: retcode
131 
132  !--- local variables
133  character(len=32):: h_name
134  integer:: msg_type
135  integer:: th_num
136  integer:: indx
137 
138  if (.not. affinity) return
139 
140  if (strict) then
141  msg_type = fatal
142  else
143  msg_type = note
144  endif
145 
146  h_name = 'generic '
147 
148  !--- allocate storage for cpuset
149  if (use_hyper_thread) then
150  cpuset_sz = nthreads
151  else
152  cpuset_sz = nthreads * 2
153  endif
154  allocate (cpu_set(0:cpuset_sz-1))
155 
156  !--- get cpuset for this MPI-rank
157  retcode = get_cpuset(cpuset_sz, cpu_set, mpp_pe(), debug_cpuset)
158  if (retcode == -1) then
159  call error_mesg('fms_affinity_set',trim(component)//' cpu_set size > allocated storage',fatal)
160  elseif ( (retcode == cpuset_sz/2) .and. (retcode == nthreads) ) then
161  call error_mesg('fms_affinity_set',trim(component)//' affinity assumes hyper-threading hardware disabled',note)
162  elseif (retcode < cpuset_sz) then
163  call error_mesg('fms_affinity_set',trim(component)//' cpu_set size smaller than expected',msg_type)
164  endif
165 
166  !--- set the affinity for the MPI-rank
167  retcode = set_cpu_affinity(cpu_set(0))
168  if (retcode == -1) then
169  call error_mesg('fms_affinity_set',trim(component)//': issue setting cpu affinity', fatal)
170  endif
171
