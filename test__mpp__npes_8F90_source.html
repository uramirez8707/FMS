19 program test_mpp_npes
20 !> @author Tom Robinson
21 !> @description This test first reads the environment variable NUM_PES which
22 !! must be set before running the program and should be set to the number
23 !! of PEs that are being used to run the program. MPP_NPES is then called
24 !! and the value is compared to NUM_PES. If the numbers are the same, then
25 !! the test is successful.
26  use mpp_mod, only : mpp_init, mpp_init_test_peset_allocated, mpp_pe, mpp_npes, &
27  stderr, stdout, mpp_error, fatal
28  implicit none
29  integer :: npes !< The total number of PEs returned from mpp_npes
30  character (len=20) :: string_npes !< npes converted to a string
31  integer :: test_npes !< The number of PEs used to run the program
32  character(len=:),allocatable :: env_pes !< test_npes as a string
33  integer :: len_env_var !< The length of env_pes
34  integer :: ierr !< MPI error return
35 !> Initialize MPI
36  call mpp_init(test_level=mpp_init_test_peset_allocated)
37 !> Find the number of PEs that were used to run the program
38  CALL get_environment_variable("NUM_PES",length=len_env_var) !> Get the length
39  allocate(character(len=len_env_var) :: env_pes) !> Allocate the string
40  CALL get_environment_variable("NUM_PES",value=env_pes) !> Get the number
41  read(env_pes,*) test_npes !> Convert to an integer
42 !> Get the total number of PEs
43  npes = mpp_npes()
44 !> Check that the number of PEs matches the number used to run the program
45  if (npes .ne. test_npes) then
46  write(string_npes,*) npes
47  call mpp_error(fatal, "The number of PEs used to run the program is "&
48  //env_pes//" but mpp_npes returned "//trim(string_npes) )
49  endif
50 !> Finalize MPI
51  call mpi_finalize (ierr)
52 end program test_mpp_npes
