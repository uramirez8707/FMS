243 !> @brief Used to dermine the value of a key from a keyname
244 subroutine get_value_from_key_0d(file_id, block_id, key_name, key_value, is_optional)
245  integer, intent(in) :: file_id !< File id of the yaml file to search
246  integer, intent(in) :: block_id !< ID corresponding to the block you want the key for
247  character(len=*), intent(in) :: key_name !< Name of the key you want the value for
248  class(*), intent(inout):: key_value !< Value of the key
249  logical, intent(in), optional :: is_optional !< Flag indicating if it is okay for they key to not exist.
250  !! If the key does not exist key_value will not be set, so it
251  !! is the user's responsibility to initialize it before the call
252 
253  character(len=255) :: buffer !< String buffer with the value
254 
255  type(c_ptr) :: c_buffer !< c pointer with the value
256  integer(kind=c_int) :: sucess !< Flag indicating if the value was obtained sucessfully
257  logical :: optional !< Flag indicating that the key was optional
258  integer :: err_unit !< integer with io error
259 
260  optional = .false.
261  if (present(is_optional)) optional = is_optional
262 
263  if (.not. is_valid_file_id(file_id)) call mpp_error(fatal, "The file id in your get_value_from_key call is invalid! Check your call.")
264  if (.not. is_valid_block_id(file_id, block_id)) call mpp_error(fatal, "The block id in your get_value_from_key call is invalid! Check your call.")
265 
266  c_buffer = get_value_from_key_wrap(file_id, block_id, key_name, sucess)
267  if (sucess == 1) then
268  buffer = fms_c2f_string(c_buffer)
269 
270  select type (key_value)
271  type is (integer(kind=i4_kind))
272  read(buffer,*, iostat=err_unit) key_value
273  if (err_unit .ne. 0) call mpp_error(fatal, "Key:"//trim(key_name)//" Error converting '"//trim(buffer)//"' to i4")
274  type is (integer(kind=i8_kind))
275  read(buffer,*, iostat=err_unit) key_value
276  if (err_unit .ne. 0) call mpp_error(fatal, "Key:"//trim(key_name)//" Error converting '"//trim(buffer)//"' to i8")
277  type is (real(kind=r4_kind))
278  read(buffer,*, iostat=err_unit) key_value
279  if (err_unit .ne. 0) call mpp_error(fatal, "Key:"//trim(key_name)//" Error converting '"//trim(buffer)//"' to r4")
280  type is (real(kind=r8_kind))
281  read(buffer,*, iostat=err_unit) key_value
282  if (err_unit .ne. 0) call mpp_error(fatal, "Key:"//trim(key_name)//" Error converting '"//trim(buffer)//"' to r8")
283  type is (character(len=*))
284  key_value = buffer
285  class default
286  call mpp_error(fatal, "The type of your buffer in your get_value_from_key call for key "//trim(key_name)//&
287  &" is not supported. Only i4, i8, r4, r8 and strings are supported.")
288  end select
289  else
290  if(.not. optional) call mpp_error(fatal, "Error getting the value for key:"//trim(key_name))
291  endif
292 
293 end subroutine get_value_from_key_0d
