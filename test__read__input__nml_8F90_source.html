21 !! @brief Tests the read_input_nml subroutine in four difference scenarios
22 !! @author Colin Gladue
23 !! @email gfdl.climate.model.info@noaa.gov
24 
25 program test_read_input_nml
26 
27  use mpp_mod, only : mpp_init, mpp_init_test_peset_allocated
28  use mpp_mod, only : mpp_error, fatal, note
29  use mpp_mod, only : read_input_nml, mpp_get_current_pelist_name
30  use mpp_mod, only : input_nml_file
31 #include<file_version.h>
32 
33 character(len=200) :: line !< Storage location of lines read from the input nml
34 character(len=128) :: filename !< Name of input nml file to be read
35 integer :: stat !< IOSTAT output integer
36 integer :: n, m !< Loop counting variable
37 integer :: current_pelist_name_len_plus1 !< Current pelist name length plus 1
38 integer :: ierr !< used by MPI_FINALIZE
39 character(len=:), allocatable :: toobig !< String passed as argument into read_input_nml that is
40  !!larger than pelis_name and should raise an error
41 
42 namelist /test_read_input_nml_nml/ test_numb
43 
44 open(10, file="test_numb.nml", form="formatted", status="old")
45 read(10, nml = test_read_input_nml_nml)
46 close(10)
47 
48 call mpp_init(test_level=mpp_init_test_peset_allocated)
49 
50 if (test_numb == 1 .or. test_numb == 2 .or. test_numb == 4) then
51  ! Test 1: Tests the subroutine on a valid input nml full of data,
52  ! with no arguments passed to read_input_nml()
53  ! Test 2: Tests the subroutine on a valid input nml full of data,
54  ! with a string passed to read_input_nml() in order to read a different nml
55  ! Test 4: Tests the subroutine on a valid empty input nml,
56  ! with no arguments passed to read_input_nml()
57  if (test_numb == 1) then
58  filename = "input.nml"
59  call read_input_nml()
60  else if (test_numb == 4) then
61  filename = "input_blank.nml"
62  call read_input_nml("blank")
63  else if (test_numb == 2) then
64  filename = "input_alternative.nml"
65  call read_input_nml("alternative")
66  end if
67  open(1, file=filename, iostat=stat) ! Open input nml or alternative
68  n = 1
69  do
70  read(1, '(A)', iostat=stat) line
71  if (stat.eq.-1) then
72  exit
73  end if
74  if (input_nml_file(n).ne.line) then
75  call mpp_error(fatal, "data resident in ./input.nml does not match&
76  & that read into input_nml_file by read_input_nml")
77  end if
78  n = n + 1
79  end do
80  close(1)
81 
82 else if (test_numb.eq.3) then
83  ! Test 3: Tests with an invalid pelist_name_in pass as an argument. An invalid
84  ! pelist_name_in would be one who's size is greater than local pelist_name
85  current_pelist_name_len_plus1 = len(mpp_get_current_pelist_name()) + 1
86  allocate(character(len=current_pelist_name_len_plus1) :: toobig)
87  call read_input_nml(pelist_name_in=toobig)
88  ! Call read_input_nml
89  ! with the local
90  ! pelist_name plus an
91  ! extra character "e"
92  deallocate(toobig)
93 end if
94 
95 call mpi_finalize(ierr)
96 
97 end program test_read_input_nml
