20 subroutine MPP_TYPE_CREATE_(field, array_of_subsizes, array_of_starts, &
21  dtype_out)
22  MPP_TYPE_, intent(in) :: field(:,:,:)
23  integer, intent(in) :: array_of_subsizes(:)
24  integer, intent(in) :: array_of_starts(:)
25  type(mpp_type), target, intent(out) :: dtype_out
26 
27  type(mpp_type), pointer :: dtype
28  integer :: newtype ! MPI datatype ID
29 
30  if (.NOT. module_is_initialized) &
31  call mpp_error(FATAL, 'MPP_TYPE_CREATE_: You must first call mpp_init.')
32 
33  if (current_clock .NE. 0) &
34  call SYSTEM_CLOCK(start_tick)
35 
36  if (verbose) &
37  call mpp_error(NOTE, 'MPP_TYPE_CREATE_: &
38  &using MPI_Type_create_subarray...')
39 
40  dtype => datatypes%head
41  ! TODO: Check mpp_byte
42 
43  ! Check if mpp_type already exists
44  do while (.not. associated(dtype))
45  dtype => dtype%next
46 
47  if (dtype%ndims /= rank(field)) cycle
48  if (any(dtype%sizes /= shape(field))) cycle
49  if (any(dtype%subsizes /= array_of_subsizes)) cycle
50  if (any(dtype%starts /= array_of_starts)) cycle
51  if (dtype%etype /= MPI_TYPE_) cycle
52 
53  ! If all parameters match, then the datatype exists and return dtype
54  dtype%counter = dtype%counter + 1
55  dtype_out = dtype
56  return
57  end do
58 
59  ! The type does not exist; create a new internal type
60  call MPI_Type_create_subarray( &
61  rank(field), &
62  shape(field), &
63  array_of_subsizes, &
64  array_of_starts, &
65  MPI_ORDER_FORTRAN, &
66  MPI_TYPE_, &
67  newtype, &
68  error &
69  )
70 
71  ! Register on the MPI runtime
72  call MPI_Type_commit(newtype, error)
73 
74  ! Create new entry
75  allocate(dtype)
76  allocate(dtype%sizes(rank(field)))
77  allocate(dtype%subsizes(rank(field)))
78  allocate(dtype%starts(rank(field)))
79 
80  ! Populate values
81  dtype%counter = 1
82  dtype%ndims = rank(field)
83  dtype%sizes = shape(field)
84  dtype%subsizes = array_of_subsizes
85  dtype%starts = array_of_starts
86  dtype%etype = MPI_TYPE_
87  dtype%id = newtype
88 
89  ! Add dtype to the list
90  dtype%prev => datatypes%tail
91  dtype%prev%next => dtype
92  datatypes%tail => dtype
93  datatypes%length = datatypes%length + 1
94 
95  ! Copy dtype to output
96  dtype_out = dtype
97 
98  if (current_clock .NE. 0) &
99  call increment_current_clock(EVENT_TYPE_CREATE, MPP_TYPE_BYTELEN_)
100 
101 end subroutine MPP_TYPE_CREATE_
