127  !> @brief Get the size, start, and end indices for output fields.
128  SUBROUTINE get_subfield_size(axes, outnum)
129  INTEGER, INTENT(in) :: axes(:) !< axes of the input_field
130  INTEGER, INTENT(in) :: outnum !< position in array output_fields
131 
132  REAL, ALLOCATABLE :: global_lat(:), global_lon(:), global_depth(:)
133  INTEGER :: global_axis_size, global_axis_sizey
134  INTEGER :: i,xbegin,xend,ybegin,yend,xbegin_l,xend_l,ybegin_l,yend_l
135  CHARACTER(len=1) :: cart
136  TYPE(domain2d) :: domain2, domain2_new
137  TYPE(domain1d) :: domain1, domain1x, domain1y
138  REAL :: start(3) !< start coordinates in 3 axes
139  REAL :: end(3) !< end coordinates in 3 axes
140  INTEGER :: gstart_indx(3)!< global start indices of output domain in 3 axes
141  INTEGER :: gend_indx(3) !< global end indices of output domain in 3 axes
142  REAL, ALLOCATABLE :: subaxis_x(:) !< containing local coordinates in x,y,z axes
143  REAL, ALLOCATABLE :: subaxis_y(:) !< containing local coordinates in x,y,z axes
144  REAL, ALLOCATABLE :: subaxis_z(:) !< containing local coordinates in x,y,z axes
145  CHARACTER(len=128) :: msg
146  INTEGER :: ishift, jshift
147  INTEGER :: grv !< Value used to determine if the region defined in the diag_table is for the whole axis, or a sub-axis
148  CHARACTER(len=128), DIMENSION(2) :: axis_domain_name
149 
150  !initilization for local output
151  ! initially out of (lat/lon/depth) range
152  start = -1.e10
153  end = -1.e10
154  gstart_indx = -1
155  gend_indx = -1
156 
157  ! get the value to compare to determine if writing full axis data
158  IF ( region_out_use_alt_value ) THEN
159  grv = glo_reg_val_alt
160  ELSE
161  grv = glo_reg_val
162  END IF
163 
164  ! get axis data (lat, lon, depth) and indices
165  start = output_fields(outnum)%output_grid%start
166  end = output_fields(outnum)%output_grid%end
167 
168  CALL get_diag_axis_domain_name(axes(1), axis_domain_name(1))
169  CALL get_diag_axis_domain_name(axes(2), axis_domain_name(2))
170 
171  IF ( index(lowercase(axis_domain_name(1)), 'cubed') == 0 .AND. &
172  & index(lowercase(axis_domain_name(2)), 'cubed') == 0 ) THEN
173  DO i = 1, SIZE(axes(:))
174  global_axis_size = get_axis_global_length(axes(i))
175  output_fields(outnum)%output_grid%subaxes(i) = -1
176  CALL get_diag_axis_cart(axes(i), cart)
177  SELECT CASE(cart)
178  CASE ('X')
179  ! <ERROR STATUS="FATAL">wrong order of axes. X should come first.</ERROR>
180  IF( i.NE.1 ) CALL error_mesg('diag_util_mod::get_subfield_size',&
181  & 'wrong order of axes, X should come first',fatal)
182  ALLOCATE(global_lon(global_axis_size))
183  CALL get_diag_axis_data(axes(i),global_lon)
184  IF( int(start(i)) == grv .AND. int(end(i)) == grv ) THEN
185  gstart_indx(i) = 1
186  gend_indx(i) = global_axis_size
187  output_fields(outnum)%output_grid%subaxes(i) = axes(i)
188  ELSE
189  gstart_indx(i) = get_index(start(i),global_lon)
190  gend_indx(i) = get_index(end(i),global_lon)
191  END IF
192  ALLOCATE(subaxis_x(gstart_indx(i):gend_indx(i)))
193  subaxis_x=global_lon(gstart_indx(i):gend_indx(i))
194  CASE ('Y')
195  ! <ERROR STATUS="FATAL">wrong order of axes, Y should come second.</ERROR>
196  IF( i.NE.2 ) CALL error_mesg('diag_util_mod::get_subfield_size',&
197  & 'wrong order of axes, Y should come second',fatal)
198  ALLOCATE(global_lat(global_axis_size))
199  CALL get_diag_axis_data(axes(i),global_lat)
200  IF( int(start(i)) == grv .AND. int(end(i)) == grv ) THEN
201  gstart_indx(i) = 1
202  gend_indx(i) = global_axis_size
203  output_fields(outnum)%output_grid%subaxes(i) = axes(i)
204  ELSE
205  gstart_indx(i) = get_index(start(i),global_lat)
206  gend_indx(i) = get_index(end(i),global_lat)
207  END IF
208  ALLOCATE(subaxis_y(gstart_indx(i):gend_indx(i)))
209  subaxis_y=global_lat(gstart_indx(i):gend_indx(i))
210  CASE ('Z')
211  ! <ERROR STATUS="FATAL">wrong values in vertical axis of region</ERROR>
212  IF ( start(i)*end(i)<0. ) CALL error_mesg('diag_util_mod::get_subfield_size',&
213  & 'wrong values in vertical axis of region',fatal)
214  IF ( start(i)>=0. .AND. end(i)>0. ) THEN
215  ALLOCATE(global_depth(global_axis_size))
216  CALL get_diag_axis_data(axes(i),global_depth)
217  gstart_indx(i) = get_index(start(i),global_depth)
218  gend_indx(i) = get_index(end(i),global_depth)
219  ALLOCATE(subaxis_z(gstart_indx(i):gend_indx(i)))
220  subaxis_z=global_depth(gstart_indx(i):gend_indx(i))
221  output_fields(outnum)%output_grid%subaxes(i) =&
222  & diag_subaxes_init(axes(i),subaxis_z, gstart_indx(i),gend_indx(i))
223  DEALLOCATE(subaxis_z,global_depth)
224  ELSE ! regional vertical axis is the same as global vertical axis
225  gstart_indx(i) = 1
226  gend_indx(i) = global_axis_size
227  output_fields(outnum)%output_grid%subaxes(i) = axes(i)
228  ! <ERROR STATUS="FATAL">i should equal 3 for z axis</ERROR>
229  IF( i /= 3 ) CALL error_mesg('diag_util_mod::get_subfield_size',&
230  & 'i should equal 3 for z axis', fatal)
231  END IF
232  CASE default
233  ! <ERROR STATUS="FATAL">Wrong axis_cart</ERROR>
234  CALL error_mesg('diag_util_mod::get_subfield_size', 'Wrong axis_cart', fatal)
235  END SELECT
236  END DO
237 
238  DO i = 1, SIZE(axes(:))
239  IF ( gstart_indx(i) == -1 .OR. gend_indx(i) == -1 ) THEN
240  ! <ERROR STATUS="FATAL">
241  ! can not find gstart_indx/gend_indx for <output_fields(outnum)%output_name>,
242  ! check region bounds for axis <i>.
243  ! </ERROR>
244  WRITE(msg,'(A,I2)') ' check region bounds for axis ', i
245  CALL error_mesg('diag_util_mod::get_subfield_size', 'can not find gstart_indx/gend_indx for '&
246  & //trim(output_fields(outnum)%output_name)//','//trim(msg), fatal)
247  END IF
248  END DO
249  ELSE ! cubed sphere
250  ! get the i and j start and end indexes
251  CALL get_local_indexes(lonstart=start(1), lonend=end(1), &
252  & latstart=start(2), latend=end(2), &
253  & istart=gstart_indx(1), iend=gend_indx(1), &
254  & jstart=gstart_indx(2), jend=gend_indx(2))
255  global_axis_size = get_axis_global_length(axes(1))
256  ALLOCATE(global_lon(global_axis_size))
257  global_axis_sizey = get_axis_global_length(axes(2))
258  ALLOCATE(global_lat(global_axis_sizey))
259  CALL get_diag_axis_data(axes(1),global_lon)
260  CALL get_diag_axis_data(axes(2),global_lat)
261 
262  !Potential fix for out-of-bounds error for global_lon and global_lat.
263  IF ((gstart_indx(1) .GT. 0 .AND. gstart_indx(2) .GT. 0) .AND. &
264  (gstart_indx(1) .LE. global_axis_size .AND. gstart_indx(2) .LE. global_axis_sizey) .AND. &
265  (gend_indx(1) .GT. 0 .AND. gend_indx(2) .GT. 0) .AND. &
266  (gend_indx(1) .LE. global_axis_size .AND. gend_indx(2) .LE. global_axis_sizey)) THEN
267  ALLOCATE(subaxis_x(gstart_indx(1):gend_indx(1)))
268  ALLOCATE(subaxis_y(gstart_indx(2):gend_indx(2)))
269  subaxis_x=global_lon(gstart_indx(1):gend_indx(1))
270  subaxis_y=global_lat(gstart_indx(2):gend_indx(2))
271  END IF
272 
273  ! Now deal with the Z component
274  IF ( SIZE(axes(:)) > 2 ) THEN
275  global_axis_size = get_axis_global_length(axes(3))
276  output_fields(outnum)%output_grid%subaxes(3) = -1
277  CALL get_diag_axis_cart(axes(3), cart)
278  ! <ERROR STATUS="FATAL">
279  ! axis(3) should be Z-axis
280  ! </ERROR>
281  IF ( lowercase(cart) /= 'z' ) CALL error_mesg('diag_util_mod::get_subfield_size', &
282  &'axis(3) should be Z-axis', fatal)
283  ! <ERROR STATUS="FATAL">
284  ! wrong values in vertical axis of region
285  ! </ERROR>
286  IF ( start(3)*end(3)<0. ) CALL error_mesg('diag_util_mod::get_subfield_size',&
287  & 'wrong values in vertical axis of region',fatal)
288  IF ( start(3)>=0. .AND. end(3)>0. ) THEN
289  ALLOCATE(global_depth(global_axis_size))
290  CALL get_diag_axis_data(axes(3),global_depth)
291  gstart_indx(3) = get_index(start(3),global_depth)
292  IF( start(3) == 0.0 ) gstart_indx(3) = 1
293  gend_indx(3) = get_index(end(3),global_depth)
294  IF( start(3) >= maxval(global_depth) ) gstart_indx(3)= global_axis_size
295  IF( end(3) >= maxval(global_depth) ) gend_indx(3) = global_axis_size
296 
297  ALLOCATE(subaxis_z(gstart_indx(3):gend_indx(3)))
298  subaxis_z=global_depth(gstart_indx(3):gend_indx(3))
299  output_fields(outnum)%output_grid%subaxes(3) =&
300  & diag_subaxes_init(axes(3),subaxis_z, gstart_indx(3),gend_indx(3))
301  DEALLOCATE(subaxis_z,global_depth)
302  ELSE ! regional vertical axis is the same as global vertical axis
303  gstart_indx(3) = 1
304  gend_indx(3) = global_axis_size
305  output_fields(outnum)%output_grid%subaxes(3) = axes(3)
306  END IF
307  END IF
308  END IF
309 
310  ! get domain and compute_domain(xbegin,xend,ybegin,yend)
311  xbegin = -1
312  xend = -1
313  ybegin = -1
314  yend = -1
315 
316  domain2 = get_domain2d(axes)
317  IF ( domain2 .NE. null_domain2d ) THEN
318  CALL mpp_get_compute_domain(domain2, xbegin, xend, ybegin, yend)
319  CALL mpp_get_domain_components(domain2, domain1x, domain1y)
320  ELSE
321  DO i = 1, min(SIZE(axes(:)),2)
322  domain1 = get_domain1d(axes(i))
323  IF ( domain1 .NE. null_domain1d ) THEN
324  CALL get_diag_axis_cart(axes(i),cart)
325  SELECT CASE(cart)
326  CASE ('X')
327  domain1x = get_domain1d(axes(i))
328  CALL mpp_get_compute_domain(domain1x, xbegin, xend)
329  CASE ('Y')
330  domain1y = get_domain1d(axes(i))
331  CALL mpp_get_compute_domain(domain1y, ybegin, yend)
332  CASE default ! do nothing here
333  END SELECT
334  ELSE
335  ! <ERROR STATUS="FATAL">No domain available</ERROR>
336  CALL error_mesg('diag_util_mod::get_subfield_size', 'NO domain available', fatal)
337  END IF
338  END DO
339  END IF
340 
341  CALL get_axes_shift(axes, ishift, jshift)
342  xend = xend+ishift
343  yend = yend+jshift
344 
345  IF ( xbegin == -1 .OR. xend == -1 .OR. ybegin == -1 .OR. yend == -1 ) THEN
346  ! <ERROR STATUS="FATAL">wrong compute domain indices</ERROR>
347  CALL error_mesg('diag_util_mod::get_subfield_size', 'wrong compute domain indices',fatal)
348  END IF
349 
350  ! get the area containing BOTH compute domain AND local output area
351  IF( gstart_indx(1) > xend .OR. xbegin > gend_indx(1) ) THEN
352  output_fields(outnum)%output_grid%l_start_indx(1) = -1
353  output_fields(outnum)%output_grid%l_end_indx(1) = -1
354  output_fields(outnum)%need_compute = .false. ! not involved
355  ELSEIF ( gstart_indx(2) > yend .OR. ybegin > gend_indx(2) ) THEN
356  output_fields(outnum)%output_grid%l_start_indx(2) = -1
357  output_fields(outnum)%output_grid%l_end_indx(2) = -1
358  output_fields(outnum)%need_compute = .false. ! not involved
359  ELSE
360  output_fields(outnum)%output_grid%l_start_indx(1) = max(xbegin, gstart_indx(1))
361  output_fields(outnum)%output_grid%l_start_indx(2) = max(ybegin, gstart_indx(2))
362  output_fields(outnum)%output_grid%l_end_indx(1) = min(xend, gend_indx(1))
363  output_fields(outnum)%output_grid%l_end_indx(2) = min(yend, gend_indx(2))
364  output_fields(outnum)%need_compute = .true. ! involved in local output
365  END IF
366 
367  IF ( output_fields(outnum)%need_compute ) THEN
368  ! need to modify domain1d and domain2d for subaxes
369  xbegin_l = output_fields(outnum)%output_grid%l_start_indx(1)
370  xend_l = output_fields(outnum)%output_grid%l_end_indx(1)
371  ybegin_l = output_fields(outnum)%output_grid%l_start_indx(2)
372  yend_l = output_fields(outnum)%output_grid%l_end_indx(2)
373  CALL mpp_modify_domain(domain2, domain2_new, xbegin_l,xend_l, ybegin_l,yend_l,&
374  & gstart_indx(1),gend_indx(1), gstart_indx(2),gend_indx(2))
375 
376  output_fields(outnum)%output_grid%subaxes(1) =&
377  & diag_subaxes_init(axes(1),subaxis_x, gstart_indx(1),gend_indx(1),domain2_new)
378  output_fields(outnum)%output_grid%subaxes(2) =&
379  & diag_subaxes_init(axes(2),subaxis_y, gstart_indx(2),gend_indx(2),domain2_new)
380  DO i = 1, SIZE(axes(:))
381  IF ( output_fields(outnum)%output_grid%subaxes(i) == -1 ) THEN
382  ! <ERROR STATUS="FATAL">
383  ! <output_fields(outnum)%output_name> error at i = <i>
384  ! </ERROR>
385  WRITE(msg,'(a,"/",I4)') 'at i = ',i
386  CALL error_mesg('diag_util_mod::get_subfield_size '//trim(output_fields(outnum)%output_name),&
387  'error '//trim(msg), fatal)
388  END IF
389  END DO
390 
391  ! local start index should start from 1
392  output_fields(outnum)%output_grid%l_start_indx(1) = max(xbegin, gstart_indx(1)) - xbegin + 1
393  output_fields(outnum)%output_grid%l_start_indx(2) = max(ybegin, gstart_indx(2)) - ybegin + 1
394  output_fields(outnum)%output_grid%l_end_indx(1) = min(xend, gend_indx(1)) - xbegin + 1
395  output_fields(outnum)%output_grid%l_end_indx(2) = min(yend, gend_indx(2)) - ybegin + 1
396  IF ( SIZE(axes(:))>2 ) THEN
397  output_fields(outnum)%output_grid%l_start_indx(3) = gstart_indx(3)
398  output_fields(outnum)%output_grid%l_end_indx(3) = gend_indx(3)
399  ELSE
400  output_fields(outnum)%output_grid%l_start_indx(3) = 1
401  output_fields(outnum)%output_grid%l_end_indx(3) = 1
402  END IF
403  END IF
404  IF ( ALLOCATED(subaxis_x) ) DEALLOCATE(subaxis_x, global_lon)
405  IF ( ALLOCATED(subaxis_y) ) DEALLOCATE(subaxis_y, global_lat)
406  END SUBROUTINE get_subfield_size

408  !> @brief Get size, start and end indices for output fields.
409  SUBROUTINE get_subfield_vert_size(axes, outnum)
410  INTEGER, DIMENSION(:), INTENT(in) :: axes !< axes of the input_field
411  INTEGER, INTENT(in) :: outnum !< position in array output_fields
412 
413  REAL, DIMENSION(3) :: start !< start and end coordinates in 3 axes
414  REAL, DIMENSION(3) :: end !< start and end coordinates in 3 axes
415  REAL, ALLOCATABLE, DIMENSION(:) :: global_depth
416  REAL, ALLOCATABLE, DIMENSION(:) :: subaxis_z !< containing local coordinates in x,y,z axes
417  INTEGER :: i, global_axis_size
418  INTEGER, DIMENSION(3) :: gstart_indx !< global start and end indices of output domain in 3 axes
419  INTEGER, DIMENSION(3) :: gend_indx !< global start and end indices of output domain in 3 axes

539  !> @brief Find index <TT>i</TT> of array such that <TT>array(i)</TT> is closest to number.
540  INTEGER FUNCTION get_index(number, array)
541  REAL, INTENT(in) :: number !<
542  REAL, INTENT(in), DIMENSION(:) :: array !<
543 
544  INTEGER :: i, n
545  LOGICAL :: found
546 
547  n = SIZE(array(:))
548  ! check if array is monotonous
549  DO i = 2, n-1
550  IF( (array(i-1)<array(i).AND.array(i)>array(i+1)) .OR. (array(i-1)>array(i).AND.array(i)<array(i+1))) THEN
551  ! <ERROR STATUS="FATAL">array NOT monotonously ordered</ERROR>
552  CALL error_mesg('diag_util_mod::get_index', 'array NOT monotonously ordered',fatal)
553  END IF
554  END DO
555  get_index = -1
556  found = .false.
557  ! search in increasing array
558  DO i = 1, n-1
559  IF ( (array(i)<=number).AND.(array(i+1)>= number) ) THEN
560  IF( number - array(i) <= array(i+1) - number ) THEN
561  get_index = i
562  found=.true.
563  ELSE
564  get_index = i+1
565  found=.true.
566  ENDIF
567  EXIT
568  END IF
569  END DO
570  ! if not found, search in decreasing array
571  IF( .NOT.found ) THEN
572  DO i = 1, n-1
573  IF ( (array(i)>=number).AND.(array(i+1)<= number) ) THEN
574  IF ( array(i)-number <= number-array(i+1) ) THEN
575  get_index = i
576  found = .true.
577  ELSE
578  get_index = i+1
579  found = .true.
580  END IF
581  EXIT
582  END IF
583  END DO
584  END IF
585  ! if still not found, is it less than the first element
586  ! or greater than last element? (Increasing Array)
587  ! But it must be within 2x the axis spacing
588  ! i.e. array(1)-(array(3)-array(1)).LT.number .AND. or 2*array(1)-array(3).LT.number
589  IF ( .NOT. found ) THEN
590  IF ( 2*array(1)-array(3).LT.number .AND. number.LT.array(1) ) THEN
591  get_index = 1
592  found = .true.
593  ELSE IF ( array(n).LT.number .AND. number.LT.2*array(n)-array(n-2) ) THEN
594  get_index = n
595  found = .true.
596  ELSE
597  found = .false.
598  END IF
599  END IF
600 
601  ! if still not found, is it greater than the first element
602  ! or less than the last element? (Decreasing Array)
603  ! But it must be within 2x the axis spacing (see above)
604  IF ( .NOT. found ) THEN
605  IF ( 2*array(1)-array(3).GT.number .AND. number.GT.array(1) ) THEN
606  get_index = 1
607  found = .true.
608  ELSE IF ( array(n).GT.number .AND. number.GT.2*array(n)-array(n-2) ) THEN
609  get_index = n
610  found = .true.
611  ELSE
612  found = .false.
613  END IF
614  END IF
615  END FUNCTION get_index
