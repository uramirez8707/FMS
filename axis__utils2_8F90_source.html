148  subroutine axis_edges(fileobj, name, edge_data, reproduce_null_char_bug_flag)
149 
150  class(fmsnetcdffile_t), intent(in) :: fileobj !< File object to read from
151  character(len=*), intent(in) :: name !< Name of a given axis
152  class(*), dimension(:), intent(out) :: edge_data !< Returned edge data from given axis name
153  logical, intent(in), optional :: reproduce_null_char_bug_flag !< Flag indicating to reproduce
154  !! the mpp_io bug where the null characters were not removed
155  !! after reading a string attribute
156 
157  integer :: ndims
158  character(len=128) :: buffer
159  integer, dimension(:), allocatable :: dim_sizes
160  real(kind=r4_kind), dimension(:), allocatable :: r32
161  real(kind=r4_kind), dimension(:,:), allocatable :: r322d
162  real(kind=r8_kind), dimension(:), allocatable :: r64
163  real(kind=r8_kind), dimension(:,:), allocatable :: r642d
164  integer :: i
165  integer :: n
166  logical :: reproduce_null_char_bug !< Local flag indicating to reproduce the mpp_io bug where
167  !! the null characters were not removed after reading a string attribute
168 
169  ndims = get_variable_num_dimensions(fileobj, name)
170  allocate(dim_sizes(ndims))
171  call get_variable_size(fileobj, name, dim_sizes)
172  n = dim_sizes(1)
173  if (size(edge_data) .ne. n+1) then
174  call mpp_error(fatal, "axis_edge: incorrect size of edge_data array.")
175  endif
176  deallocate(dim_sizes)
177 
178  reproduce_null_char_bug = .false.
179  if (present(reproduce_null_char_bug_flag)) reproduce_null_char_bug = reproduce_null_char_bug_flag
180 
181  buffer = ""
182  if (variable_att_exists(fileobj, name, "edges")) then
183  !! If the reproduce_null_char_bug flag is turned on fms2io will not remove the null character
184  call get_variable_attribute(fileobj, name, "edges", buffer(1:128), &
185  reproduce_null_char_bug_flag=reproduce_null_char_bug)
186 
187  !! Check for a null character here, if it exists *_bnds will be calculated instead of read in
188  if (reproduce_null_char_bug) then
189  i = 0
190  i = index(buffer, char(0))
191  if (i > 0) buffer = ""
192  endif
193  elseif (variable_att_exists(fileobj, name, "bounds")) then
194  !! If the reproduce_null_char_bug flag is turned on fms2io will not remove the null character
195  call get_variable_attribute(fileobj, name, "bounds", buffer(1:128), &
196  reproduce_null_char_bug_flag=reproduce_null_char_bug)
197 
198  !! Check for a null character here, if it exists *_bnds will be calculated instead of read in
199  if (reproduce_null_char_bug) then
200  i = 0
201  i = index(buffer, char(0))
202  if (i > 0) buffer = ""
203  endif
204  endif
205  if (trim(buffer) .ne. "") then
206  ndims = get_variable_num_dimensions(fileobj, buffer)
207  allocate(dim_sizes(ndims))
208  call get_variable_size(fileobj, buffer, dim_sizes)
209  if (size(dim_sizes) .eq. 1) then
210  if (dim_sizes(1) .ne. n+1) then
211  call mpp_error(fatal, "axis_edges: incorrect size of edge data.")
212  endif
213  call read_data(fileobj, buffer, edge_data)
214  elseif (size(dim_sizes) .eq. 2) then
215  if (dim_sizes(1) .ne. 2) then
216  call mpp_error(fatal, "axis_edges: first dimension of edge must be of size 2")
217  endif
218  if (dim_sizes(2) .ne. n) then
219  call mpp_error(fatal, "axis_edges: incorrect size of edge data.")
220  endif
221  select type (edge_data)
222  type is (real(kind=r4_kind))
223  allocate(r322d(dim_sizes(1), dim_sizes(2)))
224  call read_data(fileobj, buffer, r322d)
225  edge_data(1:dim_sizes(2)) = r322d(1,:)
226  edge_data(dim_sizes(2)+1) = r322d(2,dim_sizes(2))
227  deallocate(r322d)
228  type is (real(kind=r8_kind))
229  allocate(r642d(dim_sizes(1), dim_sizes(2)))
230  call read_data(fileobj, buffer, r642d)
231  edge_data(1:dim_sizes(2)) = r642d(1,:)
232  edge_data(dim_sizes(2)+1) = r642d(2,dim_sizes(2))
233  deallocate(r642d)
234  class default
235  call mpp_error(fatal, "axis_edges: unsupported kind.")
236  end select
237  endif
238  deallocate(dim_sizes)
239  else
240  select type (edge_data)
241  type is (real(kind=r4_kind))
242  allocate(r32(n))
243  call read_data(fileobj, name, r32)
244  do i = 2, n
245  edge_data(i) = r32(i-1) + 0.5_r4_kind*(r32(i) - r32(i-1))
246  enddo
247  edge_data(1) = r32(1) - 0.5_r4_kind*(r32(2) - r32(1))
248  if (abs(edge_data(1)) .lt. 1.e-10) then
249  edge_data(1) = 0._r4_kind
250  endif
251  edge_data(n+1) = r32(n) + 0.5_r4_kind*(r32(n) - r32(n-1))
252  deallocate(r32)
253  type is (real(kind=r8_kind))
254  allocate(r64(n))
255  call read_data(fileobj, name, r64)
256  do i = 2, n
257  edge_data(i) = r64(i-1) + 0.5_r8_kind*(r64(i) - r64(i-1))
258  enddo
259  edge_data(1) = r64(1) - 0.5_r8_kind*(r64(2) - r64(1))
260  if (abs(edge_data(1)) .lt. 1.d-10) then
261  edge_data(1) = 0._r8_kind
262  endif
263  edge_data(n+1) = r64(n) + 0.5_r8_kind*(r64(n) - r64(n-1))
264  deallocate(r64)
265  class default
266  call mpp_error(fatal, "axis_edges: unsupported kind.")
267  end select
268  endif
269 end subroutine axis_edges

311  !> @brief Returns lon_strt <= longitude <= lon_strt+360
312  !! @return real lon_in_range
313  function lon_in_range(lon, l_strt)
314  real, intent(in) :: lon, l_strt
315  real :: lon_in_range
316  real :: l_end
317 
318  lon_in_range = lon
319  l_end = l_strt+360.
320 
321  if (abs(lon_in_range - l_strt) < 1.e-4) then
322  lon_in_range = l_strt
323  return
324  endif
325 
326  if (abs(lon_in_range - l_end) < 1.e-4) then
327  lon_in_range = l_strt
328  return
329  endif
330 
331  do
332  if (lon_in_range < l_strt) then
333  lon_in_range = lon_in_range + f360
334  else if (lon_in_range > l_end) then
335  lon_in_range = lon_in_range - f360
336  else
337  exit
338  end if
339  end do
340 
341  end function lon_in_range

343  !> @brief Returns monotonic array of longitudes s.t., lon_strt <= lon(:) <= lon_strt+360.
344  !!
345  !> <br>The first istrt-1 entries are moved to the end of the array:
346  !!
347  !! e.g.
348  !! lon = 0 1 2 3 4 5 ... 358 359; lon_strt = 3 ==>
349  !! tranlon = 3 4 5 6 7 8 ... 359 360 361 362; istrt = 4
350  subroutine tranlon(lon, lon_start, istrt)
351 
352  ! returns array of longitudes s.t. lon_strt <= lon < lon_strt+360.
353  ! also, the first istrt-1 entries are moved to the end of the array
354  !
355  ! e.g.
356  ! lon = 0 1 2 3 4 5 ... 358 359; lon_strt = 3 ==>
357  ! tranlon = 3 4 5 6 7 8 ... 359 360 361 362; istrt = 4
358 
359  real, intent(inout), dimension(:) :: lon
360  real, intent(in) :: lon_start
361  integer, intent(out) :: istrt
362 
363 
364  integer :: len, i
365  real :: lon_strt, tmp(size(lon(:))-1)
366 
367  len = size(lon(:))
368 
369  do i=1,len
370  lon(i) = lon_in_range(lon(i),lon_start)
371  enddo
372 
373  istrt=0
374  do i=1,len-1
375  if (lon(i+1) < lon(i)) then
376  istrt=i+1
377  exit
378  endif
379  enddo
380 
381  if (istrt>1) then ! grid is not monotonic
382  if (abs(lon(len)-lon(1)) < epsln) then
383  tmp = cshift(lon(1:len-1),istrt-1)
384  lon(1:len-1) = tmp
385  lon(len) = lon(1)
386  else
387  lon = cshift(lon,istrt-1)
388  endif
389  lon_strt = lon(1)
390  do i=2,len+1
391  lon(i) = lon_in_range(lon(i),lon_strt)
392  lon_strt = lon(i)
393  enddo
394  endif
395 
396  return
397  end subroutine tranlon

399  !> nearest_index = index of nearest data point within "array" corresponding to
400  !! "value".
401  !!
402  !! inputs:
403  !!
404  !! value = arbitrary data...same units as elements in "array"
405  !! array = array of data points (must be monotonically increasing)
406  !!
407  !! output:
408  !!
409  !! nearest_index = index of nearest data point to "value"
410  !! if "value" is outside the domain of "array" then nearest_index = 1
411  !! or "ia" depending on whether array(1) or array(ia) is
412  !! closest to "value"
413  !!
414  !! note: if "array" is dimensioned array(0:ia) in the calling
415  !! program, then the returned index should be reduced
416  !! by one to account for the zero base.
417  !!
418  !! example:
419  !!
420  !! let model depths be defined by the following:
421  !! parameter (km=5)
422  !! dimension z(km)
423  !! data z /5.0, 10.0, 50.0, 100.0, 250.0/
424  !!
425  !! k1 = nearest_index (12.5, z, km)
426  !! k2 = nearest_index (0.0, z, km)
427  !!
428  !! k1 would be set to 2, and k2 would be set to 1 so that
429  !! z(k1) would be the nearest data point to 12.5 and z(k2) would
430  !! be the nearest data point to 0.0
431  !!
432  !! @return real frac_index
433  function frac_index (value, array)
434 
435  integer :: ia, i, ii, unit
436  real :: value !< arbitrary data...same units as elements in "array"
437  real :: frac_index
438  real, dimension(:) :: array !< array of data points (must be monotonically increasing)
439  logical keep_going
440 
441  ia = size(array(:))
442 
443  do i=2,ia
444  if (array(i) < array(i-1)) then
445  unit = stdout()
446  write (unit,*) '=> Error: "frac_index" array must be monotonically increasing when searching for nearest value to ',&
447  value
448  write (unit,*) ' array(i) < array(i-1) for i=',i
449  write (unit,*) ' array(i) for i=1..ia follows:'
450  do ii=1,ia
451  write (unit,*) 'i=',ii, ' array(i)=',array(ii)
452  enddo
453  call mpp_error(fatal,' "frac_index" array must be monotonically increasing.')
454  endif
455  enddo
456  if (value < array(1) .or. value > array(ia)) then
457 ! if (value < array(1)) frac_index = 1.
458 ! if (value > array(ia)) frac_index = float(ia)
459  frac_index = -1.0
460  else
461  i=1
462  keep_going = .true.
463  do while (i <= ia .and. keep_going)
464  i = i+1
465  if (value <= array(i)) then
466  frac_index = float(i-1) + (value-array(i-1))/(array(i)-array(i-1))
467  keep_going = .false.
468  endif
469  enddo
470  endif
471  end function frac_index

473  !> @brief Return index of nearest point along axis
474  !!
475  !> nearest_index = index of nearest data point within "array" corresponding to
476  !! "value".
477  !!
478  !! inputs:
479  !!
480  !! value = arbitrary data...same units as elements in "array"
481  !! array = array of data points (must be monotonically increasing)
482  !! ia = dimension of "array"
483  !!
484  !! output:
485  !!
486  !! nearest_index = index of nearest data point to "value"
487  !! if "value" is outside the domain of "array" then nearest_index = 1
488  !! or "ia" depending on whether array(1) or array(ia) is
489  !! closest to "value"
490  !!
491  !! note: if "array" is dimensioned array(0:ia) in the calling
492  !! program, then the returned index should be reduced
493  !! by one to account for the zero base.
494  !!
495  !! example:
496  !!
497  !! let model depths be defined by the following:
498  !! parameter (km=5)
499  !! dimension z(km)
500  !! data z /5.0, 10.0, 50.0, 100.0, 250.0/
501  !!
502  !! k1 = nearest_index (12.5, z, km)
503  !! k2 = nearest_index (0.0, z, km)
504  !!
505  !! k1 would be set to 2, and k2 would be set to 1 so that
506  !! z(k1) would be the nearest data point to 12.5 and z(k2) would
507  !! be the nearest data point to 0.0
508  !! @return integer nearest_index
509  function nearest_index (value, array)
510 
511  integer :: nearest_index
512  integer :: ia !< dimension of "array"
513  integer :: i, ii, unit
514  real :: value !< arbitrary data...same units as elements in "array"
515  real, dimension(:) :: array !< array of data points (must be monotonically increasing)
516  logical keep_going
517 
518  ia = size(array(:))
519 
520  do i=2,ia
521  if (array(i) < array(i-1)) then
522  unit = stdout()
523  write (unit,*) '=> Error: "nearest_index" array must be monotonically increasing &
524  &when searching for nearest value to ',value
525  write (unit,*) ' array(i) < array(i-1) for i=',i
526  write (unit,*) ' array(i) for i=1..ia follows:'
527  do ii=1,ia
528  write (unit,*) 'i=',ii, ' array(i)=',array(ii)
529  enddo
530  call mpp_error(fatal,' "nearest_index" array must be monotonically increasing.')
531  endif
532  enddo
533  if (value < array(1) .or. value > array(ia)) then
534  if (value < array(1)) nearest_index = 1
535  if (value > array(ia)) nearest_index = ia
536  else
537  i=1
538  keep_going = .true.
539  do while (i <= ia .and. keep_going)
540  i = i+1
541  if (value <= array(i)) then
542  nearest_index = i
543  if (array(i)-value > value-array(i-1)) nearest_index = i-1
544  keep_going = .false.
545  endif
546  enddo
547  endif
548  end function nearest_index

552  subroutine interp_1d_linear(grid1,grid2,data1,data2)
553 
554  real, dimension(:), intent(in) :: grid1, data1, grid2
555  real, dimension(:), intent(inout) :: data2
556 
557  integer :: n1, n2, i, n, ext
558  real :: w
559 
560  n1 = size(grid1(:))
561  n2 = size(grid2(:))
562 
563 
564  do i=2,n1
565  if (grid1(i) <= grid1(i-1)) call mpp_error(fatal, 'grid1 not monotonic')
566  enddo
567 
568  do i=2,n2
569  if (grid2(i) <= grid2(i-1)) call mpp_error(fatal, 'grid2 not monotonic')
570  enddo
571 
572  if (grid1(1) > grid2(1) ) call mpp_error(fatal, 'grid2 lies outside grid1')
573  if (grid1(n1) < grid2(n2) ) call mpp_error(fatal, 'grid2 lies outside grid1')
574 
575  do i=1,n2
576  n = nearest_index(grid2(i),grid1)
577 
578  if (grid1(n) < grid2(i)) then
579  w = (grid2(i)-grid1(n))/(grid1(n+1)-grid1(n))
580  data2(i) = (1.-w)*data1(n) + w*data1(n+1)
581  else
582  if(n==1) then
583  data2(i) = data1(n)
584  else
585  w = (grid2(i)-grid1(n-1))/(grid1(n)-grid1(n-1))
586  data2(i) = (1.-w)*data1(n-1) + w*data1(n)
587  endif
588  endif
589  enddo

592  return

597  subroutine interp_1d_cubic_spline(grid1, grid2, data1, data2, yp1, ypn)
598 
599  real, dimension(:), intent(in) :: grid1, grid2, data1
600  real, dimension(:), intent(inout) :: data2
601  real, intent(in) :: yp1, ypn
602 
603  real, dimension(size(grid1)) :: y2, u
604  real :: sig, p, qn, un, h, a ,b
605  integer :: n, m, i, k, klo, khi
606 
607  n = size(grid1(:))
608  m = size(grid2(:))
609 
610  do i=2,n
611  if (grid1(i) <= grid1(i-1)) call mpp_error(fatal, 'grid1 not monotonic')
612  enddo
613 
614  do i=2,m
615  if (grid2(i) <= grid2(i-1)) call mpp_error(fatal, 'grid2 not monotonic')
616  enddo
617 
618  if (grid1(1) > grid2(1) ) call mpp_error(fatal, 'grid2 lies outside grid1')
619  if (grid1(n) < grid2(m) ) call mpp_error(fatal, 'grid2 lies outside grid1')
620 
621  if (yp1 >.99e30) then
622  y2(1)=0.
623  u(1)=0.
624  else
625  y2(1)=-0.5
626  u(1)=(3./(grid1(2)-grid1(1)))*((data1(2)-data1(1))/(grid1(2)-grid1(1))-yp1)
627  endif
628 
629  do i=2,n-1
630  sig=(grid1(i)-grid1(i-1))/(grid1(i+1)-grid1(i-1))
631  p=sig*y2(i-1)+2.
632  y2(i)=(sig-1.)/p
633  u(i)=(6.*((data1(i+1)-data1(i))/(grid1(i+1)-grid1(i))-(data1(i)-data1(i-1)) &
634  /(grid1(i)-grid1(i-1)))/(grid1(i+1)-grid1(i-1))-sig*u(i-1))/p
635  enddo
636 
637  if (ypn > .99e30) then
638  qn=0.
639  un=0.
640  else
641  qn=0.5
642  un=(3./(grid1(n)-grid1(n-1)))*(ypn-(data1(n)-data1(n-1))/(grid1(n)-grid1(n-1)))
643  endif
644 
645  y2(n)=(un-qn*u(n-1))/(qn*y2(n-1)+1.)
646 
647  do k=n-1,1,-1
648  y2(k)=y2(k)*y2(k+1)+u(k)
649  enddo
650 
651  do k = 1, m
652  n = nearest_index(grid2(k),grid1)
653  if (grid1(n) < grid2(k)) then
654  klo = n
655  else
656  if(n==1) then
657  klo = n
658  else
659  klo = n -1
660  endif
661  endif
662  khi = klo+1
663  h = grid1(khi)-grid1(klo)
664  a = (grid1(khi) - grid2(k))/h
665  b = (grid2(k) - grid1(klo))/h
666  data2(k) = a*data1(klo) + b*data1(khi)+ ((a**3-a)*y2(klo) + (b**3-b)*y2(khi))*(h**2)/6.
667  enddo
668 
669  end subroutine interp_1d_cubic_spline
