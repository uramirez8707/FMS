23  subroutine MPP_DO_GLOBAL_FIELD_3D_AD_( domain, local, global, tile, ishift, jshift, flags, default_data)
24 !get a global field from a local field
25 !local field may be on compute OR data domain
26  type(domain2D), intent(in) :: domain
27  MPP_TYPE_, intent(inout) :: local(:,:,:)
28  integer, intent(in) :: tile, ishift, jshift
29  MPP_TYPE_, intent(in) :: global(domain%x(tile)%global%begin:,domain%y(tile)%global%begin:,:)
30  integer, intent(in), optional :: flags
31  MPP_TYPE_, intent(in), optional :: default_data
32  integer :: i, j, k, m, n, nd, num_words, lpos, rpos, ioff, joff, from_pe, root_pe, tile_id
33  integer :: ke, isc, iec, jsc, jec, is, ie, js, je, num_word_me
34  integer :: ipos, jpos
35  logical :: xonly, yonly, root_only, global_on_this_pe
36  MPP_TYPE_ :: clocal ((domain%x(1)%compute%size+ishift) *(domain%y(1)%compute%size+jshift) *size(local,3))
37  MPP_TYPE_ :: cremote((domain%x(1)%compute%max_size+ishift)*(domain%y(1)%compute%max_size+jshift)*size(local,3))
38  integer :: stackuse
39  character(len=8) :: text
40 
41  pointer( ptr_local, clocal )
42  pointer( ptr_remote, cremote )
43 
44  stackuse = size(clocal(:))+size(cremote(:))
45  if( stackuse.GT.mpp_domains_stack_size )then
46  write( text, '(i8)' )stackuse
47  call mpp_error( FATAL, &
48  'MPP_DO_GLOBAL_FIELD user stack overflow: call mpp_domains_set_stack_size('//trim(text)// &
49  & ') from all PEs.' )
50  end if
51  mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, stackuse )
52 
53  ptr_local = LOC(mpp_domains_stack)
54  ptr_remote = LOC(mpp_domains_stack(size(clocal(:))+1))
55 
56  if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_GLOBAL_FIELD: must first call mpp_domains_init.' )
57 
58  xonly = .FALSE.
59  yonly = .FALSE.
60  root_only = .FALSE.
61  if( PRESENT(flags) ) then
62  xonly = BTEST(flags,EAST)
63  yonly = BTEST(flags,SOUTH)
64  if( .NOT.xonly .AND. .NOT.yonly )call mpp_error( WARNING, &
65  'MPP_GLOBAL_FIELD: you must have flags=XUPDATE, YUPDATE or XUPDATE+YUPDATE' )
66  if(xonly .AND. yonly) then
67  xonly = .false.; yonly = .false.
68  endif
69  root_only = BTEST(flags, ROOT_GLOBAL)
70  if( (xonly .or. yonly) .AND. root_only ) then
71  call mpp_error( WARNING, 'MPP_GLOBAL_FIELD: flags = XUPDATE+GLOBAL_ROOT_ONLY or ' // &
72  'flags = YUPDATE+GLOBAL_ROOT_ONLY is not supported, will ignore GLOBAL_ROOT_ONLY' )
73  root_only = .FALSE.
74  endif
75  endif
76 
77  global_on_this_pe = .NOT. root_only .OR. domain%pe == domain%tile_root_pe
78  ipos = 0; jpos = 0
79  if(global_on_this_pe ) then
80  if(size(local,3).NE.size(global,3) ) call mpp_error( FATAL, &
81  'MPP_GLOBAL_FIELD: mismatch of third dimension size of global and local')
82  if( size(global,1).NE.(domain%x(tile)%global%size+ishift) .OR. size(global, &
83  & 2).NE.(domain%y(tile)%global%size+jshift))then
84  if(xonly) then
85  if(size(global,1).NE.(domain%x(tile)%global%size+ishift) .OR. &
86  size(global,2).NE.(domain%y(tile)%compute%size+jshift)) &
87  call mpp_error( FATAL, &
88  & 'MPP_GLOBAL_FIELD: incoming arrays do not match domain for xonly global field.' )
89  jpos = -domain%y(tile)%compute%begin + 1
90  else if(yonly) then
91  if(size(global,1).NE.(domain%x(tile)%compute%size+ishift) .OR. &
92  size(global,2).NE.(domain%y(tile)%global%size+jshift)) &
93  call mpp_error( FATAL, &
94  & 'MPP_GLOBAL_FIELD: incoming arrays do not match domain for yonly global field.' )
95  ipos = -domain%x(tile)%compute%begin + 1
96  else
97  call mpp_error( FATAL, 'MPP_GLOBAL_FIELD: incoming arrays do not match domain.' )
98  endif
99  endif
100  endif
101 
102  if( size(local,1).EQ.(domain%x(tile)%compute%size+ishift) .AND. size(local, &
103  & 2).EQ.(domain%y(tile)%compute%size+jshift) )then
104  !local is on compute domain
105  ioff = -domain%x(tile)%compute%begin + 1
106  joff = -domain%y(tile)%compute%begin + 1
107  else if( size(local,1).EQ.(domain%x(tile)%memory%size+ishift) .AND. size(local, &
108  & 2).EQ.(domain%y(tile)%memory%size+jshift) )then
109  !local is on data domain
110  ioff = -domain%x(tile)%data%begin + 1
111  joff = -domain%y(tile)%data%begin + 1
112  else
113  call mpp_error( FATAL, &
114  & 'MPP_GLOBAL_FIELD_: incoming field array must match either compute domain or memory domain.')
115  end if
116 
117  ke = size(local,3)
118  isc = domain%x(tile)%compute%begin; iec = domain%x(tile)%compute%end+ishift
119  jsc = domain%y(tile)%compute%begin; jec = domain%y(tile)%compute%end+jshift
120 
121  num_word_me = (iec-isc+1)*(jec-jsc+1)*ke
122 
123 ! make contiguous array from compute domain
124  m = 0
125 
126 ! if there is more than one tile on this pe, then no decomposition for all tiles on this pe, so we can just return
127  if(size(domain%x(:))>1) then
128  !--- the following is needed to avoid deadlock.
129  if( tile == size(domain%x(:)) ) call mpp_sync_self( )
130  return
131  end if
132 
133  root_pe = mpp_root_pe()
134 
135 !fill off-domains (note loops begin at an offset of 1)
136  if( xonly )then
137  nd = size(domain%x(1)%list(:))
138  do n = 1,nd-1
139  lpos = mod(domain%x(1)%pos+nd-n,nd)
140  rpos = mod(domain%x(1)%pos +n,nd)
141  from_pe = domain%x(1)%list(rpos)%pe
142  rpos = from_pe - root_pe ! for concurrent run, root_pe may not be 0.
143  num_words = (domain%list(rpos)%x(1)%compute%size+ishift) * (domain%list(rpos)%y(1)%compute%size+jshift) * ke
144  ! Force use of scalar, integer ptr interface
145  m = 0
146  is = domain%list(rpos)%x(1)%compute%begin; ie = domain%list(rpos)%x(1)%compute%end+ishift
147  do k = 1, ke
148  do j = jsc, jec
149  do i = is, ie
150  m = m + 1
151  cremote(m) = global(i,j+jpos,k)
152  end do
153  end do
154  end do
155 
156  call mpp_transmit( put_data=cremote(1), plen=num_words, to_pe=from_pe, &
157  get_data=clocal(1), glen=num_word_me, from_pe=domain%x(1)%list(lpos)%pe )
158 
159  call mpp_sync_self() !-ensure MPI_ISEND is done.
160  end do
161  else if( yonly )then
162  nd = size(domain%y(1)%list(:))
163  do n = 1,nd-1
164  lpos = mod(domain%y(1)%pos+nd-n,nd)
165  rpos = mod(domain%y(1)%pos +n,nd)
166  from_pe = domain%y(1)%list(rpos)%pe
167  rpos = from_pe - root_pe
168  num_words = (domain%list(rpos)%x(1)%compute%size+ishift) &
169  * (domain%list(rpos)%y(1)%compute%size+jshift) * ke
170  ! Force use of scalar, integer pointer interface
171  m = 0
172  js = domain%list(rpos)%y(1)%compute%begin; je = domain%list(rpos)%y(1)%compute%end+jshift
173  do k = 1,ke
174  do j = js, je
175  do i = isc, iec
176  m = m + 1
177  cremote(m) = global(i+ipos,j,k)
178  end do
179  end do
180  end do
181 
182  call mpp_transmit( put_data=cremote(1), plen=num_words, to_pe=from_pe, &
183  get_data=clocal(1), glen=num_word_me, from_pe=domain%y(1)%list(lpos)%pe )
184 
185  call mpp_sync_self() !-ensure MPI_ISEND is done.
186  end do
187  else
188  tile_id = domain%tile_id(1)
189  nd = size(domain%list(:))
190  if(root_only) then
191  if(domain%pe .NE. domain%tile_root_pe) then
192  call mpp_recv( clocal(1), glen=num_words, from_pe=domain%tile_root_pe, tag=COMM_TAG_1 )
193  else
194  do n = 1,nd-1
195  rpos = mod(domain%pos+n,nd)
196  if( domain%list(rpos)%tile_id(1) .NE. tile_id ) cycle
197  num_words = (domain%list(rpos)%x(1)%compute%size+ishift) * &
198  & (domain%list(rpos)%y(1)%compute%size+jshift) * ke
199  m = 0
200  is = domain%list(rpos)%x(1)%compute%begin; ie = domain%list(rpos)%x(1)%compute%end+ishift
201  js = domain%list(rpos)%y(1)%compute%begin; je = domain%list(rpos)%y(1)%compute%end+jshift
202 
203  do k = 1,ke
204  do j = js, je
205  do i = is, ie
206  m = m + 1
207  cremote(m) = global(i,j,k)
208  end do
209  end do
210  end do
211 
212  call mpp_send(cremote(1), plen=num_word_me, to_pe=domain%list(rpos)%pe, tag=COMM_TAG_1 )
213 
214  end do
215  endif
216  else
217  do n = 1,nd-1
218  rpos = mod(domain%pos+n,nd)
219  if( domain%list(rpos)%tile_id(1) .NE. tile_id ) cycle ! global field only within tile
220  num_words = (domain%list(rpos)%x(1)%compute%size+ishift) * &
221  & (domain%list(rpos)%y(1)%compute%size+jshift) * ke
222  m = 0
223  is = domain%list(rpos)%x(1)%compute%begin; ie = domain%list(rpos)%x(1)%compute%end+ishift
224  js = domain%list(rpos)%y(1)%compute%begin; je = domain%list(rpos)%y(1)%compute%end+jshift
225 
226  do k = 1,ke
227  do j = js, je
228  do i = is, ie
229  m = m + 1
230  cremote(m) = global(i,j,k)
231  end do
232  end do
233  end do
234 
235  call mpp_send( cremote(1), plen=num_words, to_pe=domain%list(rpos)%pe, tag=COMM_TAG_2 )
236 
237  end do
238 
239  do n = 1,nd-1
240  lpos = mod(domain%pos+nd-n,nd)
241  if( domain%list(lpos)%tile_id(1).NE. tile_id ) cycle ! global field only within tile
242  call mpp_recv( clocal(1), glen=num_word_me, from_pe=domain%list(lpos)%pe, tag=COMM_TAG_2 )
243  end do
244 
245 
246  endif
247  end if
248 
249  call mpp_sync_self()
250 
251  ! make contiguous array from compute domain
252  m = 0
253 #ifdef LOGICAL_VARIABLE
254  local = .false.
255 #else
256  local = 0
257 #endif
258  if(global_on_this_pe) then
259  do k = 1, ke
260  do j = jsc, jec
261  do i = isc, iec
262  m = m + 1
263  clocal(m) = global(i+ipos,j+jpos,k) !always fill local domain directly
264  local(i+ioff,j+joff,k) = clocal(m)
265  end do
266  end do
267  end do
268  else
269  do k = 1, ke
270  do j = jsc, jec
271  do i = isc, iec
272  m = m + 1
273  local(i+ioff,j+joff,k) = clocal(m)
274  end do
275  end do
276  end do
277  endif
278 
279  return
280  end subroutine MPP_DO_GLOBAL_FIELD_3D_AD_
