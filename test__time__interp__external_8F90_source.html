21 program test_time_interp_external
22 
23 use constants_mod, only : constants_init
24 use fms_mod, only : check_nml_error
25 use mpp_mod, only : mpp_init, mpp_exit, mpp_npes, stdout, stdlog, fatal, mpp_error
26 use mpp_mod, only : input_nml_file, mpp_pe, mpp_root_pe, mpp_sync
27 use mpp_domains_mod, only : mpp_domains_init, domain2d, mpp_define_layout, mpp_define_domains
28 use mpp_domains_mod, only : mpp_global_sum, mpp_global_max, mpp_global_min, bitwise_exact_sum, mpp_get_compute_domain
29 use mpp_domains_mod, only : mpp_domains_set_stack_size
30 use time_interp_external2_mod, only : time_interp_external, time_interp_external_init, get_external_fileobj
31 use time_interp_external2_mod, only : time_interp_external_exit, time_interp_external, init_external_field, get_external_field_size
32 use time_manager_mod, only : get_date, set_date, time_manager_init, set_calendar_type, julian, time_type, increment_time
33 use time_manager_mod, only : noleap
34 use horiz_interp_mod, only : horiz_interp, horiz_interp_init, horiz_interp_new, horiz_interp_del, horiz_interp_type
35 use axis_utils2_mod, only : axis_edges
36 use fms2_io_mod, only : fmsnetcdffile_t, fms2_io_init, open_file, close_file, write_data, register_axis
37 use fms2_io_mod, only : register_field, unlimited, register_variable_attribute
38 
39 
40 implicit none
41 
42 integer :: id !< Time_interp_external id
43 integer :: i !< Index for loops
44 character(len=128) :: filename='INPUT/aerosol.climatology.nc'
45 character(len=128) :: fieldname='so4_anthro'
46 type(time_type) :: time !< "model" time
47 real, allocatable :: data_d(:,:,:) !< interpolated data in compute domain
48 real, allocatable :: data_g(:,:,:) !< interpolated global data
49 type(domain2d) :: domain !<Domain to interpolate data to
50 integer :: layout(2) !< Domain layout
51 integer :: fld_size(4) !< Size of fieldname
52 integer :: isc, iec, jsc, jec !< starting (s)/ending (e) x/y indexes of the compute domain
53 integer :: isd, ied, jsd, jed !< starting (s)/ending (e) x/y indexes of the data domain
54 real :: sm !< Sum of a data array
55 real :: mx !< Max value of a data array
56 real :: mn !< Min value of a data_array
57 character(len=12) :: cal_type="julian" !< Calendar type
58 type(horiz_interp_type) :: Hinterp !< horix interp type
59 real :: lon_out(180,89) !< lat grid to interpolate to (global)
60 real :: lat_out(180,89) !< lon grid to interpolate to (global)
61 real, allocatable :: lon_local_out(:,:) !< lat grid to interpolate to (compute)
62 real, allocatable :: lat_local_out(:,:) !< lon grid to interpolate to (compute)
63 real, allocatable :: lon_in(:) !< lat grid in file
64 real, allocatable :: lat_in(:) !< lat grid in file
65 integer :: outunit !< stdout unit number
66 type(FmsNetcdfFile_t) :: fileobj !< fileobj
67 character(len=12) :: axis_names(4) !< axis_names
68 real, allocatable :: data_in(:,:) !< data added to file
69 
70 call constants_init
71 call mpp_init
72 call fms2_io_init
73 call mpp_domains_init
74 call time_interp_external_init
75 call time_manager_init
76 call horiz_interp_init
77 
78 !< Create a file to test with:
79 if (mpp_pe() .eq. mpp_root_pe()) then
80  if (open_file(fileobj, filename, "overwrite")) then
81  call register_axis(fileobj, "lon", 179)
82  call register_axis(fileobj, "lat", 89)
83  call register_axis(fileobj, "time", unlimited)
84 
85  call register_field(fileobj, "lon", "double", dimensions=(/"lon"/))
86  call register_field(fileobj, "lat", "double", dimensions=(/"lat"/))
87  call register_field(fileobj, "time", "double", dimensions=(/"time"/))
88 
89  call register_field(fileobj, fieldname, "double", dimensions=(/"lon ", "lat ", "time"/))
90  call register_field(fileobj, trim(fieldname)//"_random", "double", dimensions=(/"lon ", "lat ", "time"/))
91 
92  call register_variable_attribute(fileobj, "lon", "cartesian_axis", "X", str_len=1)
93  call register_variable_attribute(fileobj, "lat", "cartesian_axis", "Y", str_len=1)
94 
95  call register_variable_attribute(fileobj, "time", "cartesian_axis", "T", str_len=1)
96  call register_variable_attribute(fileobj, "time", "units", "days since 1800-01-01 00:00:00", str_len=30)
97  call register_variable_attribute(fileobj, "time", "calendar", "julian", str_len=6)
98 
99  call write_data(fileobj, "lat", (/(-90+i*2.0,i=1,89)/))
100  call write_data(fileobj, "lon", (/(-180+i*2.0,i=1,179)/))
101  call write_data(fileobj, "time", (/(1+i*2, i=0,2)/))
102 
103  allocate(data_in(179, 89))
104  do i=0, 2
105  data_in = real(1+i*2)
106  call write_data(fileobj, fieldname, data_in, unlim_dim_level=i+1)
107 
108  call random_number(data_in)
109  call write_data(fileobj, trim(fieldname)//"_random", data_in, unlim_dim_level=i+1)
110  enddo
111  call close_file(fileobj)
112  endif
113 endif
114 
115 !< Wait for the root pe to catch up
116 call mpp_sync()
117 
118 select case (trim(cal_type))
119 case ('julian')
120  call set_calendar_type(julian)
121 case ('no_leap')
122  call set_calendar_type(noleap)
123 case default
124  call mpp_error(fatal,'invalid calendar type')
125 end select
126 
127 outunit = stdout()
128 write(outunit,*) 'INTERPOLATING NON DECOMPOSED FIELDS'
129 write(outunit,*) '======================================'
130 !< Here every rank is on the same x/y grid
131 
132 id = init_external_field(filename,fieldname,verbose=.false.)
133 fld_size = get_external_field_size(id)
134 allocate(data_g(fld_size(1),fld_size(2),fld_size(3)))
135 data_g = 0
136 
137 !< Corresponds to time=1 in the file, so no time_interpolation
138 time = set_date(1800,1,2,0,0,0)
139 call time_interp_external(id,time,data_g,verbose=.true.)
140 sm = sum(data_g)
141 mn = minval(data_g)
142 mx = maxval(data_g)
143 write(outunit,*) 'sum= ', sm
144 write(outunit,*) 'max= ', mx
145 write(outunit,*) 'min= ', mn
146 
147 !< Corresponds to time=2, so there will be time_interpolation between time=1 and
148 !time =3 in the file
149 time = set_date(1800,1,3,0,0,0)
150 call time_interp_external(id,time,data_g,verbose=.true.)
151 sm = sum(data_g)
152 mn = minval(data_g)
153 mx = maxval(data_g)
154 write(outunit,*) 'sum= ', sm
155 write(outunit,*) 'max= ', mx
156 write(outunit,*) 'min= ', mn
157 
158 write(outunit,*) 'INTERPOLATING DOMAIN DECOMPOSED FIELDS'
159 write(outunit,*) '======================================'
160 !< Here every rank has its own section
161 
162 call mpp_define_layout((/1,fld_size(1),1,fld_size(2)/),mpp_npes(),layout)
163 call mpp_define_domains((/1,fld_size(1),1,fld_size(2)/),layout,domain)
164 call mpp_get_compute_domain(domain,isc,iec,jsc,jec)
165 call mpp_get_compute_domain(domain,isd,ied,jsd,jed)
166 
167 call mpp_domains_set_stack_size(fld_size(1)*fld_size(2)*min(fld_size(3),1)*2)
168 allocate(data_d(isd:ied,jsd:jed,fld_size(3)))
169 data_d = 0
170 
171 id = init_external_field(filename,fieldname,domain=domain, verbose=.false.)
172 
173 time = set_date(1800,1,2,0,0,0)
174 call time_interp_external(id,time,data_d,verbose=.true.)
175 sm = mpp_global_sum(domain,data_d,flags=bitwise_exact_sum)
176 mx = mpp_global_max(domain,data_d)
177 mn = mpp_global_min(domain,data_d)
178 write(outunit,*) 'global sum= ', sm
179 write(outunit,*) 'global max= ', mx
180 write(outunit,*) 'global min= ', mn
181 
182 !< Corresponds to time=2, so there will be time_interpolation between time=1 and
183 !time =3 in the file
184 time = set_date(1800,1,3,0,0,0)
185 call time_interp_external(id,time,data_d,verbose=.true.)
186 sm = mpp_global_sum(domain,data_d,flags=bitwise_exact_sum)
187 mx = mpp_global_max(domain,data_d)
188 mn = mpp_global_min(domain,data_d)
189 write(outunit,*) 'global sum= ', sm
190 write(outunit,*) 'global max= ', mx
191 write(outunit,*) 'global min= ', mn
192 
193 write(outunit,*) 'INTERPOLATING DOMAIN DECOMPOSED FIELDS USING HORIZ INTERP'
194 write(outunit,*) '======================================'
195 !< Here there will be interpolation in x/y/time
196 
197 ! define a global 2 degree output grid
198 do i=1,180
199  lon_out(i,:) = 2.0*i*atan(1.0)/45.0
200 enddo
201 
202 do i=1,89
203  lat_out(:,i) = (i-45)*2.0*atan(1.0)/45.0
204 enddo
205 
206 id = init_external_field(filename,trim(fieldname)//"_random",domain=domain,axis_names=axis_names,&
207  verbose=.false., override=.true.)
208 
209 if (.not. get_external_fileobj(filename, fileobj)) call mpp_error(fatal, "Couldn't find the fileobj for: "//trim(filename))
210 
211 allocate (lon_local_out(isc:iec,jsc:jec))
212 allocate (lat_local_out(isc:iec,jsc:jec))
213 
214 lon_local_out(isc:iec,jsc:jec) = lon_out(isc:iec,jsc:jec)
215 lat_local_out(isc:iec,jsc:jec) = lat_out(isc:iec,jsc:jec)
216 
217 !< Read in the axis data
218 allocate(lon_in(fld_size(1)+1))
219 allocate(lat_in(fld_size(2)+1))
220 
221 call axis_edges(fileobj, axis_names(1), lon_in)
222 call axis_edges(fileobj, axis_names(2), lat_in)
223 
224 lon_in = lon_in*atan(1.0)/45
225 lat_in = lat_in*atan(1.0)/45
226 
227 call horiz_interp_new(hinterp,lon_in,lat_in, lon_local_out, lat_local_out, &
228  interp_method='bilinear')
229 
230 deallocate(data_d)
231 allocate(data_d(isc:iec,jsc:jec,fld_size(3)))
232 
233 time = set_date(1800,1,3,0,0,0)
234 data_d = 0
235 call time_interp_external(id,time,data_d,verbose=.true.,horz_interp=hinterp)
236 sm = mpp_global_sum(domain,data_d,flags=bitwise_exact_sum)
237 mx = mpp_global_max(domain,data_d)
238 mn = mpp_global_min(domain,data_d)
239 write(outunit,*) 'global sum= ', sm
240 write(outunit,*) 'global max= ', mx
241 write(outunit,*) 'global min= ', mn
242 
243 sm = mpp_global_sum(domain,data_d,flags=bitwise_exact_sum)
244 write(outunit,*) 'n valid points= ', sm
245 
246 call horiz_interp_del(hinterp)
247 call time_interp_external_exit
248 call mpp_exit
249 
250 end program test_time_interp_external
