91  !> @brief Opens the output file.
92  SUBROUTINE diag_output_init (file_name, file_title, file_unit,&
93  & domain, domainU, fileobj, fileobjU, fileobjND, fnum_domain, &
94  & attributes)
95  CHARACTER(len=*), INTENT(in) :: file_name !< Output file name
96  CHARACTER(len=*), INTENT(in) :: file_title !< Descriptive title for the file
97  INTEGER , INTENT(out) :: file_unit !< File unit number assigned to the output file.
98  !! Needed for subsuquent calls to
99  !! diag_output_mod
100  TYPE(domain2d) , INTENT(in) :: domain !< Domain associated with file, if domain decomposed
101  TYPE(domainug) , INTENT(in) :: domainu !< The unstructure domain
102  type(fmsnetcdfdomainfile_t),intent(inout),target :: fileobj !< Domain decomposed fileobj
103  type(fmsnetcdfunstructureddomainfile_t),intent(inout),target :: fileobju !< Unstructured domain fileobj
104  type(fmsnetcdffile_t),intent(inout),target :: fileobjnd !< Non domain decomposed fileobj
105  character(*),intent(out) :: fnum_domain !< String indicating the type of fileobj was used:
106  !! "2d" domain decomposed
107  !! "ug" unstrucuted domain decomposed
108  !! "nd" no domain
109  TYPE(diag_atttype), INTENT(in), OPTIONAL :: attributes(:) !< Array of global attributes to be written to file
110 
111  class(fmsnetcdffile_t), pointer :: fileob => null()
112  integer :: i !< For looping through number of attributes
113  TYPE(diag_global_att_type) :: gatt
114  integer, allocatable, dimension(:) :: current_pelist
115  integer :: mype !< The pe you are on
116  character(len=9) :: mype_string !< a string to store the pe
117  !---- initialize mpp_io ----
118  IF ( .NOT.module_is_initialized ) THEN
119  module_is_initialized = .true.
120  CALL write_version_number("DIAG_OUTPUT_MOD", version)
121  END IF
122 
123 !> Checks to make sure that only domain2D or domainUG is used. If both are not null, then FATAL
124  if (domain .NE. null_domain2d .AND. domainu .NE. null_domainug)&
125  & CALL error_mesg('diag_output_init', "Domain2D and DomainUG can not be used at the same time in "//&
126  & trim(file_name), fatal)
127 
128  !---- open output file (return file_unit id) -----
129  IF ( domain .NE. null_domain2d ) THEN
130  !> Check if there is an io_domain
131  iF ( associated(mpp_get_io_domain(domain)) ) then
132  fileob => fileobj
133  if (.not.check_if_open(fileob)) call open_check(open_file(fileobj, trim(file_name)//".nc", "overwrite", &
134  domain, is_restart=.false.))
135  fnum_domain = "2d" ! 2d domain
136  file_unit = 2
137  elSE !< No io domain, so every core is going to write its own file.
138  fileob => fileobjnd
139  mype = mpp_pe()
140  write(mype_string,'(I0.4)') mype
141  if (.not.check_if_open(fileob)) then
142  call open_check(open_file(fileobjnd, trim(file_name)//".nc."//trim(mype_string), "overwrite", &
143  is_restart=.false.))
144  !< For regional subaxis add the NumFilesInSet attribute, which is added by fms2_io for (other)
145  !< domains with sufficient decomposition info. Note mppnccombine will work with an entry of zero.
146  call register_global_attribute(fileobjnd, "NumFilesInSet", 0)
147  endif
148  fnum_domain = "nd" ! no domain
149  if (file_unit < 0) file_unit = 10
150  endiF
151  ELSE IF (domainu .NE. null_domainug) THEN
152  fileob => fileobju
153  if (.not.check_if_open(fileob)) call open_check(open_file(fileobju, trim(file_name)//".nc", "overwrite", &
154  domainu, is_restart=.false.))
155  fnum_domain = "ug" ! unstructured grid
156  file_unit=3
157  ELSE
158  fileob => fileobjnd
159  allocate(current_pelist(mpp_npes()))
160  call mpp_get_current_pelist(current_pelist)
161  if (.not.check_if_open(fileob)) then
162  call open_check(open_file(fileobjnd, trim(file_name)//".nc", "overwrite", &
163  pelist=current_pelist, is_restart=.false.))
164  endif
165  fnum_domain = "nd" ! no domain
166  if (file_unit < 0) file_unit = 10
167  deallocate(current_pelist)
168  END IF
169 
170  !---- write global attributes ----
171  IF ( file_title(1:1) /= ' ' ) THEN
172  call register_global_attribute(fileob, 'title', trim(file_title), str_len=len_trim(file_title))
173  END IF
174 
175  IF ( PRESENT(attributes) ) THEN
176  DO i=1, SIZE(attributes)
177  SELECT CASE (attributes(i)%type)
178  CASE (nf90_int)
179  call register_global_attribute(fileob, trim(attributes(i)%name), attributes(i)%iatt)
180  CASE (nf90_float)
181 
182  call register_global_attribute(fileob, trim(attributes(i)%name), attributes(i)%fatt)
183  CASE (nf90_char)
184 
185  call register_global_attribute(fileob, trim(attributes(i)%name), attributes(i)%catt, str_len=len_trim(attributes(i)%catt))
186  CASE default
187  ! <ERROR STATUS="FATAL">
188  ! Unknown attribute type for attribute <name> to module/input_field <module_name>/<field_name>.
189  ! Contact the developers.
190  ! </ERROR>
191  CALL error_mesg('diag_output_mod::diag_output_init', 'Unknown attribute type for global attribute "'&
192  &//trim(attributes(i)%name)//'" in file "'//trim(file_name)//'". Contact the developers.', fatal)
193  END SELECT
194  END DO
195  END IF
196  !---- write grid type (mosaic or regular)
197  CALL get_diag_global_att(gatt)
198 
199  call register_global_attribute(fileob, 'grid_type', trim(gatt%grid_type), str_len=len_trim(gatt%grid_type))
200 
201  call register_global_attribute(fileob, 'grid_tile', trim(gatt%tile_name), str_len=len_trim(gatt%tile_name))
202 
203  END SUBROUTINE diag_output_init

205  !> @brief Write the axis meta data to file.
206  SUBROUTINE write_axis_meta_data(file_unit, axes, fileob, time_ops, time_axis_registered)
207  INTEGER, INTENT(in) :: file_unit !< File unit number
208  INTEGER, INTENT(in) :: axes(:) !< Array of axis ID's, including the time axis
209  class(fmsnetcdffile_t) , intent(inout) :: fileob !< FMS2_io fileobj
210  LOGICAL, INTENT(in), OPTIONAL :: time_ops !< .TRUE. if this file contains any min, max, time_rms, or time_average
211  logical, intent(inout) , optional :: time_axis_registered !< .TRUE. if the time axis was already written to the file
212 
213  TYPE(domain1d) :: domain
214  TYPE(domainug) :: domainu
215 
216  CHARACTER(len=mxch) :: axis_name, axis_units, axis_name_current
217  CHARACTER(len=mxchl) :: axis_long_name
218  CHARACTER(len=1) :: axis_cart_name
219  INTEGER :: axis_direction, axis_edges
220  REAL, ALLOCATABLE :: axis_data(:)
221 integer :: domain_size, axis_length, axis_pos
222  INTEGER :: num_attributes
223  TYPE(diag_atttype), DIMENSION(:), ALLOCATABLE :: attributes
224  INTEGER :: calendar, id_axis, id_time_axis
225  INTEGER :: i, j, index, num, length, edges_index
226  INTEGER :: gend !< End index of global io_domain
227  LOGICAL :: time_ops1
228  CHARACTER(len=2048) :: err_msg
229  integer :: id_axis_current
230  logical :: is_time_axis_registered
231  integer :: istart, iend
232  integer :: gstart, cstart, cend !< Start and end of global and compute domains
233  integer :: clength !< Length of compute domain
234  integer :: data_size
235  integer, allocatable, dimension(:) :: all_indicies
236  character(len=32) :: type_str !< Str indicating the type of the axis data
237 
238  ! Make sure err_msg is initialized
239  err_msg = ''
240  IF ( PRESENT(time_ops) ) THEN
241  time_ops1 = time_ops
242  ELSE
243  time_ops1 = .false.
244  END IF
245  if (present(time_axis_registered)) then
246  is_time_axis_registered = time_axis_registered
247  else
248  is_time_axis_registered = .false.
249  endif
250  !---- save the current file_unit ----
251  IF ( num_axis_in_file == 0 ) current_file_unit = file_unit
252 
253  !---- dummy checks ----
254  num = SIZE(axes(:))
255  ! <ERROR STATUS="FATAL">number of axes < 1 </ERROR>
256  IF ( num < 1 ) CALL error_mesg('write_axis_meta_data', 'number of axes < 1.', fatal)
257 
258  ! <ERROR STATUS="FATAL">writing meta data out-of-order to different files.</ERROR>
259  IF ( file_unit /= current_file_unit ) CALL error_mesg('write_axis_meta_data',&
260  & 'writing meta data out-of-order to different files.', fatal)
261 
262  IF (pack_size .eq. 1) then
263  type_str = "double"
264  ELSE IF (pack_size .eq. 2) then
265  type_str = "float"
266  ENDIF
267 
268  !---- check all axes ----
269  !---- write axis meta data for new axes ----
270  DO i = 1, num
271  id_axis = axes(i)
272  index = get_axis_index( id_axis )
273 
274  !---- skip axes already written -----
275  IF ( index > 0 ) cycle
276 
277  !---- create new axistype (then point to) -----
278  num_axis_in_file = num_axis_in_file + 1
279  axis_in_file(num_axis_in_file) = id_axis
280  edge_axis_flag(num_axis_in_file) = .false.
281  length = get_axis_global_length(id_axis)
282  ALLOCATE(axis_data(length))
283 
284  CALL get_diag_axis(id_axis, axis_name, axis_units, axis_long_name,&
285  & axis_cart_name, axis_direction, axis_edges, domain, domainu, axis_data,&
286  & num_attributes, attributes, domain_position=axis_pos)
287 
288  IF ( domain .NE. null_domain1d ) THEN
289  select type (fileob)
290  type is (fmsnetcdffile_t)
291  !> If the axis is domain decomposed and the type is FmsNetcdfFile_t, this is regional diagnostic
292  !! So treat it as any other dimension
293  call mpp_get_global_domain(domain, begin=gstart, end=gend) !< Get the global indicies
294  call mpp_get_compute_domain(domain, begin=cstart, end=cend, size=clength) !< Get the compute indicies
295  iend = cend - gstart + 1 !< Get the array indicies for the axis data
296  istart = cstart - gstart + 1
297  call register_axis(fileob, axis_name, dimension_length=clength)
298  call register_field(fileob, axis_name, type_str, (/axis_name/) )
299 
300  !> For regional subaxis add the "domain_decomposition" attribute, which is added
301  !> fms2_io for (other) domains with sufficient decomposition info.
302  call register_variable_attribute(fileob, axis_name, "domain_decomposition", &
303  (/gstart, gend, cstart, cend/))
304  type is (fmsnetcdfdomainfile_t)
305  !> If the axis is domain decomposed and the type is FmsNetcdfDomainFile_t, this is a domain decomposed dimension
306  !! so register it as one
307  call register_axis(fileob, axis_name, lowercase(trim(axis_cart_name)), domain_position=axis_pos )
308  call get_global_io_domain_indices(fileob, trim(axis_name), istart, iend)
309  call register_field(fileob, axis_name, type_str, (/axis_name/) )
310  end select
311 
312  ELSE IF ( domainu .NE. null_domainug) THEN
313  select type(fileob)
314  type is (fmsnetcdfunstructureddomainfile_t)
315  !> If the axis is in unstructured domain and the type is FmsNetcdfUnstructuredDomainFile_t, this is an unstrucutred axis
316  !! so register it as one
317  call register_axis(fileob, axis_name )
318  end select
319  call register_field(fileob, axis_name, type_str, (/axis_name/) )
320  istart = lbound(axis_data,1)
321  iend = ubound(axis_data,1)
322  ELSE
323  !> If the axis is not in a domain, register it as a normal dimension
324  call register_axis(fileob, axis_name, dimension_length=size(axis_data))
325  call register_field(fileob, axis_name, type_str, (/axis_name/) )
326  istart = lbound(axis_data,1)
327  iend = ubound(axis_data,1)
328  ENDIF !! IF ( Domain .NE. null_domain1d )
329 
330  if (length <= 0) then
331  !> @note Check if the time variable is registered. It's possible that is_time_axis_registered is set to true if using
332  !! time-templated files because they aren't closed when done writing. An alternative to this set up would be to put
333  !! variable_exists into the if statement with an .or. so that it gets registered.
334  is_time_axis_registered = variable_exists(fileob,trim(axis_name),.true.)
335  if (.not. is_time_axis_registered) then
336  call register_axis(fileob, trim(axis_name), unlimited )
337  call register_field(fileob, axis_name, type_str, (/axis_name/) )
338  is_time_axis_registered = .true.
339  if (present(time_axis_registered)) time_axis_registered = is_time_axis_registered
340  endif !! if (.not. is_time_axis_registered)
341  endif !! if (length <= 0)
342 
343  !> Add the attributes
344  if(trim(axis_units) .ne. "none") call register_variable_attribute(fileob, axis_name, "units", trim(axis_units), str_len=len_trim(axis_units))
345  call register_variable_attribute(fileob, axis_name, "long_name", trim(axis_long_name), str_len=len_trim(axis_long_name))
346  if(trim(axis_cart_name).ne."N") call register_variable_attribute(fileob, axis_name, "axis",trim(axis_cart_name), str_len=len_trim(axis_cart_name))
347 
348  if (length > 0 ) then
349  !> If not a time axis, add the positive attribute and write the data
350  select case (axis_direction)
351  case (1)
352  call register_variable_attribute(fileob, axis_name, "positive", "up", str_len=len_trim("up"))
353  case (-1)
354  call register_variable_attribute(fileob, axis_name, "positive", "down", str_len=len_trim("down"))
355  end select
356  call write_data(fileob, axis_name, axis_data(istart:iend) )
357  endif
358 
359  !> Write additional axis attributes, from diag_axis_add_attribute calls
360  CALL write_attribute_meta(file_unit, num_attributes, attributes, err_msg, varname=axis_name, fileob=fileob)
361  IF ( len_trim(err_msg) .GT. 0 ) THEN
362  CALL error_mesg('diag_output_mod::write_axis_meta_data', trim(err_msg), fatal)
363  END IF
364 
365  !> Write additional attribute (calendar_type) for time axis ----
366  !> @note calendar attribute is compliant with CF convention
367  !! http://www.cgd.ucar.edu/cms/eaton/netcdf/CF-current.htm#cal
368  IF ( axis_cart_name == 'T' ) THEN
369  time_axis_flag(num_axis_in_file) = .true.
370  id_time_axis = num_axis_in_file
371  calendar = get_calendar_type()
372 
373 
374  call register_variable_attribute(fileob, axis_name, "calendar_type", &
375  uppercase(trim(valid_calendar_types(calendar))), str_len=len_trim(valid_calendar_types(calendar)) )
376  call register_variable_attribute(fileob, axis_name, "calendar", &
377  lowercase(trim(valid_calendar_types(calendar))), str_len=len_trim(valid_calendar_types(calendar)) )
378  IF ( time_ops1 ) THEN
379  call register_variable_attribute(fileob, axis_name, 'bounds', trim(axis_name)//'_bnds', str_len=len_trim(trim(axis_name)//'_bnds'))
380  END IF
381  call set_fileobj_time_name(fileob, axis_name)
382  ELSE
383  time_axis_flag(num_axis_in_file) = .false.
384  END IF
385 
386  DEALLOCATE(axis_data)
387 
388  !> Deallocate attributes
389  IF ( ALLOCATED(attributes) ) THEN
390  DO j=1, num_attributes
391  IF ( allocated(attributes(j)%fatt ) ) THEN
392  DEALLOCATE(attributes(j)%fatt)
393  END IF
394  IF ( allocated(attributes(j)%iatt ) ) THEN
395  DEALLOCATE(attributes(j)%iatt)
396  END IF
397  END DO
398  DEALLOCATE(attributes)
399  END IF
400 
401  !------------- write axis containing edge information ---------------
402 
403  ! --- this axis has no edges -----
404  IF ( axis_edges <= 0 ) cycle
405 
406  ! --- was this axis edge previously defined? ---
407  id_axis_current = id_axis
408  axis_name_current = axis_name
409  id_axis = axis_edges
410  edges_index = get_axis_index(id_axis)
411  IF ( edges_index > 0 ) cycle
412 
413  ! ---- get data for axis edges ----
414  length = get_axis_global_length( id_axis )
415  ALLOCATE(axis_data(length))
416  CALL get_diag_axis(id_axis, axis_name, axis_units, axis_long_name, axis_cart_name,&
417  & axis_direction, axis_edges, domain, domainu, axis_data)
418 
419  ! ---- write edges attribute to original axis ----
420  call register_variable_attribute(fileob, axis_name_current, "edges",trim(axis_name), str_len=len_trim(axis_name))
421  ! ---- add edges index to axis list ----
422  ! ---- assume this is not a time axis ----
423  num_axis_in_file = num_axis_in_file + 1
424  axis_in_file(num_axis_in_file) = id_axis
425  edge_axis_flag(num_axis_in_file) = .true.
426  time_axis_flag(num_axis_in_file) = .false.
427 
428 !> Add edges axis with fms2_io
429  if (.not.variable_exists(fileob, axis_name)) then
430  call register_axis(fileob, axis_name, size(axis_data) )
431  call register_field(fileob, axis_name, type_str, (/axis_name/) )
432  if(trim(axis_units) .ne. "none") call register_variable_attribute(fileob, axis_name, "units", trim(axis_units), str_len=len_trim(axis_units))
433  call register_variable_attribute(fileob, axis_name, "long_name", trim(axis_long_name), str_len=len_trim(axis_long_name))
434  if(trim(axis_cart_name).ne."N") call register_variable_attribute(fileob, axis_name, "axis",trim(axis_cart_name), str_len=len_trim(axis_cart_name))
435  select case (axis_direction)
436  case (1)
437  call register_variable_attribute(fileob, axis_name, "positive", "up", str_len=len_trim("up"))
438  case (-1)
439  call register_variable_attribute(fileob, axis_name, "positive", "down", str_len=len_trim("down"))
440  end select
441  call write_data(fileob, axis_name, axis_data)
442  endif !! if (.not.variable_exists(fileob, axis_name))
443 
444  DEALLOCATE (axis_data)
445  END DO
446  END SUBROUTINE write_axis_meta_data

448  !> @brief Write the field meta data to file.
449  !! @return diag_fieldtype Field
450  !! @details The meta data for the field is written to the file indicated by file_unit
451  FUNCTION write_field_meta_data ( file_unit, name, axes, units, long_name, range, pack, mval,&
452  & avg_name, time_method, standard_name, interp_method, attributes, num_attributes, &
453  & use_UGdomain, fileob) result ( Field )
454  INTEGER, INTENT(in) :: file_unit !< Output file unit number
455  INTEGER, INTENT(in) :: axes(:) !< Array of axis IDs
456  CHARACTER(len=*), INTENT(in) :: name !< Field name
457  CHARACTER(len=*), INTENT(in) :: units !< Field units
458  CHARACTER(len=*), INTENT(in) :: long_name !< Field's long name
459  REAL, OPTIONAL, INTENT(in) :: range(2) !< Valid range (min, max). If min > max, the range will be ignored
460  REAL, OPTIONAL, INTENT(in) :: mval !< Missing value, must be within valid range
461  INTEGER, OPTIONAL, INTENT(in) :: pack !< Packing flag. Only valid when range specified. Valid values:
462  !! Flag | Size
463  !! --- | ---
464  !! 1 | 64bit
465  !! 2 | 32bit
466  !! 4 | 16bit
467  !! 8 | 8bit
468  CHARACTER(len=*), OPTIONAL, INTENT(in) :: avg_name !< Name of variable containing time averaging info
