83 !> @brief Initial setup for tests.
84 subroutine init(test_params, ntiles)
85 
86  type(Params), intent(out) :: test_params
87  integer, intent(in) :: ntiles
88 
89  integer :: my_rank
90  integer :: npes
91  type(Parser_t) :: parser
92 
93  character(len=32) :: buf
94  integer :: err
95  integer :: i
96 
97  !Allocate arrays.
98  allocate(test_params%global_indices(4,ntiles))
99  allocate(test_params%layout(2,ntiles))
100  allocate(test_params%pe_start(ntiles))
101  allocate(test_params%pe_end(ntiles))
102 
103  !Initialize mpp.
104  call mpp_init()
105 
106  !Define command line arguments.
107  parser = get_parser()
108  call add_argument(parser, "-x", 'Number of points (per domain tile) in the "x" direction.', &
109  requires_val=.true.)
110  call add_argument(parser, "-y", 'Number of points (per domain tile) in the "y" direction.', &
111  requires_val=.true.)
112  call add_argument(parser, "-z", 'Number of points in the "z" direction.', &
113  requires_val=.true.)
114  call add_argument(parser, "-d", "Have each rank occupy its own I/O domain tile.", &
115  requires_val=.false.)
116  call add_argument(parser, "-v", "Increase verbosity.", requires_val=.false., &
117  long_name="--verbose")
118  call parse_args(parser)
119 
120  my_rank = mpp_pe()
121  npes = mpp_npes()
122  if (my_rank .eq. 0) then
123  write(error_unit,*) "Using ", npes, " MPI ranks."
124  endif
125  call mpi_barrier(mpi_comm_world, err)
126  call mpi_check(err)
127 
128  !Set defaults.
129  test_params%nx = 96
130  test_params%ny = 96
131  test_params%nz = 30
132  test_params%io_layout(:) = 1
133  test_params%npes_group = 1
134  test_params%debug = .false.
135 
136  !Parse command line arguments.
137  call get_argument(parser, "-x", buf)
138  if (trim(buf) .ne. "not present") then
139  read(buf,*) test_params%nx
140  endif
141  call get_argument(parser, "-y", buf)
142  if (trim(buf) .ne. "not present") then
143  read(buf,*) test_params%ny
144  endif
145  call get_argument(parser, "-z", buf)
146  if (trim(buf) .ne. "not present") then
147  read(buf,*) test_params%nz
148  endif
149  call get_argument(parser, "-d", buf)
150  if (trim(buf) .eq. "present") then
151  test_params%io_layout(2) = npes/ntiles
152  else
153  test_params%npes_group = npes/ntiles
154  endif
155  call get_argument(parser, "-v", buf)
156  if (trim(buf) .eq. "present") then
157  test_params%debug = .true.
158  endif
159 
160  !Prepare for domains creation.
161  call mpp_domains_init()
162  do i = 1, ntiles
163  test_params%global_indices(:, i) = (/1, test_params%nx, 1, test_params%ny/)
164  test_params%layout(:, i) = (/1, npes/ntiles/)
165  test_params%pe_start(i) = (i-1)*(npes/ntiles)
166  test_params%pe_end(i) = i*(npes/ntiles) - 1
167  enddo
168 end subroutine init

386 !> @brief Create an unstructured land-like domain.
387 subroutine create_unstructured_domain(cubed_sphere_domain, test_params, unstructured_domain)
388 
389  type(domain2d), intent(in) :: cubed_sphere_domain
390  type(Params), intent(in) :: test_params
391  type(domainug), intent(inout) :: unstructured_domain
392 
393  integer, parameter :: ntiles = 6
394  integer :: isc
395  integer :: iec
396  integer :: jsc
397  integer :: jec
398  integer :: isd
399  integer :: ied
400  integer :: jsd
401  integer :: jed
402  logical, dimension(:,:,:), allocatable :: lmask
403  integer, dimension(:), allocatable :: npts_tile
404  real :: rmask
405  integer :: ntotal_land
406  integer, dimension(:), allocatable :: grid_index
407  integer, dimension(:), allocatable :: isl
408  integer, dimension(:), allocatable :: iel
409  integer, dimension(:), allocatable :: jsl
410  integer, dimension(:), allocatable :: jel
411  integer, dimension(:), allocatable :: ntiles_grid
412  integer :: i
413  integer :: j
414  integer :: l
415  integer :: n
416 
417  call mpp_get_compute_domain(cubed_sphere_domain, isc, iec, jsc, jec)
418  call mpp_get_data_domain(cubed_sphere_domain, isd, ied, jsd, jed)
419  allocate(npts_tile(ntiles))
420  if (mpp_pe() .eq. mpp_root_pe()) then
421  allocate(lmask(test_params%nx, test_params%ny, ntiles))
422  lmask = .false.
423  do n = 1, ntiles
424  do j = 1, test_params%ny
425  do i = 1, test_params%nx
426  call random_number(rmask)
427  if (rmask .gt. 0.5) then
428  lmask(i, j, n) = .true.
429  endif
430  enddo
431  enddo
432  npts_tile(n) = count(lmask(:, :, n))
433  enddo
434  ntotal_land = sum(npts_tile)
435  allocate(grid_index(ntotal_land))
436  l = 0
437  allocate(isl(0:mpp_npes()-1))
438  allocate(iel(0:mpp_npes()-1))
439  allocate(jsl(0:mpp_npes()-1))
440  allocate(jel(0:mpp_npes()-1))
441  call mpp_get_compute_domains(cubed_sphere_domain, xbegin=isl, xend=iel, ybegin=jsl, yend=jel)
442  do n = 1, ntiles
443  do j = 1, test_params%ny
444  do i = 1, test_params%nx
445  if (lmask(i, j, n)) then
446  l = l + 1
447  grid_index(l) = (j-1)*test_params%nx+i
448  endif
449  enddo
450  enddo
451  enddo
452  deallocate(lmask)
453  deallocate(isl)
454  deallocate(iel)
455  deallocate(jsl)
456  deallocate(jel)
457  endif
458  call mpp_broadcast(npts_tile, ntiles, mpp_root_pe())
459  if (mpp_pe() .ne. mpp_root_pe()) then
460  ntotal_land = sum(npts_tile)
461  allocate(grid_index(ntotal_land))
462  endif
463  call mpp_broadcast(grid_index, ntotal_land, mpp_root_pe())
464  allocate(ntiles_grid(ntotal_land))
465  ntiles_grid = 1
466  call mpp_define_unstruct_domain(unstructured_domain, cubed_sphere_domain, npts_tile, ntiles_grid, &
467  mpp_npes(), test_params%npes_group, grid_index, &
468  name="Unstructured domain")
469  deallocate(npts_tile)
470  deallocate(grid_index)
471  deallocate(ntiles_grid)
472 end subroutine create_unstructured_domain
