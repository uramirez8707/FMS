21 !! @brief Routines to get data in a file whose path is described in a user-provided data_table
22 !! and do spatial and temporal interpolation if necessary to convert data to model's grid and time.
23 !! @author Z. Liang, M.J. Harrison, M. Winton
24 !!
25 !! Before using @ref data_override a data_table must be created with the following entries:
26 !! gridname, fieldname_code, fieldname_file, file_name, ongrid, factor.
27 !!
28 !! More explainations about data_table entries can be found in the source code (defining data_type)
29 !!
30 !! If user wants to override fieldname_code with a const, set fieldname_file in data_table = ""
31 !! and factor = const
32 !!
33 !! If user wants to override fieldname_code with data from a file, set fieldname_file = name in
34 !! the netCDF data file, factor then will be for unit conversion (=1 if no conversion required)
35 !!
36 !! A field can be overriden globally (by default) or users can specify one or two regions in which
37 !! data_override will take place, field values outside the region will not be affected.

187  type (domain2d), intent(in), optional :: ocean_domain_in, ice_domain_in

199  read (input_nml_file, data_override_nml, iostat=io_status)
200  ierr = check_nml_error(io_status, 'data_override_nml')
201  unit = stdlog()
202  write(unit, data_override_nml)
203 
204 ! grid_center_bug is no longer supported.
205 if (grid_center_bug) then
206  call mpp_error(fatal, "data_override_init: You have overridden the default value of " // &
207  "grid_center_bug and set it to .true. in data_override_nml. This was a temporary workaround " // &
208  "that is no longer supported. Please remove this namelist variable.")

212 
213  atm_on = PRESENT(atm_domain_in)
214  ocn_on = PRESENT(ocean_domain_in)
215  lnd_on = PRESENT(land_domain_in)
216  ice_on = PRESENT(ice_domain_in)
217  lndug_on = PRESENT(land_domainug_in)
218  if(.not. module_is_initialized) then
219  atm_domain = null_domain2d
220  ocn_domain = null_domain2d
221  lnd_domain = null_domain2d
222  ice_domain = null_domain2d
223  lnd_domainug = null_domainug
224  end if
225  if (atm_on) atm_domain = atm_domain_in
226  if (ocn_on) ocn_domain = ocean_domain_in
227  if (lnd_on) lnd_domain = land_domain_in
228  if (ice_on) ice_domain = ice_domain_in
229  if (lndug_on) lnd_domainug = land_domainug_in
230 
231  if(.not. module_is_initialized) then
232  call horiz_interp_init
233  call write_version_number("DATA_OVERRIDE_MOD", version)
234 
235 ! Initialize user-provided data table
236  default_table%gridname = 'non'
237  default_table%fieldname_code = 'none'
238  default_table%fieldname_file = 'none'
239  default_table%file_name = 'none'
240  default_table%factor = 1.
241  default_table%interpol_method = 'bilinear'
242 
243 #ifdef use_yaml
244  call read_table_yaml(data_table)
245 #else
246  do i = 1,max_table
247  data_table(i) = default_table
248  enddo
249  call read_table(data_table)
250 #endif
251 
252 ! Initialize override array
253  default_array%gridname = 'NONE'
254  default_array%fieldname = 'NONE'
255  default_array%t_index = -1
256  default_array%dims = -1
257  default_array%comp_domain = -1
258  do i = 1, max_array
259  override_array(i) = default_array
260  enddo
261  call time_interp_external_init
262  end if
263 
264  module_is_initialized = .true.
265 
266  if ( .NOT. (atm_on .or. ocn_on .or. lnd_on .or. ice_on .or. lndug_on)) return
267  if (table_size .eq. 0) then
268  call mpp_error(note, "data_table is empty, not doing any data_overrides")
269  return
270  endif
271  call fms2_io_init
272 
273 ! Test if grid_file is already opened
274  inquire (file=trim(grid_file), opened=file_open)
275  if(file_open) call mpp_error(fatal, trim(grid_file)//' already opened')
276 
277  if(.not. open_file(fileobj, grid_file, 'read' )) then
278  call mpp_error(fatal, 'data_override_mod: Error in opening file '//trim(grid_file))
279  endif
280 
281  if(variable_exists(fileobj, "x_T" ) .OR. variable_exists(fileobj, "geolon_t" ) ) then
282  use_get_grid_version = 1
283  call close_file(fileobj)
284  else if(variable_exists(fileobj, "ocn_mosaic_file" ) .OR. variable_exists(fileobj, "gridfiles" ) ) then
285  use_get_grid_version = 2
286  if(variable_exists(fileobj, "gridfiles" ) ) then
287  if(count_ne_1((ocn_on .OR. ice_on), lnd_on, atm_on)) call mpp_error(fatal, 'data_override_mod: the grid file ' //&
288  'is a solo mosaic, one and only one of atm_on, lnd_on or ice_on/ocn_on should be true')
289  end if
290  else
291  call mpp_error(fatal, 'data_override_mod: none of x_T, geolon_t, ocn_mosaic_file or gridfiles exist in '// &
292  & trim(grid_file))
293  endif
294 
295  if(use_get_grid_version .EQ. 1) then
296  if (atm_on .and. .not. allocated(lon_local_atm) ) then
297  call mpp_get_compute_domain( atm_domain,is,ie,js,je)

301  endif
302  if (ocn_on .and. .not. allocated(lon_local_ocn) ) then
303  call mpp_get_compute_domain( ocn_domain,is,ie,js,je)
304  allocate(lon_local_ocn(is:ie,js:je), lat_local_ocn(is:ie,js:je))
305  call get_grid_version_1(grid_file, 'ocn', ocn_domain, is, ie, js, je, lon_local_ocn, lat_local_ocn, &
306  min_glo_lon_ocn, max_glo_lon_ocn )
307  endif
308 
309  if (lnd_on .and. .not. allocated(lon_local_lnd) ) then
310  call mpp_get_compute_domain( lnd_domain,is,ie,js,je)
311  allocate(lon_local_lnd(is:ie,js:je), lat_local_lnd(is:ie,js:je))
312  call get_grid_version_1(grid_file, 'lnd', lnd_domain, is, ie, js, je, lon_local_lnd, lat_local_lnd, &
313  min_glo_lon_lnd, max_glo_lon_lnd )
314  endif
315 
316  if (ice_on .and. .not. allocated(lon_local_ice) ) then
317  call mpp_get_compute_domain( ice_domain,is,ie,js,je)
318  allocate(lon_local_ice(is:ie,js:je), lat_local_ice(is:ie,js:je))
319  call get_grid_version_1(grid_file, 'ice', ice_domain, is, ie, js, je, lon_local_ice, lat_local_ice, &
320  min_glo_lon_ice, max_glo_lon_ice )
321  endif
322  else
323  if (atm_on .and. .not. allocated(lon_local_atm) ) then
324  call mpp_get_compute_domain(atm_domain,is,ie,js,je)
325  allocate(lon_local_atm(is:ie,js:je), lat_local_atm(is:ie,js:je))
326  call get_grid_version_2(fileobj, 'atm', atm_domain, is, ie, js, je, lon_local_atm, lat_local_atm, &
327  min_glo_lon_atm, max_glo_lon_atm )
328  endif
329 
330  if (ocn_on .and. .not. allocated(lon_local_ocn) ) then
331  call mpp_get_compute_domain( ocn_domain,is,ie,js,je)
332  allocate(lon_local_ocn(is:ie,js:je), lat_local_ocn(is:ie,js:je))
333  call get_grid_version_2(fileobj, 'ocn', ocn_domain, is, ie, js, je, lon_local_ocn, lat_local_ocn, &
334  min_glo_lon_ocn, max_glo_lon_ocn )
335  endif
336 
337  if (lnd_on .and. .not. allocated(lon_local_lnd) ) then
338  call mpp_get_compute_domain( lnd_domain,is,ie,js,je)
339  allocate(lon_local_lnd(is:ie,js:je), lat_local_lnd(is:ie,js:je))
340  call get_grid_version_2(fileobj, 'lnd', lnd_domain, is, ie, js, je, lon_local_lnd, lat_local_lnd, &
341  min_glo_lon_lnd, max_glo_lon_lnd )
342  endif
343 
344  if (ice_on .and. .not. allocated(lon_local_ice) ) then
345  call mpp_get_compute_domain( ice_domain,is,ie,js,je)
346  allocate(lon_local_ice(is:ie,js:je), lat_local_ice(is:ie,js:je))
347  call get_grid_version_2(fileobj, 'ocn', ice_domain, is, ie, js, je, lon_local_ice, lat_local_ice, &
348  min_glo_lon_ice, max_glo_lon_ice )
349  endif
350  end if
351  if(use_get_grid_version .EQ. 2) then
352  call close_file(fileobj)
353  end if

357 #ifndef use_yaml
358 subroutine read_table(data_table)
359  type(data_type), dimension(max_table), intent(inout) :: data_table
360 
361  integer :: ntable
362  integer :: ntable_lima
363  integer :: ntable_new
364 
365  integer :: iunit
366  integer :: io_status
367  character(len=256) :: record
368  type(data_type) :: data_entry
369 
370  logical :: ongrid
371  logical :: table_exists !< Flag indicating existence of data_table
372  character(len=128) :: region, region_type
373 
374  integer :: sunit
375 
376 ! Read coupler_table
377  inquire(file='data_table', exist=table_exists)
378  if (.not. table_exists) then
379  call mpp_error(note, 'data_override_mod: File data_table does not exist.')
380  table_size = 0
381  return
382  end if
383 
384  open(newunit=iunit, file='data_table', action='READ', iostat=io_status)
385  if(io_status/=0) call mpp_error(fatal, 'data_override_mod: Error in opening file data_table.')
386 
387  ntable = 0
388  ntable_lima = 0
389  ntable_new = 0
390 
391  do while (ntable <= max_table)
392  read(iunit,'(a)',end=100) record
393  if (record(1:1) == '#') cycle
394  if (record(1:10) == ' ') cycle
395  ntable=ntable+1
396  if(index(lowercase(record), "inside_region") .ne. 0 .or. index(lowercase(record), "outside_region") .ne. 0) then
397  if(index(lowercase(record), ".false.") .ne. 0 .or. index(lowercase(record), ".true.") .ne. 0 ) then
398  ntable_lima = ntable_lima + 1
399  read(record,*,err=99) data_entry%gridname, data_entry%fieldname_code, data_entry%fieldname_file, &
400  data_entry%file_name, ongrid, data_entry%factor, region, region_type
401  if(ongrid) then
402  data_entry%interpol_method = 'none'
403  else
404  data_entry%interpol_method = 'bilinear'
405  endif
406  else
407  ntable_new=ntable_new+1
408  read(record,*,err=99) data_entry%gridname, data_entry%fieldname_code, data_entry%fieldname_file, &
409  data_entry%file_name, data_entry%interpol_method, data_entry%factor, region, &
410  & region_type
411  if (data_entry%interpol_method == 'default') then
412  data_entry%interpol_method = default_table%interpol_method
413  endif
414  if (.not.(data_entry%interpol_method == 'default' .or. &
415  data_entry%interpol_method == 'bicubic' .or. &
416  data_entry%interpol_method == 'bilinear' .or. &
417  data_entry%interpol_method == 'none')) then
418  sunit = stdout()
419  write(sunit,*)" gridname is ", trim(data_entry%gridname)
420  write(sunit,*)" fieldname_code is ", trim(data_entry%fieldname_code)
421  write(sunit,*)" fieldname_file is ", trim(data_entry%fieldname_file)
422  write(sunit,*)" file_name is ", trim(data_entry%file_name)
423  write(sunit,*)" factor is ", data_entry%factor
424  write(sunit,*)" interpol_method is ", trim(data_entry%interpol_method)
425  call mpp_error(fatal, 'data_override_mod: invalid last entry in data_override_table, ' &
426  //'its value should be "default", "bicubic", "bilinear" or "none" ')
427  endif
428  endif
429  if( trim(region_type) == "inside_region" ) then
430  data_entry%region_type = inside_region
431  else if( trim(region_type) == "outside_region" ) then
432  data_entry%region_type = outside_region
433  else
434  call mpp_error(fatal, 'data_override_mod: region type should be inside_region or outside_region')
435  endif
436  if (data_entry%file_name == "") call mpp_error(fatal, &
437  "data_override: filename not given in data_table when region_type is not NO_REGION")
438  if(data_entry%fieldname_file == "") call mpp_error(fatal, &
439  "data_override: fieldname_file must be specified in data_table when region_type is not NO_REGION")
440  if( trim(data_entry%interpol_method) == 'none') call mpp_error(fatal, &
441  "data_override(data_override_init): ongrid must be false when region_type is not NO_REGION")
442  read(region,*) data_entry%lon_start, data_entry%lon_end, data_entry%lat_start, data_entry%lat_end
443  !--- make sure data_entry%lon_end > data_entry%lon_start and data_entry%lat_end > data_entry%lat_start
444  if(data_entry%lon_end .LE. data_entry%lon_start) call mpp_error(fatal, &
445  "data_override: lon_end should be greater than lon_start")
446  if(data_entry%lat_end .LE. data_entry%lat_start) call mpp_error(fatal, &
447  "data_override: lat_end should be greater than lat_start")
448  ! old format
449  else if (index(lowercase(record), ".false.") .ne. 0 .or. index(lowercase(record), ".true.") .ne. 0 ) then
450  ntable_lima = ntable_lima + 1
451  read(record,*,err=99) data_entry%gridname, data_entry%fieldname_code, data_entry%fieldname_file, &
452  data_entry%file_name, ongrid, data_entry%factor
453  if(ongrid) then
454  data_entry%interpol_method = 'none'
455  else
456  data_entry%interpol_method = 'bilinear'
457  endif
458  data_entry%lon_start = 0.0
459  data_entry%lon_end = -1.0
460  data_entry%lat_start = 0.0
461  data_entry%lat_end = -1.0
462  data_entry%region_type = no_region
463  else ! new format
464  ntable_new=ntable_new+1
465  read(record,*,err=99) data_entry%gridname, data_entry%fieldname_code, data_entry%fieldname_file, &
466  data_entry%file_name, data_entry%interpol_method, data_entry%factor
467  if (data_entry%interpol_method == 'default') then
468  data_entry%interpol_method = default_table%interpol_method
469  endif
470  if (.not.(data_entry%interpol_method == 'default' .or. &
471  data_entry%interpol_method == 'bicubic' .or. &
472  data_entry%interpol_method == 'bilinear' .or. &

690  character(len=3), intent(in) :: gridname !< model grid ID
691  character(len=*), intent(in) :: fieldname_code !< field name as used in the model
692  logical, optional, intent(out) :: override !< true if the field has been overriden succesfully
693  type(time_type), intent(in) :: time !< (target) model time
694  integer, optional, intent(in) :: data_index

696  integer, optional, intent(in) :: is_in, ie_in, js_in, je_in
697  logical, dimension(:,:,:), allocatable :: mask_out
698 
699  character(len=512) :: filename !< file containing source data
700  character(len=512) :: filename2 !< file containing source data
701  character(len=128) :: fieldname !< fieldname used in the data file
702  integer :: i,j
703  integer :: dims(4)
704  integer :: index1 !< field index in data_table
705  integer :: id_time !< index for time interp in override array
706  integer :: axis_sizes(4)
707  character(len=32) :: axis_names(4)
708  real, dimension(:,:), pointer :: lon_local =>null() !< of output (target) grid cells
709  real, dimension(:,:), pointer :: lat_local =>null() !< of output (target) grid cells
710  real, dimension(:), allocatable :: lon_tmp, lat_tmp
711 
712  logical :: data_file_is_2D = .false. !< data in netCDF file is 2D
713  logical :: ongrid, use_comp_domain
714  type(domain2d) :: domain
715  integer :: curr_position !< position of the field currently processed in override_array
716  real :: factor
717  integer, dimension(4) :: comp_domain = 0 !< istart,iend,jstart,jend for compute domain
718  integer :: nxd, nyd, nxc, nyc, nwindows
719  integer :: nwindows_x, ipos, jpos, window_size(2)

723  logical :: need_compute
724  real :: lat_min, lat_max

726  logical :: exists
727  type(fmsnetcdffile_t) :: fileobj
728  integer :: startingi !< Starting x index for the compute domain relative to the input buffer
729  integer :: endingi !< Ending x index for the compute domain relative to the input buffer
730  integer :: startingj !< Starting y index for the compute domain relative to the input buffer
731  integer :: endingj !< Ending y index for the compute domain relative to the input buffer
732  integer :: nhalox !< Number of halos in the x direction
733  integer :: nhaloy !< Number of halos in the y direction
734 
735  use_comp_domain = .false.
736  if(.not.module_is_initialized) &
737  call mpp_error(fatal,'Error: need to call data_override_init first')
738 
739 !1 Look for the data file in data_table
740  if(PRESENT(override)) override = .false.
741  if (present(data_index)) then
742  index1 = data_index
743  else
744  index1 = -1
745  do i = 1, table_size
746  if( trim(gridname) /= trim(data_table(i)%gridname)) cycle
747  if( trim(fieldname_code) /= trim(data_table(i)%fieldname_code)) cycle
748  index1 = i ! field found
749  exit
750  enddo
751  if(index1 .eq. -1) then
752  if(debug_data_override) &
753  call mpp_error(warning,'this field is NOT found in data_table: '//trim(fieldname_code))
754  return ! NO override was performed
755  endif
756  endif
757 
758  fieldname = data_table(index1)%fieldname_file ! fieldname in netCDF data file
759  factor = data_table(index1)%factor
760 
761  if(fieldname == "") then
762  data = factor
763  if(PRESENT(override)) override = .true.
764  return
765  else
766  filename = data_table(index1)%file_name
767  if (filename == "") call mpp_error(fatal,'data_override: filename not given in data_table')
768  endif
769 
770  ongrid = (data_table(index1)%interpol_method == 'none')
771 
772 !3 Check if fieldname has been previously processed
773 !$OMP CRITICAL
774  curr_position = -1
775  if(num_fields > 0 ) then
776  do i = 1, num_fields
777  if(trim(override_array(i)%gridname) /= trim(gridname)) cycle
778  if(trim(override_array(i)%fieldname) /= trim(fieldname_code)) cycle
779  curr_position = i
780  exit
781  enddo
782  endif
783 
784  if(curr_position < 0) then ! the field has not been processed previously
785  num_fields = num_fields + 1
786  curr_position = num_fields
787 
788 ! Get working domain from model's gridname
789  call get_domain(gridname,domain,comp_domain)
790  call mpp_get_data_domain(domain, xsize=nxd, ysize=nyd)
791  nxc = comp_domain(2)-comp_domain(1) + 1
792  nyc = comp_domain(4)-comp_domain(3) + 1
793 
794 ! record fieldname, gridname in override_array
795  override_array(curr_position)%fieldname = fieldname_code
796  override_array(curr_position)%gridname = gridname
797  override_array(curr_position)%comp_domain = comp_domain
798 ! get number of threads
799  override_array(curr_position)%numthreads = 1
800 #if defined(_OPENMP)
801  override_array(curr_position)%numthreads = omp_get_num_threads()
802 #endif
803 !--- data_override may be called from physics windows. The following are possible situations
804 !--- 1. size(data,1) == nxd and size(data,2) == nyd ( on data domain and there is only one window).
805 !--- 2. nxc is divisible by size(data,1), nyc is divisible by size(data,2),
806 !--- nwindow = (nxc/size(data(1))*(nyc/size(data,2)), also we require nwindows is divisible by nthreads.
807 !--- The another restrition is that size(data,1) == ie_in - is_in + 1,
808 !--- size(data,2) == je_in - js_in + 1
809  nwindows = 1
810  if( nxd == size(data,1) .AND. nyd == size(data,2) ) then !
811  use_comp_domain = .false.
812  else if ( mod(nxc, size(data,1)) ==0 .AND. mod(nyc, size(data,2)) ==0 ) then
813  use_comp_domain = .true.
814  nwindows = (nxc/size(data,1))*(nyc/size(data,2))
815  else
816  call mpp_error(fatal, &
817  & "data_override: data is not on data domain and compute domain is not divisible by size(data)")
818  endif
819  override_array(curr_position)%window_size(1) = size(data,1)
820  override_array(curr_position)%window_size(2) = size(data,2)
821 
822  window_size = override_array(curr_position)%window_size
823  override_array(curr_position)%numwindows = nwindows
824  if( mod(nwindows, override_array(curr_position)%numthreads) .NE. 0 ) then
825  call mpp_error(fatal, "data_override: nwindow is not divisible by nthreads")
826  endif
827  allocate(override_array(curr_position)%need_compute(nwindows))
828  override_array(curr_position)%need_compute = .true.
829 
830 !4 get index for time interp
831  if(ongrid) then
832  if( data_table(index1)%region_type .NE. no_region ) then
833  call mpp_error(fatal,.NE.'data_override: ongrid must be false when region_type NO_REGION')
834  endif
835 
836 ! Allow on-grid data_overrides on cubed sphere grid
837  inquire(file=trim(filename),exist=exists)
838  if (.not. exists) then
839  call get_mosaic_tile_file(filename,filename2,.false.,domain)
840  filename = filename2
841  endif
842 
843  !--- we always only pass data on compute domain
844  id_time = init_external_field(filename,fieldname,domain=domain,verbose=.false., &
845  use_comp_domain=use_comp_domain, nwindows=nwindows, ongrid=ongrid)
846  dims = get_external_field_size(id_time)
847  override_array(curr_position)%dims = dims
848  if(id_time<0) call mpp_error(fatal,'data_override:field not found in init_external_field 1')
849  override_array(curr_position)%t_index = id_time
850  else !ongrid=false
851  id_time = init_external_field(filename,fieldname,domain=domain, axis_names=axis_names,&
852  axis_sizes=axis_sizes, verbose=.false.,override=.true.,use_comp_domain=use_comp_domain, &
853  nwindows = nwindows)
854  dims = get_external_field_size(id_time)
855  override_array(curr_position)%dims = dims
856  if(id_time<0) call mpp_error(fatal,'data_override:field not found in init_external_field 2')
857  override_array(curr_position)%t_index = id_time
858 
859  ! get lon and lat of the input (source) grid, assuming that axis%data contains
860  ! lat and lon of the input grid (in degrees)
861 
862  allocate(override_array(curr_position)%horz_interp(nwindows))
863  allocate(override_array(curr_position)%lon_in(axis_sizes(1)+1))
864  allocate(override_array(curr_position)%lat_in(axis_sizes(2)+1))
865  if(get_external_fileobj(filename, fileobj)) then
866  call axis_edges(fileobj, axis_names(1), override_array(curr_position)%lon_in, &
867  reproduce_null_char_bug_flag=reproduce_null_char_bug)
868  call axis_edges(fileobj, axis_names(2), override_array(curr_position)%lat_in, &
869  reproduce_null_char_bug_flag=reproduce_null_char_bug)
870  else
871  call mpp_error(fatal,'data_override: file '//trim(filename)//' is not opened in time_interp_external')
872  end if
873 ! convert lon_in and lat_in from deg to radian
874  override_array(curr_position)%lon_in = override_array(curr_position)%lon_in * deg_to_radian
875  override_array(curr_position)%lat_in = override_array(curr_position)%lat_in * deg_to_radian
876 
877  !--- find the region of the source grid that cover the local model grid.
878  !--- currently we only find the index range for j-direction because
879  !--- of the cyclic condition in i-direction. The purpose of this is to
880  !--- decrease the memory usage and increase the IO performance.
881  select case(gridname)
882  case('OCN')
883  lon_local => lon_local_ocn; lat_local => lat_local_ocn
884  case('ICE')
885  lon_local => lon_local_ice; lat_local => lat_local_ice
886  case('ATM')
887  lon_local => lon_local_atm; lat_local => lat_local_atm
888  case('LND')
889  lon_local => lon_local_lnd; lat_local => lat_local_lnd
890  case default
891  call mpp_error(fatal,'error: gridname not recognized in data_override')
892  end select
893 
894  lat_min = minval(lat_local)
895  lat_max = maxval(lat_local)
896  is_src = 1
897  ie_src = axis_sizes(1)
898  js_src = 1
899  je_src = axis_sizes(2)
900  do j = 1, axis_sizes(2)+1
901  if( override_array(curr_position)%lat_in(j) > lat_min ) exit
902  js_src = j
903  enddo
904  do j = 1, axis_sizes(2)+1
905  je_src = j
906  if( override_array(curr_position)%lat_in(j) > lat_max ) exit
907  enddo
908 
909  !--- bicubic interpolation need one extra point in each direction. Also add
910  !--- one more point for because lat_in is in the corner but the interpolation
911  !--- use center points.
912  select case (data_table(index1)%interpol_method)
913  case ('bilinear')
914  js_src = max(1, js_src-1)
915  je_src = min(axis_sizes(2), je_src+1)
916  case ('bicubic')
917  js_src = max(1, js_src-2)
918  je_src = min(axis_sizes(2), je_src+2)
919  end select
920  override_array(curr_position)%is_src = is_src
921  override_array(curr_position)%ie_src = ie_src
922  override_array(curr_position)%js_src = js_src
923  override_array(curr_position)%je_src = je_src
924  call reset_src_data_region(id_time, is_src, ie_src, js_src, je_src)
925 
926 ! Find the index of lon_start, lon_end, lat_start and lat_end in the input grid (nearest points)
927  if( data_table(index1)%region_type .NE. no_region ) then
928  allocate( lon_tmp(axis_sizes(1)), lat_tmp(axis_sizes(2)) )
929  call read_data(fileobj, axis_names(1), lon_tmp)
930  call read_data(fileobj, axis_names(2), lat_tmp)

932  ! lat_start, lat_end are inside lat_in
933  if(data_table(index1)%lon_start < lon_tmp(1) .OR. data_table(index1)%lon_start .GT. lon_tmp(axis_sizes(1)))&
934  call mpp_error(fatal, "data_override: lon_start is outside lon_T")
