19 !> @defgroup horiz_interp_conserve_mod horiz_interp_conserve_mod
20 !> @ingroup horiz_interp
21 !> @brief Performs spatial interpolation between grids using conservative interpolation
22 !!
23 !> @author Bruce Wyman, Zhi Liang
24 !!
25 !> This module can conservatively interpolate data from any logically rectangular grid
26 !! to any rectangular grid. The interpolation scheme is area-averaging
27 !! conservative scheme. There is an optional mask field for missing input data in both
28 !! horiz_interp__conserveinit and horiz_interp_conserve. For efficiency purpose, mask should only be
29 !! kept in horiz_interp_init (will remove the mask in horiz_interp in the future).
30 !! There are 1-D and 2-D version of horiz_interp_conserve_init for 1-D and 2-D grid.
31 !! There is a optional argument mask in horiz_interp_conserve_init_2d and no mask should
32 !! to passed into horiz_interp_conserv. optional argument mask will not be passed into
33 !! horiz_interp_conserve_init_1d and optional argument mask may be passed into
34 !! horiz_interp_conserve (For the purpose of reproduce Memphis??? results).
35 !! An optional output mask field may be used in conjunction with the input mask to show
36 !! where output data exists.
37

133  subroutine horiz_interp_conserve_new_1dx1d ( Interp, lon_in, lat_in, lon_out, lat_out, verbose)
134  type(horiz_interp_type), intent(inout) :: interp
135  real, intent(in), dimension(:) :: lon_in , lat_in
136  real, intent(in), dimension(:) :: lon_out, lat_out
137  integer, intent(in), optional :: verbose
138 
139  !-----------------------------------------------------------------------
140  real, dimension(size(lat_out(:))-1,2) :: sph
141  real, dimension(size(lon_out(:))-1,2) :: theta
142  real, dimension(size(lat_in(:))) :: slat_in
143  real, dimension(size(lon_in(:))-1) :: dlon_in
144  real, dimension(size(lat_in(:))-1) :: dsph_in
145  real, dimension(size(lon_out(:))-1) :: dlon_out
146  real, dimension(size(lat_out(:))-1) :: dsph_out
147  real :: blon, fac, hpi, tpi, eps
148  integer :: num_iters = 4
149  integer :: i, j, m, n, nlon_in, nlat_in, nlon_out, nlat_out, &
150  iverbose, m2, n2, iter
151  logical :: s2n
152  character(len=64) :: mesg
153 
154  if(.not. module_is_initialized) call mpp_error(fatal, &
155  'horiz_interp_conserve_new_1dx1d: horiz_interp_conserve_init is not called')
156 
157  if(great_circle_algorithm) call mpp_error(fatal, &
158  'horiz_interp_conserve_new_1dx1d: great_circle_algorithm is not implemented, contact developer')
159  !-----------------------------------------------------------------------
160  iverbose = 0; if (present(verbose)) iverbose = verbose
161 
162  pe = mpp_pe()
163  root_pe = mpp_root_pe()
164  !-----------------------------------------------------------------------
165  hpi = 0.5*pi
166  tpi = 4.*hpi
167  interp%version = 1
168  nlon_in = size(lon_in(:))-1; nlat_in = size(lat_in(:))-1
169  nlon_out = size(lon_out(:))-1; nlat_out = size(lat_out(:))-1
170 
171  allocate ( interp % facj (nlat_out,2), interp % jlat (nlat_out,2), &
172  interp % faci (nlon_out,2), interp % ilon (nlon_out,2), &
173  interp % area_src (nlon_in, nlat_in), &
174  interp % area_dst (nlon_out, nlat_out) )
175 
176  !-----------------------------------------------------------------------
177  ! --- set-up for input grid boxes ---
178 
179  do j = 1, nlat_in+1
180  slat_in(j) = sin(lat_in(j))
181  enddo
182 
183  do j = 1, nlat_in
184  dsph_in(j) = abs(slat_in(j+1)-slat_in(j))
185  enddo
186 
187  do i = 1,nlon_in
188  dlon_in(i) = abs(lon_in(i+1)-lon_in(i))
189  enddo
190 
191  ! set south to north flag
192  s2n = .true.
193  if (lat_in(1) > lat_in(nlat_in+1)) s2n = .false.
194 
195  !-----------------------------------------------------------------------
196  ! --- set-up for output grid boxes ---
197 
198  do n = 1, nlat_out
199  dsph_out(n) = abs(sin(lat_out(n+1))-sin(lat_out(n)))
200  enddo
201 
202  do m = 1,nlon_out
203  theta(m,1) = lon_out(m)
204  theta(m,2) = lon_out(m+1)
205  dlon_out(m) = abs(lon_out(m+1)-lon_out(m))
206  enddo
207 
208  interp%nlon_src = nlon_in; interp%nlat_src = nlat_in
209  interp%nlon_dst = nlon_out; interp%nlat_dst = nlat_out
210  !***********************************************************************
211 
212  !------ set up latitudinal indexing ------
213  !------ make sure output grid goes south to north ------
214 
215  do n = 1, nlat_out
216  if (lat_out(n) < lat_out(n+1)) then
217  sph(n,1) = sin(lat_out(n))
218  sph(n,2) = sin(lat_out(n+1))
219  else
220  sph(n,1) = sin(lat_out(n+1))
221  sph(n,2) = sin(lat_out(n))
222  endif
223  enddo
224 
225  interp%jlat = 0
226  do n2 = 1, 2 ! looping on grid box edges
227  do n = 1, nlat_out ! looping on output latitudes
228  eps = 0.0
229  do iter=1,num_iters
230  ! find indices from input latitudes
231  do j = 1, nlat_in
232  if ( (s2n .and. (slat_in(j)-sph(n,n2)) <= eps .and. &
233  (sph(n,n2)-slat_in(j+1)) <= eps) .or. &
234  (.not.s2n .and. (slat_in(j+1)-sph(n,n2)) <= eps .and. &
235  (sph(n,n2)-slat_in(j)) <= eps) ) then
236  interp%jlat(n,n2) = j
237  ! weight with sin(lat) to exactly conserve area-integral
238  fac = (sph(n,n2)-slat_in(j))/(slat_in(j+1)-slat_in(j))
239  if (s2n) then
240  if (n2 == 1) interp%facj(n,n2) = 1.0 - fac
241  if (n2 == 2) interp%facj(n,n2) = fac
242  else
243  if (n2 == 1) interp%facj(n,n2) = fac
244  if (n2 == 2) interp%facj(n,n2) = 1.0 - fac
245  endif
246  exit
247  endif
248  enddo
249  if ( interp%jlat(n,n2) /= 0 ) exit
250  ! did not find this output grid edge in the input grid
251  ! increase tolerance for multiple passes
252  eps = epsilon(sph)*real(10**iter)
253  enddo
254  ! no match
255  if ( interp%jlat(n,n2) == 0 ) then
256  write (mesg,710) n,sph(n,n2)
257 710 format (': n,sph=',i3,f14.7,40x)
258  call mpp_error(fatal, 'horiz_interp_conserve_mod:no latitude index found'//trim(mesg))
259  endif
260  enddo
261  enddo
262 
263  !------ set up longitudinal indexing ------
264 
265  interp%ilon = 0
266  do m2 = 1, 2 ! looping on grid box edges
267  do m = 1, nlon_out ! looping on output longitudes
268  blon = theta(m,m2)
269  if ( blon < lon_in(1) ) blon = blon + tpi
270  if ( blon > lon_in(nlon_in+1) ) blon = blon - tpi
271  eps = 0.0
272  do iter=1,num_iters
273  ! find indices from input longitudes
274  do i = 1, nlon_in
275  if ( (lon_in(i)-blon) <= eps .and. &
276  (blon-lon_in(i+1)) <= eps ) then
277  interp%ilon(m,m2) = i
278  fac = (blon-lon_in(i))/(lon_in(i+1)-lon_in(i))
279  if (m2 == 1) interp%faci(m,m2) = 1.0 - fac
280  if (m2 == 2) interp%faci(m,m2) = fac
281  exit
282  endif
283  enddo
284  if ( interp%ilon(m,m2) /= 0 ) exit
285  ! did not find this output grid edge in the input grid
286  ! increase tolerance for multiple passes
287  eps = epsilon(blon)*real(10**iter)
288  enddo
289  ! no match
290  if ( interp%ilon(m,m2) == 0 ) then
291  print *, 'lon_out,blon,blon_in,eps=', &
292  theta(m,m2),blon,lon_in(1),lon_in(nlon_in+1),eps
293  call mpp_error(fatal, 'horiz_interp_conserve_mod: no longitude index found')
294  endif
295  enddo
296  enddo
297 
298  ! --- area of input grid boxes ---
299 
300  do j = 1,nlat_in
301  do i = 1,nlon_in
302  interp%area_src(i,j) = dlon_in(i) * dsph_in(j)
303  enddo
304  enddo
305 
306  ! --- area of output grid boxes ---
307 
308  do n = 1, nlat_out
309  do m = 1, nlon_out
310  interp%area_dst(m,n) = dlon_out(m) * dsph_out(n)
311  enddo
312  enddo
313 
314  !-----------------------------------------------------------------------
315  ! this output may be quite lengthy and is not recommended
316  ! when using more than one processor
317  if (iverbose > 2) then
318  write (*,801) (i,interp%ilon(i,1),interp%ilon(i,2), &
319  interp%faci(i,1),interp%faci(i,2),i=1,nlon_out)
320  write (*,802) (j,interp%jlat(j,1),interp%jlat(j,2), &
321  interp%facj(j,1),interp%facj(j,2),j=1,nlat_out)
322 801 format (/,2x,'i',4x,'is',5x,'ie',4x,'facis',4x,'facie', &
323  /,(i4,2i7,2f10.5))
324 802 format (/,2x,'j',4x,'js',5x,'je',4x,'facjs',4x,'facje', &
325  /,(i4,2i7,2f10.5))
326  endif
327  !-----------------------------------------------------------------------
328 
329  end subroutine horiz_interp_conserve_new_1dx1d

333  subroutine horiz_interp_conserve_new_1dx2d ( Interp, lon_in, lat_in, lon_out, lat_out, &
334  mask_in, mask_out, verbose)
335  type(horiz_interp_type), intent(inout) :: interp
336  real, intent(in), dimension(:) :: lon_in , lat_in
337  real, intent(in), dimension(:,:) :: lon_out, lat_out
338  real, intent(in), optional, dimension(:,:) :: mask_in
339  real, intent(inout), optional, dimension(:,:) :: mask_out
340  integer, intent(in), optional :: verbose
341 
342 
343  integer :: create_xgrid_1dx2d_order1, get_maxxgrid, maxxgrid
344  integer :: create_xgrid_great_circle
345  integer :: nlon_in, nlat_in, nlon_out, nlat_out, nxgrid, i, j
346  real(double_kind), dimension(size(lon_in(:))-1, size(lat_in(:))-1) :: mask_src
347  integer, allocatable, dimension(:) :: i_src, j_src, i_dst, j_dst
348  real(double_kind), allocatable, dimension(:) :: xgrid_area, clon, clat
349  real(double_kind), allocatable, dimension(:,:) :: dst_area, lon_src, lat_src
350  real(double_kind), allocatable, dimension(:) :: lat_in_flip
351  real(double_kind), allocatable, dimension(:,:) :: mask_src_flip
352  real(double_kind), allocatable, dimension(:) :: lon_in_r8, lat_in_r8
353  real(double_kind), allocatable, dimension(:,:) :: lon_out_r8, lat_out_r8
354 
355  integer :: nincrease, ndecrease
356  logical :: flip_lat
357  integer :: wordsz
358  integer(kind=1) :: one_byte(8)
359 
360  if(.not. module_is_initialized) call mpp_error(fatal, &
361  'horiz_interp_conserve_new_1dx2d: horiz_interp_conserve_init is not called')
362 
363  wordsz=size(transfer(lon_in(1), one_byte))
364  if(wordsz .NE. 4 .AND. wordsz .NE. 8) call mpp_error(fatal, &
365  'horiz_interp_conserve_new_1dx2d: wordsz should be 4 or 8')
366 
367  if( (size(lon_out,1) .NE. size(lat_out,1)) .OR. (size(lon_out,2) .NE. size(lat_out,2)) ) &
368  call mpp_error(fatal, 'horiz_interp_conserve_mod: size mismatch between lon_out and lat_out')
369  nlon_in = size(lon_in(:)) - 1; nlat_in = size(lat_in(:)) - 1
370  nlon_out = size(lon_out,1) - 1; nlat_out = size(lon_out,2) - 1
371 
372  mask_src = 1.
373  if(present(mask_in)) then
374  if( (size(mask_in,1) .NE. nlon_in) .OR. (size(mask_in,2) .NE. nlat_in)) call mpp_error(fatal, &
375  'horiz_interp_conserve_mod: size mismatch between mask_in and lon_in/lat_in')
376  mask_src = mask_in
377  end if
378 
379  maxxgrid = get_maxxgrid()
380  allocate(i_src(maxxgrid), j_src(maxxgrid), i_dst(maxxgrid), j_dst(maxxgrid) )
381  allocate( xgrid_area(maxxgrid), dst_area(nlon_out, nlat_out) )
382 
383  !--- check if source latitude is flipped
384  nincrease = 0
385  ndecrease = 0
386  do j = 1, nlat_in
387  if( lat_in(j+1) > lat_in(j) ) then
388  nincrease = nincrease + 1
389  else if ( lat_in(j+1) < lat_in(j) ) then
390  ndecrease = ndecrease + 1
391  endif
392  enddo
393 
394  if(nincrease == nlat_in) then
395  flip_lat = .false.
396  else if(ndecrease == nlat_in) then
397  flip_lat = .true.
398  else
399  call mpp_error(fatal, 'horiz_interp_conserve_mod: nlat_in should be equal to nincreaase or ndecrease')
400  endif
401 
402  allocate(lon_out_r8(size(lon_out,1),size(lon_out,2)))
403  allocate(lat_out_r8(size(lat_out,1),size(lat_out,2)))
404  lon_out_r8 = lon_out
405  lat_out_r8 = lat_out
406 
407  if( .not. great_circle_algorithm ) then
408  if(flip_lat) then
409  allocate(lat_in_flip(nlat_in+1), mask_src_flip(nlon_in,nlat_in))
410  do j = 1, nlat_in+1
411  lat_in_flip(j) = lat_in(nlat_in+2-j)
412  enddo
413  do j = 1, nlat_in
414  mask_src_flip(:,j) = mask_src(:,nlat_in+1-j)
415  enddo
416  allocate(lon_in_r8(size(lon_in)))
417  lon_in_r8 = lon_in
418  nxgrid = create_xgrid_1dx2d_order1(nlon_in, nlat_in, nlon_out, nlat_out, lon_in_r8, lat_in_flip, &
419  lon_out_r8, lat_out_r8, mask_src_flip, i_src, j_src, i_dst, j_dst, xgrid_area)
420  deallocate(lon_in_r8, lat_in_flip, mask_src_flip)
421  else
422  allocate(lon_in_r8(size(lon_in)))
423  allocate(lat_in_r8(size(lat_in)))
424  lon_in_r8 = lon_in
425  lat_in_r8 = lat_in
426  nxgrid = create_xgrid_1dx2d_order1(nlon_in, nlat_in, nlon_out, nlat_out, lon_in_r8, lat_in_r8, lon_out_r8, lat_out_r8, &
427  mask_src, i_src, j_src, i_dst, j_dst, xgrid_area)
428  deallocate(lon_in_r8,lat_in_r8)
429  endif
430  else
431  allocate(lon_src(nlon_in+1,nlat_in+1), lat_src(nlon_in+1,nlat_in+1))
432  allocate(clon(maxxgrid), clat(maxxgrid))
433  if(flip_lat) then
434  allocate(mask_src_flip(nlon_in,nlat_in))
435  do j = 1, nlat_in+1
436  do i = 1, nlon_in+1
437  lon_src(i,j) = lon_in(i)
438  lat_src(i,j) = lat_in(nlat_in+2-j)
439  enddo
440  enddo
441  do j = 1, nlat_in
442  mask_src_flip(:,j) = mask_src(:,nlat_in+1-j)
443  enddo
444  nxgrid = create_xgrid_great_circle(nlon_in, nlat_in, nlon_out, nlat_out, lon_src, lat_src, lon_out_r8, lat_out_r8, &
445  mask_src_flip, i_src, j_src, i_dst, j_dst, xgrid_area, clon, clat)
446  deallocate(mask_src_flip)
447  else
448  do j = 1, nlat_in+1
449  do i = 1, nlon_in+1
450  lon_src(i,j) = lon_in(i)
451  lat_src(i,j) = lat_in(j)
452  enddo
453  enddo
454  nxgrid = create_xgrid_great_circle(nlon_in, nlat_in, nlon_out, nlat_out, lon_src, lat_src, lon_out_r8, lat_out_r8, &
455  mask_src, i_src, j_src, i_dst, j_dst, xgrid_area, clon, clat)
456  endif
457  deallocate(lon_src, lat_src, clon, clat)
458  endif
459 
460  deallocate(lon_out_r8, lat_out_r8)
461 
462  allocate(interp%i_src(nxgrid), interp%j_src(nxgrid) )
463  allocate(interp%i_dst(nxgrid), interp%j_dst(nxgrid) )
464  allocate(interp%area_frac_dst(nxgrid) )
465  interp%version = 2
466  interp%nxgrid = nxgrid
467  interp%i_src = i_src(1:nxgrid)+1 ! in C, the starting index is 0
468  interp%j_src = j_src(1:nxgrid)+1
469  if(flip_lat) interp%j_src = nlat_in+1-interp%j_src
470  interp%i_dst = i_dst(1:nxgrid)+1
471  interp%j_dst = j_dst(1:nxgrid)+1
472 
473  ! sum over exchange grid area to get destination grid area
474  dst_area = 0.
475  do i = 1, nxgrid
476  dst_area(interp%i_dst(i), interp%j_dst(i)) = dst_area(interp%i_dst(i), interp%j_dst(i)) + xgrid_area(i)
477  end do
478 
479  do i = 1, nxgrid
480  interp%area_frac_dst(i) = xgrid_area(i)/dst_area(interp%i_dst(i), interp%j_dst(i) )
481  end do
482  interp%nlon_src = nlon_in; interp%nlat_src = nlat_in
483  interp%nlon_dst = nlon_out; interp%nlat_dst = nlat_out
484  if(present(mask_out)) then
485  if( (size(mask_out,1) .NE. nlon_out) .OR. (size(mask_out,2) .NE. nlat_out) ) call mpp_error(fatal, &
486  'horiz_interp_conserve_mod: size mismatch between mask_out and lon_out/lat_out')
487  mask_out = 0.0
488  do i = 1, nxgrid
489  mask_out(interp%i_dst(i),interp%j_dst(i)) = mask_out(interp%i_dst(i),interp%j_dst(i)) + interp%area_frac_dst(i)
490  end do
491  end if
492 
493  deallocate(i_src, j_src, i_dst, j_dst, xgrid_area, dst_area )
494 
495  end subroutine horiz_interp_conserve_new_1dx2d

499  subroutine horiz_interp_conserve_new_2dx1d ( Interp, lon_in, lat_in, lon_out, lat_out, &
500  mask_in, mask_out, verbose)
501  type(horiz_interp_type), intent(inout) :: interp
502  real, intent(in), dimension(:,:) :: lon_in , lat_in

722  !! the input data. The real value of mask_in must be in the range (0.,1.).
723  !! Set mask_in=0.0 for data points that should not be used or have missing
724  !! data. mask_in will be applied only when horiz_interp_conserve_new_1d is

733  call mpp_error(fatal, 'horiz_interp_conserve_mod: size of input array incorrect')

741  case (2)
742  if(present(mask_in) .OR. present(mask_out) ) call mpp_error(fatal, &
743  'horiz_interp_conserve: for version 2, mask_in and mask_out must be passed in horiz_interp_new, not in horiz_interp')

750  subroutine horiz_interp_conserve_version1 ( Interp, data_in, data_out, verbose, &
751  mask_in, mask_out)
752  !-----------------------------------------------------------------------
753  type (horiz_interp_type), intent(in) :: interp
754  real, intent(in), dimension(:,:) :: data_in
755  real, intent(out), dimension(:,:) :: data_out
756  integer, intent(in), optional :: verbose
757  real, intent(in), dimension(:,:), optional :: mask_in
758  real, intent(out), dimension(:,:), optional :: mask_out
759  !----------local variables----------------------------------------------------
760  integer :: m, n, nlon_in, nlat_in, nlon_out, nlat_out, &
761  miss_in, miss_out, is, ie, js, je, &
762  np, npass, iverbose
763  real :: dsum, wsum, avg_in, min_in, max_in, &
764  avg_out, min_out, max_out, eps, asum, &
765  dwtsum, wtsum, arsum, fis, fie, fjs, fje
766  !-----------------------------------------------------------------------
767  iverbose = 0; if (present(verbose)) iverbose = verbose
768 
769  eps = epsilon(wtsum)
770 
771  nlon_in = interp%nlon_src; nlat_in = interp%nlat_src
772  nlon_out = interp%nlon_dst; nlat_out = interp%nlat_dst
773 
774  if (present(mask_in)) then
775  if ( count(mask_in < -.0001 .or. mask_in > 1.0001) > 0 ) &
776  call mpp_error(fatal, 'horiz_interp_conserve_mod: input mask not between 0,1')
777  endif
778 
779  !-----------------------------------------------------------------------
780  !---- loop through output grid boxes ----
781 
782  data_out = 0.0
783  do n = 1, nlat_out
784  ! latitude window
785  ! setup ascending latitude indices and weights
786  if (interp%jlat(n,1) <= interp%jlat(n,2)) then
787  js = interp%jlat(n,1); je = interp%jlat(n,2)
788  fjs = interp%facj(n,1); fje = interp%facj(n,2)
789  else
790  js = interp%jlat(n,2); je = interp%jlat(n,1)
791  fjs = interp%facj(n,2); fje = interp%facj(n,1)
792  endif
793 
794  do m = 1, nlon_out
795  ! longitude window
796  is = interp%ilon(m,1); ie = interp%ilon(m,2)
797  fis = interp%faci(m,1); fie = interp%faci(m,2)
798  npass = 1
799  dwtsum = 0.
800  wtsum = 0.
801  arsum = 0.
802 
803  ! wrap-around on input grid
804  ! sum using 2 passes (pass 1: end of input grid)
805  if ( ie < is ) then
806  ie = nlon_in
807  fie = 1.0
808  npass = 2
809  endif
810 
811  do np = 1, npass
812  ! pass 2: beginning of input grid
813  if ( np == 2 ) then
814  is = 1
815  fis = 1.0
816  ie = interp%ilon(m,2)
817  fie = interp%faci(m,2)
818  endif
819 
820  ! summing data*weight and weight for single grid point
821  if (present(mask_in)) then
822  call data_sum ( data_in(is:ie,js:je), interp%area_src(is:ie,js:je), &
823  fis, fie, fjs,fje, dwtsum, wtsum, arsum, mask_in(is:ie,js:je) )
824  else if( ASSOCIATED(interp%mask_in) ) then
825  call data_sum ( data_in(is:ie,js:je), interp%area_src(is:ie,js:je), &
826  fis, fie, fjs,fje, dwtsum, wtsum, arsum, interp%mask_in(is:ie,js:je) )
827  else
828  call data_sum ( data_in(is:ie,js:je), interp%area_src(is:ie,js:je), &
829  fis, fie, fjs,fje, dwtsum, wtsum, arsum )
830  endif
831  enddo
832 
833  if (wtsum > eps) then
834  data_out(m,n) = dwtsum/wtsum
835  if (present(mask_out)) mask_out(m,n) = wtsum/arsum
836  else
837  data_out(m,n) = 0.
838  if (present(mask_out)) mask_out(m,n) = 0.0
839  endif
840 
841  enddo
842  enddo
843 
844  !***********************************************************************
845  ! compute statistics: minimum, maximum, and mean
846  !-----------------------------------------------------------------------
847 
848  if (iverbose > 0) then
849 
850  ! compute statistics of input data
851 
852  call stats(data_in, interp%area_src, asum, dsum, wsum, min_in, max_in, miss_in, mask_in)
853  ! diagnostic messages
854  ! on the root_pe, we can calculate the global mean, minimum and maximum.
855  if(pe == root_pe) then
856  if (wsum > 0.0) then
857  avg_in=dsum/wsum
858  else
859  print *, 'horiz_interp stats: input area equals zero '
860  avg_in=0.0
861  endif
862  if (iverbose > 1) print '(2f16.11)', 'global sum area_in = ', asum, wsum
863  endif
864 
865  ! compute statistics of output data
866  call stats(data_out, interp%area_dst, asum, dsum, wsum, min_out, max_out, miss_out, mask_out)
867  ! diagnostic messages
868  if(pe == root_pe) then
869  if (wsum > 0.0) then
870  avg_out=dsum/wsum
871  else
872  print *, 'horiz_interp stats: output area equals zero '
873  avg_out=0.0
874  endif
875  if (iverbose > 1) print '(2f16.11)', 'global sum area_out = ', asum, wsum
876  endif
877  !---- output statistics ----
878  ! the global mean, min and max are calculated on the root pe.
879  if(pe == root_pe) then
880  write (*,900)
881  write (*,901) min_in ,max_in ,avg_in
882  if (present(mask_in)) write (*,903) miss_in
883  write (*,902) min_out,max_out,avg_out
884  if (present(mask_out)) write (*,903) miss_out
885  endif
886 
887 900 format (/,1x,10('-'),' output from horiz_interp ',10('-'))
888 901 format (' input: min=',f16.9,' max=',f16.9,' avg=',f22.15)
889 902 format (' output: min=',f16.9,' max=',f16.9,' avg=',f22.15)
890 903 format (' number of missing points = ',i6)
891 
892  endif
893 
894  !-----------------------------------------------------------------------
895  end subroutine horiz_interp_conserve_version1

898  subroutine horiz_interp_conserve_version2 ( Interp, data_in, data_out, verbose )
899  !-----------------------------------------------------------------------
900  type (horiz_interp_type), intent(in) :: interp
901  real, intent(in), dimension(:,:) :: data_in
902  real, intent(out), dimension(:,:) :: data_out
903  integer, intent(in), optional :: verbose
904  integer :: i, i_src, j_src, i_dst, j_dst
