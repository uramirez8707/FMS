66 !> @brief Calculates what a given calendar time would be after a interval of time
67 !!
68 !! @param time_increment A time interval
69 !! @param units
70 !! Examples of acceptable values of units:
71 !! - 'days since 1980-01-01 00:00:00'
72 !! - 'hours since 1980-1-1 0:0:0'
73 !! - 'minutes since 0001-4-12'
74 !! The first word in the string must be
75 !! 'years', 'months', 'days', 'hours', 'minutes' or 'seconds'.
76 !! The second word must be 'since'.
77 !! year number must occupy 4 spaces.
78 !! Number of months, days, hours, minutes, seconds may occupy 1 or 2 spaces
79 !! year, month and day must be separated by a '-'
80 !! hour, minute, second must be separated by a ':'
81 !! hour, minute, second are optional. If not present then zero is assumed.
82 !!
83 !! Because months are not equal increments of time, and, for julian calendar,
84 !! neither are years, the 'years since' and 'month since' cases deserve
85 !! further explaination.
86 !!
87 !! When 'years since' is used:
88 !! The year number is increased by floor(time_increment) to obtain a time T1.
89 !! The year number is increased by floor(time_increment)+1 to obtain a time T2.
90 !! The time returned is T1 + (time_increment-floor(time_increment))*(T2-T1).
91 !!
92 !! When 'months since' is used:
93 !! The month number is increased by floor(time_increment). If it falls outside
94 !! to range 1 to 12 then it is adjusted along with the year number to convert
95 !! to a valid date. The number of days in the month of this date is used to
96 !! compute the time interval of the fraction.
97 !! That is:
98 !! The month number is increased by floor(time_increment) to obtain a time T1.
99 !! delt = the number of days in the month in which T1 falls.
100 !! The time returned is T1 + ((time_increment-floor(time_increment))*delt.
101 !! Two of the consequences of this scheme should be kept in mind.
102 !! -- The time since should not be from the 29'th to 31'st of a month,
103 !! since an invalid date is likely to result, triggering an error stop.
104 !! -- When time since is from the begining of a month, the fraction of a month
105 !! will never advance into the month after that which results from only
106 !! the whole number.
107 !!
108 !! When NO_CALENDAR is in effect, units attribute must specify a starting
109 !! day and second, with day number appearing first
110 !!
111 !! Example: 'days since 100 0' Indicates 100 days 0 seconds
112 !!
113 !! @param calendar
114 !! Acceptable values of calendar are:
115 !! 'noleap'
116 !! '365_day'
117 !! '360_day'
118 !! 'julian'
119 !! 'thirty_day_months'
120 !! 'no_calendar'
121 !!
122 !! @param optional permit_calendar_conversion
123 !! It is sometimes desirable to allow the value of the intent(in) argument
124 !! "calendar" to be different than the calendar in use by time_manager_mod.
125 !! If this is not desirable, then the optional variable "permit_calendar_conversion"
126 !! should be set to .false. so as to allow an error check.
127 !! When calendar conversion is done, the time returned is the time in the
128 !! time_manager's calendar, but corresponds to the date computed using the input calendar.
129 !! For example, suppose the time_manager is using the julian calendar and
130 !! the values of the input arguments of get_cal_time are:
131 !! time_increment = 59.0
132 !! units = 'days since 1980-1-1 00:00:00'
133 !! calendar = 'noleap'
134 !! Because it will use the noleap calendar to calculate the date, get_cal_time will return
135 !! value of time for midnight March 1 1980, but it will be time in the julian calendar
136 !! rather than the noleap calendar. It will never return a value of time corresponding
137 !! to anytime during the day Feb 29.
138 !!
139 !! Another example:
140 !! Suppose the time_manager is using either the noleap or julian calendars,
141 !! and the values of the input arguments are:
142 !! time_increment = 30.0
143 !! units = 'days since 1980-1-1'
144 !! calendar = 'thirty_day_months'
145 !! In this case get_cal_time will return the value of time for Feb 1 1980 00:00:00,
146 !! but in the time_manager's calendar.
147 !!
148 !! Calendar conversion may result in a fatal error when the input calendar type is
149 !! a calendar that has more days per year than that of the time_manager's calendar.
150 !! For example, if the input calendar type is julian and the time_manager's calendar
151 !! is thirty_day_months, then get_cal_time will try to convert Jan 31 to a time in
152 !! the thirty_day_months calendar, resulting in a fatal error.
153 !!
154 !! @note This option was originally coded to allow noleap calendar as input when
155 !! the julian calendar was in effect by the time_manager.
156 function get_cal_time(time_increment, units, calendar, permit_calendar_conversion)

164 real :: month_fraction

166 logical :: correct_form

168 character(len=64) :: err_msg
169 type(time_type) :: base_time, base_time_plus_one_yr

173 if(.not.module_is_initialized) then
174  read (input_nml_file, get_cal_time_nml, iostat=io)
175  ierr = check_nml_error(io, 'get_cal_time_nml')
176 
177  call write_version_number("GET_CAL_TIME_MOD", version)
178  logunit = stdlog()
179  if(mpp_pe() == mpp_root_pe()) write (logunit, nml=get_cal_time_nml)
180  module_is_initialized = .true.
181 endif

183 if(present(permit_calendar_conversion)) then
184  permit_conversion_local = permit_calendar_conversion
185 else
186  permit_conversion_local = allow_calendar_conversion
187 endif
188 
189 calendar_in_c = lowercase(trim(cut0(calendar)))
190 
191 correct_form = (trim(calendar_in_c)) == 'noleap' .or. (trim(calendar_in_c)) == '365_day' .or. &
192  (trim(calendar_in_c)) == '365_days' .or. &
193  (trim(calendar_in_c)) == '360_day' .or. (trim(calendar_in_c)) == 'julian' .or. &
194  (trim(calendar_in_c)) == 'no_calendar'.or. (trim(calendar_in_c)) == 'thirty_day_months' .or. &
195  (trim(calendar_in_c)) == 'gregorian'
196 
197 if(.not.correct_form) then
198  call error_mesg('get_cal_time','"'//trim(calendar_in_c)//'"'// &
199  ' is not an acceptable calendar attribute. acceptable calendars are: '// &
200  ' noleap, 365_day, 365_days, 360_day, julian, no_calendar, thirty_day_months, gregorian',fatal)
201 endif
202 
203 calendar_tm_i = get_calendar_type()
204 
205 if(.not.permit_conversion_local) then
206  correct_form = (trim(calendar_in_c) == 'noleap' .and. calendar_tm_i == noleap) .or. &
207  (trim(calendar_in_c) == '365_day' .and. calendar_tm_i == noleap) .or. &
208  (trim(calendar_in_c) == '365_days' .and. calendar_tm_i == noleap) .or. &
209  (trim(calendar_in_c) == '360_day' .and. calendar_tm_i == thirty_day_months) .or. &
210  (trim(calendar_in_c) == 'thirty_day_months' .and. calendar_tm_i == thirty_day_months) .or. &
211  (trim(calendar_in_c) == 'julian' .and. calendar_tm_i == julian) .or. &
212  (trim(calendar_in_c) == 'no_calendar' .and. calendar_tm_i == no_calendar) .or. &
213  (trim(calendar_in_c) == 'gregorian' .and. calendar_tm_i == gregorian)
214  if(.not.correct_form) then
215  call error_mesg('get_cal_time','calendar not consistent with calendar type in use by time_manager.'// &
216  ' calendar='//trim(calendar_in_c)//'. Type in use by time_manager='// &
217  & valid_calendar_types(calendar_tm_i),fatal)
218  endif
219 endif
220 
221 if (permit_conversion_local) then
222  select case (trim(calendar_in_c))
223  case ('noleap')
224  calendar_in_i = noleap
225  case ('365_day')
226  calendar_in_i = noleap
227  case ('365_days')
228  calendar_in_i = noleap
229  case ('360_day')
230  calendar_in_i = thirty_day_months
231  case ('thirty_day_months')
232  calendar_in_i = thirty_day_months
233  case ('julian')
234  calendar_in_i = julian
235  case ('no_calendar')
236  calendar_in_i = no_calendar
237  case ('gregorian')
238  calendar_in_i = gregorian
239  case default
240  call error_mesg('get_cal_time', &
241  trim(calendar_in_c)//' is an invalid calendar type (specified in call to get_cal_time)',fatal)
242  end select
243 else
244  calendar_in_i = calendar_tm_i
245 end if
246 
247 correct_form = lowercase(units(1:10)) == 'days since' .or. &
248  lowercase(units(1:11)) == 'hours since' .or. &
249  lowercase(units(1:13)) == 'minutes since' .or. &
250  lowercase(units(1:13)) == 'seconds since'
251 
252 if(calendar_in_i /= no_calendar) then
253  correct_form = correct_form .or. &
254  lowercase(units(1:11)) == 'years since' .or. &
255  lowercase(units(1:12)) == 'months since'
256 endif
257 
258 if(.not.correct_form) then
259  call error_mesg('get_cal_time',trim(units)//' is an invalid string for units.' // &
260  ' units must begin with a time unit then the word "since"' // &
261  ' Valid time units are: "seconds" "minutes", "hours", "days", and, ' // &
262  ' except when NO_CALENDAR is in effect, "months" and "years"',fatal)
263 endif
264 
265 if(calendar_in_i /= calendar_tm_i) then
266 ! switch to calendar type specified as input argument,
267 ! will switch back before returning.
268  call set_calendar_type(calendar_in_i)
269 endif
270 
271 ! index(string, substring[,back])
272 ! Returns the starting position of substring as a substring of string,
273 ! or zero if it does not occur as a substring. Default value of back is
274 ! .false. If back is .false., the starting position of the first such
275 ! substring is returned. If back is .true., the starting position of the
276 ! last such substring is returned.
277 ! Returns zero if substring is not a substring of string (regardless of value of back)
278 
279 i1 = index(units,'since') + 5
280 if(calendar_in_i == no_calendar) then
281  base_time = set_time(units(i1:len_trim(units)))
282 else
283  base_time = set_date(units(i1:len_trim(units)))
284 endif
285 
286 if(lowercase(units(1:10)) == 'days since') then
287  increment_days = floor(time_increment)
288  increment_seconds = int(86400*(time_increment - increment_days))
289 else if(lowercase(units(1:11)) == 'hours since') then
290  increment_days = floor(time_increment/24)
291  increment_seconds = int(86400*(time_increment/24 - increment_days))
292 else if(lowercase(units(1:13)) == 'minutes since') then
293  increment_days = floor(time_increment/1440)
294  increment_seconds = int(86400*(time_increment/1440 - increment_days))
295 else if(lowercase(units(1:13)) == 'seconds since') then
296  increment_days = floor(time_increment/86400)
297  increment_seconds = int(86400*(time_increment/86400 - increment_days))
298 else if(lowercase(units(1:11)) == 'years since') then
299 ! The time period between between (base_time + time_increment) and
300 ! (base_time + time_increment + 1 year) may be 360, 365, or 366 days.
301 ! This must be determined to handle time increments with year fractions.
302  call get_date(base_time, year,month,day,hour,minute,second)
303  base_time = set_date(year+floor(time_increment) ,month,day,hour,minute,second)
304  base_time_plus_one_yr = set_date(year+floor(time_increment)+1,month,day,hour,minute,second)
305  call get_time(base_time_plus_one_yr - base_time, second, day)
306  dt = (day*86400+second)*(time_increment-floor(time_increment))
307  increment_days = floor(dt/86400)
308  increment_seconds = int(dt - increment_days*86400)
309 else if(lowercase(units(1:12)) == 'months since') then
310  month_fraction = time_increment - floor(time_increment)
311  increment_years = floor(time_increment/12)
312  increment_months = floor(time_increment) - 12*increment_years
313  call get_date(base_time, year,month,day,hour,minute,second)
314  base_time = set_date(year+increment_years,month+increment_months ,day,hour,minute,second)
315  dt = 86400*days_in_month(base_time) * month_fraction
316  increment_days = floor(dt/86400)
317  increment_seconds = int(dt - increment_days*86400)
318 else
319  call error_mesg('get_cal_time','"'//trim(units)//'" is not an acceptable units attribute of time.'// &
320  & ' It must begin with: "years since", "months since", "days since", "hours since", "minutes since",'// &
321  & ' or "seconds since"',fatal)
322 endif
323 
324 if (calendar_in_i /= calendar_tm_i) then
325  if(calendar_in_i == no_calendar .or. calendar_tm_i == no_calendar) then
326  call error_mesg('get_cal_time','Cannot do calendar conversion because input calendar is '// &
327  trim(valid_calendar_types(calendar_in_i))//' and time_manager is using '// &
328  trim(valid_calendar_types(calendar_tm_i))// &
329  ' Conversion cannot be done if either is NO_CALENDAR',fatal)
330  endif
331  call get_date(base_time,year, month, day, hour, minute, second)
332  get_cal_time = set_date(year,month,day,hour,minute,second) + set_time(increment_seconds, increment_days)
333  call get_date(get_cal_time,year,month,day,hour,minute,second)
334  call set_calendar_type(calendar_tm_i)
335  get_cal_time = set_date(year,month,day,hour,minute,second, err_msg=err_msg)
336  if(err_msg /= '') then
337  call error_mesg('get_cal_time','Error in function get_cal_time: '//trim(err_msg)// &
338  ' Note that the time_manager is using the '//trim(valid_calendar_types(calendar_tm_i))//' calendar '// &
339  'while the calendar type passed to function get_cal_time is '//calendar_in_c,fatal)
340  endif
341 else
342  get_cal_time = base_time + set_time(increment_seconds, increment_days)
343 endif
344 
345 end function get_cal_time
