20 !> @brief This programs tests calls to get_mosaic_ntiles, get_mosaic_ncontacts,
21 !! get_mosaic_grid_sizes, get_mosaic_contact
22 program test_mosaic
23 
24 use mosaic2_mod, only : get_mosaic_ntiles, get_mosaic_ncontacts
25 use mosaic2_mod, only : get_mosaic_grid_sizes, get_mosaic_contact
26 use mpp_mod, only : mpp_init, mpp_error, fatal, mpp_sync, mpp_npes, mpp_get_current_pelist
27 use fms2_io_mod, only : open_file, close_file, fmsnetcdffile_t
28 use fms2_io_mod, only : register_axis, register_field, write_data
29 use fms_mod, only : fms_init, fms_end
30 
31 implicit none
32 
33 integer :: ntiles !< Number of tiles
34 integer :: ncontacts !< Number of contacts
35 integer :: n !< For do loops
36 integer, allocatable :: tile1(:) !< tile number for first contact
37 integer, allocatable :: tile2(:) !< tile number of the second contact
38 integer, allocatable :: nx(:), ny(:) !< Number of x/y points for each tile
39 integer, allocatable :: istart1(:), iend1(:), jstart1(:), jend1(:) !< Indexes of first contact point
40 integer, allocatable :: istart2(:), iend2(:), jstart2(:), jend2(:) !< Indexes of second contact point
41 character(len=128) :: mosaic_file !< Mosaic filename
42 type(FmsNetcdfFile_t):: mosaic_fileobj !< Fileobj for the file read by the test
43 integer :: answers(2, 8) !< Expected results
44 integer, allocatable :: pes(:) !< List of pes in the current pelist
45 
46 call mpp_init()
47 call fms_init()
48 
49 mosaic_file = "INPUT/ocean_mosaic.nc"
50 answers(1,:) = (/1440, 1440, 1, 1080, 1, 1, 1, 1080 /)
51 answers(2,:) = (/1, 720, 1080, 1080, 1440, 721, 1080, 1080 /)
52 
53 allocate(pes(mpp_npes()))
54 call mpp_get_current_pelist(pes)
55 
56 call create_files(pes)
57 
58 !< Open the mosaic file
59 if(.not. open_file(mosaic_fileobj, mosaic_file, 'read', pelist=pes)) then
60  call mpp_error(fatal, 'test_mosaic: error in opening file '//trim(mosaic_file))
61 endif
62 
63 ntiles = get_mosaic_ntiles(mosaic_fileobj)
64 ncontacts = get_mosaic_ncontacts(mosaic_fileobj)
65 allocate(nx(ntiles), ny(ntiles))
66 allocate(tile1(ncontacts), tile2(ncontacts) )
67 allocate(istart1(ncontacts), iend1(ncontacts), jstart1(ncontacts), jend1(ncontacts) )
68 allocate(istart2(ncontacts), iend2(ncontacts), jstart2(ncontacts), jend2(ncontacts) )
69 
70 call get_mosaic_grid_sizes(mosaic_fileobj, nx, ny )
71 call get_mosaic_contact(mosaic_fileobj, tile1, tile2, istart1, iend1, jstart1, jend1, istart2, iend2, jstart2, jend2)
72 
73 !< Compare with expected results:
74 if (ntiles .ne. 1) call mpp_error(fatal, "ntiles is not equal to 1")
75 
76 do n = 1, ntiles
77  if (nx(n) .ne. 2880/2) call mpp_error(fatal, "nx is not the expected result")
78  if (ny(n) .ne. 2160/2) call mpp_error(fatal, "ny is not the expected result")
79 end do
80 
81 if (ncontacts .ne. 2) call mpp_error(fatal, "ncontacts is not the expected result")
82 do n = 1, ncontacts
83  if (istart1(n) .ne. answers(n,1)) call mpp_error(fatal, "istart1 is not the expected result")
84  if (iend1(n) .ne. answers(n,2)) call mpp_error(fatal, "iend1 is not the expected result")
85 
86  if (jstart1(n) .ne. answers(n,3)) call mpp_error(fatal, "jstart1 is not the expected result")
87  if (jend1(n) .ne. answers(n,4)) call mpp_error(fatal, "jend1 is not the expected result")
88 
89  if (istart2(n) .ne. answers(n,5)) call mpp_error(fatal, "istart2 is not the expected result")
90  if (iend2(n) .ne. answers(n,6)) call mpp_error(fatal, "iend2 is not the expected result")
91 
92  if (jstart2(n) .ne. answers(n,7)) call mpp_error(fatal, "jstart2 is not the expected result")
93  if (jend2(n) .ne. answers(n,8)) call mpp_error(fatal, "jend2 is not the expected result")
94 end do
95 
96 deallocate(tile1, tile2, nx, ny)
97 deallocate(istart1, iend1, jstart1, jend1)
98 deallocate(istart2, iend2, jstart2, jend2)
99 
100 call close_file(mosaic_fileobj)
101 call fms_end()
102 
103 contains
104 
105 subroutine create_files(pes)
106  integer, intent(in) :: pes(:) !< List of pes
107 
108  type(FmsNetcdfFile_t):: fileobj !< Fileobj for the files written by the test
109  character(len=255) :: str_array(2) !< Array of strings because GNU
110 
111  if( open_file(fileobj, mosaic_file, 'overwrite', pelist=pes)) then
112  call register_axis(fileobj, "ntiles", 1)
113  call register_axis(fileobj, "ncontact", 2)
114  call register_axis(fileobj, "string", 255)
115 
116  str_array(1) = "string"
117  str_array(2) = "ncontact"
118  call register_field(fileobj, "contacts", "char", dimensions=str_array)
119  call register_field(fileobj, "contact_index", "char", dimensions=str_array)
120  call register_field(fileobj, "gridfiles", "char", dimensions=(/"string", "ntiles"/))
121  call register_field(fileobj, "gridtiles", "char", dimensions=(/"string", "ntiles"/))
122 
123  call write_data(fileobj, "gridfiles", (/"ocean_hgrid.nc"/))
124  call write_data(fileobj, "gridtiles", (/"tile1"/))
125 
126  str_array(1) = "2880:2880,1:2160::1:1,1:2160"
127  str_array(2) = "1:1440,2160:2160::2880:1441,2160:2160"
128  call write_data(fileobj, "contact_index", str_array)
129  call write_data(fileobj, "contacts", &
130  & (/"ocean_mosaic:tile1::ocean_mosaic:tile1", "ocean_mosaic:tile1::ocean_mosaic:tile1" /))
131 
132  call close_file(fileobj)
133  endif
134  call mpp_sync()
135 
136  if( open_file(fileobj, "INPUT/ocean_hgrid.nc", "overwrite", pelist=pes)) then
137  call register_axis(fileobj, "nx", 2880)
138  call register_axis(fileobj, "ny", 2160)
139 
140  call close_file(fileobj)
141  endif
142  call mpp_sync()
143 end subroutine create_files
144 
145 end program test_mosaic
