58 ! <INTERFACE NAME="time_interp">
59 
60 ! <OVERVIEW>
61 ! Returns a weight and dates or indices for interpolating between two dates. The
62 ! interface fraction_of_year is provided for backward compatibility with the
63 ! previous version.
64 ! </OVERVIEW>
65 ! <DESCRIPTION>
66 ! Returns weight by interpolating Time between Time1 and Time2.
67 ! i.e. weight = (Time-Time1)/(Time2-Time1)
68 ! Time1 and Time2 may be specified by any of several different ways,
69 ! which is the reason for multiple interfaces.
70 
71 ! If Time1 and Time2 are the begining and end of the year in which
72 ! Time falls, use first interface.
73 
74 ! If Time1 and Time2 fall on year boundaries, use second interface.
75 
76 ! If Time1 and Time2 fall on month boundaries, use third.
77 
78 ! If Time1 and Time2 fall on day boundaries, use fourth.
79 
80 ! If Time1 and Time2 are consecutive elements of an assending list, use fifth.
81 ! The fifth also returns the indices of Timelist between which Time falls.
82 
83 ! The sixth interface is for cyclical data. Time_beg and Time_end specify the
84 ! begining and end of a repeating period. In this case:
85 ! weight = (Time_adjusted - Time1) / (Time2 - Time1)
86 ! Where:
87 ! Time1 = Timelist(index1)
88 ! Time2 = Timelist(index2)
89 ! Time_adjusted = Time - N*Period
90 ! Period = Time_end-Time_beg
91 ! N is between (Time-Time_end)/Period and (Time-Time_beg)/Period
92 ! That is, N is the integer that results in Time_adjusted that is between Time_beg and Time_end.
93 !
94 ! </DESCRIPTION>
95 ! <TEMPLATE>
96 ! 1. call time_interp( Time, weight )
97 ! </TEMPLATE>
98 ! <TEMPLATE>
99 ! 2. call time_interp( Time, weight, year1, year2 )
100 ! </TEMPLATE>
101 ! <TEMPLATE>
102 ! 3. call time_interp( Time, weight, year1, year2, month1, month2 )
103 ! </TEMPLATE>
104 ! <TEMPLATE>
105 ! 4. call time_interp( Time, weight, year1, year2, month1, month2, day1, day2 )
106 ! </TEMPLATE>
107 ! <TEMPLATE>
108 ! 5. call time_interp( Time, Timelist, weight, index1, index2 [, modtime] )
109 ! </TEMPLATE>
110 ! <TEMPLATE>
111 ! 6. call time_interp( Time, Time_beg, Time_end, Timelist, weight, index1, index2 [,correct_leap_year_inconsistency])
112 ! </TEMPLATE>
113 ! <IN NAME="Time">
114 ! The time at which the the weight is computed.
115 ! </IN>
116 ! <IN NAME="Time_beg">
117 ! For cyclical interpolation: Time_beg specifies the begining time of a cycle.
118 ! </IN>
119 ! <IN NAME="Time_end">
120 ! For cyclical interpolation: Time_end specifies the ending time of a cycle.
121 ! </IN>
122 ! <IN NAME="Timelist">
123 ! For cyclical interpolation: Timelist is an array of times between Time_beg and Time_end.
124 ! Must be monotonically increasing.
125 ! </IN>
126 ! <IN NAME="modtime">
127 ! </IN>
128 ! <IN NAME="index1">
129 ! Timelist(index1) = The largest value of Timelist which is less than mod(Time,Time_end-Time_beg)
130 ! </IN>
131 ! <IN NAME="index2">
132 ! Timelist(index2) = The smallest value of Timelist which is greater than mod(Time,Time_end-Time_beg)
133 ! </IN>
134 ! <IN NAME="correct_leap_year_inconsistency">
135 ! Turns on a kluge for an inconsistency which may occur in a special case.
136 ! When the modulo time period (i.e. Time_end - Time_beg) is a whole number of years
137 ! and is not a multiple of 4, and the calendar in use has leap years, then it is
138 ! likely that the interpolation will involve mapping a common year onto a leap year.
139 ! In this case it is often desirable, but not absolutely necessary, to use data for
140 ! Feb 28 of the leap year when it is mapped onto a common year.
141 ! To turn this on, set correct_leap_year_inconsistency=.true.
142 ! </IN>
143 ! <OUT NAME="weight">
144 ! weight = (mod(Time,Time_end-Time_beg) - Timelist(index1)) / (Timelist(index2) - Timelist(index1))
145 ! </OUT>
146 ! <OUT NAME="year1"> </OUT>
147 ! <OUT NAME="year2"> </OUT>
148 ! <OUT NAME="month1"> </OUT>
149 ! <OUT NAME="month2"> </OUT>
150 ! <OUT NAME="day1"> </OUT>
151 ! <OUT NAME="day2"> </OUT>
152 ! <OUT NAME="index1"> </OUT>
153 ! <OUT NAME="index2"> </OUT>
154 ! <ERROR MSG="input time list not ascending order" STATUS="ERROR">
155 ! The list of input time types must have ascending dates.
156 ! </ERROR>
157 ! <ERROR MSG="modulo months must have same length" STATUS="ERROR">
158 ! The length of the current month for input Time and Time_list
159 ! must be the same when using the modulo month option. The
160 ! modulo month option is available but not supported.
161 ! </ERROR>
162 ! <ERROR MSG="invalid value for argument modtime" STATUS="ERROR">
163 ! The optional argument modtime must have a value set by one
164 ! of the public parameters: NONE, YEAR, MONTH, DAY. The
165 ! MONTH and DAY options are available but not supported.
166 ! </ERROR>
167 ! <ERROR MSG="period of list exceeds modulo period" STATUS="ERROR">
168 ! The difference between the last and first values in the input
169 ! Time list/array exceeds the length of the modulo period.
170 ! </ERROR>
171 ! <ERROR MSG="time before range of list or time after range of list" STATUS="ERROR">
172 ! The difference between the last and first values in the input
173 ! These errors occur when you are not using a modulo axis and
174 ! the input Time occurs before the first value in the Time
175 ! list/array or after the last value in the Time list/array.
176 ! </ERROR>
177 ! <NOTE>
178 ! Examples:
179 ! <PRE>
180 ! Time: Jan 01 00z weight = 0.0
181 ! Time: Jul 01 weight ~ 0.5
182 ! Time: Dec 31 23z weight ~ 1.0
183 ! </PRE>
184 ! </NOTE>

334 ! returns fractional time between mid points of consecutive years
335 
336  !> @brief Calculates fractional time between mid points of consecutive years
337  subroutine time_interp_year ( Time, weight, year1, year2 )
338 
339  type(time_type), intent(in) :: Time
340  real , intent(out) :: weight !< fractional time between midpoints of year1 and year2
341  integer , intent(out) :: year1, year2
342 
343  integer :: year, month, day, hour, minute, second
344  type (time_type) :: Mid_year, Mid_year1, Mid_year2
345 
346 
347  if ( .not. module_is_initialized ) call time_interp_init()

350 
351  ! mid point of current year
352  mid_year = year_midpt(year)
353 
354  if ( time >= mid_year ) then
355  ! current time is after mid point of current year
356  year1 = year
357  year2 = year+1
358  mid_year2 = year_midpt(year2)
359  weight = (time - mid_year) // (mid_year2 - mid_year)
360  else
361  ! current time is before mid point of current year
362  year2 = year
363  year1 = year-1
364  mid_year1 = year_midpt(year1)
365  weight = (time - mid_year1) // (mid_year - mid_year1)
366  endif
367 
368  end subroutine time_interp_year

371 ! <SUBROUTINE NAME="time_interp_month" INTERFACE="time_interp">
372 ! <IN NAME="Time" TYPE="time_type" > </IN>
373 ! <OUT NAME="weight" TYPE="real"> </OUT>
374 ! <OUT NAME="year1" TYPE="integer"> </OUT>
375 ! <OUT NAME="year2" TYPE="integer"> </OUT>
376 ! <OUT NAME="month1" TYPE="integer"> </OUT>
377 ! <OUT NAME="month2" TYPE="integer"> </OUT>
378 ! </SUBROUTINE>
379  !> @brief Calculates fractional time between mid points of consecutive months
380  subroutine time_interp_month ( Time, weight, year1, year2, month1, month2 )
381 
382  type(time_type), intent(in) :: Time
383  real , intent(out) :: weight
384  integer , intent(out) :: year1, year2, month1, month2
385 
386  integer :: year, month, day, hour, minute, second, &
387  mid_month, cur_month, mid1, mid2
388 
389  if ( .not. module_is_initialized ) call time_interp_init()
390 
391  call get_date (time, year, month, day, hour, minute, second)
392 
393  ! mid point of current month in seconds
394  mid_month = days_in_month(time) * halfday
395  ! time into current month in seconds
396  cur_month = second + secmin*minute + sechour*hour + secday*(day-1)
397 
398  if ( cur_month >= mid_month ) then
399  ! current time is after mid point of current month
400  year1 = year; month1 = month
401  year2 = year; month2 = month+1
402  if (month2 > monyear) then
403  year2 = year2+1; month2 = 1
404  endif
405  mid1 = mid_month
406  mid2 = days_in_month(set_date(year2,month2,2)) * halfday
407  weight = real(cur_month - mid1) / real(mid1+mid2)
408  else
409  ! current time is before mid point of current month
410  year2 = year; month2 = month
411  year1 = year; month1 = month-1
412  if (month1 < 1) then
413  year1 = year1-1; month1 = monyear
414  endif
415  if (year1>0) then
416  mid1 = days_in_month(set_date(year1,month1,2)) * halfday
417  else
418  ! this can happen if we are at the beginning of year 1. In this case
419  ! use December 0001 to calculate the duration of December 0000.
420  ! This should work for all calendars
421  mid1 = days_in_month(set_date(1,month1,2)) * halfday
422  endif
423  mid2 = mid_month
424  weight = real(cur_month + mid1) / real(mid1+mid2)
425  endif
426 
427  end subroutine time_interp_month

432 ! <OUT NAME="weight" TYPE="real"> </OUT>
433 ! <OUT NAME="year1" TYPE="integer"> </OUT>

438 ! <OUT NAME="day2" TYPE="integer"> </OUT>

440  !> @brief Calculates fractional time between mid points of consecutive days
441  subroutine time_interp_day ( Time, weight, year1, year2, month1, month2, day1, day2 )
442 
443  type(time_type), intent(in) :: Time
444  real , intent(out) :: weight
445  integer , intent(out) :: year1, year2, month1, month2, day1, day2
446 
447  integer :: year, month, day, hour, minute, second, sday
448 
449  if ( .not. module_is_initialized ) call time_interp_init()
450 
451  call get_date (time, year, month, day, hour, minute, second)
452 
453  ! time into current day in seconds
454  sday = second + secmin*minute + sechour*hour
455 
456  if ( sday >= halfday ) then
457  ! current time is after mid point of day
458  year1 = year; month1 = month; day1 = day
459  year2 = year; month2 = month; day2 = day + 1
460  weight = real(sday - halfday) / real(secday)
461 
462  if (day2 > days_in_month(time)) then
463  month2 = month2 + 1
464  day2 = 1
465  if (month2 > monyear) then
466  month2 = 1; year2 = year2+1
467  endif
468  endif
469  else
470  ! current time is before mid point of day
471  year2 = year; month2 = month; day2 = day
472  year1 = year; month1 = month; day1 = day - 1
473  weight = real(sday + halfday) / real(secday)
474 
475  if (day1 < 1) then
476  month1 = month1 - 1
477  if (month1 < 1) then
478  month1 = monyear; year1 = year1-1
479  endif
480  day1 = days_in_month(set_date(year1,month1,2))
481  endif
482  endif
483 
484  end subroutine time_interp_day

486 !#######################################################################
487 ! <SUBROUTINE NAME="time_interp_modulo" INTERFACE="time_interp">
488 ! <IN NAME="Time" TYPE="time_type" > </IN>
489 ! <IN NAME="Time_beg" TYPE="time_type"> </IN>
490 ! <IN NAME="Time_end" TYPE="time_type"> </IN>
491 ! <IN NAME="Timelist" TYPE="time_type" DIM="(:)"> </IN>
492 ! <IN NAME="correct_leap_year_inconsistency" TYPE="logical, optional" DEFAULT=".false.">
493 ! Turns on a kluge for an inconsistency which may occur in a special case.
494 ! When the modulo time period (i.e. Time_end - Time_beg) is a whole number of years
495 ! and is not a multiple of 4, and the calendar in use has leap years, then it is
496 ! likely that the interpolation will involve mapping a common year onto a leap year.
497 ! In this case it is often desirable, but not absolutely necessary, to use data for
498 ! Feb 28 of the leap year when it is mapped onto a common year.
499 ! To turn this on, set correct_leap_year_inconsistency=.true. </IN>
500 ! <OUT NAME="weight" TYPE="real"> </OUT>
501 ! <OUT NAME="index1" TYPE="real"> </OUT>
502 ! <OUT NAME="index2" TYPE="real"> </OUT>
503 ! </SUBROUTINE>
504 
505 subroutine time_interp_modulo(Time, Time_beg, Time_end, Timelist, weight, index1, index2, &
506  correct_leap_year_inconsistency, err_msg)
507 type(time_type), intent(in) :: Time, Time_beg, Time_end, Timelist(:)
508 real , intent(out) :: weight
509 integer , intent(out) :: index1, index2
510 logical, intent(in), optional :: correct_leap_year_inconsistency!< When true turns on a kluge for an
511  !! inconsistency which may occur in a special case.
512  !! When the modulo time period (i.e. Time_end - Time_beg) is a
513  !! whole number of years and is not a multiple of 4, and the calendar
514  !! in use has leap years, then it is likely that the interpolation
515  !! will involve mapping a common year onto a leap year. In this case
516  !! it is often desirable, but not absolutely necessary, to use data
517  !! for Feb 28 of the leap year when it is mapped onto a common year.
518 character(len=*), intent(out), optional :: err_msg
519 
520  type(time_type) :: Period, T
521  integer :: is, ie,i1,i2
522  integer :: ys,ms,ds,hs,mins,ss ! components of the starting date
523  integer :: ye,me,de,he,mine,se ! components of the ending date
524  integer :: yt,mt,dt,ht,mint,st ! components of the current date
525  integer :: dt1 ! temporary value for day
526  integer :: n ! size of Timelist
527  integer :: stdoutunit
528  logical :: correct_lyr, calendar_has_leap_years, do_the_lyr_correction
529 
530  if ( .not. module_is_initialized ) call time_interp_init
531  if( present(err_msg) ) err_msg = ''
532 
533  stdoutunit = stdout()
534  n = size(timelist)
535 
536  if (time_beg>=time_end) then
537  if(fms_error_handler('time_interp_modulo', &
538  'end of the specified time loop interval must be later than its beginning',err_msg)) return
539  endif
540 
541  calendar_has_leap_years = (get_calendar_type() == julian .or. get_calendar_type() == gregorian)
542 
543  period = time_end-time_beg ! period of the time axis
544 
545  if(present(correct_leap_year_inconsistency)) then
546  correct_lyr = correct_leap_year_inconsistency
547  else
548  correct_lyr = .false.
549  endif
550 
551  ! bring the requested time inside the specified time period
552  t = time
553 
554  do_the_lyr_correction = .false.
555 
556  ! Determine if the leap year correction needs to be done.
557  ! It never needs to be done unless 3 conditions are met:
558  ! 1) We are using a calendar with leap years
559  ! 2) optional argument correct_leap_year_inconsistency is present and equals .true.
560  ! 3) The modulo time period is an integer number of years
561  ! If all of these are true then set do_the_lyr_correction to .true.
562 
563  if(calendar_has_leap_years .and. correct_lyr) then
564  call get_date(time_beg,ys,ms,ds,hs,mins,ss)
565  call get_date(time_end,ye,me,de,he,mine,se)
566  if(ms==me.and.ds==de.and.hs==he.and.mins==mine.and.ss==se) then
567  ! whole number of years
568  do_the_lyr_correction = .true.
569  endif
570  endif

574  yt = ys+modulo(yt-ys,ye-ys)

576  ! If it is Feb 29, but we map into a common year, use Feb 28
577  if(mt==2.and.dt==29.and..not.leap_year(set_date(yt,1,1))) dt1=28

581  ! but before the starting date. So we shift it to the last year.
582  if(mt==2.and.dt==29.and..not.leap_year(set_date(ye,1,1))) dt=28
583  t = set_date(ye,mt,dt,ht,mint,st)
584  endif

588  enddo
589  do while ( t < time_beg )

591  enddo
592  endif
593 
594  ! find indices of the first and last records in the Timelist that are within
595  ! the requested time period.
596  if (time_end<=timelist(1).or.time_beg>=timelist(n)) then
597  if(get_calendar_type() == no_calendar) then
598  call print_time(time_beg, 'Time_beg' )
599  call print_time(time_end, 'Time_end' )
600  call print_time(timelist(1), 'Timelist(1)' )
601  call print_time(timelist(n), 'Timelist(n)' )
602  else
603  call print_date(time_beg, 'Time_beg' )
604  call print_date(time_end, 'Time_end' )
605  call print_date(timelist(1), 'Timelist(1)' )
606  call print_date(timelist(n), 'Timelist(n)' )
607  endif
608  write(stdoutunit,*)'where n = size(Timelist) =',n
609  if(fms_error_handler('time_interp_modulo', &
610  'the entire time list is outside the specified time loop interval',err_msg)) return
611  endif
612 
613  call bisect(timelist,time_beg,index1=i1,index2=i2)
614  if (i1 < 1) then
615  is = 1 ! Time_beg before lower boundary
616  else if (time_beg == timelist(i1)) then
617  is = i1 ! Time_beg right on the lower boundary
618  else
619  is = i2 ! Time_beg inside the interval or on upper boundary
620  endif
621  call bisect(timelist,time_end,index1=i1,index2=i2)
622  if (time_end > timelist(i1)) then
623  ie = i1
624  else if (time_end == timelist(i1)) then
625  if(time_beg == timelist(is)) then
626  ! Timelist includes time levels at both the lower and upper ends of the period.
627  ! The endpoints of Timelist specify the same point in the cycle.
628  ! This ambiguity is resolved by ignoring the last time level.
629  ie = i1-1
630  else
631  ie = i1
632  endif
633  else
634 ! This should never happen because bisect does not return i1 such that Time_end < Timelist(i1)
635  endif
636  if (is>=ie) then
637  if(get_calendar_type() == no_calendar) then
638  call print_time(time_beg, 'Time_beg =')
639  call print_time(time_end, 'Time_end =')
640  call print_time(timelist(1), 'Timelist(1)=')
641  call print_time(timelist(n), 'Timelist(n)=')
642  else
643  call print_date(time_beg, 'Time_beg =')
644  call print_date(time_end, 'Time_end =')
645  call print_date(timelist(1), 'Timelist(1)=')
646  call print_date(timelist(n), 'Timelist(n)=')
647  endif
648  write(stdoutunit,*)'where n = size(Timelist) =',n
649  write(stdoutunit,*)'is =',is,'ie =',ie
650  if(fms_error_handler('time_interp_modulo', &
651  'error in calculation of time list bounds within the specified time loop interval',err_msg)) return
652  endif
653 
654  ! handle special cases:
655  if( t>=timelist(ie) ) then
656  ! time is after the end of the portion of the time list within the requested period
657  index1 = ie; index2 = is
658  weight = (t-timelist(ie))//(period-(timelist(ie)-timelist(is)))
659  else if (t<timelist(is)) then
660  ! time is before the beginning of the portion of the time list within the requested period
661  index1 = ie; index2 = is
662  weight = 1.0-((timelist(is)-t)//(period-(timelist(ie)-timelist(is))))
663  else
664  call bisect(timelist,t,index1,index2)
665  weight = (t-timelist(index1)) // (timelist(index2)-timelist(index1))
666  endif
667 
668 end subroutine time_interp_modulo

670 !#######################################################################
671 !> Given an array of times in ascending order and a specific time returns
672 !! values of index1 and index2 such that the Timelist(index1)<=Time and
673 !! Time<=Timelist(index2), and index2=index1+1
674 !! index1=0, index2=1 or index=n, index2=n+1 are returned to indicate that
675 !! the time is out of range
676 subroutine bisect(Timelist,Time,index1,index2)
677  type(time_type) , intent(in) :: Timelist(:)
678  type(time_type) , intent(in) :: Time
679  integer, optional, intent(out) :: index1, index2
680 
681  integer :: i,il,iu,n,i1,i2
682 
683  n = size(timelist(:))
684 
685  if (time==timelist(1)) then
686  i1 = 1 ; i2 = 2
687  else if (time==timelist(n)) then
688  i1 = n ; i2 = n+1
689  else
690  il = 0; iu=n+1
691  do while(iu-il > 1)
692  i = (iu+il)/2
693  if(timelist(i) > time) then
694  iu = i
695  else
696  il = i
697  endif
698  enddo
699  i1 = il ; i2 = il+1
700  endif
701 
702  if(PRESENT(index1)) index1 = i1
703  if(PRESENT(index2)) index2 = i2

763 ! the modulo period, then the endpoints of Timelist specify the same point in the cycle.
764 ! This ambiguity is resolved by ignoring the last time level.

787  weight = (t-timelist(index1)) // (timelist(index2)-timelist(index1))
788 
789 ! time falls before starting list value
790  else if ( t < ts ) then

793  call time_list_error(ts,tserr)
794  call time_list_error(te,teerr)

798  err_msg)) return
799  deallocate(terr,tserr,teerr)
800  endif
801  td = te-ts
802  weight = 1. - ((ts-t) // (period-td))
803  index1 = n
804  index2 = 1
805 
806 ! time falls on ending list value
807  else if ( t == te ) then
808  if(perthlike_behavior) then
809  weight = 1.0
810  index1 = n-1
811  index2 = n
812  else
813  weight = 0.
814  index1 = n
815  if (mtime == none) then
816  index2 = n
817  else
818  index2 = 1
819  endif
820  endif
821 
822 ! time falls after ending list value
823  else if ( t > te ) then
824  if (mtime == none) then
825  call time_list_error(t,terr)
826  call time_list_error(ts,tserr)
827  call time_list_error(te,teerr)
828  if(fms_error_handler('time_interp_list',&

831  err_msg)) return
832  deallocate(terr,tserr,teerr)
833  endif
834  td = te-ts
835  weight = (t-te) // (period-td)
836  index1 = n
837  index2 = 1
838  endif
839 
840 end subroutine time_interp_list
