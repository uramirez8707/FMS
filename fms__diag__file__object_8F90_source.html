177 !< @brief Allocates the number of files and sets an ID based for each file
178 !! @return true if there are files allocated in the YAML object
179 logical function fms_diag_files_object_init (files_array)
180  class(fmsDiagFileContainer_type), allocatable, target, intent(inout) :: files_array (:) !< array of diag files
181  class(fmsDiagFile_type), pointer :: obj => null() !< Pointer for each member of the array
182  integer :: nFiles !< Number of files in the diag yaml
183  integer :: i !< Looping iterator
184  if (diag_yaml%has_diag_files()) then
185  nfiles = diag_yaml%size_diag_files()
186  allocate (files_array(nfiles))
187  set_ids_loop: do i= 1,nfiles
188  !> If the file has a sub_regional, define it as one and allocate the sub_axis_ids array.
189  !! This will be set in a add_axes
190  if (diag_yaml%diag_files(i)%has_file_sub_region()) then
191  allocate(subregionalfile_type :: files_array(i)%FMS_diag_file)
192  obj => files_array(i)%FMS_diag_file
193  select type (obj)
194  type is (subregionalfile_type)
195  allocate(obj%sub_axis_ids(max_axes))
196  obj%sub_axis_ids = diag_null
197  obj%write_on_this_pe = .false.
198  obj%is_subaxis_defined = .false.
199  obj%number_of_axis = 0
200  end select
201  else
202  allocate(fmsdiagfile_type::files_array(i)%FMS_diag_file)
203  obj => files_array(i)%FMS_diag_file
204  endif
205  !!
206  obj%diag_yaml_file => diag_yaml%diag_files(i)
207  obj%id = i
208  allocate(obj%field_ids(diag_yaml%diag_files(i)%size_file_varlist()))
209  allocate(obj%yaml_ids(diag_yaml%diag_files(i)%size_file_varlist()))
210  allocate(obj%field_registered(diag_yaml%diag_files(i)%size_file_varlist()))
211  !! Initialize the integer arrays
212  obj%field_ids = diag_not_registered
213  obj%yaml_ids = diag_not_registered
214  obj%field_registered = .false.
215  obj%num_registered_fields = 0
216 
217  !> These will be set in a set_file_domain
218  obj%type_of_domain = no_domain
219  obj%domain => null()
220 
221  !> This will be set in a add_axes
222  allocate(obj%axis_ids(max_axes))
223  obj%number_of_axis = 0
224 
225  !> Set the start_time of the file to the base_time and set up the *_output variables
226  obj%start_time = get_base_time()
227  obj%last_output = get_base_time()
228  obj%next_output = diag_time_inc(obj%start_time, obj%get_file_freq(), obj%get_file_frequnit())
229  obj%next_next_output = diag_time_inc(obj%next_output, obj%get_file_freq(), obj%get_file_frequnit())
230 
231  if (obj%has_file_new_file_freq()) then
232  obj%next_close = diag_time_inc(obj%start_time, obj%get_file_new_file_freq(), &
233  obj%get_file_new_file_freq_units())
234  else
235  obj%next_close = diag_time_inc(obj%start_time, very_large_file_freq, diag_days)
236  endif
237  obj%is_file_open = .false.
238 
239  if(obj%has_file_duration()) then
240  obj%no_more_data = diag_time_inc(obj%start_time, obj%get_file_duration(), &
241  obj%get_file_duration_units())
242  else
243  obj%no_more_data = diag_time_inc(obj%start_time, very_large_file_freq, diag_days)
244  endif
245 
246  obj%time_ops = .false.
247  obj%unlimited_dimension = 0
248  obj%is_static = obj%get_file_freq() .eq. -1
249 
250  nullify(obj)
251  enddo set_ids_loop
252  fms_diag_files_object_init = .true.
253  else
254  fms_diag_files_object_init = .false.
255 ! mpp_error("fms_diag_files_object_init: The diag_table.yaml file has not been correctly parsed.",&
256 ! FATAL)
257  endif
258 end function fms_diag_files_object_init

286 !> \brief Initializes a diurnal axis for a fileobj
287 !! \note This is going to be called for every variable in the file, if the variable is not a diurnal variable
288 !! it will do nothing. It only defined a diurnal axis once.
289 subroutine init_diurnal_axis(this, diag_axis, naxis, yaml_id)
290  class(fmsDiagFile_type), intent(inout) :: this !< The file object
291  class(fmsDiagAxisContainer_type), intent(inout) :: diag_axis(:) !< Array of diag_axis object
292  integer, intent(inout) :: naxis !< Number of diag_axis that heve been defined
293  integer, intent(in) :: yaml_id !< The ID to the variable's yaml
294 
295  integer :: i !< For do loops
296  type(diagYamlFilesVar_type), pointer :: field_yaml !< pointer to the yaml entry
297 
298  field_yaml => diag_yaml%get_diag_field_from_id(yaml_id)
299 
300  !< Go away if the file does not need a diurnal axis
301  if (.not. field_yaml%has_n_diurnal()) return
302 
303  !< Check if the diurnal axis is already defined for this number of diurnal samples
304  do i = 1, this%number_of_axis
305  select type(axis=>diag_axis(this%axis_ids(i))%axis)
306  type is (fmsdiagdiurnalaxis_type)
307  if(field_yaml%get_n_diurnal() .eq. axis%get_diurnal_axis_samples()) return
308  end select
309  end do
310 
311  !< If it is not already defined, define it
312  call define_diurnal_axis(diag_axis, naxis, field_yaml%get_n_diurnal(), .true.)
313  call define_diurnal_axis(diag_axis, naxis, field_yaml%get_n_diurnal(), .false.)
314 
315  !< Add it to the list of axis for the file
316  this%number_of_axis = this%number_of_axis + 1
317  this%axis_ids(this%number_of_axis) = naxis !< This is the diurnal axis edges
318 
319  this%number_of_axis = this%number_of_axis + 1
320  this%axis_ids(this%number_of_axis) = naxis - 1 !< This the diurnal axis
321 
322 end subroutine init_diurnal_axis

863  character(len=128) :: base_name !< The file name as defined in the yaml
864  !! without the wildcard definition
865  character(len=128) :: file_name !< The file name as it will be written to disk
866  character(len=128) :: temp_name !< Temp variable to store the file_name
867  character(len=128) :: start_date !< The start_time as a string that will be added to
868  !! the begining of the filename (start_date.filename)
869  character(len=128) :: suffix !< The current time as a string that will be added to
870  !! the end of filename

872  INTEGER :: year !< The year of the start_date

876  INTEGER :: minute !< The minute of the start_date
877  INTEGER :: second !< The second of the start_date

887  if (diag_file%is_file_open) return
888

895  allocate(fmsnetcdffile_t :: diag_file%fileobj)
896  is_regional = .true.
897  type is (fmsdiagfile_type)

902  case (two_d_domain)

908  endif

912 
913  !< If using the new_file_freq figure out what the name is based on the current time
914  if (diag_file%has_file_new_file_freq()) then
915  !< If using a wildcard file name (i.e ocn%4yr%2mo%2dy%2hr), get the basename (i.e ocn)
916  pos = index(diag_file_name, '%')

932  write (start_date, '(1I20.4, 2I2.2)') year, month, day
933 
934  file_name = trim(adjustl(start_date))//'.'//trim(file_name)
935  END IF
936 
937  file_name = trim(file_name)//".nc"
938 
939  !< If this is a regional file add the PE and the tile_number to the filename
940  if (is_regional) then
941  !< Get the pe number that will be appended to the end of the file
942  write(mype_string,'(I0.4)') mpp_pe()
943 
944  !< Add the tile number if appropriate
945  select type (domain)
946  type is (diagdomain2d_t)
947  temp_name = file_name
948  call get_mosaic_tile_file(temp_name, file_name, .true., domain%domain2)
949  end select
950 
951  file_name = trim(file_name)//"."//trim(mype_string)
952  endif
953 
954  !< Open the file!
955  select type (fileobj => diag_file%fileobj)
956  type is (fmsnetcdffile_t)
957  if (is_regional) then
958  if (.not. open_file(fileobj, file_name, "overwrite", pelist=(/mpp_pe()/))) &
959  &call mpp_error(fatal, "Error opening the file:"//file_name)
960  else
961  allocate(pes(mpp_npes()))
962  call mpp_get_current_pelist(pes)
963 
964  if (.not. open_file(fileobj, file_name, "overwrite", pelist=pes)) &
965  &call mpp_error(fatal, "Error opening the file:"//file_name)
966  endif
967  type is (fmsnetcdfdomainfile_t)
968  select type (domain)
969  type is (diagdomain2d_t)
970  if (.not. open_file(fileobj, file_name, "overwrite", domain%Domain2)) &
971  &call mpp_error(fatal, "Error opening the file:"//file_name)
972  end select
973  type is (fmsnetcdfunstructureddomainfile_t)
974  select type (domain)
975  type is (diagdomainug_t)
976  if (.not. open_file(fileobj, file_name, "overwrite", domain%DomainUG)) &
977  &call mpp_error(fatal, "Error opening the file:"//file_name)
978  end select
979  end select
980 
981  file_is_opened = .true.
982  diag_file%is_file_open = file_is_opened
983  domain => null()
984  diag_file => null()
