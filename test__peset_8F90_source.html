19 program test_peset
20  !> @author Jessica Liptak
21  !> @description Indirectly test the functionality of get_peset and expand_peset in mpp_util_sma.inc by passing
22  !! processor IDs among PEs via mpp_send and mpp_recv, then calling mpp_sync_self.
23  use mpp_mod, only : mpp_init, mpp_sync_self, mpp_exit, mpp_npes, &
24  mpp_error, fatal, mpp_pe, mpp_init_test_peset_allocated, &
25  mpp_broadcast, mpp_root_pe, mpp_declare_pelist, input_nml_file
26  implicit none
27 
28  integer :: npes !< The total number of PEs returned from mpp_npes
29  integer, dimension(:), allocatable :: pelist !< List of pes that will communicate
30  integer, dimension(:), allocatable :: numlist !< List of integers assigned to each pe
31  integer :: i, ierr, sum_id
32  !> Initialize mpp
33  call mpp_init(test_level=mpp_init_test_peset_allocated)
34  !> Get the total number of PEs
35  npes = mpp_npes()
36  !> create a pelist and a list of numbers to populate
37  allocate(pelist(npes))
38  allocate(numlist(npes))
39  pelist=(/(i,i=0,npes-1)/)
40  call mpp_declare_pelist(pelist)
41  ! sum used to verify that all PEs received all numbers
42  sum_id=sum(pelist)
43  numlist(:)=0
44  !> loop through number of pes and populate number list if the current pe is the root pe
45  do i = 1, npes
46  if (mpp_pe() .eq. mpp_root_pe()) &
47  numlist(i) = i-1
48  enddo
49  !> root pe sends the numlist to all of the other pes
50  call mpp_broadcast(numlist, npes, mpp_root_pe())
51  !> check that calling routines are complete
52  !! @note: mpp_sync_self calls get_peset; get pe_set calls expand_peset
53  call mpp_sync_self(pelist)
54  if (sum(numlist) .ne. sum_id) call mpp_error(fatal,"numlist sum does not match pelist sum")
55  if (allocated(numlist)) deallocate(numlist)
56  if (allocated(pelist)) deallocate(pelist)
57  !> Finalize mpp
58  call mpi_finalize(ierr)
59 end program test_peset
